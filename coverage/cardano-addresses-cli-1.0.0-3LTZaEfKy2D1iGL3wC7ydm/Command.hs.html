   90 -- | Run a given command
   91 run :: CLI -> IO ()
   92 run = handle prettyIOException . \case
   93     RecoveryPhrase sub -> RecoveryPhrase.run sub
   94     Key sub -> Key.run sub
   95     Address sub -> Address.run sub
   96 
   97 -- | Parse command line options and arguments
   98 parse :: IO CLI
   99 parse = customExecParser (prefs showHelpOnEmpty) cli

  101 -- | Enable ANSI colors on Windows and correct output buffering
  102 setup :: IO ()
  103 setup =
  104     mapM_ hSetup [stderr, stdout, stdin]
  105   where
  106     hSetup :: Handle -> IO ()
  107     hSetup h = do
  108       void $ hSupportsANSIWithoutEmulation h
  109       hSetBuffering h NoBuffering

  111 -- | Force the locale text encoding to UTF-8. This is needed because the CLI
  112 -- prints UTF-8 characters regardless of the @LANG@ environment variable or any
  113 -- other settings.
  114 --
  115 -- On Windows the current console code page is changed to UTF-8.
  116 withUtf8Encoding :: IO a -> IO a
  117 withUtf8Encoding action = withCP65001 (setUtf8EncodingHandles >> action)
  118   where
  119     setUtf8EncodingHandles :: IO ()
  120     setUtf8EncodingHandles = do
  121         utf8' <- mkTextEncoding "UTF-8//TRANSLIT"
  122         mapM_ (`hSetEncoding` utf8') [stdin, stdout, stderr]
