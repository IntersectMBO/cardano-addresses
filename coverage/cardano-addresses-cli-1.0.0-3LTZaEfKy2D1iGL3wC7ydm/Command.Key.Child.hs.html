   44 data Cmd = Child
   45     { encoding :: Encoding
   46     , scheme :: DerivationScheme
   47     , path :: DerivationPath
   48     } deriving (Show)
   49 
   50 mod :: (Cmd -> parent) -> Mod CommandFields parent
   51 mod liftCmd = command "child" $
   52     info (helper <*> fmap liftCmd parser) $ mempty
   53         <> progDesc "Derive child keys from a parent public/private key"
   54         <> footerDoc (Just $ string $ mconcat
   55             [ "The parent key is read from stdin."
   56             ])
   57   where
   58     parser = Child
   59         <$> encodingOpt [humanReadablePart|???|]
   60         <*> derivationSchemeOpt
   61         <*> derivationPathArg
   62 
   63 run :: Cmd -> IO ()
   64 run Child{encoding,path,scheme} = do
   65     xkey <- hGetXP__ stdin
   66 
   67     (child,encoding') <- case xkey of
   68         Left xpub -> do
   69             let ixs = castDerivationPath path
   70             case foldM (deriveXPub scheme) xpub ixs of
   71                 Nothing ->
   72                     fail
   73                         "Couldn't derive child key. If you're trying to derive \
   74                         \children on a PUBLIC key, you must use soft indexes only."
   75                 Just child ->
   76                     pure
   77                         ( xpubToBytes child
   78                         , fmap (const [humanReadablePart|xpub|] ) encoding
   79                         )
   80 
   81         Right xprv -> do
   82             let ixs = castDerivationPath path
   83             let Identity child = foldM (\k -> pure . deriveXPrv scheme k) xprv ixs
   84             pure
   85                 ( xprvToBytes child
   86                 , fmap (const [humanReadablePart|xprv|]) encoding
   87                 )
   88 
   89     hPutBytes stdout child encoding'
