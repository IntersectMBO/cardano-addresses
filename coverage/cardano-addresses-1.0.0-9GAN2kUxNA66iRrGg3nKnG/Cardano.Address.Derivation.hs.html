  122 -- | Construct an 'XPrv' from raw 'ByteString'.
  123 --
  124 -- @since 1.0.0
  125 xprvFromBytes :: ByteString -> Maybe XPrv
  126 xprvFromBytes bytes
  127     | BS.length bytes /= 96 = Nothing
  128     | otherwise = do
  129         let (prv, cc) = BS.splitAt 64 bytes
  130         pub <- ed25519ScalarMult (BS.take 32 prv)
  131         eitherToMaybe $ CC.xprv $ prv <> pub <> cc
  132   where
  133     ed25519ScalarMult :: ByteString -> Maybe ByteString
  134     ed25519ScalarMult bs = do
  135         scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
  136         pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar
  137 
  138 -- | Convert an 'XPrv' to a raw 'ByteString'.
  139 --
  140 -- @since 1.0.0
  141 xprvToBytes :: XPrv -> ByteString
  142 xprvToBytes =
  143     stripPub . CC.unXPrv
  144   where
  145     -- From  < xprv | pub | cc >
  146     -- â†³ To  < xprv |     | cc >
  147     stripPub :: ByteString -> ByteString
  148     stripPub xprv' = prv <> chainCode
  149       where
  150         (prv, rest) = BS.splitAt 64 xprv'
  151         (_pub, chainCode) = BS.splitAt 32 rest

  234 -- | Key Depth in the derivation path, according to BIP-0039 / BIP-0044
  235 --
  236 -- @
  237 -- root | purpose' | cointype' | account' | change | address@
  238 -- 0th      1st         2nd        3rd        4th      5th
  239 -- @
  240 --
  241 -- We do not manipulate purpose, cointype and change paths directly, so there
  242 -- are no constructors for these.
  243 --
  244 -- @since 1.0.0
  245 data Depth = RootK | AccountK | AddressK
  246 
  247 -- | Marker for addresses type engaged. We want to handle three cases here.
  248 -- The first two are pertinent to UTxO accounting
  249 -- and the last one handles rewards from participation in staking.
  250 -- (a) external chain is used for addresses that are part of the 'advertised'
  251 --     targets of a given transaction
  252 -- (b) internal change is for addresses used to handle the change of a
  253 --     the transaction within a given wallet
  254 -- (c) the addresses for a reward (chimeric) account
  255 data AccountingStyle
  256     = UTxOExternal
  257     | UTxOInternal
  258     | MutableAccount
  259     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  260 
  261 instance NFData AccountingStyle
  262 
  263 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  264 -- to assign the wrong index to the corresponding constructor (by swapping
  265 -- around the constructor above for instance).
  266 instance Enum AccountingStyle where
  267     toEnum = \case
  268         0 -> UTxOExternal
  269         1 -> UTxOInternal
  270         2 -> MutableAccount
  271         _ -> error "AccountingStyle.toEnum: bad argument"
  272     fromEnum = \case
  273         UTxOExternal -> 0
  274         UTxOInternal -> 1
  275         MutableAccount -> 2

  277 -- | A derivation index, with phantom-types to disambiguate derivation type.
  278 --
  279 -- @
  280 -- let accountIx = Index 'Hardened 'AccountK
  281 -- let addressIx = Index 'Soft 'AddressK
  282 -- @
  283 --
  284 -- @since 1.0.0
  285 newtype Index (derivationType :: DerivationType) (depth :: Depth) = Index
  286     { getIndex :: Word32 }
  287     deriving stock (Generic, Show, Eq, Ord)
  288 
  289 instance NFData (Index derivationType depth)
  290 
  291 instance Bounded (Index 'Hardened depth) where
  292     minBound = Index 0x80000000
  293     maxBound = Index maxBound
  294 
  295 instance Bounded (Index 'Soft depth) where
  296     minBound = Index minBound
  297     maxBound = let (Index ix) = minBound @(Index 'Hardened _) in Index (ix - 1)
  298 
  299 instance Bounded (Index 'WholeDomain depth) where
  300     minBound = Index minBound
  301     maxBound = Index maxBound
  302 
  303 instance Enum (Index 'Hardened depth) where
  304     fromEnum (Index ix) = fromIntegral ix
  305     toEnum ix
  306         | Index (fromIntegral ix) < minBound @(Index 'Hardened _) =
  307             error "Index@Hardened.toEnum: bad argument"
  308         | otherwise =
  309             Index (fromIntegral ix)
  310 
  311 instance Enum (Index 'Soft depth) where
  312     fromEnum (Index ix) = fromIntegral ix
  313     toEnum ix
  314         | Index (fromIntegral ix) > maxBound @(Index 'Soft _) =
  315             error "Index@Soft.toEnum: bad argument"
  316         | otherwise =
  317             Index (fromIntegral ix)
  318 
  319 instance Enum (Index 'WholeDomain depth) where
  320     fromEnum (Index ix) = fromIntegral ix
  321     toEnum ix
  322         | Index (fromIntegral ix) > maxBound @(Index 'WholeDomain _) =
  323             error "Index@WholeDomain.toEnum: bad argument"
  324         | otherwise =
  325             Index (fromIntegral ix)
  326 
  327 instance Buildable (Index derivationType depth) where
  328     build (Index ix) = fromString (show ix)

  331 -- | Type of derivation that should be used with the given indexes.
  332 --
  333 -- In theory, we should only consider two derivation types: soft and hard.
  334 --
  335 -- However, historically, addresses in Cardano used to be generated across both
  336 -- the soft and the hard domain. We therefore introduce a 'WholeDomain' derivation
  337 -- type that is the exact union of `Hardened` and `Soft`.
  338 --
  339 -- @since 1.0.0
  340 data DerivationType = Hardened | Soft | WholeDomain
  341
