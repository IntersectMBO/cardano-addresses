  156 -- $overview
  157 --
  158 -- This module provides an implementation of:
  159 --
  160 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Shelley master keys from mnemonic sentences
  161 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  162 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  163 -- - 'Cardano.Address.PaymentAddress': for constructing payment addresses from a address public key
  164 -- - 'Cardano.Address.DelegationAddress': for constructing delegation addresses from address and staking public keys

  166 -- | A cryptographic key for sequential-scheme address derivation, with
  167 -- phantom-types to disambiguate key types.
  168 --
  169 -- @
  170 -- let rootPrivateKey = Shelley 'RootK XPrv
  171 -- let accountPubKey  = Shelley 'AccountK XPub
  172 -- let addressPubKey  = Shelley 'AddressK XPub
  173 -- @
  174 --
  175 -- @since 2.0.0
  176 newtype Shelley (depth :: Depth) key = Shelley
  177     { getKey :: key
  178         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  179         --
  180         -- @since 1.0.0
  181     }
  182     deriving stock (Generic, Show, Eq)
  183 
  184 deriving instance (Functor (Shelley depth))
  185 instance (NFData key) => NFData (Shelley depth key)
  186 
  187 data Role
  188     = UTxOExternal
  189     | UTxOInternal
  190     | Stake
  191     | Multisig
  192     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  193 
  194 instance NFData Role
  195 
  196 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  197 -- to assign the wrong index to the corresponding constructor (by swapping
  198 -- around the constructor above for instance).
  199 instance Enum Role where
  200     toEnum = \case
  201         0 -> UTxOExternal
  202         1 -> UTxOInternal
  203         2 -> Stake
  204         3 -> Multisig
  205         _ -> error "Role.toEnum: bad argument"
  206     fromEnum = \case
  207         UTxOExternal -> 0
  208         UTxOInternal -> 1
  209         Stake -> 2
  210         Multisig -> 3

  213 -- Key Derivation
  214 --
  215 -- $keyDerivation
  216 --
  217 -- === Generating a root key from 'SomeMnemonic'
  218 -- > :set -XOverloadedStrings
  219 -- > :set -XTypeApplications
  220 -- > :set -XDataKinds
  221 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  222 -- >
  223 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  224 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  225 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shelley 'RootK XPrv
  226 --
  227 -- === Deriving child keys
  228 --
  229 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  230 --
  231 -- > let accIx = toEnum 0x80000000
  232 -- > let acctK = deriveAccountPrivateKey rootK accIx
  233 -- >
  234 -- > let addIx = toEnum 0x00000014
  235 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  236 --
  237 -- > let stakeK = deriveStakingPrivateKey acctK
  238 
  239 instance Internal.GenMasterKey Shelley where
  240     type SecondFactor Shelley = ScrubbedBytes
  241 
  242     genMasterKeyFromXPrv = liftXPrv
  243     genMasterKeyFromMnemonic fstFactor sndFactor =
  244         Shelley $ generateNew seedValidated sndFactor
  245         where
  246             seed  = someMnemonicToBytes fstFactor
  247             seedValidated = assert
  248                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  249                 seed

  251 instance Internal.HardDerivation Shelley where
  252     type AccountIndexDerivationType Shelley = 'Hardened
  253     type AddressIndexDerivationType Shelley = 'Soft
  254     type WithRole Shelley = Role
  255 
  256     deriveAccountPrivateKey (Shelley rootXPrv) accIx =
  257         let
  258             purposeIx =
  259                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  260             coinTypeIx =
  261                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  262             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  263                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  264             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  265                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  266             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  267                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  268         in
  269             Shelley acctXPrv
  270 
  271     deriveAddressPrivateKey (Shelley accXPrv) role addrIx =
  272         let
  273             roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  274             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  275                 deriveXPrv DerivationScheme2 accXPrv roleCode
  276             addrXPrv = -- lvl5 derivation; soft derivation of address index
  277                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  278         in
  279             Shelley addrXPrv
  280 
  281 instance Internal.SoftDerivation Shelley where
  282     deriveAddressPublicKey (Shelley accXPub) role addrIx =
  283         fromMaybe errWrongIndex $ do
  284             let roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  285             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  286                 deriveXPub DerivationScheme2 accXPub roleCode
  287             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  288                 deriveXPub DerivationScheme2 changeXPub addrIx
  289             return $ Shelley addrXPub
  290       where
  291         errWrongIndex = error $
  292             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  293             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  294             \either a programmer error, or, we may have reached the maximum \
  295             \number of addresses for a given wallet."

  355 -- Re-export from 'Cardano.Address.Derivation' to have it documented specialized in Haddock
  356 --
  357 -- | Derive a staking key for a corresponding 'AccountK'. Note that wallet
  358 -- software are by convention only using one staking key per account, and always
  359 -- the first account (with index 0').
  360 --
  361 -- Deriving staking keys for something else than the initial account is not
  362 -- recommended and can lead to incompatibility with existing wallet softwares
  363 -- (Daedalus, Yoroi, Adalite...).

  365 -- @since 2.0.0
  366 deriveStakingPrivateKey
  367     :: Shelley 'AccountK XPrv
  368     -> Shelley 'StakingK XPrv
  369 deriveStakingPrivateKey accXPrv =
  370     let (Shelley stakeXPrv) =
  371             deriveAddressPrivateKey accXPrv Stake (minBound @(Index 'Soft _))
  372     in Shelley stakeXPrv

  428 -- > let ptr = ChainPointer 123 1 2
  429 -- > bech32 $ pointerAddress tag (toXPub <$> addrK) ptr

  432 instance Internal.StakeAddress Shelley where
  433     stakeAddress discrimination k = unsafeMkAddress $
  434         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  435             putWord8 firstByte
  436             putByteString (blake2b224 k)
  437       where
  438           -- First 4 bits are `1110` for keyhash28 reward account address.
  439           -- Next 4 bits are network discriminator.
  440           -- `1110 0000` is 224 in decimal.
  441           firstByte =
  442             let addrType = 224
  443                 netTagLimit = 16
  444             in addrType + invariantNetworkTag netTagLimit (networkTag @Shelley discrimination)
  445           expectedLength =
  446               let headerSizeBytes = 1
  447               in headerSizeBytes + hashSize
  448 
  449 instance Internal.PaymentAddress Shelley where
  450     paymentAddress discrimination k = unsafeMkAddress $
  451         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  452             putWord8 firstByte
  453             putByteString (blake2b224 k)
  454       where
  455           -- we use here the fact that payment address stands for what is named
  456           -- as enterprise address, ie., address carrying no stake rights. For
  457           -- rationale why we may need such addresses refer to delegation
  458           -- specification - Section 3.2.3. What is important here is that the
  459           -- address is composed of discrimination byte and 28 bytes hashed public key.
  460           -- Moreover, it was decided that first 4 bits for enterprise address
  461           -- will be `0110`. The next for 4 bits are reserved for network discriminator.
  462           -- `0110 0000` is 96 in decimal.
  463           firstByte =
  464             let addrType = 96
  465                 netTagLimit = 16
  466             in addrType + invariantNetworkTag netTagLimit (networkTag @Shelley discrimination)
  467           expectedLength =
  468               let headerSizeBytes = 1
  469               in headerSizeBytes + hashSize

  491 -- | Analyze an 'Address' to know whether it's a Shelley address or not.
  492 --
  493 -- Throws 'AddrError' if it's not a valid Shelley address, or a ready-to-print
  494 -- string giving details about the 'Address'.
  495 --
  496 -- @since 2.0.0
  497 inspectShelleyAddress
  498     :: (Alternative m, MonadThrow m)
  499     => Maybe XPub
  500     -> Address
  501     -> m Json.Value
  502 inspectShelleyAddress mRootPub addr
  503     | BS.length bytes < 1 + hashSize = throwM (ShWrongInputSize (BS.length bytes))
  504     | otherwise =
  505         let
  506             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  507             addrType = fstByte .&. 0b11110000
  508             network  = fstByte .&. 0b00001111
  509         in
  510             case addrType of
  511                -- 0000: base address: keyhash28,keyhash28
  512                 0b00000000 | BS.length rest == 2 * hashSize ->
  513                     pure $ Json.object
  514                         [ "address_style"     .= Json.String "Shelley"
  515                         , "stake_reference"   .= Json.String "by value"
  516                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  517                         , "stake_key_hash"    .= base16 (BS.drop hashSize rest)
  518                         , "network_tag"       .= network
  519                         ]
  520                -- 0001: base address: scripthash32,keyhash28
  521                 0b00010000 | BS.length rest == hashSize + hashSize ->
  522                     pure $ Json.object
  523                         [ "address_style"     .= Json.String "Shelley"
  524                         , "stake_reference"   .= Json.String "by value"
  525                         , "script_hash"       .= base16 (BS.take hashSize rest)
  526                         , "stake_key_hash"    .= base16 (BS.drop hashSize rest)
  527                         , "network_tag"       .= network
  528                         ]
  529                -- 0010: base address: keyhash28,scripthash32
  530                 0b00100000 | BS.length rest == hashSize + hashSize ->
  531                     pure $ Json.object
  532                         [ "address_style"     .= Json.String "Shelley"
  533                         , "stake_reference"   .= Json.String "by value"
  534                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  535                         , "stake_script_hash" .= base16 (BS.drop hashSize rest)
  536                         , "network_tag"       .= network
  537                         ]
  538                -- 0011: base address: scripthash32,scripthash32
  539                 0b00110000 | BS.length rest == 2 * hashSize ->
  540                     pure $ Json.object
  541                         [ "address_style"     .= Json.String "Shelley"
  542                         , "stake_reference"   .= Json.String "by value"
  543                         , "script_hash"       .= base16 (BS.take hashSize rest)
  544                         , "stake_script_hash" .= base16 (BS.drop hashSize rest)
  545                         , "network_tag"       .= network
  546                         ]
  547                -- 0100: pointer address: keyhash28, 3 variable length uint
  548                -- TODO Could fo something better for pointer and try decoding
  549                --      the pointer
  550                 0b01000000 | BS.length rest > hashSize -> do
  551                     ptr <- getPtr (BS.drop hashSize rest)
  552                     pure $ Json.object
  553                         [ "address_style"     .= Json.String "Shelley"
  554                         , "stake_reference"   .= Json.String "by pointer"
  555                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  556                         , "pointer"           .= ptrToJSON ptr
  557                         , "network_tag"       .= network
  558                         ]
  559                -- 0101: pointer address: scripthash32, 3 variable length uint
  560                -- TODO Could fo something better for pointer and try decoding
  561                --      the pointer
  562                 0b01010000 | BS.length rest > hashSize -> do
  563                     ptr <- getPtr (BS.drop hashSize rest)
  564                     pure $ Json.object
  565                         [ "address_style"     .= Json.String "Shelley"
  566                         , "stake_reference"   .= Json.String "by pointer"
  567                         , "script_hash"       .= base16 (BS.take hashSize rest)
  568                         , "pointer"           .= ptrToJSON ptr
  569                         , "network_tag"       .= network
  570                         ]
  571                -- 0110: enterprise address: keyhash28
  572                 0b01100000 | BS.length rest == hashSize ->
  573                     pure $ Json.object
  574                         [ "address_style"     .= Json.String "Shelley"
  575                         , "stake_reference"   .= Json.String "none"
  576                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  577                         , "network_tag"       .= network
  578                         ]
  579                -- 0111: enterprise address: scripthash32
  580                 0b01110000 | BS.length rest == hashSize ->
  581                     pure $ Json.object
  582                         [ "address_style"     .= Json.String "Shelley"
  583                         , "stake_reference"   .= Json.String "none"
  584                         , "script_hash"       .= base16 (BS.take hashSize rest)
  585                         , "network_tag"       .= network
  586                         ]
  587                -- 1000: byron address
  588                 0b10000000 ->
  589                     inspectIcarusAddress addr <|> inspectByronAddress mRootPub addr
  590                -- 1110: reward account: keyhash28
  591                 0b11100000 | BS.length rest == hashSize ->
  592                     pure $ Json.object
  593                         [ "address_style"     .= Json.String "Shelley"
  594                         , "stake_reference"   .= Json.String "by value"
  595                         , "stake_key_hash"    .= base16 (BS.take hashSize rest)
  596                         , "network_tag"       .= network
  597                         ]
  598                -- 1111: reward account: scripthash32
  599                 0b11110000 | BS.length rest == hashSize ->
  600                     pure $ Json.object
  601                         [ "address_style"     .= Json.String "Shelley"
  602                         , "stake_reference"   .= Json.String "by value"
  603                         , "script_hash"       .= base16 (BS.take hashSize rest)
  604                         , "network_tag"       .= network
  605                         ]
  606                 _ -> throwM ShUnknownAddrType
  607   where
  608     bytes  = unAddress addr
  609     base16 = T.unpack . T.decodeUtf8 . encode EBase16

  611     ptrToJSON :: ChainPointer -> Json.Value
  612     ptrToJSON ChainPointer{slotNum,transactionIndex,outputIndex} = Json.object
  613         [ "slot_num" .= slotNum

  641 -- | Extend an existing payment 'Address' to make it a delegation address.
  642 --
  643 -- @since 2.0.0
  644 extendAddress
  645     :: Address
  646     -> Either (Shelley 'StakingK XPub) ChainPointer
  647     -> Either ErrExtendAddress Address
  648 extendAddress addr stakeReference = do
  649     when (isNothing (inspectShelleyAddress Nothing addr)) $
  650         Left $ ErrInvalidAddressStyle "Given address isn't a Shelley address"
  651 
  652     let bytes = unAddress addr
  653     let (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  654 
  655     when ((fstByte .&. 0b11110000) /= 0b01100000) $ do
  656         Left $ ErrInvalidAddressType "Only payment addresses can be extended"
  657 
  658     case stakeReference of
  659         Left stakingKey -> do
  660             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  661                 putWord8 $ fstByte .&. 0b00001111
  662                 putByteString rest
  663                 putByteString . blake2b224 $ stakingKey
  664 
  665         Right pointer -> do
  666             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  667                 putWord8 $ fstByte .&. 0b01001111
  668                 putByteString rest
  669                 putPointer pointer
  670   where
  671     putPointer (ChainPointer a b c) = do
  672         putVariableLengthNat a
  673         putVariableLengthNat b
  674         putVariableLengthNat c

  749 -- | Construct 'NetworkDiscriminant' for Cardano 'Shelley' from a number.

  751 -- 'MkNetworkDiscriminantError' is thrown.

  755     :: Integer

  761 -- | Retrieve the network discriminant of a given 'Address'.

  764 -- @since 2.0.0
  765 inspectNetworkDiscriminant
  766     :: Address
  767     -> Maybe (NetworkDiscriminant Shelley)
  768 inspectNetworkDiscriminant addr =
  769     inspectShelleyAddress Nothing addr $>
  770         let
  771             bytes = unAddress addr

  841 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
