   37 data ShelleyAddrError
   38     = ShUnknownAddrType
   39     | ShWrongInputSize Int -- the actual size
   40     | ShPtrRetrieveError String -- human readable error of underlying operation
   41 
   42 deriving instance Show ShelleyAddrError
   43 
   44 instance Exception ShelleyAddrError where
   45   displayException = prettyShError
   46 
   47 prettyShError :: ShelleyAddrError -> String
   48 prettyShError ShUnknownAddrType = "Unknown address type"
   49 prettyShError (ShWrongInputSize i) = format "Wrong input size of {}" i
   50 prettyShError (ShPtrRetrieveError s) = format "Failed to retrieve pointer (underlying errors was: {})" s
   51

   53 data ByronAddrError
   54     = BrMissingExpectedDerivationPath
   55     | forall e . (Exception e, Show e) => BrDeserialiseError e
   56     | BrFailedToDecryptPath
   57 
   58 deriving instance Show ByronAddrError
   59 
   60 instance Exception ByronAddrError where
   61   displayException = prettyBrError
   62 
   63 prettyBrError :: ByronAddrError -> String
   64 prettyBrError BrMissingExpectedDerivationPath = "Missing expected derivation path"
   65 prettyBrError (BrDeserialiseError e) = format "Deserialisation error (was: {})" (show e)
   66 prettyBrError BrFailedToDecryptPath = "Failed to decrypt derivation path"
   67 
   68 
   69 data IcarusAddrError
   70     = IcUnexpectedDerivationPath
   71     | forall e . (Exception e, Show e) => IcDeserialiseError e
   72 
   73 deriving instance Show IcarusAddrError
   74 
   75 instance Exception IcarusAddrError where
   76   displayException = prettyIcError
   77 
   78 prettyIcError :: IcarusAddrError -> String
   79 prettyIcError IcUnexpectedDerivationPath = "Unexpected derivation path"
   80 prettyIcError (IcDeserialiseError e) = format "Deserialisation error (was: {})" (show e)
