   39 -- | The script embodies combination of signing keys that need to be met to make
   40 -- it valid. We assume here that the script could
   41 -- delivered from standard input. The examples below are self-explanatory:
   42 --
   43 -- 1. requiring signature
   44 -- 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe
   45 --
   46 -- 2. 'any' for signature required
   47 -- any [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   48 --
   49 -- 3. 'all' signatures required
   50 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   51 --
   52 -- 4. 'at_least' 1 signature required
   53 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   54 --
   55 -- 5. Nested script are supported
   56 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]]
   57 --
   58 -- Parser is insensitive to whitespaces.
   59 --
   60 -- @since 3.0.0
   61 scriptParser :: ReadP Script
   62 scriptParser =
   63     requireAllOfParser <++
   64     requireAnyOfParser <++
   65     requireAtLeastOfParser <++
   66     requireSignatureOfParser
   67 
   68 requireSignatureOfParser :: ReadP Script
   69 requireSignatureOfParser = do
   70     P.skipSpaces
   71     verKeyH <- P.munch1 (\c -> isDigit c || isLetter c)
   72     case length verKeyH of
   73         56 -> do
   74             let (Right bytes) = fromBase16 $ T.encodeUtf8 $ T.pack verKeyH
   75             let (Just keyHash) = keyHashFromBytes bytes
   76             return $ RequireSignatureOf keyHash
   77         len ->
   78             fail $ "Verification key hash should be 28 bytes, but received "
   79             <> show len <> " bytes."
   80 
   81 requireAllOfParser :: ReadP Script
   82 requireAllOfParser = do
   83     P.skipSpaces
   84     _identifier <- P.string "all"
   85     RequireAllOf <$> commonPart
   86 
   87 requireAnyOfParser :: ReadP Script
   88 requireAnyOfParser = do
   89     P.skipSpaces
   90     _identifier <- P.string "any"
   91     RequireAnyOf <$> commonPart
   92 
   93 requireAtLeastOfParser :: ReadP Script
   94 requireAtLeastOfParser = do
   95     P.skipSpaces
   96     _identifier <- P.string "at_least"
   97     RequireMOf <$> naturalParser <*> commonPart
   98 
   99 naturalParser :: ReadP Word8
  100 naturalParser = do
  101     P.skipSpaces
  102     fromInteger . read <$> P.munch1 isDigit
  103 
  104 commonPart :: ReadP [Script]
  105 commonPart = do
  106     P.skipSpaces
  107     _open <- P.string "["
  108     P.skipSpaces
  109     content <- P.sepBy1 scriptParser (P.string ",")
  110     P.skipSpaces
  111     _close <- P.string "]"
  112     P.skipSpaces
  113     return content
