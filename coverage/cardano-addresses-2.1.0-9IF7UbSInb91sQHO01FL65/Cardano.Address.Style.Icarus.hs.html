  129 -- $overview
  130 --
  131 -- This module provides an implementation of:
  132 --
  133 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Icarus master keys from mnemonic sentences
  134 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  135 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  136 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  137 --
  138 -- We call 'Icarus' addresses the new format of Cardano addresses which came
  139 -- after 'Cardano.Address.Style.Byron.Byron'. This is the format initially used in /Yoroi/
  140 -- and now also used by /Daedalus/.
  141 
  142 -- | A cryptographic key for sequential-scheme address derivation, with
  143 -- phantom-types to disambiguate key types.
  144 --
  145 -- @
  146 -- let rootPrivateKey = Icarus 'RootK XPrv
  147 -- let accountPubKey  = Icarus 'AccountK XPub
  148 -- let addressPubKey  = Icarus 'AddressK XPub
  149 -- @
  150 --
  151 -- @since 1.0.0
  152 newtype Icarus (depth :: Depth) key = Icarus
  153     { getKey :: key
  154         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  155         --
  156         -- @since 1.0.0
  157     }
  158     deriving stock (Generic, Show, Eq)
  159 
  160 deriving instance (Functor (Icarus depth))
  161 instance (NFData key) => NFData (Icarus depth key)
  162 
  163 data Role
  164     = UTxOExternal
  165     | UTxOInternal
  166     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  167 
  168 instance NFData Role
  169 
  170 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  171 -- to assign the wrong index to the corresponding constructor (by swapping
  172 -- around the constructor above for instance).
  173 instance Enum Role where
  174     toEnum = \case
  175         0 -> UTxOExternal
  176         1 -> UTxOInternal
  177         _ -> error "Role.toEnum: bad argument"
  178     fromEnum = \case
  179         UTxOExternal -> 0
  180         UTxOInternal -> 1

  182 --
  183 -- Key Derivation
  184 --
  185 -- $keyDerivation
  186 --
  187 -- === Generating a root key from 'SomeMnemonic'
  188 -- > :set -XOverloadedStrings
  189 -- > :set -XTypeApplications
  190 -- > :set -XDataKinds
  191 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  192 -- >
  193 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  194 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  195 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Icarus 'RootK XPrv
  196 --
  197 -- === Deriving child keys
  198 --
  199 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  200 --
  201 -- > let accIx = toEnum 0x80000000
  202 -- > let acctK = deriveAccountPrivateKey rootK accIx
  203 -- >
  204 -- > let addIx = toEnum 0x00000014
  205 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  206 
  207 instance Internal.GenMasterKey Icarus where
  208     type SecondFactor Icarus = ScrubbedBytes
  209 
  210     genMasterKeyFromXPrv = liftXPrv
  211     genMasterKeyFromMnemonic (SomeMnemonic mw) sndFactor =
  212         let
  213             seed  = entropyToBytes $ mnemonicToEntropy mw
  214             seedValidated = assert
  215                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  216                 seed
  217         in Icarus $ generateNew seedValidated sndFactor

  219 instance Internal.HardDerivation Icarus where
  220     type AccountIndexDerivationType Icarus = 'Hardened
  221     type AddressIndexDerivationType Icarus = 'Soft
  222     type WithRole Icarus = Role
  223 
  224     deriveAccountPrivateKey (Icarus rootXPrv) accIx =
  225         let
  226             purposeIx =
  227                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  228             coinTypeIx =
  229                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  230             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  231                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  232             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  233                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  234             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  235                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  236         in
  237             Icarus acctXPrv
  238 
  239     deriveAddressPrivateKey (Icarus accXPrv) role addrIx =
  240         let
  241             roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  242             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  243                 deriveXPrv DerivationScheme2 accXPrv roleCode
  244             addrXPrv = -- lvl5 derivation; soft derivation of address index
  245                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  246         in
  247             Icarus addrXPrv
  248 
  249 instance Internal.SoftDerivation Icarus where
  250     deriveAddressPublicKey (Icarus accXPub) role addrIx =
  251         fromMaybe errWrongIndex $ do
  252             let roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  253             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  254                 deriveXPub DerivationScheme2 accXPub roleCode
  255             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  256                 deriveXPub DerivationScheme2 changeXPub addrIx
  257             return $ Icarus addrXPub
  258       where
  259         errWrongIndex = error $
  260             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  261             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  262             \either a programmer error, or, we may have reached the maximum \
  263             \number of addresses for a given wallet."

  324 --
  325 -- Addresses
  326 --
  327 -- $addresses
  328 -- === Generating a 'PaymentAddress'
  329 --
  330 -- > import Cardano.Address ( bech32 )
  331 -- > import Cardano.Address.Derivation ( toXPub(..) )
  332 -- >
  333 -- > bech32 $ paymentAddress icarusMainnet (toXPub <$> addrK)
  334 -- > "addr1vxpfffuj3zkp5g7ct6h4va89caxx9ayq2gvkyfvww48sdncxsce5t"
  335 --
  336 -- | Analyze an 'Address' to know whether it's an Icarus address or not.
  337 -- Throws 'IcarusAddrError' if the address isn't a byron address, or return a
  338 -- structured JSON that gives information about an address.
  339 --
  340 -- @since 2.0.0
  341 inspectIcarusAddress :: MonadThrow m => Address -> m Json.Value
  342 inspectIcarusAddress addr = do
  343     payload <- either (throwM . IcDeserialiseError) pure
  344         $ CBOR.deserialiseCbor CBOR.decodeAddressPayload bytes
  345     (root, attrs) <- either (throwM . IcDeserialiseError) pure
  346         $ CBOR.deserialiseCbor decodePayload payload
  347     when (elem 1 . fmap fst $ attrs) $ throwM IcUnexpectedDerivationPath
  348     ntwrk <- either (throwM . IcDeserialiseError) pure
  349         $ CBOR.deserialiseCbor CBOR.decodeProtocolMagicAttr payload
  350     pure $ Json.object
  351         [ "address_style"   .= Json.String "Icarus"
  352         , "stake_reference" .= Json.String "none"
  353         , "address_root"    .= T.unpack (T.decodeUtf8 $ encode EBase16 root)
  354         , "network_tag"     .= maybe Json.Null toJSON ntwrk
  355         ]
  356   where
  357     bytes :: ByteString
  358     bytes = unAddress addr
  359 
  360     decodePayload :: forall s. CBOR.Decoder s (ByteString, [(Word8, ByteString)])
  361     decodePayload = do
  362         _ <- CBOR.decodeListLenCanonicalOf 3
  363         root <- CBOR.decodeBytes
  364         (root,) <$> CBOR.decodeAllAttributes

  449 -- Purpose is a constant set to 44' (or 0x8000002C) following the original
  450 -- BIP-44 specification.
  451 --
  452 -- It indicates that the subtree of this node is used according to this
  453 -- specification.
  454 --
  455 -- Hardened derivation is used at this level.
  456 purposeIndex :: Word32
  457 purposeIndex = 0x8000002C
  458 
  459 -- One master node (seed) can be used for unlimited number of independent
  460 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  461 -- same space for various cryptocoins has some disadvantages.
  462 --
  463 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  464 -- addresses across cryptocoins and improving privacy issues.
  465 --
  466 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  467 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  468 -- Lovelace.
  469 --
  470 -- Hardened derivation is used at this level.
  471 coinTypeIndex :: Word32
  472 coinTypeIndex = 0x80000717

  474 -- The minimum seed length for 'generateKeyFromMnemonic' and 'unsafeGenerateKeyFromMnemonic'.
  475 minSeedLengthBytes :: Int
  476 minSeedLengthBytes = 16

  478 -- Hardware Ledger devices generates keys from mnemonic using a different
  479 -- approach (different from the rest of Cardano).
  480 --
  481 -- It is a combination of:
  482 --
  483 -- - [SLIP 0010](https://github.com/satoshilabs/slips/blob/master/slip-0010.md)
  484 -- - [BIP 0032](https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki)
  485 -- - [BIP 0039](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki)
  486 -- - [RFC 8032](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  487 -- - What seems to be arbitrary changes from Ledger regarding the calculation of
  488 --   the initial chain code and generation of the root private key.
  489 unsafeGenerateKeyFromHardwareLedger
  490     :: SomeMnemonic
  491         -- ^ The root mnemonic
  492     -> Icarus 'RootK XPrv
  493 unsafeGenerateKeyFromHardwareLedger (SomeMnemonic mw) = unsafeFromRight $ do
  494     let seed = pbkdf2HmacSha512
  495             $ T.encodeUtf8
  496             $ T.intercalate " "
  497             $ mnemonicToText mw
  498 
  499     -- NOTE
  500     -- SLIP-0010 refers to `iR` as the chain code. Here however, the chain code
  501     -- is obtained as a hash of the initial seed whereas iR is used to make part
  502     -- of the root private key itself.
  503     let cc = hmacSha256 (BS.pack [1] <> seed)
  504     let (iL, iR) = first pruneBuffer $ hashRepeatedly seed
  505 
  506     prv <- maybe (Left "invalid xprv") pure $ xprvFromBytes $ iL <> iR <> cc
  507     pure $ Icarus prv
  508   where
  509     -- Errors yielded in the body of 'unsafeGenerateKeyFromHardwareLedger' are
  510     -- programmer errors (out-of-range byte buffer access or, invalid length for
  511     -- cryptographic operations). Therefore, we throw badly if we encounter any.
  512     unsafeFromRight :: Either String a -> a
  513     unsafeFromRight = either error id
  514 
  515     -- This is the algorithm described in SLIP 0010 for master key generation
  516     -- with an extra step to discard _some_ of the potential private keys. Why
  517     -- this extra step remains a mystery as of today.
  518     --
  519     --      1. Generate a seed byte sequence S of 512 bits according to BIP-0039.
  520     --         (done in a previous step, passed as argument).
  521     --
  522     --      2. Calculate I = HMAC-SHA512(Key = "ed25519 seed", Data = S)
  523     --
  524     --      3. Split I into two 32-byte sequences, IL and IR.
  525     --
  526     -- extra *******************************************************************
  527     -- *                                                                       *
  528     -- *    3.5 If the third highest bit of the last byte of IL is not zero    *
  529     -- *        S = I and go back to step 2.                                   *
  530     -- *                                                                       *
  531     -- *************************************************************************
  532     --
  533     --      4. Use parse256(IL) as master secret key, and IR as master chain code.
  534     hashRepeatedly :: ByteString -> (ByteString, ByteString)
  535     hashRepeatedly bytes = case BS.splitAt 32 (hmacSha512 bytes) of
  536         (iL, iR) | isInvalidKey iL -> hashRepeatedly (iL <> iR)
  537         (iL, iR) -> (iL, iR)
  538       where
  539         isInvalidKey k = testBit (k `BS.index` 31) 5
  540 
  541     -- - Clear the lowest 3 bits of the first byte
  542     -- - Clear the highest bit of the last byte
  543     -- - Set the second highest bit of the last byte
  544     --
  545     -- As described in [RFC 8032 - 5.1.5](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  546     pruneBuffer :: ByteString -> ByteString
  547     pruneBuffer bytes =
  548         let
  549             (firstByte, rest) = fromMaybe (error "pruneBuffer: no first byte") $
  550                 BS.uncons bytes
  551 
  552             (rest', lastByte) = fromMaybe (error "pruneBuffer: no last byte") $
  553                 BS.unsnoc rest
  554 
  555             firstPruned = firstByte
  556                 & (`clearBit` 0)
  557                 & (`clearBit` 1)
  558                 & (`clearBit` 2)
  559 
  560             lastPruned = lastByte
  561                 & (`setBit` 6)
  562                 & (`clearBit` 7)
  563         in
  564             (firstPruned `BS.cons` BS.snoc rest' lastPruned)
  565 
  566     -- As described in [BIP 0039 - From Mnemonic to Seed](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki#from-mnemonic-to-seed)
  567     pbkdf2HmacSha512 :: ByteString -> ByteString
  568     pbkdf2HmacSha512 bytes = PBKDF2.generate
  569         (PBKDF2.prfHMAC SHA512)
  570         (PBKDF2.Parameters 2048 64)
  571         bytes
  572         ("mnemonic" :: ByteString)
  573 
  574     hmacSha256 :: ByteString -> ByteString
  575     hmacSha256 =
  576         BA.convert @(HMAC SHA256) . hmac salt
  577 
  578     -- As described in [SLIP 0010 - Master Key Generation](https://github.com/satoshilabs/slips/blob/master/slip-0010.md#master-key-generation)
  579     hmacSha512 :: ByteString -> ByteString
  580     hmacSha512 =
  581         BA.convert @(HMAC SHA512) . hmac salt
  582 
  583     salt :: ByteString
  584     salt = "ed25519 seed"
