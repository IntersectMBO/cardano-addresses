   49 -- | A 'Script' type represents multi signature script. The script embodies conditions
   50 -- that need to be satisfied to make it valid.
   51 --
   52 -- @since 3.0.0
   53 data Script =
   54       RequireSignatureOf KeyHash
   55     | RequireAllOf [Script]
   56     | RequireAnyOf [Script]
   57     | RequireMOf Word8 [Script]
   58     deriving stock (Generic, Show, Eq)
   59 
   60 instance NFData Script
   61 
   62 -- | A 'KeyHash' type represents verification key hash
   63 -- that participate in building MultisigScript. The hash is expected to have size of
   64 -- 28-byte.
   65 --
   66 -- @since 3.0.0
   67 newtype KeyHash = KeyHash ByteString
   68     deriving (Generic, Show, Eq)
   69 
   70 instance NFData KeyHash
   71 
   72 -- | Apply Blake2b224 hashing on Shelley 'XPub'.
   73 -- This results in 28-byte hash.
   74 --
   75 -- @since 3.0.0
   76 hashKey :: Shelley key XPub -> KeyHash
   77 hashKey = KeyHash . blake2b224
   78 
   79 -- | Construct an 'KeyHash' from raw 'ByteString' (28 bytes).
   80 --
   81 -- @since 3.0.0
   82 keyHashFromBytes :: ByteString -> Maybe KeyHash
   83 keyHashFromBytes bytes
   84     | BS.length bytes /= hashSize = Nothing
   85     | otherwise = Just $ KeyHash bytes

   87 -- | Validate 'Script'
   88 --
   89 -- @since 3.0.0
   90 validateScript :: Script -> Bool
   91 validateScript (RequireSignatureOf _) = True
   92 validateScript (RequireAllOf content) =
   93     not (L.null content) && L.all ((== True) . validateScript) content
   94 validateScript (RequireAnyOf content) =
   95     not (L.null content) && L.all ((== True) . validateScript) content
   96 validateScript (RequireMOf m content) =
   97     m > 0 && L.length content >= fromIntegral (toInteger m) &&
   98     L.all ((== True) . validateScript) content

  100 toCBOR' :: Script -> CBOR.Encoding
  101 toCBOR' (RequireSignatureOf (KeyHash verKeyHash)) =
  102     encodeMultiscriptCtr 0 2 <> CBOR.encodeBytes verKeyHash
  103 toCBOR' (RequireAllOf contents) =
  104     encodeMultiscriptCtr 1 2 <> encodeFoldable toCBOR' contents
  105 toCBOR' (RequireAnyOf contents) =
  106     encodeMultiscriptCtr 2 2 <> encodeFoldable toCBOR' contents
  107 toCBOR' (RequireMOf m contents) =
  108     encodeMultiscriptCtr 3 3 <> CBOR.encodeInt (fromInteger $ toInteger m)
  109     <> encodeFoldable toCBOR' contents

  111 -- | This function realizes what cardano-node's `Api.serialiseToCBOR script` realizes
  112 -- This is basically doing the symbolically following:
  113 -- toCBOR [0,multisigScript]
  114 toCBOR :: Script -> ByteString
  115 toCBOR script =
  116     CBOR.toStrictByteString $ encodeMultisigBeginning <> toCBOR' script
  117 
  118 encodeMultisigBeginning :: CBOR.Encoding
  119 encodeMultisigBeginning = CBOR.encodeListLen 2 <> CBOR.encodeWord8 0
  120 
  121 encodeMultiscriptCtr :: Word -> Word -> CBOR.Encoding
  122 encodeMultiscriptCtr ctrIndex listLen =
  123     CBOR.encodeListLen listLen <> CBOR.encodeWord ctrIndex
  124 
  125 encodeFoldable :: (Foldable f) => (a -> CBOR.Encoding) -> f a -> CBOR.Encoding
  126 encodeFoldable encode xs = wrapArray len contents
  127   where
  128     (len, contents) = foldl' go (0, mempty) xs
  129     go (!l, !enc) next = (l + 1, enc <> encode next)
  130 
  131     wrapArray :: Word -> CBOR.Encoding -> CBOR.Encoding
  132     wrapArray len' contents' =
  133         if len' <= 23
  134         then CBOR.encodeListLen len' <> contents'
  135         else CBOR.encodeListLenIndef <> contents' <> CBOR.encodeBreak
  136 
  137 -- | This function realizes what cardano-node's
  138 -- `Api.serialiseToRawBytes $ Api.scriptHash script` realizes
  139 -- This is basically doing the symbolically the following (using symbols from toCBOR):
  140 -- digest $ (nativeMultisigTag <> toCBOR multisigScript )
  141 toScriptHash :: Script -> ByteString
  142 toScriptHash script = digest $ nativeMultiSigTag <> toCBOR'' script
  143   where
  144       nativeMultiSigTag :: ByteString
  145       nativeMultiSigTag = "\00"
  146 
  147       digest = BA.convert . hash @_ @Blake2b_224
  148 
  149       toCBOR'' :: Script -> ByteString
  150       toCBOR'' script' = CBOR.toStrictByteString $ toCBOR' script'
