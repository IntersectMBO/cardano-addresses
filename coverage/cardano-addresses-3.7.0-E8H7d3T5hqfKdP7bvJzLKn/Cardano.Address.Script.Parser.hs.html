   42 -- | Run 'scriptParser' on string input.
   43 --
   44 -- @since 3.0.0
   45 scriptFromString :: String -> Either ErrValidateScript (Script KeyHash)
   46 scriptFromString str =
   47     case readP_to_S scriptParser str of
   48          [(script, "")] -> pure script
   49          _ -> Left Malformed
   50 
   51 -- | The script embodies combination of signing keys that need to be met to make
   52 -- it valid. We assume here that the script could
   53 -- delivered from standard input. The examples below are self-explanatory:
   54 --
   55 -- 1. requiring signature
   56 -- 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe
   57 --
   58 -- 2. 'any' for signature required
   59 -- any [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   60 --
   61 -- 3. 'all' signatures required
   62 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   63 --
   64 -- 4. 'at_least' 1 signature required
   65 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   66 --
   67 -- 5. Nested script are supported
   68 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]]
   69 -- 6. 1 signature required after slot number 120
   70 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, active_from 120]
   71 -- 7. 1 signature required until slot number 150
   72 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, active_until 150]
   73 -- 8. 1 signature required in slot interval <145, 150)
   74 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, active_from 145, active_until 150]
   75 --
   76 -- Parser is insensitive to whitespaces.
   77 --
   78 -- @since 3.0.0
   79 scriptParser :: ReadP (Script KeyHash)
   80 scriptParser =
   81     requireAllOfParser <++
   82     requireAnyOfParser <++
   83     requireAtLeastOfParser <++
   84     requireSignatureOfParser <++
   85     activeFromSlotParser <++
   86     activeUntilSlotParser
   87 
   88 requireSignatureOfParser :: ReadP (Script KeyHash)
   89 requireSignatureOfParser = do
   90     P.skipSpaces
   91     verKeyStr <- P.munch1 (\c -> isDigit c || isLetter c || c == '_')
   92     case keyHashFromText (T.pack verKeyStr) of
   93         Left e  -> fail (prettyErrKeyHashFromText e)
   94         Right h -> pure (RequireSignatureOf h)
   95 
   96 requireAllOfParser :: ReadP (Script KeyHash)
   97 requireAllOfParser = do
   98     P.skipSpaces
   99     _identifier <- P.string "all"
  100     RequireAllOf <$> commonPart
  101 
  102 requireAnyOfParser :: ReadP (Script KeyHash)
  103 requireAnyOfParser = do
  104     P.skipSpaces
  105     _identifier <- P.string "any"
  106     RequireAnyOf <$> commonPart
  107 
  108 requireAtLeastOfParser :: ReadP (Script KeyHash)
  109 requireAtLeastOfParser = do
  110     P.skipSpaces
  111     _identifier <- P.string "at_least"
  112     RequireSomeOf <$> naturalParser <*> commonPart
  113 
  114 activeFromSlotParser :: ReadP (Script KeyHash)
  115 activeFromSlotParser = do
  116     P.skipSpaces
  117     _identifier <- P.string "active_from"
  118     ActiveFromSlot <$> slotParser
  119 
  120 activeUntilSlotParser :: ReadP (Script KeyHash)
  121 activeUntilSlotParser = do
  122     P.skipSpaces
  123     _identifier <- P.string "active_until"
  124     ActiveUntilSlot <$> slotParser
  125 
  126 naturalParser :: ReadP Word8
  127 naturalParser = do
  128     P.skipSpaces
  129     fromInteger . read <$> P.munch1 isDigit
  130 
  131 slotParser :: ReadP Natural
  132 slotParser = do
  133     P.skipSpaces
  134     fromInteger . read <$> P.munch1 isDigit
  135 
  136 commonPart :: ReadP [(Script KeyHash)]
  137 commonPart = do
  138     P.skipSpaces
  139     _open <- P.string "["
  140     P.skipSpaces
  141     content <- P.sepBy scriptParser (P.string ",")
  142     P.skipSpaces
  143     _close <- P.string "]"
  144     P.skipSpaces
  145     return content
