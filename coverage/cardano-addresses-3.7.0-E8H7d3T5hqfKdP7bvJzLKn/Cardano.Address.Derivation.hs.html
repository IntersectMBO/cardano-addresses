  117 -- | An opaque type representing an extended private key.
  118 --
  119 -- __Properties:__
  120 --
  121 -- ===== Roundtripping
  122 --
  123 -- @forall xprv. 'xprvFromBytes' ('xprvToBytes' xprv) == 'Just' xprv@
  124 --
  125 -- ===== Chain Code Invariance
  126 --
  127 -- @forall xprv. 'xprvChainCode' xprv == 'xpubChainCode' ('toXPub' xprv)@
  128 --
  129 -- ===== Public Key Signature
  130 --
  131 -- @forall xprv msg. 'verify' ('toXPub' xprv) msg ('sign' xprv msg) == 'True'@
  132 --
  133 -- @since 1.0.0
  134 type XPrv = CC.XPrv
  135 
  136 -- | An opaque type representing an extended public key.

  138 -- __Properties:__

  152 -- | Construct an 'XPub' from raw 'ByteString' (64 bytes).
  153 --
  154 -- @since 1.0.0
  155 xpubFromBytes :: ByteString -> Maybe XPub
  156 xpubFromBytes = eitherToMaybe . CC.xpub

  158 -- | Convert an 'XPub' to a raw 'ByteString' (64 bytes).
  159 --
  160 -- @since 1.0.0
  161 xpubToBytes :: XPub -> ByteString
  162 xpubToBytes xpub = xpubPublicKey xpub <> xpubChainCode xpub
  163 
  164 -- | Extract the public key from an 'XPub' as a raw 'ByteString' (32 bytes).
  165 --
  166 -- @since 2.0.0
  167 xpubPublicKey :: XPub -> ByteString
  168 xpubPublicKey (CC.XPub pub _cc) = pub
  169 
  170 -- | Extract the chain code from an 'XPub' as a raw 'ByteString' (32 bytes).
  171 --
  172 -- @since 2.0.0
  173 xpubChainCode :: XPub -> ByteString
  174 xpubChainCode (CC.XPub _pub (CC.ChainCode cc)) = cc
  175 
  176 -- | Construct an 'XPrv' from raw 'ByteString' (96 bytes).
  177 --
  178 -- @since 1.0.0
  179 xprvFromBytes :: ByteString -> Maybe XPrv
  180 xprvFromBytes bytes
  181     | BS.length bytes /= 96 = Nothing
  182     | otherwise = do
  183         let (prv, cc) = BS.splitAt 64 bytes
  184         pub <- ed25519ScalarMult (BS.take 32 prv)
  185         eitherToMaybe $ CC.xprv $ prv <> pub <> cc
  186   where
  187     ed25519ScalarMult :: ByteString -> Maybe ByteString
  188     ed25519ScalarMult bs = do
  189         scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
  190         pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar
  191 
  192 -- From  < xprv | pub | cc >
  193 -- â†³ To  < xprv |     | cc >
  194 --
  195 -- | Convert an 'XPrv' to a raw 'ByteString' (96 bytes).
  196 --
  197 -- @since 1.0.0
  198 xprvToBytes :: XPrv -> ByteString
  199 xprvToBytes xprv =
  200     xprvPrivateKey xprv <> xprvChainCode xprv

  202 -- | Extract the private key from an 'XPrv' as a raw 'ByteString' (64 bytes).

  204 -- @since 2.0.0
  205 xprvPrivateKey :: XPrv -> ByteString
  206 xprvPrivateKey = BS.take 64 . CC.unXPrv

  208 -- | Extract the chain code from an 'XPrv' as a raw 'ByteString' (32 bytes).

  210 -- @since 2.0.0
  211 xprvChainCode :: XPrv -> ByteString
  212 xprvChainCode = BS.drop 96 . CC.unXPrv

  308 -- | Key Depth in the derivation path, according to BIP-0039 / BIP-0044
  309 --
  310 -- @
  311 -- root | purpose' | cointype' | account' | role | address@
  312 -- 0th      1st         2nd        3rd       4th     5th
  313 -- @
  314 --
  315 -- We do not manipulate purpose, cointype and change paths directly, so there
  316 -- are no constructors for these.
  317 --
  318 -- @since 1.0.0
  319 data Depth = RootK | AccountK | PaymentK | DelegationK | ScriptK
  320 
  321 -- | A derivation index, with phantom-types to disambiguate derivation type.
  322 --
  323 -- @
  324 -- let accountIx = Index 'Hardened 'AccountK
  325 -- let addressIx = Index 'Soft 'PaymentK
  326 -- @
  327 --
  328 -- @since 1.0.0
  329 newtype Index (derivationType :: DerivationType) (depth :: Depth) = Index
  330     { indexToWord32 :: Word32
  331     -- ^ Get the index as a 'Word32'
  332     -- @since 3.3.0
  333     }
  334     deriving stock (Generic, Show, Eq, Ord)
  335 
  336 instance NFData (Index derivationType depth)
  337 
  338 instance Bounded (Index 'Hardened depth) where
  339     minBound = Index 0x80000000
  340     maxBound = Index maxBound
  341 
  342 instance Bounded (Index 'Soft depth) where
  343     minBound = Index minBound
  344     maxBound = let (Index ix) = minBound @(Index 'Hardened _) in Index (ix - 1)
  345 
  346 instance Bounded (Index 'WholeDomain depth) where
  347     minBound = Index minBound
  348     maxBound = Index maxBound
  349 
  350 -- Construct an 'Index' from any Word32 value, without any validation, for
  351 -- internal use only.
  352 --
  353 -- Always use 'indexFromWord32' or 'wholeDomainIndex' instead of this function.
  354 unsafeMkIndex :: Word32 -> Index ty depth
  355 unsafeMkIndex = Index
  356 
  357 -- | Construct derivation path indices from raw 'Word32' values.
  358 indexFromWord32
  359     :: forall ix derivationType depth.
  360        (ix ~ Index derivationType depth, Bounded ix)
  361     => Word32 -> Maybe ix
  362 indexFromWord32 ix
  363     | ix >= indexToWord32 (minBound @ix) && ix <= indexToWord32 (maxBound @ix) =
  364         Just (Index ix)
  365     | otherwise =
  366         Nothing
  367 
  368 -- | Increment an index, if possible.
  369 --
  370 -- @since 3.3.0
  371 nextIndex
  372     :: forall ix derivationType depth.
  373        (ix ~ Index derivationType depth, Bounded ix)
  374     => ix -> Maybe ix
  375 nextIndex (Index ix) = indexFromWord32 (ix + 1)
  376 
  377 -- | Constructs a full domain 'Index'. This can't fail, unlike 'fromWord32'.
  378 --
  379 -- @since 3.3.0
  380 wholeDomainIndex :: Word32 -> Index 'WholeDomain depth
  381 wholeDomainIndex = Index

  383 -- | Upcasts an 'Index' to one with the full 'Word32' domain.
  384 --
  385 -- @since 3.3.0
  386 coerceWholeDomainIndex :: Index ty depth0 -> Index 'WholeDomain depth1

  393 -- | Type of derivation that should be used with the given indexes.
  394 --
  395 -- In theory, we should only consider two derivation types: soft and hard.
  396 --
  397 -- However, historically, addresses in Cardano used to be generated across both
  398 -- the soft and the hard domain. We therefore introduce a 'WholeDomain' derivation
  399 -- type that is the exact union of `Hardened` and `Soft`.
  400 --
  401 -- @since 1.0.0
  402 data DerivationType = Hardened | Soft | WholeDomain
  403
