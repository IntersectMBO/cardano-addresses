    6 module Command.Key.Hash
    7     ( Cmd (..)
    8     , mod
    9     , run
   10     ) where
   11 
   12 import Prelude hiding
   13     ( mod )
   14 
   15 import Cardano.Address.Derivation
   16     ( hashCredential )
   17 import Codec.Binary.Encoding
   18     ( AbstractEncoding (..) )
   19 import Control.Monad
   20     ( when )
   21 import Options.Applicative
   22     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   23 import Options.Applicative.Help.Pretty
   24     ( string )
   25 import System.IO
   26     ( stdin, stdout )
   27 import System.IO.Extra
   28     ( hGetBech32, hPutBytes )
   29 
   30 import qualified Cardano.Codec.Bech32.Prefixes as CIP5
   31 import qualified Data.ByteString as BS
   32 
   33 data Cmd = Hash
   34     deriving (Show)
   35 
   36 mod :: (Cmd -> parent) -> Mod CommandFields parent
   37 mod liftCmd = command "hash" $
   38     info (helper <*> fmap liftCmd parser) $ mempty
   39         <> progDesc "Get the hash of a public key"
   40         <> footerDoc (Just $ string $ mconcat
   41             [ "The public key is read from stdin."
   42             ])
   43   where
   44     parser = pure Hash
   45 
   46 run :: Cmd -> IO ()
   47 run Hash = do
   48     (hrp, bytes) <- hGetBech32 stdin allowedPrefixes
   49     guardBytes hrp bytes
   50     hPutBytes stdout (hashCredential $ BS.take 32 bytes) (EBech32 $ prefixFor hrp)
   51   where
   52     allowedPrefixes =
   53         [ CIP5.addr_vk
   54         , CIP5.addr_xvk
   55         , CIP5.stake_vk
   56         , CIP5.stake_xvk
   57         , CIP5.script_vk
   58         , CIP5.script_xvk
   59         ]
   60 
   61     guardBytes hrp bytes
   62         | hrp `elem` [CIP5.addr_xvk, CIP5.stake_xvk, CIP5.script_xvk] = do
   63             when (BS.length bytes /= 64) $
   64                 fail "data should be a 32-byte public key with a 32-byte chain-code appended"
   65 
   66         | otherwise = do
   67             when (BS.length bytes /= 32) $
   68                 fail "data should be a 32-byte public key."
   69 
   70     prefixFor hrp
   71         | hrp == CIP5.addr_vk    = CIP5.addr_vkh
   72         | hrp == CIP5.addr_xvk   = CIP5.addr_vkh
   73 
   74         | hrp == CIP5.stake_vk   = CIP5.stake_vkh
   75         | hrp == CIP5.stake_xvk  = CIP5.stake_vkh
   76 
   77         | hrp == CIP5.script_vk  = CIP5.script_vkh
   78         | hrp == CIP5.script_xvk = CIP5.script_vkh
   79 
   80         | otherwise = error "impossible: pattern-match not coverage all cases."
