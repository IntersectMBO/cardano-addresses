   52 data ScriptError
   53     = MalformedScript
   54     | InvalidScript ErrValidateScript
   55     deriving (Eq, Show)
   56 
   57 prettyScriptError :: ScriptError -> String
   58 prettyScriptError = \case
   59     MalformedScript ->
   60         "Parsing of the script failed. The script should be composed of nested \
   61         \lists, and the verification keys should be either encoded as base16, \
   62         \bech32 or base58."
   63     InvalidScript e ->
   64         prettyErrValidateScript e
   65 
   66 --
   67 -- Applicative Parsers
   68 --
   69 
   70 scriptArg :: Parser Script
   71 scriptArg = argument (eitherReader reader) $ mempty
   72     <> metavar "SCRIPT"
   73     <> help
   74         "Script string."
   75   where
   76     reader :: String -> Either String Script
   77     reader
   78         = left prettyScriptError
   79         . maybe (Left malformedScript) (\s -> bimap InvalidScript (const s) $ validateScript s)
   80         . scriptFromString
   81       where
   82         malformedScript = MalformedScript
   83 
   84 stakeCredentialArg  :: String -> Parser (Credential 'StakingK)
   85 stakeCredentialArg str =
   86     (StakingFromKey . liftXPub <$> xpubOpt "from-key" str)
   87     <|>
   88     (StakingFromScript <$> scriptHashArg str)
   89 
   90 scriptHashArg :: String -> Parser ScriptHash
   91 scriptHashArg helpDoc =
   92     option (eitherReader reader) $ mempty
   93         <> long "from-script"
   94         <> metavar "SCRIPT_HASH"
   95         <> help helpDoc
   96   where
   97     reader :: String -> Either String ScriptHash
   98     reader str = do
   99         bytes <- encodingReader str
  100         case scriptHashFromBytes bytes of
  101             Just scriptHash -> pure scriptHash
  102             Nothing -> Left "Failed to convert bytes into a valid script hash."
