    7 module Command.Key.Public
    8     ( Cmd (..)
    9     , mod
   10     , run
   11     ) where
   12 
   13 import Prelude hiding
   14     ( mod )
   15 
   16 import Cardano.Address.Derivation
   17     ( toXPub, xpubPublicKey, xpubToBytes )
   18 import Codec.Binary.Encoding
   19     ( AbstractEncoding (..) )
   20 import Options.Applicative
   21     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   22 import Options.Applicative.Help.Pretty
   23     ( string )
   24 import Options.Applicative.Public
   25     ( PublicType (..), publicOpt )
   26 import System.IO
   27     ( stdin, stdout )
   28 import System.IO.Extra
   29     ( hGetXPrv, hPutBytes )
   30 
   31 import qualified Cardano.Codec.Bech32.Prefixes as CIP5
   32 
   33 newtype Cmd = Public
   34     { chainCode :: PublicType
   35     } deriving (Show)
   36 
   37 mod :: (Cmd -> parent) -> Mod CommandFields parent
   38 mod liftCmd = command "public" $
   39     info (helper <*> fmap liftCmd parser) $ mempty
   40         <> progDesc "Get the public counterpart of a private key"
   41         <> footerDoc (Just $ string $ mconcat
   42             [ "The private key is read from stdin."
   43             , "To get extended public key pass '--with-chain-code'."
   44             , "To get public key pass '--without-chain-code'."
   45             ])
   46   where
   47     parser = Public
   48         <$> publicOpt
   49 
   50 run :: Cmd -> IO ()
   51 run Public{chainCode} = do
   52     (hrp, xprv) <- hGetXPrv stdin allowedPrefixes
   53     let xpub = toXPub xprv
   54     let bytes = case chainCode of
   55             WithChainCode    -> xpubToBytes xpub
   56             WithoutChainCode -> xpubPublicKey xpub
   57     hPutBytes stdout bytes (EBech32 $ prefixToPublic chainCode hrp)
   58   where
   59     allowedPrefixes =
   60         [ CIP5.root_xsk
   61         , CIP5.acct_xsk
   62         , CIP5.addr_xsk
   63         , CIP5.stake_xsk
   64         , CIP5.script_xsk
   65         ]
   66 
   67     prefixToPublic WithChainCode hrp
   68         | hrp == CIP5.root_xsk   = CIP5.root_xvk
   69         | hrp == CIP5.acct_xsk   = CIP5.acct_xvk
   70         | hrp == CIP5.addr_xsk   = CIP5.addr_xvk
   71         | hrp == CIP5.stake_xsk  = CIP5.stake_xvk
   72         | hrp == CIP5.script_xsk = CIP5.script_xvk
   73 
   74     prefixToPublic WithoutChainCode hrp
   75         | hrp == CIP5.root_xsk   = CIP5.root_vk
   76         | hrp == CIP5.acct_xsk   = CIP5.acct_vk
   77         | hrp == CIP5.addr_xsk   = CIP5.addr_vk
   78         | hrp == CIP5.stake_xsk  = CIP5.stake_vk
   79         | hrp == CIP5.script_xsk = CIP5.script_vk
   80 
   81     prefixToPublic _ _ =
   82         error "impossible: pattern-match not covering all cases."
