    8 module Command.Key.Hash
    9     ( Cmd (..)
   10     , mod
   11     , run
   12     ) where
   13 
   14 import Prelude hiding
   15     ( mod )
   16 
   17 import Cardano.Address.Style.Shelley
   18     ( hashKey, liftXPub )
   19 import Cardano.Script
   20     ( KeyHash (..) )
   21 import Codec.Binary.Bech32.TH
   22     ( humanReadablePart )
   23 import Options.Applicative
   24     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   25 import Options.Applicative.Encoding
   26     ( Encoding, encodingOpt )
   27 import Options.Applicative.Help.Pretty
   28     ( string )
   29 import System.IO
   30     ( stdin, stdout )
   31 import System.IO.Extra
   32     ( hGetXPub, hPutBytes )
   33 
   34 newtype Cmd = Hash
   35     { encoding :: Encoding
   36     } deriving (Show)
   37 
   38 mod :: (Cmd -> parent) -> Mod CommandFields parent
   39 mod liftCmd = command "hash" $
   40     info (helper <*> fmap liftCmd parser) $ mempty
   41         <> progDesc "Get the hash of a public key"
   42         <> footerDoc (Just $ string $ mconcat
   43             [ "The public key is read from stdin."
   44             ])
   45   where
   46     parser = Hash
   47         <$> encodingOpt [humanReadablePart|xpub_hash|]
   48 
   49 run :: Cmd -> IO ()
   50 run Hash{encoding} = do
   51     xpub <- hGetXPub stdin
   52     let (KeyHash hash) = hashKey (liftXPub xpub)
   53     hPutBytes stdout hash encoding
