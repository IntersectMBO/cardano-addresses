   45 data Cmd = Cmd
   46     {  credentialType :: CredentialType
   47     ,  networkTag :: NetworkTag
   48     } deriving (Show)
   49 
   50 mod :: (Cmd -> parent) -> Mod CommandFields parent
   51 mod liftCmd = command "stake" $
   52     info (helper <*> fmap liftCmd parser) $ mempty
   53         <> progDesc "Create a stake address"
   54         <> header "Create a stake address \
   55             \that references a staking key (1-1)."
   56         <> footerDoc (Just $ vsep
   57             [ string "The public key is read from stdin."
   58             , string ""
   59             , string "Example:"
   60             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   61             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   62             , indent 2 $ string ""
   63             , indent 2 $ bold $ string "$ cat root.prv \\"
   64             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/2/0 > stake.prv"
   65             , indent 2 $ string ""
   66             , indent 2 $ bold $ string "$ cat stake.prv \\"
   67             , indent 4 $ bold $ string $ "| "<>progName<>" key public \\"
   68             , indent 4 $ bold $ string $ "| "<>progName<>" address stake --network-tag testnet"
   69             , indent 2 $ string "stake_test1uzp7swuxjx7wmpkkvat8kpgrmjl8ze0dj9lytn25qv2tm4g6n5c35"
   70             ])
   71   where
   72     parser = Cmd
   73         <$> credentialOpt
   74         <*> networkTagOpt Shelley
   75 
   76 run :: Cmd -> IO ()
   77 run Cmd{networkTag,credentialType} = do
   78     case (mkNetworkDiscriminant . fromIntegral . unNetworkTag) networkTag of
   79         Left e -> die (fmt $ build e)
   80         Right discriminant -> do
   81             addr <- case credentialType of
   82                 CredentialFromKey -> do
   83                     xpub <- hGetXPub stdin
   84                     let credential = StakingFromKey $ Shelley.liftXPub xpub
   85                     pure $ unsafeFromRight $ Shelley.stakeAddress discriminant credential
   86                 CredentialFromScript -> do
   87                     scriptHash <- hGetScriptHash stdin
   88                     let credential = StakingFromScript scriptHash
   89                     pure $ unsafeFromRight $ Shelley.stakeAddress discriminant credential
   90             B8.hPutStr stdout $ T.encodeUtf8 $ bech32With hrp addr
   91   where
   92     hrp | networkTag == shelleyTestnet = [humanReadablePart|stake_test|]
   93         | otherwise = [humanReadablePart|stake|]
