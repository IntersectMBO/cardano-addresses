   48 data Cmd = Cmd
   49     {  credentialType :: CredentialType
   50     ,  networkTag :: NetworkTag
   51     } deriving (Show)
   52 
   53 mod :: (Cmd -> parent) -> Mod CommandFields parent
   54 mod liftCmd = command "payment" $
   55     info (helper <*> fmap liftCmd parser) $ mempty
   56         <> progDesc "Create a payment address"
   57         <> header "Payment addresses carry no delegation rights whatsoever."
   58         <> footerDoc (Just $ vsep
   59             [ string "Example:"
   60             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   61             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   62             , indent 2 $ string ""
   63             , indent 2 $ bold $ string "$ cat root.prv \\"
   64             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/0/0 > addr.prv"
   65             , indent 2 $ string ""
   66             , indent 2 $ bold $ string "$ cat addr.prv \\"
   67             , indent 4 $ bold $ string $ "| "<>progName<>" key public --with-chain-code \\"
   68             , indent 4 $ bold $ string $ "| "<>progName<>" address payment --from-key --network-tag testnet"
   69             , indent 2 $ string "addr_test1vqrlltfahghjxl5sy5h5mvfrrlt6me5fqphhwjqvj5jd88cccqcek"
   70             ])
   71   where
   72     parser = Cmd
   73         <$> credentialOpt
   74         <*> networkTagOpt Shelley
   75 
   76 run :: Cmd -> IO ()
   77 run Cmd{networkTag,credentialType} = do
   78     case (mkNetworkDiscriminant . fromIntegral . unNetworkTag) networkTag of
   79         Left ErrWrongNetworkTag{} -> do
   80             fail "Invalid network tag. Must be between [0, 15]"
   81         Right discriminant -> do
   82             addr <- case credentialType of
   83                 CredentialFromKey -> do
   84                     xpub <- hGetXPub stdin
   85                     pure $ Shelley.paymentAddress discriminant (PaymentFromKey $ Shelley.liftXPub xpub)
   86                 CredentialFromScript -> do
   87                     scriptHash <- hGetScriptHash stdin
   88                     pure $ Shelley.paymentAddress discriminant (PaymentFromScript scriptHash)
   89             B8.hPutStr stdout $ T.encodeUtf8 $ bech32With hrp addr
   90   where
   91     hrp | networkTag == shelleyTestnet = [humanReadablePart|addr_test|]
   92         | otherwise = [humanReadablePart|addr|]
