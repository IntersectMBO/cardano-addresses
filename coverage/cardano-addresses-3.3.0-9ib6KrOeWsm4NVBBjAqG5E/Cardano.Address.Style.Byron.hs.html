  128 -- $overview
  129 --
  130 -- This module provides an implementation of:
  131 --
  132 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Byron master keys from mnemonic sentences
  133 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical derivation of parent to child keys
  134 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  135 --
  136 -- We call 'Byron' addresses the old address type used by Daedalus in the early
  137 -- days of Cardano. Using this type of addresses and underlying key scheme is
  138 -- now considered __deprecated__ because of some security implications.
  139 --
  140 -- The internals of the 'Byron' does not matter for the reader, but basically
  141 -- contains what is necessary to perform key derivation and generate addresses
  142 -- from a 'Byron' type.
  143 --
  144 -- == Deprecation Notice
  145 --
  146 -- Unless you have good reason to do so (like writing backward-compatible code
  147 -- with an existing piece), any new implementation __should use__ the
  148 -- 'Cardano.Address.Style.Icarus.Icarus' style for key and addresses.
  149

  151 -- | Material for deriving HD random scheme keys, which can be used for making
  152 -- addresses.
  153 --
  154 -- @since 1.0.0
  155 data Byron (depth :: Depth) key = Byron
  156     { getKey :: key
  157     -- ^ The raw private or public key.
  158     --
  159     -- @since 1.0.0
  160     , derivationPath :: DerivationPath depth
  161     -- ^ The address derivation indices for the level of this key.
  162     --
  163     -- @since 1.0.0
  164     , payloadPassphrase :: ScrubbedBytes
  165     -- ^ Used for encryption of the derivation path payload within an address.
  166     --
  167     -- @since 1.0.0
  168     } deriving stock (Generic)

  170 {-# DEPRECATED getKey "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  179 -- | The hierarchical derivation indices for a given level/depth.
  180 --
  181 -- @since 1.0.0
  182 type family DerivationPath (depth :: Depth) :: * where
  183     -- The root key is generated from the seed.
  184     DerivationPath 'RootK =
  185         ()
  186     -- The account key is generated from the root key and account index.
  187     DerivationPath 'AccountK =
  188         Index 'WholeDomain 'AccountK
  189     -- The address key is generated from the account key and address index.
  190     DerivationPath 'PaymentK =
  191         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'PaymentK)
  192 {-# DEPRECATED DerivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  197 -- === Generating a root key from 'SomeMnemonic'
  198 -- > :set -XOverloadedStrings
  199 -- > :set -XTypeApplications
  200 -- > :set -XDataKinds
  201 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  202 -- >
  203 -- > let (Right mw) = mkSomeMnemonic @'[12] ["moon","fox","ostrich","quick","cactus","raven","wasp","intact","first","ring","crumble","error"]
  204 -- > let rootK = genMasterKeyFromMnemonic mw :: Byron 'RootK XPrv

  207 --
  208 -- > let Just accIx = fromWord32 0x80000000
  209 -- > let acctK = deriveAccountPrivateKey rootK accIx
  210 -- >
  211 -- > let Just addIx = fromWord32 0x80000014
  212 -- > let addrK = deriveAddressPrivateKey acctK addIx

  214 instance Internal.GenMasterKey Byron where
  215     type SecondFactor Byron = ()
  216 
  217     genMasterKeyFromXPrv xprv =
  218         liftXPrv (toXPub xprv) () xprv
  219     genMasterKeyFromMnemonic (SomeMnemonic mw) () =
  220         liftXPrv (toXPub xprv) () xprv
  221       where
  222         xprv = generate (hashSeed seedValidated)
  223         seed  = entropyToBytes $ mnemonicToEntropy mw
  224         seedValidated = assert
  225             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  226             seed

  228 instance Internal.HardDerivation Byron where
  229     type AddressIndexDerivationType Byron = 'WholeDomain
  230     type AccountIndexDerivationType Byron = 'WholeDomain
  231     type WithRole Byron = ()

  233     deriveAccountPrivateKey rootXPrv accIx = Byron
  234         { getKey = deriveXPrv DerivationScheme1 (getKey rootXPrv) accIx
  235         , derivationPath = accIx
  236         , payloadPassphrase = payloadPassphrase rootXPrv
  237         }

  239     deriveAddressPrivateKey accXPrv () addrIx = Byron
  240         { getKey = deriveXPrv DerivationScheme1 (getKey accXPrv) addrIx
  241         , derivationPath = (derivationPath accXPrv, addrIx)
  242         , payloadPassphrase = payloadPassphrase accXPrv
  243         }

  338 -- | Analyze an 'Address' to know whether it's a Byron address or not.
  339 -- Throws 'ByronAddrError' if the address isn't a byron address, or return a
  340 -- structured JSON that gives information about an address.
  341 --
  342 -- @since 3.0.0
  343 inspectAddress :: forall m. MonadThrow m => Maybe XPub -> Address -> m Json.Value
  344 inspectAddress mRootPub addr = do
  345     payload <- either (throwM . DeserialiseError) pure
  346         $ CBOR.deserialiseCbor CBOR.decodeAddressPayload bytes
  347 
  348     (root, attrs) <- either (throwM . DeserialiseError) pure
  349         $ CBOR.deserialiseCbor decodePayload payload
  350 
  351     path <- do
  352         attr <- maybe (throwM MissingExpectedDerivationPath) pure
  353             (find ((== 1) . fst) attrs)
  354         case mRootPub of
  355             Nothing ->
  356                 pure $ toJSON $ T.unpack $ T.decodeUtf8 $ encode EBase16 $ snd attr
  357             Just rootPub ->
  358                 decryptPath rootPub attr
  359 
  360     ntwrk <- either (throwM . DeserialiseError) pure
  361         $ CBOR.deserialiseCbor CBOR.decodeProtocolMagicAttr payload
  362 
  363     pure $ Json.object
  364         [ "address_style"   .= Json.String "Byron"
  365         , "stake_reference" .= Json.String "none"
  366         , "address_root"    .= T.unpack (T.decodeUtf8 $ encode EBase16 root)
  367         , "derivation_path" .= path
  368         , "network_tag"     .= maybe Json.Null toJSON ntwrk
  369         ]
  370   where
  371     bytes :: ByteString
  372     bytes = unAddress addr
  373 
  374     decodePayload :: forall s. CBOR.Decoder s (ByteString, [(Word8, ByteString)])
  375     decodePayload = do
  376         _ <- CBOR.decodeListLenCanonicalOf 3
  377         root <- CBOR.decodeBytes
  378         (root,) <$> CBOR.decodeAllAttributes
  379 
  380     decryptPath :: XPub -> (Word8, ByteString) -> m Json.Value
  381     decryptPath rootPub attr = do
  382         let pwd = hdPassphrase rootPub
  383         path <- either (const (throwM FailedToDecryptPath)) pure
  384             $ CBOR.deserialiseCbor (CBOR.decodeDerivationPathAttr pwd [attr]) mempty
  385         case path of
  386             Nothing -> throwM FailedToDecryptPath
  387             Just (acctIx, addrIx) -> pure $ Json.object
  388                 [ "account_index" .= prettyIndex acctIx
  389                 , "address_index" .= prettyIndex addrIx
  390                 ]
  391 
  392     prettyIndex :: Word32 -> String
  393     prettyIndex ix
  394         | ix >= firstHardened = show (ix - firstHardened) <> "H"
  395         | otherwise = show ix
  396       where
  397         firstHardened = 0x80000000

  399 instance Internal.PaymentAddress Byron where
  400     paymentAddress discrimination k = unsafeMkAddress
  401         $ CBOR.toStrictByteString
  402         $ CBOR.encodeAddress (getKey k) attrs
  403       where
  404         (acctIx, addrIx) = bimap indexToWord32 indexToWord32 $ derivationPath k
  405         pwd = payloadPassphrase k
  406         NetworkTag magic = networkTag @Byron discrimination
  407         attrs = case addressDiscrimination @Byron discrimination of
  408             RequiresNetworkTag ->
  409                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  410                 , CBOR.encodeProtocolMagicAttr magic
  411                 ]
  412             RequiresNoTag ->
  413                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  414                 ]

  460 -- | Backdoor for generating a new key from a raw 'XPrv'.
  461 --
  462 -- Note that the @depth@ is left open so that the caller gets to decide what type
  463 -- of key this is. This is mostly for testing, in practice, seeds are used to
  464 -- represent root keys, and one should 'genMasterKeyFromXPrv'
  465 --
  466 -- The first argument is a type-family 'DerivationPath' and its type depends on
  467 -- the 'depth' of the key.

  469 -- __examples:__
  470 --
  471 -- >>> liftXPrv rootPrv () prv
  472 -- _ :: Byron RootK XPrv
  473 --
  474 -- >>> liftXPrv rootPrv minBound prv

  476 --
  477 -- >>> liftXPrv rootPrv (minBound, minBound) prv
  478 -- _ :: Byron PaymentK XPrv

  482     :: XPub -- ^ A root public key
  483     -> DerivationPath depth
  484     -> XPrv
  485     -> Byron depth XPrv
  486 liftXPrv rootPub derivationPath getKey = Byron
  487     { getKey
  488     , derivationPath
  489     , payloadPassphrase = hdPassphrase rootPub
  490     }

  518 -- The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  519 minSeedLengthBytes :: Int
  520 minSeedLengthBytes = 16
  521 
  522 -- Hash the seed entropy (generated from mnemonic) used to initiate a HD
  523 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  524 -- minimum for 'generate' (32 bytes).
  525 --
  526 -- Note that our current implementation deviates from BIP-39 because we use a
  527 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  528 --
  529 -- There are two methods of hashing the seed entropy, for different use cases.
  530 --
  531 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  532 --    Blake2b_256, inside a double CBOR serialization sandwich.
  533 --
  534 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  535 --    Blake2b_256, without any serialization.
  536 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  537 hashSeed = serialize . blake2b256 . serialize
  538   where
  539     serialize = BA.convert . cbor . BA.convert
  540     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  541 
  542 -- Hash a byte string through blake2b 256
  543 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  544 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  545 
  546 -- Derive a symmetric key for encrypting and authenticating the address
  547 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  548 -- is employed.
  549 hdPassphrase :: XPub -> ScrubbedBytes
  550 hdPassphrase masterKey =
  551     PBKDF2.generate
  552     (PBKDF2.prfHMAC SHA512)
  553     (PBKDF2.Parameters 500 32)
  554     (xpubToBytes masterKey)
  555     ("address-hashing" :: ByteString)
