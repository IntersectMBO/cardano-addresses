  142 -- | Construct an 'XPrv' from raw 'ByteString'.
  143 --
  144 -- @since 1.0.0
  145 xprvFromBytes :: ByteString -> Maybe XPrv
  146 xprvFromBytes bytes
  147     | BS.length bytes /= 96 = Nothing
  148     | otherwise = do
  149         let (prv, cc) = BS.splitAt 64 bytes
  150         pub <- ed25519ScalarMult (BS.take 32 prv)
  151         eitherToMaybe $ CC.xprv $ prv <> pub <> cc
  152   where
  153     ed25519ScalarMult :: ByteString -> Maybe ByteString
  154     ed25519ScalarMult bs = do
  155         scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
  156         pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar
  157 
  158 -- | Convert an 'XPrv' to a raw 'ByteString'.
  159 --
  160 -- @since 1.0.0
  161 xprvToBytes :: XPrv -> ByteString
  162 xprvToBytes =
  163     stripPub . CC.unXPrv
  164   where
  165     -- From  < xprv | pub | cc >
  166     -- â†³ To  < xprv |     | cc >
  167     stripPub :: ByteString -> ByteString
  168     stripPub xprv' = prv <> chainCode
  169       where
  170         (prv, rest) = BS.splitAt 64 xprv'
  171         (_pub, chainCode) = BS.splitAt 32 rest

  254 -- | Key Depth in the derivation path, according to BIP-0039 / BIP-0044
  255 --
  256 -- @
  257 -- root | purpose' | cointype' | account' | change | address@
  258 -- 0th      1st         2nd        3rd        4th      5th
  259 -- @
  260 --
  261 -- We do not manipulate purpose, cointype and change paths directly, so there
  262 -- are no constructors for these.
  263 --
  264 -- @since 1.0.0
  265 data Depth = RootK | AccountK | AddressK | StakingK
  266 
  267 -- | Marker for addresses type engaged. We want to handle three cases here.
  268 -- The first two are pertinent to UTxO accounting
  269 -- and the last one handles rewards from participation in staking.
  270 -- (a) external chain is used for addresses that are part of the 'advertised'
  271 --     targets of a given transaction
  272 -- (b) internal change is for addresses used to handle the change of a
  273 --     the transaction within a given wallet
  274 data AccountingStyle
  275     = UTxOExternal
  276     | UTxOInternal
  277     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  278 
  279 instance NFData AccountingStyle
  280 
  281 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  282 -- to assign the wrong index to the corresponding constructor (by swapping
  283 -- around the constructor above for instance).
  284 instance Enum AccountingStyle where
  285     toEnum = \case
  286         0 -> UTxOExternal
  287         1 -> UTxOInternal
  288         _ -> error "AccountingStyle.toEnum: bad argument"
  289     fromEnum = \case
  290         UTxOExternal -> 0
  291         UTxOInternal -> 1
  292 
  293 -- | A derivation index, with phantom-types to disambiguate derivation type.
  294 --
  295 -- @
  296 -- let accountIx = Index 'Hardened 'AccountK
  297 -- let addressIx = Index 'Soft 'AddressK
  298 -- @
  299 --
  300 -- @since 1.0.0
  301 newtype Index (derivationType :: DerivationType) (depth :: Depth) = Index
  302     { getIndex :: Word32 }
  303     deriving stock (Generic, Show, Eq, Ord)
  304 
  305 instance NFData (Index derivationType depth)
  306 
  307 instance Bounded (Index 'Hardened depth) where
  308     minBound = Index 0x80000000
  309     maxBound = Index maxBound
  310 
  311 instance Bounded (Index 'Soft depth) where
  312     minBound = Index minBound
  313     maxBound = let (Index ix) = minBound @(Index 'Hardened _) in Index (ix - 1)
  314 
  315 instance Bounded (Index 'WholeDomain depth) where
  316     minBound = Index minBound
  317     maxBound = Index maxBound
  318 
  319 instance Enum (Index 'Hardened depth) where
  320     fromEnum (Index ix) = fromIntegral ix
  321     toEnum ix
  322         | Index (fromIntegral ix) < minBound @(Index 'Hardened _) =
  323             error "Index@Hardened.toEnum: bad argument"
  324         | otherwise =
  325             Index (fromIntegral ix)
  326 
  327 instance Enum (Index 'Soft depth) where
  328     fromEnum (Index ix) = fromIntegral ix
  329     toEnum ix
  330         | Index (fromIntegral ix) > maxBound @(Index 'Soft _) =
  331             error "Index@Soft.toEnum: bad argument"
  332         | otherwise =
  333             Index (fromIntegral ix)
  334 
  335 instance Enum (Index 'WholeDomain depth) where
  336     fromEnum (Index ix) = fromIntegral ix
  337     toEnum ix
  338         | Index (fromIntegral ix) > maxBound @(Index 'WholeDomain _) =
  339             error "Index@WholeDomain.toEnum: bad argument"
  340         | otherwise =
  341             Index (fromIntegral ix)
  342 
  343 instance Buildable (Index derivationType depth) where
  344     build (Index ix) = fromString (show ix)

  347 -- | Type of derivation that should be used with the given indexes.
  348 --
  349 -- In theory, we should only consider two derivation types: soft and hard.
  350 --
  351 -- However, historically, addresses in Cardano used to be generated across both
  352 -- the soft and the hard domain. We therefore introduce a 'WholeDomain' derivation
  353 -- type that is the exact union of `Hardened` and `Soft`.
  354 --
  355 -- @since 1.0.0
  356 data DerivationType = Hardened | Soft | WholeDomain
  357 
  358 -- | An interface for doing hard derivations from the root private key, /Master Key/

  370     -- @since 1.0.0
