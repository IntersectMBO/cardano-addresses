   99 -- $overview
  100 --
  101 -- This module provides an implementation of:
  102 --
  103 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Jormungandr master keys from mnemonic sentences
  104 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  105 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  106 -- - 'Cardano.Address.PaymentAddress': for constructing payment addresses from a address public key
  107 -- - 'Cardano.Address.DelegationAddress': for constructing payment addresses from a address and stake public keys
  108 --
  109 -- == Disclaimer
  110 --
  111 -- Beware that this format of address have been specially crafted for the
  112 -- incentivized testnet (ITN). This will soon be __deprecated__ in favor of
  113 -- 'Cardano.Address.Style.Shelley.Shelley'.
  114 --
  115 -- __New integrations should not use this address style__.
  116 
  117 -- | A cryptographic key for sequential-scheme address derivation, with
  118 -- phantom-types to disambiguate key types.
  119 --
  120 -- @
  121 -- let rootPrivateKey = Jormungandr 'RootK XPrv
  122 -- let accountPubKey  = Jormungandr 'AccountK XPub
  123 -- let addressPubKey  = Jormungandr 'AddressK XPub
  124 -- @
  125 --
  126 -- @since 2.0.0
  127 newtype Jormungandr (depth :: Depth) key = Jormungandr
  128     { getKey :: key
  129         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  130         --
  131         -- @since 2.0.0
  132     }
  133     deriving stock (Generic, Show, Eq)
  134 
  135 deriving instance (Functor (Jormungandr depth))
  136 instance (NFData key) => NFData (Jormungandr depth key)

  138 --
  139 -- Key Derivation
  140 --
  141 -- $keyDerivation
  142 --
  143 -- === Generating a root key from 'SomeMnemonic'
  144 -- > :set -XOverloadedStrings
  145 -- > :set -XTypeApplications
  146 -- > :set -XDataKinds
  147 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  148 -- >
  149 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  150 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  151 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Jormungandr 'RootK XPrv
  152 --
  153 -- === Deriving child keys
  154 --
  155 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  156 --
  157 -- > import Cardano.Address.Derivation ( AccountingStyle(..) )
  158 -- >
  159 -- > let accIx = toEnum 0x80000000
  160 -- > let acctK = deriveAccountPrivateKey rootK accIx
  161 -- >
  162 -- > let addIx = toEnum 0x00000014
  163 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  164 --
  165 -- > let stakeK = deriveStakingPrivateKey acctK
  166 
  167 instance Internal.GenMasterKey Jormungandr where
  168     type SecondFactor Jormungandr = ScrubbedBytes
  169 
  170     genMasterKeyFromXPrv = liftXPrv
  171     genMasterKeyFromMnemonic fstFactor sndFactor =
  172         Jormungandr $ generateNew seedValidated sndFactor
  173         where
  174             seed  = someMnemonicToBytes fstFactor
  175             seedValidated = assert
  176                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  177                 seed

  179 instance Internal.HardDerivation Jormungandr where
  180     type AccountIndexDerivationType Jormungandr = 'Hardened
  181     type AddressIndexDerivationType Jormungandr = 'Soft
  182     type WithAccountStyle Jormungandr = AccountingStyle
  183 
  184     deriveAccountPrivateKey (Jormungandr rootXPrv) accIx =
  185         let
  186             purposeIx =
  187                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  188             coinTypeIx =
  189                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  190             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  191                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  192             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  193                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  194             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  195                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  196         in
  197             Jormungandr acctXPrv
  198 
  199     deriveAddressPrivateKey (Jormungandr accXPrv) accountingStyle addrIx =
  200         let
  201             changeCode =
  202                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  203             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  204                 deriveXPrv DerivationScheme2 accXPrv changeCode
  205             addrXPrv = -- lvl5 derivation; soft derivation of address index
  206                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  207         in
  208             Jormungandr addrXPrv
  209 
  210 instance Internal.SoftDerivation Jormungandr where
  211     deriveAddressPublicKey (Jormungandr accXPub) accountingStyle addrIx =
  212         fromMaybe errWrongIndex $ do
  213             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  214             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  215                 deriveXPub DerivationScheme2 accXPub changeCode
  216             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  217                 deriveXPub DerivationScheme2 changeXPub addrIx
  218             return $ Jormungandr addrXPub
  219       where
  220         errWrongIndex = error $
  221             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  222             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  223             \either a programmer error, or, we may have reached the maximum \
  224             \number of addresses for a given wallet."
  225 
  226 instance Internal.StakingDerivation Jormungandr where
  227     deriveStakingPrivateKey (Jormungandr accXPrv) =
  228         let
  229             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  230                 deriveXPrv DerivationScheme2 accXPrv (toEnum @(Index 'Soft _) 2)
  231             stakeXPrv = -- lvl5 derivation; soft derivation of address index
  232                 deriveXPrv DerivationScheme2 changeXPrv (minBound @(Index 'Soft _))
  233         in
  234             Jormungandr stakeXPrv

  331 instance Internal.PaymentAddress Jormungandr where
  332     paymentAddress discrimination k = unsafeMkAddress $
  333         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  334             putWord8 (invariantNetworkTag 255 firstByte)
  335             putByteString (xpubPublicKey $ getKey k)
  336       where
  337           firstByte = networkTag @Jormungandr discrimination
  338           expectedLength = 1 + publicKeySize
  339 
  340 instance Internal.DelegationAddress Jormungandr where
  341     delegationAddress discrimination paymentKey stakingKey = unsafeMkAddress $
  342         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  343             putWord8 (invariantNetworkTag 255 $ NetworkTag $ firstByte + 1)
  344             putByteString . xpubPublicKey . getKey $ paymentKey
  345             putByteString . xpubPublicKey . getKey $ stakingKey
  346       where
  347           (NetworkTag firstByte) = networkTag @Jormungandr discrimination
  348           expectedLength = 1 + 2*publicKeySize

  425 -- Purpose is a constant set to 1852' (or 0x8000073c) following the BIP-44
  426 -- extension for Cardano:
  427 --
  428 -- https://github.com/input-output-hk/implementation-decisions/blob/e2d1bed5e617f0907bc5e12cf1c3f3302a4a7c42/text/1852-hd-chimeric.md
  429 --
  430 -- It indicates that the subtree of this node is used according to this
  431 -- specification.
  432 --
  433 -- Hardened derivation is used at this level.
  434 purposeIndex :: Word32
  435 purposeIndex = 0x8000073c
  436 
  437 -- One master node (seed) can be used for unlimited number of independent
  438 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  439 -- same space for various cryptocoins has some disadvantages.
  440 --
  441 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  442 -- addresses across cryptocoins and improving privacy issues.
  443 --
  444 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  445 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  446 -- Lovelace.
  447 --
  448 -- Hardened derivation is used at this level.
  449 coinTypeIndex :: Word32
  450 coinTypeIndex = 0x80000717
