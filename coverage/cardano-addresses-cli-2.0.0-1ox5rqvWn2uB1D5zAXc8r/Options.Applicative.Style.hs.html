   40 -- | Represent a style of wallet.
   41 data Style
   42     = Byron
   43     | Icarus
   44     | Jormungandr
   45     | Shelley
   46     deriving (Eq, Show, Enum, Bounded)
   47 
   48 -- TODO
   49 -- Allow passphrase here.
   50 --
   51 -- | Generate an extended root private key from a mnemonic sentence, in the
   52 -- given style.
   53 generateRootKey :: SomeMnemonic -> Style -> IO XPrv
   54 generateRootKey mw = \case
   55     Byron -> do
   56         let rootK = Byron.genMasterKeyFromMnemonic mw
   57         pure $ Byron.getKey rootK
   58     Icarus -> do
   59         let sndFactor = mempty
   60         let rootK = Icarus.genMasterKeyFromMnemonic mw sndFactor
   61         pure $ Icarus.getKey rootK
   62     Jormungandr -> do
   63         let sndFactor = mempty
   64         let rootK = Jormungandr.genMasterKeyFromMnemonic mw sndFactor
   65         pure $ Jormungandr.getKey rootK
   66     Shelley -> do
   67         let sndFactor = mempty
   68         let rootK = Shelley.genMasterKeyFromMnemonic mw sndFactor
   69         pure $ Shelley.getKey rootK
   70 
   71 --
   72 -- Applicative Parser
   73 --
   74 
   75 -- | Parse a 'Style' from the command-line, as an argument.
   76 styleArg :: Parser Style
   77 styleArg = argument (eitherReader reader) $ mempty
   78     <> metavar "STYLE"
   79     <> help styles'
   80     <> completer (listCompleter styles)
   81   where
   82     styles :: [String]
   83     styles = show @Style <$> [minBound .. maxBound]
   84 
   85     styles' = intercalate " | " styles
   86 
   87     reader :: String -> Either String Style
   88     reader str = case toLower <$> str of
   89         "byron"       -> Right Byron
   90         "icarus"      -> Right Icarus
   91         "jormungandr" -> Right Jormungandr
   92         "shelley"     -> Right Shelley
   93         _             -> Left $ "Unknown style; expecting one of " <> styles'
