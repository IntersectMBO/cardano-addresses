   44 -- | Parse a 'NetworkTag' from the command-line, as an option
   45 networkTagOpt :: Style -> Parser NetworkTag
   46 networkTagOpt style = option (eitherReader reader) $ mempty
   47     <> metavar "NETWORK-TAG"
   48     <> long "network-tag"
   49     <> helpDoc  (Just (vsep (string <$> doc style)))
   50     <> completer (listCompleter $ show <$> tagsFor style)
   51   where
   52     doc style' =
   53         [ "A tag which identifies a Cardano network."
   54         , ""
   55         ] ++ case style' of
   56         Byron ->
   57             [ "┌ Byron / Icarus ──────────"
   58             , "│ mainnet: " <> show (unNetworkTag (snd Byron.byronMainnet))
   59             , "│ staging: " <> show (unNetworkTag (snd Byron.byronStaging))
   60             , "│ testnet: " <> show (unNetworkTag (snd Byron.byronTestnet))
   61             ]
   62         Icarus ->
   63             doc Byron
   64         Jormungandr ->
   65             [ "┌ Jormungandr ─────────────"
   66             , "│ testnet: " <> show (unNetworkTag Jormungandr.incentivizedTestnet)
   67             ]
   68         Shelley ->
   69             [ "┌ Shelley ─────────────────"
   70             , "│ mainnet: " <> show (unNetworkTag Shelley.shelleyMainnet)
   71             , "│ testnet: " <> show (unNetworkTag Shelley.shelleyTestnet)
   72             ]
   73 
   74     tagsFor = \case
   75         Byron ->
   76             [ unNetworkTag (snd Byron.byronMainnet)
   77             , unNetworkTag (snd Byron.byronStaging)
   78             , unNetworkTag (snd Byron.byronTestnet)
   79             ]
   80         Icarus ->
   81             tagsFor Byron
   82         Jormungandr ->
   83             [ unNetworkTag Jormungandr.incentivizedTestnet
   84             ]
   85         Shelley ->
   86             [ unNetworkTag Shelley.shelleyMainnet
   87             , unNetworkTag Shelley.shelleyTestnet
   88             ]
   89 
   90     reader =
   91         maybe
   92             (Left "Invalid network tag. Must be a integer value.")
   93             (Right . NetworkTag)
   94             . readMaybe
