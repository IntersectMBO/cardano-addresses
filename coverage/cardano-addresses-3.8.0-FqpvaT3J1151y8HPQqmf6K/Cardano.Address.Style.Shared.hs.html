   75 -- $overview
   76 --
   77 -- This module provides an implementation of:
   78 --
   79 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Shared master keys from mnemonic sentences
   80 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
   81 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
   82 --
   83 -- - 'paymentAddress': for constructing payment addresses from a address public key or a script
   84 -- - 'delegationAddress': for constructing delegation addresses from payment credential (public key or script) and stake credential (public key or script)
   85 -- - 'pointerAddress': for constructing delegation addresses from payment credential (public key or script) and chain pointer
   86 -- - 'stakeAddress': for constructing reward accounts from stake credential (public key or script)
   87 
   88 -- | A cryptographic key for sequential-scheme address derivation, with
   89 -- phantom-types to disambiguate key types. The derivation is mostly like Shelley, except the used purpose index
   90 -- (here 1854H rather than Shelley's 1852H)
   91 --
   92 -- @
   93 -- let rootPrivateKey = Shared 'RootK XPrv
   94 -- let accountPubKey  = Shared 'AccountK XPub
   95 -- let addressPubKey  = Shared 'PaymentK XPub
   96 -- @
   97 --
   98 -- @since 3.4.0
   99 newtype Shared (depth :: Depth) key = Shared
  100     { getKey :: key
  101         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  102         --
  103         -- @since 3.4.0
  104     }
  105     deriving stock (Generic, Show, Eq)
  106 
  107 deriving instance (Functor (Shared depth))
  108 instance (NFData key) => NFData (Shared depth key)
  109 
  110 --
  111 -- Key Derivation
  112 --
  113 -- $keyDerivation
  114 --
  115 -- === Generating a root key from 'SomeMnemonic'
  116 -- > :set -XOverloadedStrings
  117 -- > :set -XTypeApplications
  118 -- > :set -XDataKinds
  119 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  120 -- >
  121 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  122 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  123 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shared 'RootK XPrv
  124 --
  125 -- === Deriving child keys
  126 --
  127 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  128 --
  129 -- > let Just accIx = indexFromWord32 0x80000000
  130 -- > let acctK = deriveAccountPrivateKey rootK accIx
  131 -- >
  132 -- > let Just addIx = indexFromWord32 0x00000014
  133 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  134 --
  135 -- > let stakeK = deriveDelegationPrivateKey acctK
  136 
  137 instance Internal.GenMasterKey Shared where
  138     type SecondFactor Shared = ScrubbedBytes
  139 
  140     genMasterKeyFromXPrv = liftXPrv
  141     genMasterKeyFromMnemonic fstFactor sndFactor =
  142         Shared $ genMasterKeyFromMnemonicShelley fstFactor sndFactor
  143 
  144 instance Internal.HardDerivation Shared where
  145     type AccountIndexDerivationType Shared = 'Hardened
  146     type AddressIndexDerivationType Shared = 'Soft
  147     type WithRole Shared = Role
  148 
  149     deriveAccountPrivateKey (Shared rootXPrv) accIx =
  150         Shared $ deriveAccountPrivateKeyShelley rootXPrv accIx purposeIndex
  151 
  152     deriveAddressPrivateKey (Shared accXPrv) keyRole addrIx =
  153         Shared $ deriveAddressPrivateKeyShelley accXPrv keyRole addrIx
  154 
  155 instance Internal.SoftDerivation Shared where
  156     deriveAddressPublicKey (Shared accXPub) keyRole addrIx =
  157         Shared $ deriveAddressPublicKeyShelley accXPub keyRole addrIx

  265 --- | Computes a 28-byte Blake2b224 digest of a Shared 'XPub'.
  266 ---
  267 --- @since 3.4.0
  268 hashKey :: KeyRole -> Shared key XPub -> KeyHash
  269 hashKey cred = KeyHash cred . hashCredential . xpubPublicKey . getKey
