  171 -- $overview
  172 --
  173 -- This module provides an implementation of:
  174 --
  175 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Shelley master keys from mnemonic sentences
  176 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  177 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  178 --
  179 -- - 'paymentAddress': for constructing payment addresses from a address public key or a script
  180 -- - 'delegationAddress': for constructing delegation addresses from payment credential (public key or script) and stake credential (public key or script)
  181 -- - 'pointerAddress': for constructing delegation addresses from payment credential (public key or script) and chain pointer
  182 -- - 'stakeAddress': for constructing reward accounts from stake credential (public key or script)
  183 
  184 -- | A cryptographic key for sequential-scheme address derivation, with
  185 -- phantom-types to disambiguate key types.
  186 --
  187 -- @
  188 -- let rootPrivateKey = Shelley 'RootK XPrv
  189 -- let accountPubKey  = Shelley 'AccountK XPub
  190 -- let addressPubKey  = Shelley 'PaymentK XPub
  191 -- @
  192 --
  193 -- @since 2.0.0
  194 newtype Shelley (depth :: Depth) key = Shelley
  195     { getKey :: key
  196         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  197         --
  198         -- @since 1.0.0
  199     }
  200     deriving stock (Generic, Show, Eq)
  201 
  202 deriving instance (Functor (Shelley depth))
  203 instance (NFData key) => NFData (Shelley depth key)
  204 
  205 -- | Describe what the keys within an account are used for.
  206 --
  207 -- - UTxOExternal: used for public addresses sent to other parties for receiving money.
  208 -- - UTxOInternal: generated by wallet software to send change back to the wallet.
  209 -- - Stake: used for stake key(s) and delegation.
  210 --
  211 -- @since 3.0.0
  212 data Role
  213     = UTxOExternal
  214     | UTxOInternal
  215     | Stake
  216     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  217 
  218 instance NFData Role
  219 
  220 roleFromIndex :: Index 'Soft depth -> Maybe Role
  221 roleFromIndex ix = case indexToWord32 ix of
  222     0 -> Just UTxOExternal
  223     1 -> Just UTxOInternal
  224     2 -> Just Stake
  225     _ -> Nothing
  226 
  227 roleToIndex :: Role -> Index 'Soft depth
  228 roleToIndex = unsafeMkIndex . \case
  229     UTxOExternal -> 0
  230     UTxOInternal -> 1
  231     Stake -> 2

  234 -- Key Derivation
  235 --
  236 -- $keyDerivation

  238 -- === Generating a root key from 'SomeMnemonic'
  239 -- > :set -XOverloadedStrings
  240 -- > :set -XTypeApplications
  241 -- > :set -XDataKinds
  242 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  243 -- >
  244 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  245 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  246 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shelley 'RootK XPrv
  247 --
  248 -- === Deriving child keys
  249 --
  250 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  251 --
  252 -- > let Just accIx = indexFromWord32 0x80000000
  253 -- > let acctK = deriveAccountPrivateKey rootK accIx
  254 -- >
  255 -- > let Just addIx = indexFromWord32 0x00000014
  256 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  257 --
  258 -- > let stakeK = deriveDelegationPrivateKey acctK
  259

  261     type SecondFactor Shelley = ScrubbedBytes

  264     genMasterKeyFromMnemonic fstFactor sndFactor =
  265         Shelley $ genMasterKeyFromMnemonicShelley fstFactor sndFactor

  352 -- Re-export from 'Cardano.Address.Derivation' to have it documented specialized in Haddock
  353 --
  354 -- | Derive a delegation key for a corresponding 'AccountK'. Note that wallet
  355 -- software are by convention only using one delegation key per account, and always
  356 -- the first account (with index 0').
  357 --
  358 -- Deriving delegation keys for something else than the initial account is not
  359 -- recommended and can lead to incompatibility with existing wallet softwares
  360 -- (Daedalus, Yoroi, Adalite...).
  361 --
  362 -- @since 2.0.0
  363 deriveDelegationPrivateKey
  364     :: Shelley 'AccountK XPrv
  365     -> Shelley 'DelegationK XPrv
  366 deriveDelegationPrivateKey accXPrv =
  367     let (Shelley stakeXPrv) =
  368             deriveAddressPrivateKey accXPrv Stake (minBound @(Index 'Soft _))
  369     in Shelley stakeXPrv

  488 -- | Analyze an 'Address' to know whether it's a valid address for the Cardano
  489 -- Shelley era. Shelley format addresses, as well as old-style Byron and Icarus
  490 -- addresses can be parsed by this function.
  491 --
  492 -- Returns a JSON value containing details about the 'Address', or throws
  493 -- 'ErrInspectAddress' if it's not a valid address.
  494 --
  495 -- @since 3.0.0
  496 inspectAddress
  497     :: (Alternative m, MonadThrow m)
  498     => Maybe XPub
  499     -> Address
  500     -> m Json.Value
  501 inspectAddress mRootPub addr = either throwM (pure . toJSON) $
  502     eitherInspectAddress mRootPub addr
  503 
  504 -- | Determines whether an 'Address' is a valid address for the Cardano Shelley
  505 -- era. Shelley format addresses, as well as old-style Byron and Icarus
  506 -- addresses can be parsed by this function.
  507 --
  508 -- Returns either details about the 'Address', or 'ErrInspectAddress' if it's
  509 -- not a valid address.
  510 --
  511 -- @since 3.4.0
  512 eitherInspectAddress
  513     :: Maybe XPub
  514     -> Address
  515     -> Either ErrInspectAddress InspectAddress
  516 eitherInspectAddress mRootPub addr = unpackAddress addr >>= parseInfo
  517   where
  518     parseInfo :: AddressParts -> Either ErrInspectAddress InspectAddress
  519     parseInfo parts = case addrType parts of
  520         -- 1000: byron address
  521         0b10000000 ->
  522             (bimap ErrIcarus InspectAddressIcarus (Icarus.eitherInspectAddress addr))
  523             `orElse`
  524             (bimap ErrByron InspectAddressByron (Byron.eitherInspectAddress mRootPub addr))
  525         -- Anything else: shelley address
  526         _ -> bimap ErrShelley InspectAddressShelley (parseAddressInfoShelley parts)
  527 
  528 -- | Returns either details about the 'Address', or
  529 -- 'ErrInspectAddressOnlyShelley' if it's not a valid Shelley address.
  530 parseAddressInfoShelley :: AddressParts -> Either ErrInspectAddressOnlyShelley AddressInfo
  531 parseAddressInfoShelley AddressParts{..} = case addrType of
  532     -- 0000: base address: keyhash28,keyhash28
  533     0b00000000 | addrRestLength == credentialHashSize + credentialHashSize ->
  534         Right addressInfo
  535             { infoStakeReference = Just ByValue
  536             , infoSpendingKeyHash = Just addrHash1
  537             , infoStakeKeyHash = Just addrHash2
  538             }
  539     -- 0001: base address: scripthash28,keyhash28
  540     0b00010000 | addrRestLength == credentialHashSize + credentialHashSize ->
  541         Right addressInfo
  542             { infoStakeReference = Just ByValue
  543             , infoScriptHash = Just addrHash1
  544             , infoStakeKeyHash = Just addrHash2
  545             }
  546     -- 0010: base address: keyhash28,scripthash28
  547     0b00100000 | addrRestLength == credentialHashSize + credentialHashSize ->
  548         Right addressInfo
  549             { infoStakeReference = Just ByValue
  550             , infoSpendingKeyHash = Just addrHash1
  551             , infoStakeScriptHash = Just addrHash2
  552             }
  553     -- 0011: base address: scripthash28,scripthash28
  554     0b00110000 | addrRestLength == 2 * credentialHashSize ->
  555         Right addressInfo
  556             { infoStakeReference = Just ByValue
  557             , infoScriptHash = Just addrHash1
  558             , infoStakeScriptHash = Just addrHash2
  559             }
  560     -- 0100: pointer address: keyhash28, 3 variable length uint
  561     0b01000000 | addrRestLength > credentialHashSize -> do
  562         ptr <- getPtr addrHash2
  563         pure addressInfo
  564             { infoStakeReference = Just $ ByPointer ptr
  565             , infoSpendingKeyHash = Just addrHash1
  566             }
  567     -- 0101: pointer address: scripthash28, 3 variable length uint
  568     0b01010000 | addrRestLength > credentialHashSize -> do
  569         ptr <- getPtr addrHash2
  570         pure addressInfo
  571             { infoStakeReference = Just $ ByPointer ptr
  572             , infoScriptHash = Just addrHash1
  573             }
  574     -- 0110: enterprise address: keyhash28
  575     0b01100000 | addrRestLength == credentialHashSize ->
  576         Right addressInfo
  577             { infoStakeReference = Nothing
  578             , infoSpendingKeyHash = Just addrHash1
  579             }
  580     -- 0111: enterprise address: scripthash28
  581     0b01110000 | addrRestLength == credentialHashSize ->
  582         Right addressInfo
  583             { infoStakeReference = Nothing
  584             , infoScriptHash = Just addrHash1
  585             }
  586     -- 1110: reward account: keyhash28
  587     0b11100000 | addrRestLength == credentialHashSize ->
  588         Right addressInfo
  589             { infoStakeReference = Just ByValue
  590             , infoStakeKeyHash = Just addrHash1
  591             }
  592     -- 1111: reward account: scripthash28
  593     0b11110000 | addrRestLength == credentialHashSize ->
  594         Right addressInfo
  595             { infoStakeReference = Just ByValue
  596             , infoScriptHash = Just addrHash1
  597             }
  598     unknown -> Left (UnknownType unknown)
  599 
  600   where
  601     addressInfo = AddressInfo
  602         { infoNetworkTag = NetworkTag $ fromIntegral addrNetwork
  603         , infoStakeReference = Nothing
  604         , infoSpendingKeyHash = Nothing
  605         , infoStakeKeyHash = Nothing
  606         , infoScriptHash = Nothing
  607         , infoStakeScriptHash = Nothing
  608         , infoAddressType = shiftR (addrType .&. 0b11110000) 4
  609         }

  827 -- | Extend an existing payment 'Address' to make it a delegation address.
  828 --
  829 -- @since 2.0.0
  830 extendAddress
  831     :: Address
  832     -> Credential 'DelegationK
  833     -> Either ErrExtendAddress Address
  834 extendAddress addr infoStakeReference = do
  835     when (isNothing (inspectAddress Nothing addr)) $
  836         Left $ ErrInvalidAddressStyle "Given address isn't a Shelley address"
  837 
  838     let bytes = unAddress addr
  839     let (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  840 
  841     let paymentFirstByte = fstByte .&. 0b11110000
  842     let extendableTypes = addressType <$>
  843             [ EnterpriseAddress CredentialFromKey
  844             , EnterpriseAddress CredentialFromScript
  845             ]
  846     unless (paymentFirstByte `elem` extendableTypes) $ do
  847         Left $ ErrInvalidAddressType "Only payment addresses can be extended"
  848 
  849     case infoStakeReference of
  850         -- base address: keyhash28,keyhash28    : 00000000 -> 0
  851         -- base address: scripthash32,keyhash28 : 00010000 -> 16
  852         DelegationFromKey delegationKey -> do
  853             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  854                 -- 0b01100000 .&. 0b00011111 = 0
  855                 -- 0b01110000 .&. 0b00011111 = 16
  856                 putWord8 $ fstByte .&. 0b00011111
  857                 putByteString rest
  858                 putByteString . hashCredential . xpubPublicKey . getKey $ delegationKey
  859 
  860         -- base address: keyhash28,scripthash32    : 00100000 -> 32
  861         -- base address: scripthash32,scripthash32 : 00110000 -> 48
  862         DelegationFromScript (ScriptHash scriptBytes) -> do
  863             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  864                 -- 0b01100000 .&. 0b00111111 = 32
  865                 -- 0b01110000 .&. 0b00111111 = 48
  866                 putWord8 $ fstByte .&. 0b00111111
  867                 putByteString rest
  868                 putByteString scriptBytes
  869 
  870         -- pointer address: keyhash28, 3 variable length uint    : 01000000 -> 64
  871         -- pointer address: scripthash32, 3 variable length uint : 01010000 -> 80
  872         DelegationFromPointer pointer -> do
  873             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  874                 -- 0b01100000 .&. 0b01011111 = 64
  875                 -- 0b01110000 .&. 0b01011111 = 80
  876                 putWord8 $ fstByte .&. 0b01011111
  877                 putByteString rest
  878                 putPointer pointer
  879   where
  880     putPointer (ChainPointer a b c) = do
  881         putVariableLengthNat a
  882         putVariableLengthNat b
  883         putVariableLengthNat c

 1052 -- Helper to constructs appropriate address headers. Rest of the payload is left
 1053 -- to the caller as a raw 'ByteString'.
 1054 constructPayload
 1055     :: AddressType
 1056     -> NetworkDiscriminant Shelley
 1057     -> ByteString
 1058     -> Address
 1059 constructPayload addrType discrimination bytes = unsafeMkAddress $
 1060     invariantSize expectedLength $ BL.toStrict $ runPut $ do
 1061         putWord8 firstByte
 1062         putByteString bytes
 1063   where
 1064     firstByte =
 1065         let netTagLimit = 16
 1066         in addressType addrType + invariantNetworkTag netTagLimit (networkTag @Shelley discrimination)
 1067     expectedLength =
 1068         let headerSizeBytes = 1
 1069         in headerSizeBytes + credentialHashSize

 1088     -> Word32
 1089     -> XPrv
 1090 deriveAccountPrivateKeyShelley rootXPrv accIx purpose =

 1094         Just coinTypeIx =

 1097             deriveXPrv DerivationScheme2 rootXPrv purposeIx
 1098         coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
 1099             deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
 1100         acctXPrv = -- lvl3 derivation; hardened derivation of account' index
 1101             deriveXPrv DerivationScheme2 coinTypeXPrv accIx
