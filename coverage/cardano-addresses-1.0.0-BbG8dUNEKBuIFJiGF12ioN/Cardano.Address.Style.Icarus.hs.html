  122 -- $overview
  123 --
  124 -- This module provides an implementation of:
  125 --
  126 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Icarus master keys from mnemonic sentences
  127 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  128 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  129 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  130 --
  131 -- We call 'Icarus' addresses the new format of Cardano addresses which came
  132 -- after 'Cardano.Address.Style.Byron.Byron'. This is the format initially used in /Yoroi/
  133 -- and now also used by /Daedalus/.
  134 
  135 -- | A cryptographic key for sequential-scheme address derivation, with
  136 -- phantom-types to disambiguate key types.
  137 --
  138 -- @
  139 -- let rootPrivateKey = Icarus 'RootK XPrv
  140 -- let accountPubKey  = Icarus 'AccountK XPub
  141 -- let addressPubKey  = Icarus 'AddressK XPub
  142 -- @
  143 --
  144 -- @since 1.0.0
  145 newtype Icarus (depth :: Depth) key = Icarus
  146     { getKey :: key
  147         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  148         --
  149         -- @since 1.0.0
  150     }
  151     deriving stock (Generic, Show, Eq)
  152 
  153 deriving instance (Functor (Icarus depth))
  154 instance (NFData key) => NFData (Icarus depth key)

  156 --
  157 -- Key Derivation
  158 --
  159 -- $keyDerivation
  160 --
  161 -- === Generating a root key from 'SomeMnemonic'
  162 -- > :set -XOverloadedStrings
  163 -- > :set -XTypeApplications
  164 -- > :set -XDataKinds
  165 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  166 -- >
  167 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  168 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  169 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Icarus 'RootK XPrv
  170 --
  171 -- === Deriving child keys
  172 --
  173 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  174 --
  175 -- > import Cardano.Address.Derivation ( AccountingStyle(..) )
  176 -- >
  177 -- > let accIx = toEnum 0x80000000
  178 -- > let acctK = deriveAccountPrivateKey rootK accIx
  179 -- >
  180 -- > let addIx = toEnum 0x00000014
  181 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  182 
  183 instance Internal.GenMasterKey Icarus where
  184     type SecondFactor Icarus = ScrubbedBytes
  185 
  186     genMasterKeyFromXPrv = liftXPrv
  187     genMasterKeyFromMnemonic (SomeMnemonic mw) sndFactor =
  188         let
  189             seed  = entropyToBytes $ mnemonicToEntropy mw
  190             seedValidated = assert
  191                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  192                 seed
  193         in Icarus $ generateNew seedValidated sndFactor

  195 instance Internal.HardDerivation Icarus where
  196     type AccountIndexDerivationType Icarus = 'Hardened
  197     type AddressIndexDerivationType Icarus = 'Soft
  198     type WithAccountStyle Icarus = AccountingStyle
  199 
  200     deriveAccountPrivateKey (Icarus rootXPrv) accIx =
  201         let
  202             purposeIx =
  203                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  204             coinTypeIx =
  205                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  206             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  207                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  208             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  209                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  210             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  211                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  212         in
  213             Icarus acctXPrv
  214 
  215     deriveAddressPrivateKey (Icarus accXPrv) accountingStyle addrIx =
  216         let
  217             changeCode =
  218                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  219             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  220                 deriveXPrv DerivationScheme2 accXPrv changeCode
  221             addrXPrv = -- lvl5 derivation; soft derivation of address index
  222                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  223         in
  224             Icarus addrXPrv
  225 
  226 instance Internal.SoftDerivation Icarus where
  227     deriveAddressPublicKey (Icarus accXPub) accountingStyle addrIx =
  228         fromMaybe errWrongIndex $ do
  229             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  230             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  231                 deriveXPub DerivationScheme2 accXPub changeCode
  232             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  233                 deriveXPub DerivationScheme2 changeXPub addrIx
  234             return $ Icarus addrXPub
  235       where
  236         errWrongIndex = error $
  237             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  238             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  239             \either a programmer error, or, we may have reached the maximum \
  240             \number of addresses for a given wallet."

  305 -- === Generating a 'PaymentAddress'
  306 --
  307 -- > import Cardano.Address ( bech32 )
  308 -- > import Cardano.Address.Derivation ( AccountingStyle(..), toXPub(..) )
  309 -- >
  310 -- > bech32 $ paymentAddress icarusMainnet (toXPub <$> addrK)
  311 -- > "addr1vxpfffuj3zkp5g7ct6h4va89caxx9ayq2gvkyfvww48sdncxsce5t"
  312 --
  313 -- | Analyze an 'Address' to know whether it's an Icarus address or not.
  314 -- Returns 'Nothing' if the address isn't a byron address, or return a
  315 -- structured JSON that gives information about an address.
  316 --
  317 -- @since 2.0.0
  318 inspectIcarusAddress :: Address -> Maybe Json.Value
  319 inspectIcarusAddress addr = do
  320     payload <- CBOR.deserialiseCbor CBOR.decodeAddressPayload bytes
  321     (root, attrs) <- CBOR.deserialiseCbor decodePayload payload
  322     guard $ 1 `notElem` (fst <$> attrs)
  323     ntwrk <- CBOR.deserialiseCbor CBOR.decodeProtocolMagicAttr payload
  324     pure $ Json.object
  325         [ "address_style"   .= Json.String "Icarus"
  326         , "stake_reference" .= Json.String "none"
  327         , "address_root"    .= T.unpack (T.decodeUtf8 $ encode EBase16 root)
  328         , "network_tag"     .= maybe Json.Null toJSON ntwrk
  329         ]
  330   where
  331     bytes :: ByteString
  332     bytes = unAddress addr
  333 
  334     decodePayload :: forall s. CBOR.Decoder s (ByteString, [(Word8, ByteString)])
  335     decodePayload = do
  336         _ <- CBOR.decodeListLenCanonicalOf 3
  337         root <- CBOR.decodeBytes
  338         (root,) <$> CBOR.decodeAllAttributes

  423 -- Purpose is a constant set to 44' (or 0x8000002C) following the original
  424 -- BIP-44 specification.
  425 --
  426 -- It indicates that the subtree of this node is used according to this
  427 -- specification.
  428 --
  429 -- Hardened derivation is used at this level.
  430 purposeIndex :: Word32
  431 purposeIndex = 0x8000002C
  432 
  433 -- One master node (seed) can be used for unlimited number of independent
  434 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  435 -- same space for various cryptocoins has some disadvantages.
  436 --
  437 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  438 -- addresses across cryptocoins and improving privacy issues.
  439 --
  440 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  441 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  442 -- Lovelace.
  443 --
  444 -- Hardened derivation is used at this level.
  445 coinTypeIndex :: Word32
  446 coinTypeIndex = 0x80000717

  448 -- The minimum seed length for 'generateKeyFromMnemonic' and 'unsafeGenerateKeyFromMnemonic'.
  449 minSeedLengthBytes :: Int
  450 minSeedLengthBytes = 16
  451 
  452 -- Hardware Ledger devices generates keys from mnemonic using a different
  453 -- approach (different from the rest of Cardano).
  454 --
  455 -- It is a combination of:
  456 --
  457 -- - [SLIP 0010](https://github.com/satoshilabs/slips/blob/master/slip-0010.md)
  458 -- - [BIP 0032](https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki)
  459 -- - [BIP 0039](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki)
  460 -- - [RFC 8032](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  461 -- - What seems to be arbitrary changes from Ledger regarding the calculation of
  462 --   the initial chain code and generation of the root private key.
  463 unsafeGenerateKeyFromHardwareLedger
  464     :: SomeMnemonic
  465         -- ^ The root mnemonic
  466     -> Icarus 'RootK XPrv
  467 unsafeGenerateKeyFromHardwareLedger (SomeMnemonic mw) = unsafeFromRight $ do
  468     let seed = pbkdf2HmacSha512
  469             $ T.encodeUtf8
  470             $ T.intercalate " "
  471             $ mnemonicToText mw
  472 
  473     -- NOTE
  474     -- SLIP-0010 refers to `iR` as the chain code. Here however, the chain code
  475     -- is obtained as a hash of the initial seed whereas iR is used to make part
  476     -- of the root private key itself.
  477     let cc = hmacSha256 (BS.pack [1] <> seed)
  478     let (iL, iR) = first pruneBuffer $ hashRepeatedly seed
  479 
  480     prv <- maybe (Left "invalid xprv") pure $ xprvFromBytes $ iL <> iR <> cc
  481     pure $ Icarus prv
  482   where
  483     -- Errors yielded in the body of 'unsafeGenerateKeyFromHardwareLedger' are
  484     -- programmer errors (out-of-range byte buffer access or, invalid length for
  485     -- cryptographic operations). Therefore, we throw badly if we encounter any.
  486     unsafeFromRight :: Either String a -> a
  487     unsafeFromRight = either error id
  488 
  489     -- This is the algorithm described in SLIP 0010 for master key generation
  490     -- with an extra step to discard _some_ of the potential private keys. Why
  491     -- this extra step remains a mystery as of today.
  492     --
  493     --      1. Generate a seed byte sequence S of 512 bits according to BIP-0039.
  494     --         (done in a previous step, passed as argument).
  495     --
  496     --      2. Calculate I = HMAC-SHA512(Key = "ed25519 seed", Data = S)
  497     --
  498     --      3. Split I into two 32-byte sequences, IL and IR.
  499     --
  500     -- extra *******************************************************************
  501     -- *                                                                       *
  502     -- *    3.5 If the third highest bit of the last byte of IL is not zero    *
  503     -- *        S = I and go back to step 2.                                   *
  504     -- *                                                                       *
  505     -- *************************************************************************
  506     --
  507     --      4. Use parse256(IL) as master secret key, and IR as master chain code.
  508     hashRepeatedly :: ByteString -> (ByteString, ByteString)
  509     hashRepeatedly bytes = case BS.splitAt 32 (hmacSha512 bytes) of
  510         (iL, iR) | isInvalidKey iL -> hashRepeatedly (iL <> iR)
  511         (iL, iR) -> (iL, iR)
  512       where
  513         isInvalidKey k = testBit (k `BS.index` 31) 5
  514 
  515     -- - Clear the lowest 3 bits of the first byte
  516     -- - Clear the highest bit of the last byte
  517     -- - Set the second highest bit of the last byte
  518     --
  519     -- As described in [RFC 8032 - 5.1.5](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  520     pruneBuffer :: ByteString -> ByteString
  521     pruneBuffer bytes =
  522         let
  523             (firstByte, rest) = fromMaybe (error "pruneBuffer: no first byte") $
  524                 BS.uncons bytes
  525 
  526             (rest', lastByte) = fromMaybe (error "pruneBuffer: no last byte") $
  527                 BS.unsnoc rest
  528 
  529             firstPruned = firstByte
  530                 & (`clearBit` 0)
  531                 & (`clearBit` 1)
  532                 & (`clearBit` 2)
  533 
  534             lastPruned = lastByte
  535                 & (`setBit` 6)
  536                 & (`clearBit` 7)
  537         in
  538             (firstPruned `BS.cons` BS.snoc rest' lastPruned)
  539 
  540     -- As described in [BIP 0039 - From Mnemonic to Seed](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki#from-mnemonic-to-seed)
  541     pbkdf2HmacSha512 :: ByteString -> ByteString
  542     pbkdf2HmacSha512 bytes = PBKDF2.generate
  543         (PBKDF2.prfHMAC SHA512)
  544         (PBKDF2.Parameters 2048 64)
  545         bytes
  546         ("mnemonic" :: ByteString)
  547 
  548     hmacSha256 :: ByteString -> ByteString
  549     hmacSha256 =
  550         BA.convert @(HMAC SHA256) . hmac salt
  551 
  552     -- As described in [SLIP 0010 - Master Key Generation](https://github.com/satoshilabs/slips/blob/master/slip-0010.md#master-key-generation)
  553     hmacSha512 :: ByteString -> ByteString
  554     hmacSha512 =
  555         BA.convert @(HMAC SHA512) . hmac salt
  556 
  557     salt :: ByteString
  558     salt = "ed25519 seed"
