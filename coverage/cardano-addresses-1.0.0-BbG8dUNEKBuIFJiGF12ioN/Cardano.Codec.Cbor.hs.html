   85 In the composition of a Cardano address, the following functions concern the
   86 "Derivation Path" box.
   87 
   88 +-------------------------------------------------------------------------------+
   89 |                                                                               |
   90 |                        CBOR-Serialized Object with CRC¹                       |
   91 |                                                                               |
   92 +-------------------------------------------------------------------------------+
   93                                         |
   94                                         |
   95                                         v
   96 +-------------------------------------------------------------------------------+
   97 |     Address Root    |     Address Attributes    |           AddrType          |
   98 |                     |                           |                             |
   99 |   Hash (224 bits)   |  Der. Path² + Stake + NM  |  PubKey | (Script) | Redeem |
  100 |                     |    (open for extension)   |     (open for extension)    |
  101 +-------------------------------------------------------------------------------+
  102              |                 |
  103              |                 |     +----------------------------------+
  104              v                 |     |        Derivation Path           |
  105 +---------------------------+  |---->|                                  |
  106 | SHA3-256                  |  |     | ChaChaPoly⁴ AccountIx/AddressIx  |
  107 |   |> Blake2b 224          |  |     +----------------------------------+
  108 |   |> CBOR                 |  |
  109 |                           |  |
  110 |  -AddrType                |  |     +----------------------------------+
  111 |  -ASD³ (~AddrType+PubKey) |  |     |       Stake Distribution         |
  112 |  -Address Attributes      |  |     |                                  |
  113 +---------------------------+  |---->|  BootstrapEra | (Single | Multi) |
  114                                |     +----------------------------------+
  115                                |
  116                                |
  117                                |     +----------------------------------+
  118                                |     |          Network Magic           |
  119                                |---->|                                  |
  120                                      | Addr Discr: MainNet vs TestNet   |
  121                                      +----------------------------------+
  122 
  123 -------------------------------------------------------------------------------}

  125 -- * Encoding

  127 -- | Encode a public key to a corresponding Cardano Address. The encoding of the
  128 -- attributes part of an address is left out to the caller; This allows for
  129 -- distinguishing between Sequential and Random addresses (the former doesn't
  130 -- have any attributes to encode).
  131 --
  132 -- @
  133 -- -- Old / Random Addresses
  134 -- let encodeAddrAttributes = mempty
  135 --      <> CBOR.encodeMapLen 1
  136 --      <> CBOR.encodeWord8 1
  137 --      <> encodeDerivationPath (hdPassphrase rootXPub) accIx addrIx
  138 -- let addr = encodeAddress xpub encodeAddrAttributes
  139 --
  140 -- -- New / Sequential Addresses
  141 -- let encodeAddrAttributes = mempty <> CBOR.encodeMapLen 0
  142 -- let addr = encodeAddress xpub encodeAddrAttributes
  143 -- @
  144 --
  145 -- Note that we are passing the behavior to encode attributes as a parameter
  146 -- here and do not handle multiple cases in 'encodeAddress' itself for multiple
  147 -- reasons:
  148 --
  149 -- - Inversion of control gives us a nicer implementation overall
  150 --
  151 -- - Encoding attributes for Random addresses requires more context than just
  152 --   the public key (like the wallet root id and some extra logic for encoding
  153 --   passphrases). This is just scheme-specific and is better left out of this
  154 --   particular function
  155 encodeAddress :: XPub -> [CBOR.Encoding] -> CBOR.Encoding
  156 encodeAddress (XPub pub (ChainCode cc)) attrs =
  157     encodeAddressPayload payload
  158   where
  159     blake2b224 = hash @_ @Blake2b_224
  160     sha3256 = hash @_ @SHA3_256
  161     payload = CBOR.toStrictByteString $ mempty
  162         <> CBOR.encodeListLen 3
  163         <> CBOR.encodeBytes root
  164         <> encodeAttributes attrs
  165         <> CBOR.encodeWord8 0 -- Address Type, 0 = Public Key
  166     root = BA.convert $ blake2b224 $ sha3256 $ CBOR.toStrictByteString $ mempty
  167         <> CBOR.encodeListLen 3
  168         <> CBOR.encodeWord8 0 -- Address Type, 0 = Public Key
  169         <> encodeSpendingData
  170         <> encodeAttributes attrs
  171     encodeXPub =
  172         CBOR.encodeBytes (pub <> cc)
  173     encodeSpendingData = CBOR.encodeListLen 2
  174         <> CBOR.encodeWord8 0
  175         <> encodeXPub
  176 
  177 encodeAddressPayload :: ByteString -> CBOR.Encoding
  178 encodeAddressPayload payload = mempty
  179     <> CBOR.encodeListLen 2
  180     <> CBOR.encodeTag 24 -- Hard-Coded Tag value in cardano-sl
  181     <> CBOR.encodeBytes payload
  182     <> CBOR.encodeWord32 (crc32 payload)
  183 
  184 encodeAttributes :: [CBOR.Encoding] -> CBOR.Encoding
  185 encodeAttributes attrs = CBOR.encodeMapLen l <> mconcat attrs
  186   where
  187     l = fromIntegral (length attrs)
  188 
  189 encodeProtocolMagicAttr :: Word32 -> CBOR.Encoding
  190 encodeProtocolMagicAttr pm = mempty
  191     <> CBOR.encodeWord 2 -- Tag for 'ProtocolMagic' attribute
  192     <> CBOR.encodeBytes (CBOR.toStrictByteString $ CBOR.encodeWord32 pm)

  194 -- This is the opposite of 'decodeDerivationPathAttr'.
  195 --
  196 -- NOTE: The caller must ensure that the passphrase length is 32 bytes.
  197 encodeDerivationPathAttr
  198     :: ScrubbedBytes
  199     -> Word32
  200     -> Word32
  201     -> CBOR.Encoding
  202 encodeDerivationPathAttr pwd acctIx addrIx = mempty
  203     <> CBOR.encodeWord8 1 -- Tag for 'DerivationPath' attribute
  204     <> CBOR.encodeBytes (encryptDerivationPath pwd path)
  205   where
  206     path = encodeDerivationPath acctIx addrIx
  207 
  208 encodeDerivationPath
  209     :: Word32
  210     -> Word32
  211     -> CBOR.Encoding
  212 encodeDerivationPath acctIx addrIx = mempty
  213     <> CBOR.encodeListLenIndef
  214     <> CBOR.encodeWord32 acctIx
  215     <> CBOR.encodeWord32 addrIx
  216     <> CBOR.encodeBreak

  218 -- | ChaCha20/Poly1305 encrypting and signing the HD payload of addresses.
  219 --
  220 -- NOTE: The caller must ensure that the passphrase length is 32 bytes.
  221 encryptDerivationPath
  222     :: ScrubbedBytes
  223        -- ^ Symmetric key / passphrase, 32-byte long
  224     -> CBOR.Encoding
  225         -- ^ Payload to be encrypted
  226     -> ByteString
  227         -- ^ Ciphertext with a 128-bit crypto-tag appended.
  228 encryptDerivationPath pwd payload = unsafeSerialize $ do
  229     nonce <- Poly.nonce12 cardanoNonce
  230     st1 <- Poly.finalizeAAD <$> Poly.initialize pwd nonce
  231     let (out, st2) = Poly.encrypt (CBOR.toStrictByteString payload) st1
  232     return $ out <> BA.convert (Poly.finalize st2)
  233   where
  234     unsafeSerialize :: CryptoFailable ByteString -> ByteString
  235     unsafeSerialize =
  236         CBOR.toStrictByteString . CBOR.encodeBytes . useInvariant
  237 
  238     -- Encryption will fail if the key is the wrong size, but that won't happen
  239     -- if the key was created with 'generateKeyFromSeed'.
  240     useInvariant = \case
  241         CryptoPassed res -> res
  242         CryptoFailed err -> error $ "encodeAddressKey: " ++ show err
  243 
  244 -- | Hard-coded nonce from the legacy code-base.
  245 cardanoNonce :: ByteString
  246 cardanoNonce = "serokellfore"

  248 decodeAddress :: CBOR.Decoder s ByteString
  249 decodeAddress = do
  250     _ <- CBOR.decodeListLenCanonicalOf 2
  251         -- CRC Protection Wrapper
  252     tag <- CBOR.decodeTag
  253         -- Mysterious hard-coded tag cardano-sl seems to so much like
  254     bytes <- CBOR.decodeBytes
  255         -- Addr Root + Attributes + Type
  256     crc <- CBOR.decodeWord32 -- CRC
  257     -- NOTE 1:
  258     -- Treating addresses as a blob here, so we just re-encode them as such
  259     -- Ultimately for us, addresses are nothing more than a bunch of bytes that
  260     -- we display in a Base58 format when we have to.
  261     --
  262     -- NOTE 2:
  263     -- We may want to check the CRC at this level as-well... maybe not.
  264     return $ CBOR.toStrictByteString $ mempty
  265         <> CBOR.encodeListLen 2
  266         <> CBOR.encodeTag tag
  267         <> CBOR.encodeBytes bytes
  268         <> CBOR.encodeWord32 crc
  269 
  270 decodeAddressPayload :: CBOR.Decoder s ByteString
  271 decodeAddressPayload = do
  272     _ <- CBOR.decodeListLenCanonicalOf 2
  273     _ <- CBOR.decodeTag
  274     bytes <- CBOR.decodeBytes
  275     _ <- CBOR.decodeWord32 -- CRC
  276     return bytes

  278 decodeAddressDerivationPath
  279     :: ScrubbedBytes
  280     -> CBOR.Decoder s (Maybe (Word32, Word32))
  281 decodeAddressDerivationPath pwd = do
  282     _ <- CBOR.decodeListLenCanonicalOf 3
  283     _ <- CBOR.decodeBytes
  284     path <- decodeAllAttributes >>= decodeDerivationPathAttr pwd
  285     addrType <- CBOR.decodeWord8 -- Type
  286     when (addrType /= 0) $
  287         fail $ mconcat
  288             [ "decodeAddressDerivationPath: type is not 0 (public key), it is "
  289             , show addrType
  290             ]
  291     pure path
  292 
  293 decodeProtocolMagicAttr
  294     :: CBOR.Decoder s (Maybe Word32)
  295 decodeProtocolMagicAttr = do
  296     _ <- CBOR.decodeListLenCanonicalOf 3
  297     _ <- CBOR.decodeBytes
  298     attrs <- decodeAllAttributes
  299     case find ((== 2) . fst) attrs of
  300         Nothing -> pure Nothing
  301         Just (_, bytes) -> case deserialiseCbor CBOR.decodeWord32 bytes of
  302             Nothing -> fail "unable to decode attribute into protocol magic"
  303             Just pm -> pure (Just pm)
  304 
  305 -- | The attributes are pairs of numeric tags and bytes, where the bytes will be
  306 -- CBOR-encoded stuff. This decoder does not enforce "canonicity" of entries.
  307 decodeAllAttributes
  308     :: CBOR.Decoder s [(Word8, ByteString)]
  309 decodeAllAttributes = do
  310     n <- CBOR.decodeMapLenCanonical -- Address Attributes length
  311     replicateM n decodeAttr
  312   where
  313     decodeAttr = (,) <$> CBOR.decodeWord8 <*> CBOR.decodeBytes
  314 
  315 decodeDerivationPathAttr
  316     :: ScrubbedBytes
  317     -> [(Word8, ByteString)]
  318     -> CBOR.Decoder s (Maybe (Word32, Word32))
  319 decodeDerivationPathAttr pwd attrs = do
  320     case lookup derPathTag attrs of
  321         Just payload -> do
  322             decodeNestedBytes decoder payload
  323         Nothing -> fail $ mconcat
  324             [ "decodeDerivationPathAttr: Missing attribute "
  325             , show derPathTag
  326             ]
  327   where
  328     derPathTag = 1
  329     decoder :: CBOR.Decoder s (Maybe (Word32, Word32))
  330     decoder = do
  331         bytes <- CBOR.decodeBytes
  332         case decryptDerivationPath pwd bytes of
  333             CryptoPassed plaintext ->
  334                 Just <$> decodeNestedBytes decodeDerivationPath plaintext
  335             CryptoFailed _ ->
  336                 pure Nothing
  337 
  338 -- | ChaCha20/Poly1305 decrypting and authenticating the HD payload of
  339 -- addresses.
  340 decryptDerivationPath
  341     :: ScrubbedBytes
  342        -- ^ Symmetric key / passphrase, 32-byte long
  343     -> ByteString
  344         -- ^ Payload to be decrypted
  345     -> CryptoFailable ByteString
  346 decryptDerivationPath pwd bytes = do
  347     let (payload, tag) = BS.splitAt (BS.length bytes - 16) bytes
  348     nonce <- Poly.nonce12 cardanoNonce
  349     st1 <- Poly.finalizeAAD <$> Poly.initialize pwd nonce
  350     let (out, st2) = Poly.decrypt payload st1
  351     when (BA.convert (Poly.finalize st2) /= tag) $
  352         CryptoFailed CryptoError_MacKeyInvalid
  353     return out

  355 -- Opposite of 'encodeDerivationPath'.
  356 decodeDerivationPath
  357     :: CBOR.Decoder s (Word32, Word32)
  358 decodeDerivationPath = do
  359     ixs <- decodeListIndef CBOR.decodeWord32
  360     case ixs of
  361         [acctIx, addrIx] ->
  362             pure (acctIx, addrIx)
  363         _ ->
  364             fail $ mconcat
  365                 [ "decodeDerivationPath: invalid derivation path payload: "
  366                 , "expected two indexes but got: "
  367                 , show ixs
  368                 ]
  369 -- | Decode an arbitrary long list. CBOR introduce a "break" character to
  370 -- mark the end of the list, so we simply decode each item until we encounter
  371 -- a break character.
  372 --
  373 -- @
  374 --     myDecoder :: CBOR.Decoder s [MyType]
  375 --     myDecoder = decodeListIndef decodeOne
  376 --       where
  377 --         decodeOne :: CBOR.Decoder s MyType
  378 -- @
  379 decodeListIndef :: forall s a. CBOR.Decoder s a -> CBOR.Decoder s [a]
  380 decodeListIndef decodeOne = do
  381     _ <- CBOR.decodeListLenIndef
  382     CBOR.decodeSequenceLenIndef (flip (:)) [] reverse decodeOne

  384 -- | Byron CBOR encodings often have CBOR nested in CBOR. This helps decoding
  385 -- a particular 'ByteString' that represents a CBOR object.
  386 decodeNestedBytes
  387     :: MonadFail m
  388     => (forall s. CBOR.Decoder s r)
  389     -> ByteString
  390     -> m r
  391 decodeNestedBytes dec bytes =
  392     case CBOR.deserialiseFromBytes dec (BL.fromStrict bytes) of
  393         Right ("", res) ->
  394             pure res
  395         Right _ ->
  396             fail "Leftovers when decoding nested bytes"
  397         _ ->
  398             fail "Could not decode nested bytes"

  400 -- | Shortcut for deserialising a strict 'Bytestring' with the given decoder.
  401 deserialiseCbor
  402     :: (forall s. CBOR.Decoder s a)
  403     -> ByteString
  404     -> Maybe a
  405 deserialiseCbor dec =
  406     fmap snd . eitherToMaybe . CBOR.deserialiseFromBytes dec . BL.fromStrict
  407 
  408 -- | CBOR deserialise without error handling - handy for prototypes or testing.
  409 unsafeDeserialiseCbor
  410     :: HasCallStack
  411     => (forall s. CBOR.Decoder s a)
  412     -> BL.ByteString
  413     -> a
  414 unsafeDeserialiseCbor decoder bytes = either
  415     (\e -> error $ "unsafeSerializeCbor: " <> show e)
  416     snd
  417     (CBOR.deserialiseFromBytes decoder bytes)
