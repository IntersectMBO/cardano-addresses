  143 -- $overview
  144 --
  145 -- This module provides an implementation of:
  146 --
  147 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Shelley master keys from mnemonic sentences
  148 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  149 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  150 -- - 'Cardano.Address.PaymentAddress': for constructing payment addresses from a address public key
  151 -- - 'Cardano.Address.DelegationAddress': for constructing delegation addresses from address and staking public keys

  153 -- | A cryptographic key for sequential-scheme address derivation, with
  154 -- phantom-types to disambiguate key types.
  155 --
  156 -- @
  157 -- let rootPrivateKey = Shelley 'RootK XPrv
  158 -- let accountPubKey  = Shelley 'AccountK XPub
  159 -- let addressPubKey  = Shelley 'AddressK XPub
  160 -- @
  161 --
  162 -- @since 2.0.0
  163 newtype Shelley (depth :: Depth) key = Shelley
  164     { getKey :: key
  165         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  166         --
  167         -- @since 1.0.0
  168     }
  169     deriving stock (Generic, Show, Eq)
  170 
  171 deriving instance (Functor (Shelley depth))
  172 instance (NFData key) => NFData (Shelley depth key)

  174 --
  175 -- Key Derivation
  176 --
  177 -- $keyDerivation
  178 --
  179 -- === Generating a root key from 'SomeMnemonic'
  180 -- > :set -XOverloadedStrings
  181 -- > :set -XTypeApplications
  182 -- > :set -XDataKinds
  183 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  184 -- >
  185 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  186 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  187 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shelley 'RootK XPrv
  188 --
  189 -- === Deriving child keys
  190 --
  191 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  192 --
  193 -- > import Cardano.Address.Derivation ( AccountingStyle(..) )
  194 -- >
  195 -- > let accIx = toEnum 0x80000000
  196 -- > let acctK = deriveAccountPrivateKey rootK accIx
  197 -- >
  198 -- > let addIx = toEnum 0x00000014
  199 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  200 --
  201 -- > let stakeK = deriveStakingPrivateKey acctK
  202 
  203 instance Internal.GenMasterKey Shelley where
  204     type SecondFactor Shelley = ScrubbedBytes
  205 
  206     genMasterKeyFromXPrv = liftXPrv
  207     genMasterKeyFromMnemonic fstFactor sndFactor =
  208         Shelley $ generateNew seedValidated sndFactor
  209         where
  210             seed  = someMnemonicToBytes fstFactor
  211             seedValidated = assert
  212                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  213                 seed

  215 instance Internal.HardDerivation Shelley where
  216     type AccountIndexDerivationType Shelley = 'Hardened
  217     type AddressIndexDerivationType Shelley = 'Soft
  218     type WithAccountStyle Shelley = AccountingStyle
  219 
  220     deriveAccountPrivateKey (Shelley rootXPrv) accIx =
  221         let
  222             purposeIx =
  223                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  224             coinTypeIx =
  225                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  226             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  227                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  228             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  229                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  230             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  231                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  232         in
  233             Shelley acctXPrv
  234 
  235     deriveAddressPrivateKey (Shelley accXPrv) accountingStyle addrIx =
  236         let
  237             changeCode =
  238                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  239             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  240                 deriveXPrv DerivationScheme2 accXPrv changeCode
  241             addrXPrv = -- lvl5 derivation; soft derivation of address index
  242                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  243         in
  244             Shelley addrXPrv
  245 
  246 instance Internal.SoftDerivation Shelley where
  247     deriveAddressPublicKey (Shelley accXPub) accountingStyle addrIx =
  248         fromMaybe errWrongIndex $ do
  249             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  250             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  251                 deriveXPub DerivationScheme2 accXPub changeCode
  252             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  253                 deriveXPub DerivationScheme2 changeXPub addrIx
  254             return $ Shelley addrXPub
  255       where
  256         errWrongIndex = error $
  257             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  258             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  259             \either a programmer error, or, we may have reached the maximum \
  260             \number of addresses for a given wallet."
  261 
  262 instance Internal.StakingDerivation Shelley where
  263     deriveStakingPrivateKey (Shelley accXPrv) =
  264         let
  265             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  266                 deriveXPrv DerivationScheme2 accXPrv (toEnum @(Index 'Soft _) 2)
  267             stakeXPrv = -- lvl5 derivation; soft derivation of address index
  268                 deriveXPrv DerivationScheme2 changeXPrv (minBound @(Index 'Soft _))
  269         in
  270             Shelley stakeXPrv

  374 -- > let ptr = ChainPointer 123 1 2
  375 -- > bech32 $ pointerAddress tag (toXPub <$> addrK) ptr

  378 instance Internal.PaymentAddress Shelley where
  379     paymentAddress discrimination k = unsafeMkAddress $
  380         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  381             putWord8 firstByte
  382             putByteString (blake2b224 k)
  383       where
  384           -- we use here the fact that payment address stands for what is named
  385           -- as enterprise address, ie., address carrying no stake rights. For
  386           -- rationale why we may need such addresses refer to delegation
  387           -- specification - Section 3.2.3. What is important here is that the
  388           -- address is composed of discrimination byte and 28 bytes hashed public key.
  389           -- Moreover, it was decided that first 4 bits for enterprise address
  390           -- will be `0110`. The next for 4 bits are reserved for network discriminator.
  391           -- `0110 0000` is 96 in decimal.
  392           firstByte =
  393               96 + invariantNetworkTag 16 (networkTag @Shelley discrimination)
  394           expectedLength = 1 + pubkeyHashSize
  395 
  396 instance Internal.DelegationAddress Shelley where

  416 -- | Analyze an 'Address' to know whether it's a Shelley address or not.
  417 --
  418 -- Returns 'Nothing' if it's not a valid Shelley address, or a ready-to-print
  419 -- string giving details about the 'Address'.
  420 --
  421 -- @since 2.0.0
  422 inspectShelleyAddress :: Address -> Maybe Json.Value
  423 inspectShelleyAddress addr
  424     | BS.length bytes < 1 + pubkeyHashSize = Nothing
  425     | otherwise =
  426         let
  427             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  428             addrType = fstByte .&. 0b11110000
  429             network  = fstByte .&. 0b00001111
  430         in
  431             case addrType of
  432                -- 0000: base address: keyhash28,keyhash28
  433                 0b00000000 | BS.length rest == 2 * pubkeyHashSize ->
  434                     Just $ Json.object
  435                         [ "address_style"     .= Json.String "Shelley"
  436                         , "stake_reference"   .= Json.String "by value"
  437                         , "spending_key_hash" .= base16 (BS.take pubkeyHashSize rest)
  438                         , "stake_key_hash"    .= base16 (BS.drop pubkeyHashSize rest)
  439                         , "network_tag"       .= network
  440                         ]
  441                -- 0001: base address: scripthash32,keyhash28
  442                 0b00010000 | BS.length rest == pubkeyHashSize + scriptHashSize ->
  443                     Just $ Json.object
  444                         [ "address_style"     .= Json.String "Shelley"
  445                         , "stake_reference"   .= Json.String "by value"
  446                         , "script_hash"       .= base16 (BS.take scriptHashSize rest)
  447                         , "stake_key_hash"    .= base16 (BS.drop scriptHashSize rest)
  448                         , "network_tag"       .= network
  449                         ]
  450                -- 0010: base address: keyhash28,scripthash32
  451                 0b00100000 | BS.length rest == pubkeyHashSize + scriptHashSize ->
  452                     Just $ Json.object
  453                         [ "address_style"     .= Json.String "Shelley"
  454                         , "stake_reference"   .= Json.String "by value"
  455                         , "spending_key_hash" .= base16 (BS.take pubkeyHashSize rest)
  456                         , "stake_script_hash" .= base16 (BS.drop pubkeyHashSize rest)
  457                         , "network_tag"       .= network
  458                         ]
  459                -- 0011: base address: scripthash32,scripthash32
  460                 0b00110000 | BS.length rest == 2 * scriptHashSize ->
  461                     Just $ Json.object
  462                         [ "address_style"     .= Json.String "Shelley"
  463                         , "stake_reference"   .= Json.String "by value"
  464                         , "script_hash"       .= base16 (BS.take scriptHashSize rest)
  465                         , "stake_script_hash" .= base16 (BS.drop scriptHashSize rest)
  466                         , "network_tag"       .= network
  467                         ]
  468                -- 0100: pointer address: keyhash28, 3 variable length uint
  469                -- TODO Could fo something better for pointer and try decoding
  470                --      the pointer
  471                 0b01000000 | BS.length rest > pubkeyHashSize -> do
  472                     ptr <- getPtr (BS.drop pubkeyHashSize rest)
  473                     Just $ Json.object
  474                         [ "address_style"     .= Json.String "Shelley"
  475                         , "stake_reference"   .= Json.String "by pointer"
  476                         , "spending_key_hash" .= base16 (BS.take pubkeyHashSize rest)
  477                         , "pointer"           .= ptrToJSON ptr
  478                         , "network_tag"       .= network
  479                         ]
  480                -- 0101: pointer address: scripthash32, 3 variable length uint
  481                -- TODO Could fo something better for pointer and try decoding
  482                --      the pointer
  483                 0b01010000 | BS.length rest > scriptHashSize -> do
  484                     ptr <- getPtr (BS.drop scriptHashSize rest)
  485                     Just $ Json.object
  486                         [ "address_style"     .= Json.String "Shelley"
  487                         , "stake_reference"   .= Json.String "by pointer"
  488                         , "script_hash"       .= base16 (BS.take scriptHashSize rest)
  489                         , "pointer"           .= ptrToJSON ptr
  490                         , "network_tag"       .= network
  491                         ]
  492                -- 0110: enterprise address: keyhash28
  493                 0b01100000 | BS.length rest == pubkeyHashSize ->
  494                     Just $ Json.object
  495                         [ "address_style"     .= Json.String "Shelley"
  496                         , "stake_reference"   .= Json.String "none"
  497                         , "spending_key_hash" .= base16 (BS.take pubkeyHashSize rest)
  498                         , "network_tag"       .= network
  499                         ]
  500                -- 0111: enterprise address: scripthash32
  501                 0b01110000 | BS.length rest == scriptHashSize ->
  502                     Just $ Json.object
  503                         [ "address_style"     .= Json.String "Shelley"
  504                         , "stake_reference"   .= Json.String "none"
  505                         , "script_hash"       .= base16 (BS.take scriptHashSize rest)
  506                         , "network_tag"       .= network
  507                         ]
  508                -- 1000: byron address
  509                 0b10000000 ->
  510                     inspectByronAddress addr <|> inspectIcarusAddress addr
  511                -- 1110: reward account: keyhash28
  512                 0b11100000 | BS.length rest == pubkeyHashSize ->
  513                     Just $ Json.object
  514                         [ "address_style"     .= Json.String "Shelley"
  515                         , "stake_reference"   .= Json.String "by value"
  516                         , "stake_key_hash"    .= base16 (BS.take pubkeyHashSize rest)
  517                         , "network_tag"       .= network
  518                         ]
  519                -- 1111: reward account: scripthash32
  520                 0b11110000 | BS.length rest == scriptHashSize ->
  521                     Just $ Json.object
  522                         [ "address_style"     .= Json.String "Shelley"
  523                         , "stake_reference"   .= Json.String "by value"
  524                         , "script_hash"       .= base16 (BS.take scriptHashSize rest)
  525                         , "network_tag"       .= network
  526                         ]
  527                 _ ->
  528                     Nothing
  529   where
  530     bytes  = unAddress addr
  531     base16 = T.unpack . T.decodeUtf8 . encode EBase16

  563 -- | Extend an existing payment 'Address' to make it a delegation address.
  564 --
  565 -- @since 2.0.0
  566 extendAddress
  567     :: Address
  568     -> Either (Shelley 'StakingK XPub) ChainPointer
  569     -> Either ErrExtendAddress Address
  570 extendAddress addr stakeReference = do
  571     when (isNothing (inspectShelleyAddress addr)) $
  572         Left $ ErrInvalidAddressStyle "Given address isn't a Shelley address"
  573 
  574     let bytes = unAddress addr
  575     let (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  576 
  577     when ((fstByte .&. 0b11110000) /= 0b01100000) $ do
  578         Left $ ErrInvalidAddressType "Only payment addresses can be extended"
  579 
  580     case stakeReference of
  581         Left stakingKey -> do
  582             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  583                 putWord8 $ fstByte .&. 0b00001111
  584                 putByteString rest
  585                 putByteString . blake2b224 $ stakingKey
  586 
  587         Right pointer -> do
  588             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  589                 putWord8 $ fstByte .&. 0b01001111
  590                 putByteString rest
  591                 putPointer pointer
  592   where
  593     putPointer (ChainPointer a b c) = do
  594         putVariableLengthNat a
  595         putVariableLengthNat b
  596         putVariableLengthNat c

  733 -- One master node (seed) can be used for unlimited number of independent
  734 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  735 -- same space for various cryptocoins has some disadvantages.
  736 --
  737 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  738 -- addresses across cryptocoins and improving privacy issues.
  739 --
  740 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  741 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  742 -- Lovelace.
  743 --
  744 -- Hardened derivation is used at this level.
