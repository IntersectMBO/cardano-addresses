  115 -- $overview
  116 --
  117 -- This module provides an implementation of:
  118 --
  119 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Byron master keys from mnemonic sentences
  120 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical derivation of parent to child keys
  121 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  122 --
  123 -- We call 'Byron' addresses the old address type used by Daedalus in the early
  124 -- days of Cardano. Using this type of addresses and underlying key scheme is
  125 -- now considered __deprecated__ because of some security implications.
  126 --
  127 -- The internals of the 'Byron' does not matter for the reader, but basically
  128 -- contains what is necessary to perform key derivation and generate addresses
  129 -- from a 'Byron' type.
  130 --
  131 -- == Deprecation Notice
  132 --
  133 -- Unless you have good reason to do so (like writing backward-compatible code
  134 -- with an existing piece), any new implementation __should use__ the
  135 -- 'Cardano.Address.Style.Icarus.Icarus' style for key and addresses.
  136 
  137 
  138 -- | Material for deriving HD random scheme keys, which can be used for making
  139 -- addresses.
  140 --
  141 -- @since 1.0.0
  142 data Byron (depth :: Depth) key = Byron
  143     { getKey :: key
  144     -- ^ The raw private or public key.
  145     --
  146     -- @since 1.0.0
  147     , derivationPath :: DerivationPath depth
  148     -- ^ The address derivation indices for the level of this key.
  149     --
  150     -- @since 1.0.0
  151     , payloadPassphrase :: ScrubbedBytes
  152     -- ^ Used for encryption of the derivation path payload within an address.
  153     --
  154     -- @since 1.0.0
  155     } deriving stock (Generic)
  156 {-# DEPRECATED Byron "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  157 {-# DEPRECATED getKey "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  158 {-# DEPRECATED derivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  159 {-# DEPRECATED payloadPassphrase "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  161 instance (NFData key, NFData (DerivationPath depth)) => NFData (Byron depth key)

  166 -- | The hierarchical derivation indices for a given level/depth.
  167 --
  168 -- @since 1.0.0
  169 type family DerivationPath (depth :: Depth) :: * where
  170     -- The root key is generated from the seed.
  171     DerivationPath 'RootK =
  172         ()
  173     -- The account key is generated from the root key and account index.
  174     DerivationPath 'AccountK =
  175         Index 'WholeDomain 'AccountK
  176     -- The address key is generated from the account key and address index.
  177     DerivationPath 'AddressK =
  178         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  179 {-# DEPRECATED DerivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  182 -- Key Derivation
  183 --
  184 -- === Generating a root key from 'SomeMnemonic'
  185 -- > :set -XOverloadedStrings
  186 -- > :set -XTypeApplications
  187 -- > :set -XDataKinds
  188 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  189 -- >
  190 -- > let (Right mw) = mkSomeMnemonic @'[12] ["moon","fox","ostrich","quick","cactus","raven","wasp","intact","first","ring","crumble","error"]
  191 -- > let rootK = genMasterKeyFromMnemonic mw :: Byron 'RootK XPrv
  192 --
  193 -- === Deriving child keys
  194 --
  195 -- > let accIx = toEnum 0x80000000
  196 -- > let acctK = deriveAccountPrivateKey rootK accIx
  197 -- >
  198 -- > let addIx = toEnum 0x80000014
  199 -- > let addrK = deriveAddressPrivateKey acctK addIx
  200 
  201 instance Internal.GenMasterKey Byron where
  202     type SecondFactor Byron = ()
  203 
  204     genMasterKeyFromXPrv xprv =
  205         liftXPrv (toXPub xprv) () xprv
  206     genMasterKeyFromMnemonic (SomeMnemonic mw) () =
  207         liftXPrv (toXPub xprv) () xprv
  208       where
  209         xprv = generate (hashSeed seedValidated)
  210         seed  = entropyToBytes $ mnemonicToEntropy mw
  211         seedValidated = assert
  212             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  213             seed
  214 
  215 instance Internal.HardDerivation Byron where
  216     type AddressIndexDerivationType Byron = 'WholeDomain
  217     type AccountIndexDerivationType Byron = 'WholeDomain
  218     type WithAccountStyle Byron = ()
  219 
  220     deriveAccountPrivateKey rootXPrv accIx = Byron
  221         { getKey = deriveXPrv DerivationScheme1 (getKey rootXPrv) accIx
  222         , derivationPath = accIx
  223         , payloadPassphrase = payloadPassphrase rootXPrv
  224         }
  225 
  226     deriveAddressPrivateKey accXPrv () addrIx = Byron
  227         { getKey = deriveXPrv DerivationScheme1 (getKey accXPrv) addrIx
  228         , derivationPath = (derivationPath accXPrv, addrIx)
  229         , payloadPassphrase = payloadPassphrase accXPrv
  230         }

  291 -- | Analyze an 'Address' to know whether it's a Byron address or not.
  292 -- Returns 'Nothing' if the address isn't a byron address, or return a
  293 -- structured JSON that gives information about an address.
  294 --
  295 -- @since 2.0.0
  296 inspectByronAddress :: Address -> Maybe Json.Value
  297 inspectByronAddress addr = do
  298     payload <- CBOR.deserialiseCbor CBOR.decodeAddressPayload bytes
  299     (root, attrs) <- CBOR.deserialiseCbor decodePayload payload
  300     path  <- find ((== 1) . fst) attrs
  301     ntwrk <- CBOR.deserialiseCbor CBOR.decodeProtocolMagicAttr payload
  302     pure $ Json.object
  303         [ "address_style"   .= Json.String "Byron"
  304         , "stake_reference" .= Json.String "none"
  305         , "address_root"    .= T.unpack (T.decodeUtf8 $ encode EBase16 root)
  306         , "derivation_path" .= T.unpack (T.decodeUtf8 $ encode EBase16 $ snd path)
  307         , "network_tag"     .= maybe Json.Null toJSON ntwrk
  308         ]
  309   where
  310     bytes :: ByteString
  311     bytes = unAddress addr
  312 
  313     decodePayload :: forall s. CBOR.Decoder s (ByteString, [(Word8, ByteString)])
  314     decodePayload = do
  315         _ <- CBOR.decodeListLenCanonicalOf 3
  316         root <- CBOR.decodeBytes
  317         (root,) <$> CBOR.decodeAllAttributes

  319 instance Internal.PaymentAddress Byron where
  320     paymentAddress discrimination k = unsafeMkAddress
  321         $ CBOR.toStrictByteString
  322         $ CBOR.encodeAddress (getKey k) attrs
  323       where
  324         (acctIx, addrIx) = bimap word32 word32 $ derivationPath k
  325         pwd = payloadPassphrase k
  326         NetworkTag magic = networkTag @Byron discrimination
  327         attrs = case addressDiscrimination @Byron discrimination of
  328             RequiresNetworkTag ->
  329                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  330                 , CBOR.encodeProtocolMagicAttr magic
  331                 ]
  332             RequiresNoTag ->
  333                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  334                 ]

  380 -- | Backdoor for generating a new key from a raw 'XPrv'.
  381 --
  382 -- Note that the @depth@ is left open so that the caller gets to decide what type
  383 -- of key this is. This is mostly for testing, in practice, seeds are used to
  384 -- represent root keys, and one should 'genMasterKeyFromXPrv'
  385 --
  386 -- The first argument is a type-family 'DerivationPath' and its type depends on
  387 -- the 'depth' of the key.
  388 --
  389 -- __examples:__
  390 --
  391 -- >>> liftXPrv rootPrv () prv
  392 -- _ :: Byron RootK XPrv
  393 --
  394 -- >>> liftXPrv rootPrv minBound prv
  395 -- _ :: Byron AccountK XPrv
  396 --
  397 -- >>> liftXPrv rootPrv (minBound, minBound) prv
  398 -- _ :: Byron AddressK XPrv
  399 --
  400 -- @since 2.0.0
  401 liftXPrv
  402     :: XPub -- ^ A root public key
  403     -> DerivationPath depth
  404     -> XPrv
  405     -> Byron depth XPrv
  406 liftXPrv rootPub derivationPath getKey = Byron
  407     { getKey
  408     , derivationPath
  409     , payloadPassphrase = hdPassphrase rootPub
  410     }

  438 -- The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  439 minSeedLengthBytes :: Int
  440 minSeedLengthBytes = 16
  441 
  442 -- Hash the seed entropy (generated from mnemonic) used to initiate a HD
  443 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  444 -- minimum for 'generate' (32 bytes).
  445 --
  446 -- Note that our current implementation deviates from BIP-39 because we use a
  447 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  448 --
  449 -- There are two methods of hashing the seed entropy, for different use cases.
  450 --
  451 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  452 --    Blake2b_256, inside a double CBOR serialization sandwich.
  453 --
  454 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  455 --    Blake2b_256, without any serialization.
  456 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  457 hashSeed = serialize . blake2b256 . serialize
  458   where
  459     serialize = BA.convert . cbor . BA.convert
  460     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  461 
  462 -- Hash a byte string through blake2b 256
  463 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  464 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  465 
  466 -- Derive a symmetric key for encrypting and authenticating the address
  467 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  468 -- is employed.
  469 hdPassphrase :: XPub -> ScrubbedBytes
  470 hdPassphrase masterKey =
  471     PBKDF2.generate
  472     (PBKDF2.prfHMAC SHA512)
  473     (PBKDF2.Parameters 500 32)
  474     (xpubToBytes masterKey)
  475     ("address-hashing" :: ByteString)
