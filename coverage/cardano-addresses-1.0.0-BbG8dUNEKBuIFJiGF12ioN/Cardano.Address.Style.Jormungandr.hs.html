  114 -- $overview
  115 --
  116 -- This module provides an implementation of:
  117 --
  118 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Jormungandr master keys from mnemonic sentences
  119 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  120 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  121 -- - 'Cardano.Address.PaymentAddress': for constructing payment addresses from a address public key
  122 -- - 'Cardano.Address.DelegationAddress': for constructing payment addresses from a address and stake public keys
  123 --
  124 -- == Disclaimer
  125 --
  126 -- Beware that this format of address have been specially crafted for the
  127 -- incentivized testnet (ITN). This will soon be __deprecated__ in favor of
  128 -- 'Cardano.Address.Style.Shelley.Shelley'.
  129 --
  130 -- __New integrations should not use this address style__.
  131 
  132 -- | A cryptographic key for sequential-scheme address derivation, with
  133 -- phantom-types to disambiguate key types.
  134 --
  135 -- @
  136 -- let rootPrivateKey = Jormungandr 'RootK XPrv
  137 -- let accountPubKey  = Jormungandr 'AccountK XPub
  138 -- let addressPubKey  = Jormungandr 'AddressK XPub
  139 -- @
  140 --
  141 -- @since 2.0.0
  142 newtype Jormungandr (depth :: Depth) key = Jormungandr
  143     { getKey :: key
  144         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  145         --
  146         -- @since 2.0.0
  147     }
  148     deriving stock (Generic, Show, Eq)
  149 
  150 deriving instance (Functor (Jormungandr depth))
  151 instance (NFData key) => NFData (Jormungandr depth key)

  153 --
  154 -- Key Derivation
  155 --
  156 -- $keyDerivation
  157 --
  158 -- === Generating a root key from 'SomeMnemonic'
  159 -- > :set -XOverloadedStrings
  160 -- > :set -XTypeApplications
  161 -- > :set -XDataKinds
  162 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  163 -- >
  164 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  165 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  166 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Jormungandr 'RootK XPrv
  167 --
  168 -- === Deriving child keys
  169 --
  170 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  171 --
  172 -- > import Cardano.Address.Derivation ( AccountingStyle(..) )
  173 -- >
  174 -- > let accIx = toEnum 0x80000000
  175 -- > let acctK = deriveAccountPrivateKey rootK accIx
  176 -- >
  177 -- > let addIx = toEnum 0x00000014
  178 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  179 --
  180 -- > let stakeK = deriveStakingPrivateKey acctK
  181 
  182 instance Internal.GenMasterKey Jormungandr where
  183     type SecondFactor Jormungandr = ScrubbedBytes
  184 
  185     genMasterKeyFromXPrv = liftXPrv
  186     genMasterKeyFromMnemonic fstFactor sndFactor =
  187         Jormungandr $ generateNew seedValidated sndFactor
  188         where
  189             seed  = someMnemonicToBytes fstFactor
  190             seedValidated = assert
  191                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  192                 seed

  194 instance Internal.HardDerivation Jormungandr where
  195     type AccountIndexDerivationType Jormungandr = 'Hardened
  196     type AddressIndexDerivationType Jormungandr = 'Soft
  197     type WithAccountStyle Jormungandr = AccountingStyle
  198 
  199     deriveAccountPrivateKey (Jormungandr rootXPrv) accIx =
  200         let
  201             purposeIx =
  202                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  203             coinTypeIx =
  204                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  205             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  206                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  207             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  208                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  209             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  210                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  211         in
  212             Jormungandr acctXPrv
  213 
  214     deriveAddressPrivateKey (Jormungandr accXPrv) accountingStyle addrIx =
  215         let
  216             changeCode =
  217                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  218             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  219                 deriveXPrv DerivationScheme2 accXPrv changeCode
  220             addrXPrv = -- lvl5 derivation; soft derivation of address index
  221                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  222         in
  223             Jormungandr addrXPrv
  224 
  225 instance Internal.SoftDerivation Jormungandr where
  226     deriveAddressPublicKey (Jormungandr accXPub) accountingStyle addrIx =
  227         fromMaybe errWrongIndex $ do
  228             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  229             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  230                 deriveXPub DerivationScheme2 accXPub changeCode
  231             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  232                 deriveXPub DerivationScheme2 changeXPub addrIx
  233             return $ Jormungandr addrXPub
  234       where
  235         errWrongIndex = error $
  236             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  237             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  238             \either a programmer error, or, we may have reached the maximum \
  239             \number of addresses for a given wallet."
  240 
  241 instance Internal.StakingDerivation Jormungandr where
  242     deriveStakingPrivateKey (Jormungandr accXPrv) =
  243         let
  244             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  245                 deriveXPrv DerivationScheme2 accXPrv (toEnum @(Index 'Soft _) 2)
  246             stakeXPrv = -- lvl5 derivation; soft derivation of address index
  247                 deriveXPrv DerivationScheme2 changeXPrv (minBound @(Index 'Soft _))
  248         in
  249             Jormungandr stakeXPrv

  346 instance Internal.PaymentAddress Jormungandr where
  347     paymentAddress discrimination k = unsafeMkAddress $
  348         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  349             putWord8 (invariantNetworkTag 255 firstByte)
  350             putByteString (xpubPublicKey $ getKey k)
  351       where
  352           firstByte = networkTag @Jormungandr discrimination
  353           expectedLength = 1 + publicKeySize
  354 
  355 instance Internal.DelegationAddress Jormungandr where
  356     delegationAddress discrimination paymentKey stakingKey = unsafeMkAddress $
  357         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  358             putWord8 (invariantNetworkTag 255 $ NetworkTag $ firstByte + 1)
  359             putByteString . xpubPublicKey . getKey $ paymentKey
  360             putByteString . xpubPublicKey . getKey $ stakingKey
  361       where
  362           (NetworkTag firstByte) = networkTag @Jormungandr discrimination
  363           expectedLength = 1 + 2*publicKeySize

  365 -- | Analyze an 'Address' to know whether it's a Jormungandr address or not.
  366 --
  367 -- Returns 'Nothing' if it's not a valid Shelley address, or a ready-to-print
  368 -- string giving details about the 'Jormungandr'.
  369 --
  370 -- @since 2.0.0
  371 inspectJormungandrAddress :: Address -> Maybe Json.Value
  372 inspectJormungandrAddress addr
  373     | BS.length bytes < 1 + publicKeySize = Nothing
  374     | otherwise =
  375         let
  376             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  377             addrType = fstByte .&. 0b01111111
  378             network  = (fstByte .&. 0b10000000) `shiftR` 7
  379             size = publicKeySize
  380         in
  381             case addrType of
  382                 0x03 | BS.length rest == size ->
  383                     Just $ Json.object
  384                     [ "address_style"   .= Json.String "Jormungandr"
  385                     , "address_type"    .= Json.String "single"
  386                     , "stake_reference" .= Json.String "none"
  387                     , "spending_key"    .= base16 (BS.take size rest)
  388                     , "network_tag"     .= network
  389                     ]
  390                 0x04 | BS.length rest == 2 * size ->
  391                     Just $ Json.object
  392                     [ "address_style"   .= Json.String "Jormungandr"
  393                     , "address_type"    .= Json.String "group"
  394                     , "stake_reference" .= Json.String "by value"
  395                     , "spending_key"    .= base16 (BS.take size rest)
  396                     , "stake_key"       .= base16 (BS.drop size rest)
  397                     , "network_tag"     .= network
  398                     ]
  399                 0x05 | BS.length rest == size ->
  400                     Just $ Json.object
  401                     [ "address_style"   .= Json.String "Jormungandr"
  402                     , "address_type"    .= Json.String "account"
  403                     , "stake_reference" .= Json.String "none"
  404                     , "account_key"     .= base16 (BS.take size rest)
  405                     , "network_tag"     .= network
  406                     ]
  407                 0x06 | BS.length rest == size ->
  408                     Just $ Json.object
  409                     [ "address_style"   .= Json.String "Jormungandr"
  410                     , "address_type"    .= Json.String "multisig"
  411                     , "stake_reference" .= Json.String "none"
  412                     , "merkle_root"     .= base16 (BS.take size rest)
  413                     , "network_tag"     .= network
  414                     ]
  415                 _ ->
  416                     Nothing
  417   where
  418     bytes = unAddress addr
  419     base16 = T.unpack . T.decodeUtf8 . encode EBase16

  496 -- Purpose is a constant set to 1852' (or 0x8000073c) following the BIP-44
  497 -- extension for Cardano:
  498 --
  499 -- https://github.com/input-output-hk/implementation-decisions/blob/e2d1bed5e617f0907bc5e12cf1c3f3302a4a7c42/text/1852-hd-chimeric.md
  500 --
  501 -- It indicates that the subtree of this node is used according to this
  502 -- specification.
  503 --
  504 -- Hardened derivation is used at this level.
  505 purposeIndex :: Word32
  506 purposeIndex = 0x8000073c
  507 
  508 -- One master node (seed) can be used for unlimited number of independent
  509 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  510 -- same space for various cryptocoins has some disadvantages.
  511 --
  512 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  513 -- addresses across cryptocoins and improving privacy issues.
  514 --
  515 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  516 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  517 -- Lovelace.
  518 --
  519 -- Hardened derivation is used at this level.
  520 coinTypeIndex :: Word32
  521 coinTypeIndex = 0x80000717

  523 -- The minimum seed length for 'generateKeyFromSeed' and
  524 -- 'unsafeGenerateKeyFromSeed'.
