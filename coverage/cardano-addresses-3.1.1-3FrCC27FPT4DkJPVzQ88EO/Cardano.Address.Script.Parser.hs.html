   48 -- | Run 'scriptParser' on string input.
   49 --
   50 -- @since 3.0.0
   51 scriptFromString :: String -> Either ErrValidateScript Script
   52 scriptFromString str =
   53     case readP_to_S scriptParser str of
   54          [(script, "")] -> validateScript script $> script
   55          _ -> Left Malformed
   56 
   57 -- | The script embodies combination of signing keys that need to be met to make
   58 -- it valid. We assume here that the script could
   59 -- delivered from standard input. The examples below are self-explanatory:
   60 --
   61 -- 1. requiring signature
   62 -- 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe
   63 --
   64 -- 2. 'any' for signature required
   65 -- any [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   66 --
   67 -- 3. 'all' signatures required
   68 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   69 --
   70 -- 4. 'at_least' 1 signature required
   71 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   72 --
   73 -- 5. Nested script are supported
   74 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]]
   75 --
   76 -- Parser is insensitive to whitespaces.
   77 --
   78 -- @since 3.0.0
   79 scriptParser :: ReadP Script
   80 scriptParser =
   81     requireAllOfParser <++
   82     requireAnyOfParser <++
   83     requireAtLeastOfParser <++
   84     requireSignatureOfParser
   85 
   86 requireSignatureOfParser :: ReadP Script
   87 requireSignatureOfParser = do
   88     P.skipSpaces
   89     verKeyStr <- P.munch1 (\c -> isDigit c || isLetter c || c == '_')
   90     case keyHashFromText (T.pack verKeyStr) of
   91         Left e  -> fail (prettyErrKeyHashFromText e)
   92         Right h -> pure (RequireSignatureOf h)
   93 
   94 requireAllOfParser :: ReadP Script
   95 requireAllOfParser = do
   96     P.skipSpaces
   97     _identifier <- P.string "all"
   98     RequireAllOf <$> commonPart
   99 
  100 requireAnyOfParser :: ReadP Script
  101 requireAnyOfParser = do
  102     P.skipSpaces
  103     _identifier <- P.string "any"
  104     RequireAnyOf <$> commonPart
  105 
  106 requireAtLeastOfParser :: ReadP Script
  107 requireAtLeastOfParser = do
  108     P.skipSpaces
  109     _identifier <- P.string "at_least"
  110     RequireSomeOf <$> naturalParser <*> commonPart
  111 
  112 naturalParser :: ReadP Word8
  113 naturalParser = do
  114     P.skipSpaces
  115     fromInteger . read <$> P.munch1 isDigit
  116 
  117 commonPart :: ReadP [Script]
  118 commonPart = do
  119     P.skipSpaces
  120     _open <- P.string "["
  121     P.skipSpaces
  122     content <- P.sepBy scriptParser (P.string ",")
  123     P.skipSpaces
  124     _close <- P.string "]"
  125     P.skipSpaces
  126     return content
