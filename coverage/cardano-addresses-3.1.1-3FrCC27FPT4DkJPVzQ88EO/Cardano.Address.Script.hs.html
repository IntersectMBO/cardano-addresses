   89 -- | A 'Script' type represents multi signature script. The script embodies conditions
   90 -- that need to be satisfied to make it valid.
   91 --
   92 -- @since 3.0.0
   93 data Script
   94     = RequireSignatureOf !KeyHash
   95     | RequireAllOf ![Script]
   96     | RequireAnyOf ![Script]
   97     | RequireSomeOf Word8 ![Script]
   98     | ActiveFromSlot Natural
   99     | ActiveUntilSlot Natural
  100     deriving stock (Generic, Show, Eq)
  101 instance NFData Script
  102 
  103 -- | This function realizes what cardano-node's `Api.serialiseToCBOR script` realizes
  104 -- This is basically doing the symbolically following:
  105 -- toCBOR [0,multisigScript]
  106 --
  107 -- @since 3.0.0
  108 serializeScript :: Script -> ByteString
  109 serializeScript script =
  110     multisigTag <> CBOR.toStrictByteString (toCBOR script)
  111   where
  112     -- | Magic number representing the tag of the native multi-signature script
  113     -- language. For each script language included, a new tag is chosen.
  114     multisigTag :: ByteString
  115     multisigTag = "\00"
  116 
  117     toCBOR :: Script -> CBOR.Encoding
  118     toCBOR = \case
  119         RequireSignatureOf (KeyHash verKeyHash) ->
  120             encodeMultiscriptCtr 0 2 <> CBOR.encodeBytes verKeyHash
  121         RequireAllOf contents ->
  122             encodeMultiscriptCtr 1 2 <> encodeFoldable toCBOR contents
  123         RequireAnyOf contents ->
  124             encodeMultiscriptCtr 2 2 <> encodeFoldable toCBOR contents
  125         RequireSomeOf m contents -> mconcat
  126             [ encodeMultiscriptCtr 3 3
  127             , CBOR.encodeInt (fromInteger $ toInteger m)
  128             , encodeFoldable toCBOR contents
  129             ]
  130         ActiveFromSlot slotNum ->
  131             encodeMultiscriptCtr 4 2 <> CBOR.encodeWord64 (fromInteger $ toInteger slotNum)
  132         ActiveUntilSlot slotNum ->
  133             encodeMultiscriptCtr 5 2 <> CBOR.encodeWord64 (fromInteger $ toInteger slotNum)
  134 
  135     encodeMultiscriptCtr :: Word -> Word -> CBOR.Encoding
  136     encodeMultiscriptCtr ctrIndex listLen =
  137         CBOR.encodeListLen listLen <> CBOR.encodeWord ctrIndex
  138 
  139     encodeFoldable :: (Foldable f) => (a -> CBOR.Encoding) -> f a -> CBOR.Encoding
  140     encodeFoldable encode' xs = wrapArray len contents
  141       where
  142         (len, contents) = foldl' go (0, mempty) xs
  143         go (!l, !enc) next = (l + 1, enc <> encode' next)
  144 
  145         wrapArray :: Word -> CBOR.Encoding -> CBOR.Encoding
  146         wrapArray len' contents'
  147             | len' <= 23 = CBOR.encodeListLen len' <> contents'
  148             | otherwise  = CBOR.encodeListLenIndef <> contents' <> CBOR.encodeBreak
  149 
  150 -- | Computes the hash of a given script, by first serializing it to CBOR.
  151 --
  152 -- @since 3.0.0
  153 toScriptHash :: Script -> ScriptHash
  154 toScriptHash = ScriptHash . hashCredential . serializeScript

  168 scriptHashFromBytes bytes
  169     | BS.length bytes /= credentialHashSize = Nothing
  170     | otherwise = Just $ ScriptHash bytes

  188 -- | Encode a 'KeyHash' to bech32 'Text', using @script_vkh@ as a human readable prefix.
  189 --
  190 -- @since 3.0.0
  191 keyHashToText :: KeyHash -> Text
  192 keyHashToText (KeyHash keyHash) =
  193     T.decodeUtf8 $ encode (EBech32 CIP5.script_vkh) keyHash
  194 
  195 -- | Construct a 'KeyHash' from 'Text'. Either hex encoded text or
  196 -- Bech32 encoded text with `script_vkh`, `script_vk` or `script_xvk` hrp are
  197 -- expected. Raw keys will be hashed on the fly, whereas hash that are directly
  198 -- provided will remain as such.
  199 --
  200 -- @since 3.1.0
  201 keyHashFromText :: Text -> Either ErrKeyHashFromText KeyHash
  202 keyHashFromText txt = do
  203     (hrp, dp) <- either
  204         (const $ Left ErrKeyHashFromTextInvalidString)
  205         Right
  206         (Bech32.decodeLenient txt)
  207 
  208     maybeToRight ErrKeyHashFromTextWrongDataPart (Bech32.dataPartToBytes dp)
  209         >>= convertBytes hrp
  210         >>= maybeToRight ErrKeyHashFromTextWrongPayload . keyHashFromBytes
  211  where
  212     convertBytes hrp bytes
  213         | hrp == CIP5.script_vkh = Right bytes
  214         | hrp == CIP5.script_vk  = Right $ hashCredential bytes
  215         | hrp == CIP5.script_xvk = Right $ hashCredential $ BS.take 32 bytes
  216         | otherwise = Left ErrKeyHashFromTextWrongHrp
  217 
  218 -- Possible errors when deserializing a key hash from text.

  223     | ErrKeyHashFromTextWrongPayload
  224     | ErrKeyHashFromTextWrongHrp
  225     | ErrKeyHashFromTextWrongDataPart

  246 -- | 'Script' folding
  247 --
  248 -- @since 3.2.0
  249 foldScript :: (KeyHash -> b -> b) -> b -> Script -> b
  250 foldScript fn zero = \case
  251     RequireSignatureOf k -> fn k zero
  252     RequireAllOf xs      -> foldMScripts xs
  253     RequireAnyOf xs      -> foldMScripts xs
  254     RequireSomeOf _ xs   -> foldMScripts xs
  255     ActiveFromSlot _     -> zero
  256     ActiveUntilSlot _    -> zero
  257   where
  258     foldMScripts =
  259         runIdentity . foldM (\acc -> Identity . foldScript fn acc) zero

  266 -- | Validate a 'Script', semantically
  267 --
  268 -- @since 3.0.0
  269 validateScript :: Script -> Either ErrValidateScript ()
  270 validateScript = \case
  271     RequireSignatureOf (KeyHash bytes) -> do
  272         when (BS.length bytes /= credentialHashSize) $ Left WrongKeyHash
  273 
  274     RequireAllOf script -> do
  275         when (L.null script) $ Left EmptyList
  276         when (hasDuplicate script) $ Left DuplicateSignatures
  277         traverse_ validateScript script
  278 
  279     RequireAnyOf script -> do
  280         when (L.null script) $ Left EmptyList
  281         when (hasDuplicate script) $ Left DuplicateSignatures
  282         traverse_ validateScript script
  283 
  284     RequireSomeOf m script -> do
  285         when (m == 0) $ Left MZero
  286         when (length script < fromIntegral m) $ Left ListTooSmall
  287         when (hasDuplicate script) $ Left DuplicateSignatures
  288         traverse_ validateScript script
  289 
  290     ActiveFromSlot _ -> pure ()
  291 
  292     ActiveUntilSlot _ -> pure ()
  293   where
  294     hasDuplicate xs = do
  295         length sigs /= length (L.nub sigs)
  296       where
  297         sigs = [ sig | RequireSignatureOf sig <- xs ]
  298 
  299 -- | Possible validation errors when validating a script
  300 --
  301 -- @since 3.0.0
  302 data ErrValidateScript
  303     = EmptyList
  304     | ListTooSmall
  305     | MZero
  306     | DuplicateSignatures
  307     | WrongKeyHash
  308     | Malformed
  309     deriving (Eq, Show)
  310 
  311 -- | Pretty-print a validation error.
  312 --
  313 -- @since 3.0.0
  314 prettyErrValidateScript
  315     :: ErrValidateScript
  316     -> String
  317 prettyErrValidateScript = \case
  318     EmptyList ->
  319         "The list inside a script is empty."
  320     MZero ->
  321         "At least must be at least 1."
  322     ListTooSmall ->
  323         "At least must not be larger than the list of keys."
  324     DuplicateSignatures ->
  325         "The list inside a script has duplicate keys."
  326     WrongKeyHash ->
  327         "The hash of verification key is expected to have "<>show credentialHashSize<>" bytes."
  328     Malformed ->
  329         "Parsing of the script failed. The script should be composed of nested \
  330         \lists, and the verification keys should be either encoded as bech32."

  364 instance ToJSON Script where
  365     toJSON (RequireSignatureOf keyHash) =
  366         String $ keyHashToText keyHash
  367     toJSON (RequireAllOf content) =
  368         object ["all" .= fmap toJSON content]
  369     toJSON (RequireAnyOf content) =
  370         object ["any" .= fmap toJSON content]
  371     toJSON (RequireSomeOf count scripts) =
  372         object ["some" .= object ["at_least" .= count, "from" .= scripts]]
  373     toJSON (ActiveFromSlot slot) =
  374         object ["active_from" .= slot]
  375     toJSON (ActiveUntilSlot slot) =
  376         object ["active_until" .= slot]
  377 
  378 instance FromJSON Script where
  379     parseJSON v = do
  380         script <- asum
  381             [ parseKey v
  382             , parseAnyOf v
  383             , parseAllOf v
  384             , parseAtLeast v
  385             , parseActiveFrom v
  386             , parseActiveUntil v
  387             ] <|> backtrack v
  388 
  389         either (fail . prettyErrValidateScript) pure
  390             (validateScript script)
  391 
  392         return script
  393       where
  394         parseKey = withText "Script KeyHash" $
  395             either
  396                 (fail . prettyErrKeyHashFromText)
  397                 (pure . RequireSignatureOf)
  398                 . keyHashFromText
  399 
  400         parseAnyOf = withObject "Script AnyOf" $ \o ->
  401             RequireAnyOf <$> o .: "any"
  402 
  403         parseAllOf = withObject "Script AllOf" $ \o ->
  404             RequireAllOf <$> o .: "all"
  405 
  406         parseAtLeast = withObject "Script SomeOf" $ \o -> do
  407             some <- o .: "some"
  408             RequireSomeOf <$> some .: "at_least" <*> some .: "from"
  409 
  410         parseActiveFrom = withObject "Script ActiveFrom" $ \o ->
  411             ActiveFromSlot <$> o .: "active_from"
  412 
  413         parseActiveUntil = withObject "Script ActiveUntil" $ \o ->
  414             ActiveUntilSlot <$> o .: "active_until"
  415 
  416         -- NOTE: Because we use an alternative sum to define all parsers, in
  417         -- case all parser fails, only the last error is returned which can be
  418         -- very misleading. For example, sending {"any": []} yields an error
  419         -- telling us that the key `"some"` is missing.
  420         --
  421         -- To cope with this, we add a last parser 'backtrack' which always
  422         -- fail but with a more helpful error which tries its best at
  423         -- identifying the right constructor.
  424         backtrack = \case
  425             Object o -> do
  426                 mAny  <- o .:? "any"  :: Parser (Maybe Value)
  427                 mAll  <- o .:? "all"  :: Parser (Maybe Value)
  428                 mSome <- o .:? "some" :: Parser (Maybe Value)
  429                 case (mAny, mAll, mSome) of
  430                     (Just{}, Nothing, Nothing)  -> parseAnyOf v
  431                     (Nothing, Just{}, Nothing)  -> parseAllOf v
  432                     (Nothing, Nothing, Just{})  -> parseAtLeast v
  433                     (Nothing, Nothing, Nothing) -> fail
  434                         "Found object with no known key 'any', 'all' or 'some'"
  435                     (      _,       _,      _)  -> fail
  436                         "Found multiple keys 'any', 'all' and/or 'some' at the same level"
  437 
  438             String{} ->
  439                 parseKey v
  440 
  441             _ ->
  442                 Json.typeMismatch "Object or String" v
