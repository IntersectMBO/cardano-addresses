   51 newtype Cmd = Inspect
   52     { rootPublicKey :: Maybe XPub
   53     } deriving (Show)
   54 
   55 mod :: (Cmd -> parent) -> Mod CommandFields parent
   56 mod liftCmd = command "inspect" $
   57     info (helper <*> fmap liftCmd parser) $ mempty
   58         <> progDesc "Show information about an address"
   59         <> footerDoc (Just $ vsep
   60             [ string "The address is read from stdin."
   61             , string ""
   62             , string "Example:"
   63             , indent 2 $ bold $ string "$ cat addr.prv \\"
   64             , indent 4 $ bold $ string $ "| "<>progName<>" key public --with-chain-code \\"
   65             , indent 4 $ bold $ string $ "| "<>progName<>" address payment --network-tag testnet \\"
   66             , indent 4 $ bold $ string $ "| "<>progName<>" address delegation $(cat stake.prv | "<>progName<>" key public --with-chain-code) \\"
   67             , indent 4 $ bold $ string $ "| "<>progName<>" address inspect"
   68             , indent 2 $ string "{"
   69             , indent 2 $ string "    \"address_style\": \"Shelley\","
   70             , indent 2 $ string "    \"stake_reference\": \"by value\","
   71             , indent 2 $ string "    \"stake_key_hash\": \"6b542d6da35e6c95d95a33c6f66ec482d3f4caf3ad35e2ede09cf827\","
   72             , indent 2 $ string "    \"spending_key_hash\": \"44bc4f524f49a78a9c8a45b882d8710cb9254b3da6a978d50dc9b870\","
   73             , indent 2 $ string "    \"network_tag\": 0,"
   74             , indent 2 $ string "    \"address_type\": 0"
   75             , indent 2 $ string "}"
   76             ])
   77   where
   78     parser = Inspect
   79         <$> optional (xpubOpt [CIP5.root_xvk] "root" helpDoc)
   80     helpDoc =
   81         "A root public key. If specified, tries to decrypt the derivation path \
   82         \of Byron addresses."

   84 -- used for 'inspect'
   85 instance Error SomeException
   86 
   87 run :: Cmd -> IO ()
   88 run Inspect{rootPublicKey} = do
   89     bytes <- hGetBytes stdin
   90     case inspect (unsafeMkAddress bytes) of
   91       Right json -> BL8.hPutStrLn stdout (Json.encodePretty json)
   92       Left  e    -> die $ format "Error: {}" (displayException e)
   93   where
   94     -- We can't use IO here, because (<|>) in IO only catches IOException type,
   95     -- but MonadThrow in IO doesn't specialize to IOException.
   96     --
   97     -- Luckily, there's an existing instance for:
   98     --   instance (Error e) => Alternative (Either e)
   99     inspect :: (e ~ SomeException) => Address -> Either e Json.Value
  100     inspect = Shelley.inspectAddress rootPublicKey
