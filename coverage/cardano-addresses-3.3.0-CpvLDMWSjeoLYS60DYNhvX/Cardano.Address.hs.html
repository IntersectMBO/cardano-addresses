   75 -- | An 'Address' type representing 'Cardano' addresses. Internals are
   76 -- irrevelant to the user.

   90 -- | Encode an 'Address' to a base58 'Text'.
   91 --
   92 -- @since 1.0.0
   93 base58 :: Address -> Text
   94 base58 = T.decodeUtf8 . encode EBase58 . unAddress
   95 
   96 -- | Decode a base58-encoded 'Text' into an 'Address'
   97 --
   98 -- @since 1.0.0
   99 fromBase58 :: Text -> Maybe Address
  100 fromBase58 =
  101     deserialiseCbor (unsafeMkAddress <$> decodeAddress)
  102    <=<
  103     eitherToMaybe . E.fromBase58 . T.encodeUtf8
  104 
  105 -- | Encode a Shelley 'Address' to bech32 'Text', using @addr@ or @addr_test@ as
  106 -- a human readable prefix (depending on the network tag in the address).
  107 --
  108 -- @since 1.0.0
  109 bech32 :: Address -> Text
  110 bech32 addr = bech32With (addressHrp addr) addr
  111 
  112 -- | Encode an 'Address' to bech32 'Text', using the specified human readable
  113 -- prefix.
  114 --
  115 -- @since 2.0.0
  116 bech32With :: HumanReadablePart -> Address -> Text
  117 bech32With hrp = T.decodeLatin1 . encode (EBech32 hrp) . unAddress
  118 
  119 -- | Decode a bech32-encoded 'Text' into an 'Address'
  120 --
  121 -- @since 1.0.0
  122 fromBech32 :: Text -> Maybe Address
  123 fromBech32 = eitherToMaybe
  124     . fmap (unsafeMkAddress . snd)
  125     . E.fromBech32 (const id)
  126     . T.encodeUtf8
  127 
  128 -- | Returns the HRP for a shelley address, using the network tag.
  129 addressHrp :: Address -> HumanReadablePart
  130 addressHrp (Address bs) = case BS.uncons bs of
  131     Just (w8, _) | testBit w8 0 -> CIP5.addr
  132     _ -> CIP5.addr_test

  173 -- | A 'ChainPointer' type representing location of some object
  174 -- in the blockchain (eg., delegation certificate). This can be achieved
  175 -- unambiguously by specifying slot number, transaction index and the index
  176 -- in the object list (eg., certification list).
  177 -- For delegation certificates, alternatively, the delegation key can be used and
  178 -- then 'DelegationAddress' can be used.
  179 --
  180 -- @since 2.0.0
  181 data ChainPointer = ChainPointer
  182     { slotNum :: Natural
  183       -- ^ Pointer to the slot
  184     , transactionIndex :: Natural
  185       -- ^ transaction index
  186     , outputIndex :: Natural
  187       -- ^ output list index
  188     } deriving stock (Generic, Show, Eq, Ord)
  189 instance NFData ChainPointer

  218 -- Magic constant associated with a given network. This is mainly used in two
  219 -- places:
  220 --
  221 -- (1) In 'Address' payloads, to discriminate addresses between networks.
  222 -- (2) At the network-level, when doing handshake with nodes.
  223 newtype NetworkTag
  224     = NetworkTag { unNetworkTag :: Word32 }
  225     deriving (Generic, Show, Eq)
  226 instance NFData NetworkTag
  227 
  228 -- Describe requirements for address discrimination on the Byron era.
  229 data AddressDiscrimination
  230     = RequiresNetworkTag
  231     | RequiresNoTag
  232     deriving (Generic, Show, Eq)
  233 instance NFData AddressDiscrimination

  235 invariantSize :: HasCallStack => Int -> ByteString -> ByteString
  236 invariantSize expectedLength bytes
  237     | BS.length bytes == expectedLength = bytes
  238     | otherwise = error
  239       $ "length was "
  240       ++ show (BS.length bytes)
  241       ++ ", but expected to be "
  242       ++ (show expectedLength)
  243 
  244 invariantNetworkTag :: HasCallStack => Word32 -> NetworkTag -> Word8
  245 invariantNetworkTag limit (NetworkTag num)
  246     | num < limit = fromIntegral num
  247     | otherwise = error
  248       $ "network tag was "
  249       ++ show num
  250       ++ ", but expected to be less than "
  251       ++ show limit
