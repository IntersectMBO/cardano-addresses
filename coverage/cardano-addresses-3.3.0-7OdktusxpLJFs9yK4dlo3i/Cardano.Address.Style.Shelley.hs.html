  164 -- $overview
  165 --
  166 -- This module provides an implementation of:
  167 --
  168 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Shelley master keys from mnemonic sentences
  169 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  170 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  171 --
  172 -- - 'paymentAddress': for constructing payment addresses from a address public key or a script
  173 -- - 'delegationAddress': for constructing delegation addresses from payment credential (public key or script) and stake credential (public key or script)
  174 -- - 'pointerAddress': for constructing delegation addresses from payment credential (public key or script) and chain pointer
  175 -- - 'stakeAddress': for constructing reward accounts from stake credential (public key or script)

  177 -- | A cryptographic key for sequential-scheme address derivation, with
  178 -- phantom-types to disambiguate key types.
  179 --
  180 -- @
  181 -- let rootPrivateKey = Shelley 'RootK XPrv
  182 -- let accountPubKey  = Shelley 'AccountK XPub
  183 -- let addressPubKey  = Shelley 'PaymentK XPub
  184 -- @
  185 --
  186 -- @since 2.0.0
  187 newtype Shelley (depth :: Depth) key = Shelley
  188     { getKey :: key
  189         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  190         --
  191         -- @since 1.0.0
  192     }
  193     deriving stock (Generic, Show, Eq)
  194 
  195 deriving instance (Functor (Shelley depth))
  196 instance (NFData key) => NFData (Shelley depth key)
  197 
  198 -- | Describe what the keys within an account are used for.
  199 --
  200 -- - UTxOExternal: used for public addresses sent to other parties for receiving money.
  201 -- - UTxOInternal: generated by wallet software to send change back to the wallet.
  202 -- - Stake: used for stake key(s) and delegation.
  203 -- - MultisigForPayment: used for keys used in multi-party transactions / scripts in payment credential.
  204 -- - MultisigForDelegation: used for keys used in multi-party transactions / scripts in delegation credential.
  205 --
  206 -- @since 3.0.0
  207 data Role
  208     = UTxOExternal
  209     | UTxOInternal
  210     | Stake
  211     | MultisigForPayment
  212     | MultisigForDelegation
  213     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  214 
  215 instance NFData Role
  216 
  217 roleFromIndex :: Index 'Soft depth -> Maybe Role
  218 roleFromIndex ix = case indexToWord32 ix of
  219     0 -> Just UTxOExternal
  220     1 -> Just UTxOInternal
  221     2 -> Just Stake
  222     3 -> Just MultisigForPayment
  223     4 -> Just MultisigForDelegation
  224     _ -> Nothing
  225 
  226 roleToIndex :: Role -> Index 'Soft depth
  227 roleToIndex = unsafeMkIndex . \case
  228     UTxOExternal -> 0
  229     UTxOInternal -> 1
  230     Stake -> 2
  231     MultisigForPayment -> 3
  232     MultisigForDelegation -> 4

  237 -- $keyDerivation

  239 -- === Generating a root key from 'SomeMnemonic'
  240 -- > :set -XOverloadedStrings
  241 -- > :set -XTypeApplications
  242 -- > :set -XDataKinds
  243 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  244 -- >
  245 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  246 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  247 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shelley 'RootK XPrv
  248 --
  249 -- === Deriving child keys
  250 --
  251 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  252 --
  253 -- > let Just accIx = indexFromWord32 0x80000000
  254 -- > let acctK = deriveAccountPrivateKey rootK accIx
  255 -- >
  256 -- > let Just addIx = indexFromWord32 0x00000014
  257 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  258 --
  259 -- > let stakeK = deriveDelegationPrivateKey acctK

  261 instance Internal.GenMasterKey Shelley where
  262     type SecondFactor Shelley = ScrubbedBytes
  263 
  264     genMasterKeyFromXPrv = liftXPrv
  265     genMasterKeyFromMnemonic fstFactor sndFactor =
  266         Shelley $ generateNew seedValidated sndFactor
  267         where
  268             seed  = someMnemonicToBytes fstFactor
  269             seedValidated = assert
  270                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  271                 seed

  273 instance Internal.HardDerivation Shelley where
  274     type AccountIndexDerivationType Shelley = 'Hardened
  275     type AddressIndexDerivationType Shelley = 'Soft
  276     type WithRole Shelley = Role
  277 
  278     deriveAccountPrivateKey (Shelley rootXPrv) accIx =
  279         let
  280             Just purposeIx =
  281                 indexFromWord32 @(Index 'Hardened _) purposeIndex
  282             Just coinTypeIx =
  283                 indexFromWord32 @(Index 'Hardened _) coinTypeIndex
  284             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  285                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  286             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  287                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  288             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  289                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  290         in
  291             Shelley acctXPrv
  292 
  293     deriveAddressPrivateKey (Shelley accXPrv) role addrIx =
  294         let
  295             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  296                 deriveXPrv DerivationScheme2 accXPrv (roleToIndex role)
  297             addrXPrv = -- lvl5 derivation; soft derivation of address index
  298                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  299         in
  300             Shelley addrXPrv
  301 
  302 instance Internal.SoftDerivation Shelley where
  303     deriveAddressPublicKey (Shelley accXPub) role addrIx =
  304         fromMaybe errWrongIndex $ do
  305             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  306                 deriveXPub DerivationScheme2 accXPub (roleToIndex role)
  307             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  308                 deriveXPub DerivationScheme2 changeXPub addrIx
  309             return $ Shelley addrXPub
  310       where
  311         errWrongIndex = error $
  312             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  313             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  314             \either a programmer error, or, we may have reached the maximum \
  315             \number of addresses for a given wallet."

  375 -- Re-export from 'Cardano.Address.Derivation' to have it documented specialized in Haddock
  376 --
  377 -- | Derive a delegation key for a corresponding 'AccountK'. Note that wallet
  378 -- software are by convention only using one delegation key per account, and always
  379 -- the first account (with index 0').
  380 --
  381 -- Deriving delegation keys for something else than the initial account is not
  382 -- recommended and can lead to incompatibility with existing wallet softwares
  383 -- (Daedalus, Yoroi, Adalite...).
  384 --
  385 -- @since 2.0.0
  386 deriveDelegationPrivateKey
  387     :: Shelley 'AccountK XPrv
  388     -> Shelley 'DelegationK XPrv
  389 deriveDelegationPrivateKey accXPrv =
  390     let (Shelley stakeXPrv) =
  391             deriveAddressPrivateKey accXPrv Stake (minBound @(Index 'Soft _))
  392     in Shelley stakeXPrv

  540 -- | Analyze an 'Address' to know whether it's a Shelley address or not.
  541 --
  542 -- Throws 'AddrError' if it's not a valid Shelley address, or a ready-to-print
  543 -- string giving details about the 'Address'.
  544 --
  545 -- @since 3.0.0
  546 inspectAddress
  547     :: (Alternative m, MonadThrow m)
  548     => Maybe XPub
  549     -> Address
  550     -> m Json.Value
  551 inspectAddress mRootPub addr
  552     | BS.length bytes < 1 + credentialHashSize = throwM (WrongInputSize (BS.length bytes))
  553     | otherwise =
  554         let
  555             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  556             addrType = fstByte .&. 0b11110000
  557             network  = fstByte .&. 0b00001111
  558 
  559             (fstHash, sndHash) = BS.splitAt credentialHashSize rest
  560 
  561             addrHrp = if NetworkTag (fromIntegral network)  == shelleyMainnet
  562                 then CIP5.addr
  563                 else CIP5.addr_test
  564             bech32Spending = bech32With addrHrp
  565         in
  566             case addrType of
  567                -- 0000: base address: keyhash28,keyhash28
  568                 0b00000000 | BS.length rest == 2 * credentialHashSize ->
  569                     pure $ Json.object
  570                         [ "address_style"     .= Json.String "Shelley"
  571                         , "stake_reference"   .= Json.String "by value"
  572                         , "spending_key_hash" .= base16 fstHash
  573                         , "spending_key_hash_bech32" .= bech32Spending fstHash
  574                         , "stake_key_hash"    .= base16 sndHash
  575                         , "stake_key_hash_bech32" .= bech32Stake sndHash
  576                         , "network_tag"       .= network
  577                         ]
  578                -- 0001: base address: scripthash32,keyhash28
  579                 0b00010000 | BS.length rest == credentialHashSize + credentialHashSize ->
  580                     pure $ Json.object
  581                         [ "address_style"     .= Json.String "Shelley"
  582                         , "stake_reference"   .= Json.String "by value"
  583                         , "script_hash"       .= base16 fstHash
  584                         , "script_hash_bech32".= bech32Script fstHash
  585                         , "stake_key_hash"    .= base16 sndHash
  586                         , "stake_key_hash_bech32" .= bech32Stake sndHash
  587                         , "network_tag"       .= network
  588                         ]
  589                -- 0010: base address: keyhash28,scripthash32
  590                 0b00100000 | BS.length rest == credentialHashSize + credentialHashSize ->
  591                     pure $ Json.object
  592                         [ "address_style"     .= Json.String "Shelley"
  593                         , "stake_reference"   .= Json.String "by value"
  594                         , "spending_key_hash" .= base16 fstHash
  595                         , "spending_key_hash_bech32" .= bech32Spending fstHash
  596                         , "stake_script_hash" .= base16 sndHash

  805 -- | Extend an existing payment 'Address' to make it a delegation address.
  806 --
  807 -- @since 2.0.0
  808 extendAddress
  809     :: Address
  810     -> Credential 'DelegationK
  811     -> Either ErrExtendAddress Address
  812 extendAddress addr stakeReference = do
  813     when (isNothing (inspectShelleyAddress Nothing addr)) $
  814         Left $ ErrInvalidAddressStyle "Given address isn't a Shelley address"
  815 
  816     let bytes = unAddress addr
  817     let (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  818 
  819     let paymentFirstByte = fstByte .&. 0b11110000
  820     let extendableTypes = addressType <$>
  821             [ EnterpriseAddress CredentialFromKey
  822             , EnterpriseAddress CredentialFromScript
  823             ]
  824     unless (paymentFirstByte `elem` extendableTypes) $ do
  825         Left $ ErrInvalidAddressType "Only payment addresses can be extended"
  826 
  827     case stakeReference of
  828         -- base address: keyhash28,keyhash28    : 00000000 -> 0
  829         -- base address: scripthash32,keyhash28 : 00010000 -> 16
  830         DelegationFromKey delegationKey -> do
  831             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  832                 -- 0b01100000 .&. 0b00011111 = 0
  833                 -- 0b01110000 .&. 0b00011111 = 16
  834                 putWord8 $ fstByte .&. 0b00011111
  835                 putByteString rest
  836                 putByteString . hashCredential . xpubPublicKey . getKey $ delegationKey
  837 
  838         -- base address: keyhash28,scripthash32    : 00100000 -> 32
  839         -- base address: scripthash32,scripthash32 : 00110000 -> 48
  840         DelegationFromScript (ScriptHash scriptBytes) -> do
  841             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  842                 -- 0b01100000 .&. 0b00111111 = 32
  843                 -- 0b01110000 .&. 0b00111111 = 48
  844                 putWord8 $ fstByte .&. 0b00111111
  845                 putByteString rest
  846                 putByteString scriptBytes
  847 
  848         -- pointer address: keyhash28, 3 variable length uint    : 01000000 -> 64
  849         -- pointer address: scripthash32, 3 variable length uint : 01010000 -> 80
  850         DelegationFromPointer pointer -> do
  851             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  852                 -- 0b01100000 .&. 0b01011111 = 64
  853                 -- 0b01110000 .&. 0b01011111 = 80
  854                 putWord8 $ fstByte .&. 0b01011111
  855                 putByteString rest
  856                 putPointer pointer
  857   where
  858     putPointer (ChainPointer a b c) = do
  859         putVariableLengthNat a
  860         putVariableLengthNat b
  861         putVariableLengthNat c

 1016 addressType :: AddressType -> Word8
 1017 addressType = \case
 1018     ByronAddress                                                -> 0b10000000
 1019     BaseAddress       CredentialFromKey    CredentialFromKey    -> 0b00000000
 1020     BaseAddress       CredentialFromScript CredentialFromKey    -> 0b00010000
 1021     BaseAddress       CredentialFromKey    CredentialFromScript -> 0b00100000

 1024     PointerAddress    CredentialFromScript                      -> 0b01010000

 1030 -- Helper to constructs appropriate address headers. Rest of the payload is left
 1031 -- to the caller as a raw 'ByteString'.
 1032 constructPayload
 1033     :: AddressType
 1034     -> NetworkDiscriminant Shelley
 1035     -> ByteString
 1036     -> Address
 1037 constructPayload addrType discrimination bytes = unsafeMkAddress $
 1038     invariantSize expectedLength $ BL.toStrict $ runPut $ do
 1039         putWord8 firstByte
 1040         putByteString bytes
 1041   where
 1042     firstByte =
 1043         let netTagLimit = 16
 1044         in addressType addrType + invariantNetworkTag netTagLimit (networkTag @Shelley discrimination)
 1045     expectedLength =
 1046         let headerSizeBytes = 1
 1047         in headerSizeBytes + credentialHashSize
