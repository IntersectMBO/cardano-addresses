   61 -- | Read some bytes from the console, and decode them if the encoding is recognized.
   62 hGetBytes :: Handle -> IO ByteString
   63 hGetBytes h = do
   64     raw <- B8.filter noNewline <$> B8.hGetContents h
   65     case detectEncoding (T.unpack $ T.decodeUtf8 raw) of
   66         Just (EBase16  ) -> decode fromBase16 raw
   67         Just (EBech32{}) -> decode fromBech32 raw
   68         Just (EBase58  ) -> decode fromBase58 raw
   69         Nothing          -> fail
   70             "Couldn't detect input encoding? Data on stdin must be encoded as \
   71             \bech16, bech32 or base58."
   72   where
   73     decode :: (bin -> Either String bin) -> bin -> IO bin
   74     decode from = either fail pure . from
   75 
   76 -- | Read some English mnemonic words from the console, or fail.
   77 hGetSomeMnemonic :: Handle -> IO SomeMnemonic
   78 hGetSomeMnemonic h = do
   79     wrds <- T.words . T.filter noNewline . T.decodeUtf8 <$> B8.hGetContents h
   80     case mkSomeMnemonic @'[ 9, 12, 15, 18, 21, 24 ] wrds of
   81         Left (MkSomeMnemonicError e) -> fail e
   82         Right mw -> pure mw
   83 
   84 -- | Read an encoded private key from the console, or fail.
   85 hGetXPrv :: Handle -> IO XPrv
   86 hGetXPrv h = do
   87     bytes <- hGetBytes h
   88     case xprvFromBytes bytes of
   89         Nothing  -> fail "Couldn't convert bytes into extended private key."
   90         Just key -> pure key
   91 
   92 -- | Read an encoded public key from the console, or fail.
   93 hGetXPub :: Handle -> IO XPub
   94 hGetXPub h = do
   95     bytes <- hGetBytes h
   96     case xpubFromBytes bytes of
   97         Nothing  -> fail "Couldn't convert bytes into extended public key."
   98         Just key -> pure key
   99 
  100 -- | Read either an encoded public or private key from the console, or fail.
  101 hGetXP__ :: Handle -> IO (Either XPub XPrv)
  102 hGetXP__ h = do
  103     bytes <- hGetBytes h
  104     case (xpubFromBytes bytes, xprvFromBytes bytes) of
  105         (Just xpub,         _) -> pure (Left  xpub)
  106         (_        , Just xprv) -> pure (Right xprv)
  107         _                      -> fail
  108             "Couldn't convert bytes into neither extended public or private keys."

  115 -- | Print bytes to the console with the given encoding.
  116 hPutBytes :: Handle -> ByteString -> Encoding -> IO ()
  117 hPutBytes h bytes =
  118     B8.hPutStr h . flip encode bytes

  124 noNewline :: Char -> Bool
  125 noNewline = (`notElem` ['\n', '\r'])
  126 
  127 -- | Fail with a colored red error message.
  128 prettyIOException :: IOException -> IO a
  129 prettyIOException e = do
  130     B8.hPutStrLn stderr $ T.encodeUtf8 $ T.pack $ show e
  131     exitFailure
