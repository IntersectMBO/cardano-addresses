    8 module Command.Key.FromRecoveryPhrase
    9     ( Cmd (..)
   10     , mod
   11     , run
   12     ) where
   13 
   14 import Prelude hiding
   15     ( mod )
   16 
   17 import Cardano.Address.Derivation
   18     ( xprvToBytes )
   19 import Codec.Binary.Bech32.TH
   20     ( humanReadablePart )
   21 import Options.Applicative
   22     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   23 import Options.Applicative.Encoding
   24     ( Encoding, encodingOpt )
   25 import Options.Applicative.Help.Pretty
   26     ( bold, indent, string, vsep )
   27 import Options.Applicative.Style
   28     ( Style, generateRootKey, styleArg )
   29 import System.IO
   30     ( stdin, stdout )
   31 import System.IO.Extra
   32     ( hGetSomeMnemonic, hPutBytes, progName )
   33 
   34 
   35 data Cmd = FromRecoveryPhrase
   36     { encoding :: Encoding
   37     , style :: Style
   38     } deriving (Show)
   39 
   40 mod :: (Cmd -> parent) -> Mod CommandFields parent
   41 mod liftCmd = command "from-recovery-phrase" $
   42     info (helper <*> fmap liftCmd parser) $ mempty
   43         <> progDesc "Convert a recovery phrase to an extended private key"
   44         <> footerDoc (Just $ vsep
   45             [ string "The recovery phrase is read from stdin."
   46             , string ""
   47             , string "Example:"
   48             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate \\"
   49             , indent 2 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Icarus"
   50             ])
   51   where
   52     parser = FromRecoveryPhrase
   53         <$> encodingOpt [humanReadablePart|xprv|]
   54         <*> styleArg
   55 
   56 run :: Cmd -> IO ()
   57 run FromRecoveryPhrase{encoding,style} = do
   58     someMnemonic <- hGetSomeMnemonic stdin
   59     rootK <- generateRootKey someMnemonic style
   60     hPutBytes stdout (xprvToBytes rootK) encoding
