  135 -- $overview
  136 --
  137 -- This module provides an implementation of:
  138 --
  139 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Byron master keys from mnemonic sentences
  140 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical derivation of parent to child keys
  141 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  142 --
  143 -- We call 'Byron' addresses the old address type used by Daedalus in the early
  144 -- days of Cardano. Using this type of addresses and underlying key scheme is
  145 -- now considered __deprecated__ because of some security implications.
  146 --
  147 -- The internals of the 'Byron' does not matter for the reader, but basically
  148 -- contains what is necessary to perform key derivation and generate addresses
  149 -- from a 'Byron' type.
  150 --
  151 -- == Deprecation Notice
  152 --
  153 -- Unless you have good reason to do so (like writing backward-compatible code
  154 -- with an existing piece), any new implementation __should use__ the
  155 -- 'Cardano.Address.Style.Icarus.Icarus' style for key and addresses.
  156

  158 -- | Material for deriving HD random scheme keys, which can be used for making
  159 -- addresses.
  160 --
  161 -- @since 1.0.0
  162 data Byron (depth :: Depth) key = Byron
  163     { getKey :: key
  164     -- ^ The raw private or public key.
  165     --
  166     -- @since 1.0.0
  167     , derivationPath :: DerivationPath depth
  168     -- ^ The address derivation indices for the level of this key.
  169     --
  170     -- @since 1.0.0
  171     , payloadPassphrase :: ScrubbedBytes
  172     -- ^ Used for encryption of the derivation path payload within an address.
  173     --
  174     -- @since 1.0.0
  175     } deriving stock (Generic)
  176 {-# DEPRECATED Byron "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  177 {-# DEPRECATED getKey "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  178 {-# DEPRECATED derivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  179 {-# DEPRECATED payloadPassphrase "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  186 -- | The hierarchical derivation indices for a given level/depth.
  187 --
  188 -- @since 1.0.0
  189 type family DerivationPath (depth :: Depth) :: Type where
  190     -- The root key is generated from the seed.
  191     DerivationPath 'RootK =
  192         ()
  193     -- The account key is generated from the root key and account index.
  194     DerivationPath 'AccountK =
  195         Index 'WholeDomain 'AccountK
  196     -- The address key is generated from the account key and address index.
  197     DerivationPath 'PaymentK =
  198         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'PaymentK)
  199 {-# DEPRECATED DerivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  203 --
  204 -- === Generating a root key from 'SomeMnemonic'
  205 -- > :set -XOverloadedStrings
  206 -- > :set -XTypeApplications
  207 -- > :set -XDataKinds
  208 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  209 -- >
  210 -- > let (Right mw) = mkSomeMnemonic @'[12] ["moon","fox","ostrich","quick","cactus","raven","wasp","intact","first","ring","crumble","error"]
  211 -- > let rootK = genMasterKeyFromMnemonic mw :: Byron 'RootK XPrv
  212 --
  213 -- === Deriving child keys
  214 --
  215 -- > let Just accIx = fromWord32 0x80000000
  216 -- > let acctK = deriveAccountPrivateKey rootK accIx
  217 -- >
  218 -- > let Just addIx = fromWord32 0x80000014
  219 -- > let addrK = deriveAddressPrivateKey acctK addIx

  221 instance Internal.GenMasterKey Byron where
  222     type SecondFactor Byron = ()
  223 
  224     genMasterKeyFromXPrv xprv =
  225         liftXPrv (toXPub xprv) () xprv
  226     genMasterKeyFromMnemonic (SomeMnemonic mw) () =
  227         liftXPrv (toXPub xprv) () xprv
  228       where
  229         xprv = generate (hashSeed seedValidated)
  230         seed  = entropyToBytes $ mnemonicToEntropy mw
  231         seedValidated = assert
  232             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  233             seed

  235 instance Internal.HardDerivation Byron where
  236     type AddressIndexDerivationType Byron = 'WholeDomain
  237     type AccountIndexDerivationType Byron = 'WholeDomain
  238     type WithRole Byron = ()

  240     deriveAccountPrivateKey rootXPrv accIx = Byron
  241         { getKey = deriveXPrv DerivationScheme1 (getKey rootXPrv) accIx
  242         , derivationPath = accIx
  243         , payloadPassphrase = payloadPassphrase rootXPrv
  244         }

  247         { getKey = deriveXPrv DerivationScheme1 (getKey accXPrv) addrIx
  248         , derivationPath = (derivationPath accXPrv, addrIx)
  249         , payloadPassphrase = payloadPassphrase accXPrv

  356 -- | Determines whether an 'Address' is a Byron address.
  357 --
  358 -- Returns either details about the 'Address', or 'ErrInspectAddress' if it's
  359 -- not a valid address.
  360 --
  361 -- @since 3.4.0
  362 eitherInspectAddress :: Maybe XPub -> Address -> Either ErrInspectAddress AddressInfo
  363 eitherInspectAddress mRootPub addr = do
  364     payload <- first DeserialiseError $
  365         CBOR.deserialiseCbor CBOR.decodeAddressPayload bytes
  366 
  367     (root, attrs) <- first DeserialiseError $
  368         CBOR.deserialiseCbor decodePayload payload
  369 
  370     path <- do
  371         attr <- maybe (Left MissingExpectedDerivationPath) Right $
  372             find ((== 1) . fst) attrs
  373         case mRootPub of
  374             Nothing -> Right $ EncryptedDerivationPath $ snd attr
  375             Just rootPub -> decryptPath attr rootPub
  376 
  377     ntwrk <- bimap DeserialiseError (fmap NetworkTag) $
  378         CBOR.deserialiseCbor CBOR.decodeProtocolMagicAttr payload
  379 
  380     pure AddressInfo
  381         { infoAddressRoot = root
  382         , infoPayload = path
  383         , infoNetworkTag = ntwrk
  384         }
  385   where
  386     bytes :: ByteString
  387     bytes = unAddress addr
  388 
  389     decodePayload :: forall s. CBOR.Decoder s (ByteString, [(Word8, ByteString)])
  390     decodePayload = do
  391         _ <- CBOR.decodeListLenCanonicalOf 3
  392         root <- CBOR.decodeBytes
  393         (root,) <$> CBOR.decodeAllAttributes
  394 
  395     decryptPath :: (Word8, ByteString) -> XPub -> Either ErrInspectAddress PayloadInfo
  396     decryptPath attr rootPub = do
  397         let pwd = hdPassphrase rootPub
  398         path <- first (const FailedToDecryptPath) $
  399             CBOR.deserialiseCbor (CBOR.decodeDerivationPathAttr pwd [attr]) mempty
  400         case path of
  401             Nothing -> Left FailedToDecryptPath
  402             Just (accountIndex, addressIndex) -> Right PayloadDerivationPath{..}
  403

  413 -- | The derivation path in a Byron address payload.
  414 --
  415 -- @since 3.4.0
  416 data PayloadInfo
  417     = PayloadDerivationPath
  418         { accountIndex :: !Word32
  419         , addressIndex :: !Word32
  420         }
  421     | EncryptedDerivationPath
  422         { encryptedDerivationPath :: !ByteString
  423         }
  424     deriving (Generic, Show, Eq)

  426 instance ToJSON AddressInfo where
  427     toJSON AddressInfo{..} = Json.object
  428         [ "address_root"    .= T.decodeUtf8 (encode EBase16 infoAddressRoot)
  429         , "derivation_path" .= infoPayload
  430         , "network_tag"     .= maybe Json.Null toJSON infoNetworkTag
  431         , "address_type"    .= toJSON @Word8 8
  432         ]
  433 
  434 instance ToJSON PayloadInfo where
  435     toJSON PayloadDerivationPath{..} = Json.object
  436         [ "account_index" .= prettyIndex accountIndex
  437         , "address_index" .= prettyIndex addressIndex
  438         ]
  439       where
  440         prettyIndex :: Word32 -> String
  441         prettyIndex ix
  442             | ix >= firstHardened = show (ix - firstHardened) <> "H"
  443             | otherwise = show ix
  444           where
  445             firstHardened = 0x80000000
  446     toJSON EncryptedDerivationPath{..} = Json.String $
  447         T.decodeUtf8 $ encode EBase16 encryptedDerivationPath

  449 instance Internal.PaymentAddress Byron where
  450     paymentAddress discrimination k = unsafeMkAddress
  451         $ CBOR.toStrictByteString
  452         $ CBOR.encodeAddress (getKey k) attrs
  453       where
  454         (acctIx, addrIx) = bimap indexToWord32 indexToWord32 $ derivationPath k
  455         pwd = payloadPassphrase k
  456         NetworkTag magic = networkTag @Byron discrimination
  457         attrs = case addressDiscrimination @Byron discrimination of
  458             RequiresNetworkTag ->
  459                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  460                 , CBOR.encodeProtocolMagicAttr magic
  461                 ]
  462             RequiresNoTag ->
  463                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  464                 ]

  580 -- The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  581 minSeedLengthBytes :: Int
  582 minSeedLengthBytes = 16
  583 
  584 -- Hash the seed entropy (generated from mnemonic) used to initiate a HD
  585 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  586 -- minimum for 'generate' (32 bytes).
  587 --
  588 -- Note that our current implementation deviates from BIP-39 because we use a
  589 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  590 --
  591 -- There are two methods of hashing the seed entropy, for different use cases.
  592 --
  593 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  594 --    Blake2b_256, inside a double CBOR serialization sandwich.
  595 --
  596 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  597 --    Blake2b_256, without any serialization.
  598 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  599 hashSeed = serialize . blake2b256 . serialize
  600   where
  601     serialize = BA.convert . cbor . BA.convert
  602     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  603 
  604 -- Hash a byte string through blake2b 256
  605 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  606 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  607 
  608 -- Derive a symmetric key for encrypting and authenticating the address
  609 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  610 -- is employed.
  611 hdPassphrase :: XPub -> ScrubbedBytes
  612 hdPassphrase masterKey =
  613     PBKDF2.generate
  614     (PBKDF2.prfHMAC SHA512)
  615     (PBKDF2.Parameters 500 32)
  616     (xpubToBytes masterKey)
  617     ("address-hashing" :: ByteString)
