  144 -- $overview
  145 --
  146 -- This module provides an implementation of:
  147 --
  148 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Icarus master keys from mnemonic sentences
  149 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  150 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  151 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  152 --
  153 -- We call 'Icarus' addresses the new format of Cardano addresses which came
  154 -- after 'Cardano.Address.Style.Byron.Byron'. This is the format initially used in /Yoroi/
  155 -- and now also used by /Daedalus/.

  157 -- | A cryptographic key for sequential-scheme address derivation, with
  158 -- phantom-types to disambiguate key types.
  159 --
  160 -- @
  161 -- let rootPrivateKey = Icarus 'RootK XPrv
  162 -- let accountPubKey  = Icarus 'AccountK XPub
  163 -- let addressPubKey  = Icarus 'PaymentK XPub
  164 -- @
  165 --
  166 -- @since 1.0.0
  167 newtype Icarus (depth :: Depth) key = Icarus
  168     { getKey :: key
  169         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  170         --
  171         -- @since 1.0.0
  172     }
  173     deriving stock (Generic, Show, Eq)
  174 
  175 deriving instance (Functor (Icarus depth))
  176 instance (NFData key) => NFData (Icarus depth key)
  177 
  178 data Role
  179     = UTxOExternal
  180     | UTxOInternal
  181     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  182 
  183 instance NFData Role

  185 roleFromIndex :: Index 'Soft depth -> Maybe Role
  186 roleFromIndex ix = case indexToWord32 ix of
  187     0 -> Just UTxOExternal
  188     1 -> Just UTxOInternal
  189     _ -> Nothing
  190 
  191 roleToIndex :: Role -> Index 'Soft depth
  192 roleToIndex = unsafeMkIndex . \case
  193     UTxOExternal -> 0
  194     UTxOInternal -> 1

  196 --
  197 -- Key Derivation
  198 --
  199 -- $keyDerivation
  200 --
  201 -- === Generating a root key from 'SomeMnemonic'
  202 -- > :set -XOverloadedStrings
  203 -- > :set -XTypeApplications
  204 -- > :set -XDataKinds
  205 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  206 -- >
  207 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  208 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  209 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Icarus 'RootK XPrv
  210 --
  211 -- === Deriving child keys
  212 --
  213 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  214 --
  215 -- > let Just accIx = indexFromWord32 0x80000000
  216 -- > let acctK = deriveAccountPrivateKey rootK accIx
  217 -- >
  218 -- > let Just addIx = indexFromWord32 0x00000014
  219 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx

  221 instance Internal.GenMasterKey Icarus where
  222     type SecondFactor Icarus = ScrubbedBytes
  223 
  224     genMasterKeyFromXPrv = liftXPrv
  225     genMasterKeyFromMnemonic (SomeMnemonic mw) sndFactor =
  226         let
  227             seed  = entropyToBytes $ mnemonicToEntropy mw
  228             seedValidated = assert
  229                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  230                 seed
  231         in Icarus $ generateNew seedValidated sndFactor

  233 instance Internal.HardDerivation Icarus where
  234     type AccountIndexDerivationType Icarus = 'Hardened
  235     type AddressIndexDerivationType Icarus = 'Soft
  236     type WithRole Icarus = Role
  237 
  238     deriveAccountPrivateKey (Icarus rootXPrv) accIx =
  239         let
  240             Just purposeIx =
  241                 indexFromWord32 @(Index 'Hardened _) purposeIndex
  242             Just coinTypeIx =
  243                 indexFromWord32 @(Index 'Hardened _) coinTypeIndex
  244             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  245                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  246             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  247                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  248             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  249                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  250         in
  251             Icarus acctXPrv
  252 
  253     deriveAddressPrivateKey (Icarus accXPrv) role addrIx =
  254         let
  255             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  256                 deriveXPrv DerivationScheme2 accXPrv (roleToIndex role)
  257             addrXPrv = -- lvl5 derivation; soft derivation of address index
  258                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  259         in
  260             Icarus addrXPrv
  261 
  262 instance Internal.SoftDerivation Icarus where
  263     deriveAddressPublicKey (Icarus accXPub) role addrIx =
  264         fromMaybe errWrongIndex $ do
  265             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  266                 deriveXPub DerivationScheme2 accXPub (roleToIndex role)
  267             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  268                 deriveXPub DerivationScheme2 changeXPub addrIx
  269             return $ Icarus addrXPub
  270       where
  271         errWrongIndex = error $
  272             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  273             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  274             \either a programmer error, or, we may have reached the maximum \
  275             \number of addresses for a given wallet."

  389 -- | Determines whether an 'Address' is an Icarus address.
  390 --
  391 -- Returns either details about the 'Address', or 'ErrInspectAddress' if it's
  392 -- not a valid icarus address.
  393 --
  394 -- @since 3.4.0
  395 eitherInspectAddress :: Address -> Either ErrInspectAddress AddressInfo
  396 eitherInspectAddress addr = do
  397     payload <- first DeserialiseError $
  398         CBOR.deserialiseCbor CBOR.decodeAddressPayload $
  399         unAddress addr
  400     ntwrk <- bimap DeserialiseError (fmap NetworkTag) $
  401         CBOR.deserialiseCbor CBOR.decodeProtocolMagicAttr payload
  402     (root, attrs) <- first DeserialiseError $
  403         CBOR.deserialiseCbor decodePayload payload
  404     if (elem 1 $ fst <$> attrs)
  405         then Left UnexpectedDerivationPath
  406         else Right AddressInfo
  407             { infoAddressRoot = root
  408             , infoNetworkTag = ntwrk
  409             }
  410   where
  411     decodePayload :: forall s. CBOR.Decoder s (ByteString, [(Word8, ByteString)])
  412     decodePayload = do
  413         _ <- CBOR.decodeListLenCanonicalOf 3
  414         root <- CBOR.decodeBytes
  415         (root,) <$> CBOR.decodeAllAttributes

  527 -- Purpose is a constant set to 44' (or 0x8000002C) following the original
  528 -- BIP-44 specification.
  529 --
  530 -- It indicates that the subtree of this node is used according to this
  531 -- specification.
  532 --
  533 -- Hardened derivation is used at this level.
  534 purposeIndex :: Word32
  535 purposeIndex = 0x8000002C
  536 
  537 -- One master node (seed) can be used for unlimited number of independent
  538 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  539 -- same space for various cryptocoins has some disadvantages.
  540 --
  541 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  542 -- addresses across cryptocoins and improving privacy issues.
  543 --
  544 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  545 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  546 -- Lovelace.
  547 --
  548 -- Hardened derivation is used at this level.
  549 coinTypeIndex :: Word32
  550 coinTypeIndex = 0x80000717

  552 -- The minimum seed length for 'generateKeyFromMnemonic' and 'unsafeGenerateKeyFromMnemonic'.
  553 minSeedLengthBytes :: Int
  554 minSeedLengthBytes = 16

  556 -- Hardware Ledger devices generates keys from mnemonic using a different
  557 -- approach (different from the rest of Cardano).
  558 --
  559 -- It is a combination of:
  560 --
  561 -- - [SLIP 0010](https://github.com/satoshilabs/slips/blob/master/slip-0010.md)
  562 -- - [BIP 0032](https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki)
  563 -- - [BIP 0039](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki)
  564 -- - [RFC 8032](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  565 -- - What seems to be arbitrary changes from Ledger regarding the calculation of
  566 --   the initial chain code and generation of the root private key.
  567 unsafeGenerateKeyFromHardwareLedger
  568     :: SomeMnemonic
  569         -- ^ The root mnemonic
  570     -> Icarus 'RootK XPrv
  571 unsafeGenerateKeyFromHardwareLedger (SomeMnemonic mw) = unsafeFromRight $ do
  572     let seed = pbkdf2HmacSha512
  573             $ T.encodeUtf8
  574             $ T.intercalate " "
  575             $ mnemonicToText mw
  576 
  577     -- NOTE
  578     -- SLIP-0010 refers to `iR` as the chain code. Here however, the chain code
  579     -- is obtained as a hash of the initial seed whereas iR is used to make part
  580     -- of the root private key itself.
  581     let cc = hmacSha256 (BS.pack [1] <> seed)
  582     let (iL, iR) = first pruneBuffer $ hashRepeatedly seed
  583 
  584     prv <- maybe (Left "invalid xprv") pure $ xprvFromBytes $ iL <> iR <> cc
  585     pure $ Icarus prv
  586   where
  587     -- Errors yielded in the body of 'unsafeGenerateKeyFromHardwareLedger' are
  588     -- programmer errors (out-of-range byte buffer access or, invalid length for
  589     -- cryptographic operations). Therefore, we throw badly if we encounter any.
  590     unsafeFromRight :: Either String a -> a
  591     unsafeFromRight = either error id
  592 
  593     -- This is the algorithm described in SLIP 0010 for master key generation
  594     -- with an extra step to discard _some_ of the potential private keys. Why
  595     -- this extra step remains a mystery as of today.
  596     --
  597     --      1. Generate a seed byte sequence S of 512 bits according to BIP-0039.
  598     --         (done in a previous step, passed as argument).
  599     --
  600     --      2. Calculate I = HMAC-SHA512(Key = "ed25519 seed", Data = S)
  601     --
  602     --      3. Split I into two 32-byte sequences, IL and IR.
  603     --
  604     -- extra *******************************************************************
  605     -- *                                                                       *
  606     -- *    3.5 If the third highest bit of the last byte of IL is not zero    *
  607     -- *        S = I and go back to step 2.                                   *
  608     -- *                                                                       *
  609     -- *************************************************************************
  610     --
  611     --      4. Use parse256(IL) as master secret key, and IR as master chain code.
  612     hashRepeatedly :: ByteString -> (ByteString, ByteString)
  613     hashRepeatedly bytes = case BS.splitAt 32 (hmacSha512 bytes) of
  614         (iL, iR) | isInvalidKey iL -> hashRepeatedly (iL <> iR)
  615         (iL, iR) -> (iL, iR)
  616       where
  617         isInvalidKey k = testBit (k `BS.index` 31) 5
  618 
  619     -- - Clear the lowest 3 bits of the first byte
  620     -- - Clear the highest bit of the last byte
  621     -- - Set the second highest bit of the last byte
  622     --
  623     -- As described in [RFC 8032 - 5.1.5](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  624     pruneBuffer :: ByteString -> ByteString
  625     pruneBuffer bytes =
  626         let
  627             (firstByte, rest) = fromMaybe (error "pruneBuffer: no first byte") $
  628                 BS.uncons bytes
  629 
  630             (rest', lastByte) = fromMaybe (error "pruneBuffer: no last byte") $
  631                 BS.unsnoc rest
  632 
  633             firstPruned = firstByte
