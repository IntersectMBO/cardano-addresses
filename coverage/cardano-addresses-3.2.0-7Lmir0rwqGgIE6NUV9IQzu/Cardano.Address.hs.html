   72 -- | An 'Address' type representing 'Cardano' addresses. Internals are
   73 -- irrevelant to the user.
   74 --
   75 -- @since 1.0.0
   76 newtype Address = Address
   77     { unAddress :: ByteString
   78     } deriving stock (Generic, Show, Eq, Ord)
   79 instance NFData Address

   87 -- | Encode an 'Address' to a base58 'Text'.
   88 --
   89 -- @since 1.0.0
   90 base58 :: Address -> Text
   91 base58 = T.decodeUtf8 . encode EBase58 . unAddress
   92 
   93 -- | Decode a base58-encoded 'Text' into an 'Address'
   94 --
   95 -- @since 1.0.0
   96 fromBase58 :: Text -> Maybe Address
   97 fromBase58 =
   98     deserialiseCbor (unsafeMkAddress <$> decodeAddress)
   99    <=<
  100     eitherToMaybe . E.fromBase58 . T.encodeUtf8
  101 
  102 -- | Encode an 'Address' to bech32 'Text', using @addr@ as a human readable prefix.
  103 --
  104 -- @since 1.0.0
  105 bech32 :: Address -> Text
  106 bech32 = bech32With CIP5.addr
  107 
  108 -- | Encode an 'Address' to bech32 'Text', using a specified human readable prefix.
  109 --
  110 -- @since 2.0.0
  111 bech32With :: HumanReadablePart -> Address -> Text
  112 bech32With hrp =
  113     T.decodeUtf8 . encode (EBech32 hrp) . unAddress
  114 
  115 -- | Decode a bech32-encoded  'Text' into an 'Address'
  116 --
  117 -- @since 1.0.0
  118 fromBech32 :: Text -> Maybe Address
  119 fromBech32 =
  120     eitherToMaybe . fmap (unsafeMkAddress . snd) . E.fromBech32 (const id) . T.encodeUtf8

  161 -- | A 'ChainPointer' type representing location of some object
  162 -- in the blockchain (eg., delegation certificate). This can be achieved
  163 -- unambiguously by specifying slot number, transaction index and the index
  164 -- in the object list (eg., certification list).
  165 -- For delegation certificates, alternatively, the delegation key can be used and
  166 -- then 'DelegationAddress' can be used.
  167 --
  168 -- @since 2.0.0
  169 data ChainPointer = ChainPointer
  170     { slotNum :: Natural
  171       -- ^ Pointer to the slot
  172     , transactionIndex :: Natural
  173       -- ^ transaction index
  174     , outputIndex :: Natural
  175       -- ^ output list index
  176     } deriving stock (Generic, Show, Eq, Ord)
  177 instance NFData ChainPointer

  206 -- Magic constant associated with a given network. This is mainly used in two
  207 -- places:
  208 --
  209 -- (1) In 'Address' payloads, to discriminate addresses between networks.
  210 -- (2) At the network-level, when doing handshake with nodes.
  211 newtype NetworkTag
  212     = NetworkTag { unNetworkTag :: Word32 }
  213     deriving (Generic, Show, Eq)
  214 instance NFData NetworkTag
  215 
  216 -- Describe requirements for address discrimination on the Byron era.
  217 data AddressDiscrimination
  218     = RequiresNetworkTag
  219     | RequiresNoTag
  220     deriving (Generic, Show, Eq)
  221 instance NFData AddressDiscrimination

  223 invariantSize :: HasCallStack => Int -> ByteString -> ByteString
  224 invariantSize expectedLength bytes
  225     | BS.length bytes == expectedLength = bytes
  226     | otherwise = error
  227       $ "length was "
  228       ++ show (BS.length bytes)
  229       ++ ", but expected to be "
  230       ++ (show expectedLength)
  231 
  232 invariantNetworkTag :: HasCallStack => Word32 -> NetworkTag -> Word8
  233 invariantNetworkTag limit (NetworkTag num)
  234     | num < limit = fromIntegral num
  235     | otherwise = error
  236       $ "network tag was "
  237       ++ show num
  238       ++ ", but expected to be less than "
  239       ++ show limit
