  124 -- $introduction
  125 --
  126 -- We call 'Entropy' an arbitrary sequence of bytes that has been generated through __high quality randomness methods__.
  127 -- The allowed size of an 'Entropy' is @96-256@ bits and is __necessarily a multiple of 32 bits__ (4 bytes).
  128 --
  129 -- We call 'Mnemonic' an 'Entropy' with an appended checksum calculated by taking the first @ent / 32@ bits of the /SHA256/ hash of it,
  130 -- where ent designates the 'Entropy' size in bits.
  131 --
  132 -- The concatenated result is split into groups of @11@ bits, each encoding a number from 0 to 2047 serving as an index into a [known dictionary](https://github.com/input-output-hk/cardano-wallet/tree/master/specifications/mnemonic/english.txt).
  133 -- This makes for a __human-readable sentence__ of English words.
  134 --
  135 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  136 -- | Entropy Size        | Checksum Size | Sentence Length | Example                                                                                                                                         |
  137 -- +=====================+===============+=================+=================================================================================================================================================+
  138 -- | 96  bits (12 bytes) | 3 bits        | 9 words         | test child burst immense armed parrot company walk dog                                                                                          |
  139 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  140 -- | 128 bits (16 bytes) | 4 bits        | 12 words        | test walk nut penalty hip pave soap entry language right filter choice                                                                          |
  141 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  142 -- | 160 bits (20 bytes) | 5 bits        | 15 words        | art forum devote street sure rather head chuckle guard poverty release quote oak craft enemy                                                    |
  143 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  144 -- | 192 bits (24 bytes) | 6 bits        | 18 words        | churn shaft spoon second erode useless thrive burst group seed element sign scrub buffalo jelly grace neck useless                              |
  145 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  146 -- | 224 bits (28 bytes) | 7 bits        | 21 words        | draft ability female child jump maid roof hurt below live topple paper exclude ordinary coach churn sunset emerge blame ketchup much            |
  147 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  148 -- | 256 bits (32 bytes) | 8 bits        | 24 words        | excess behave track soul table wear ocean cash stay nature item turtle palm soccer lunch horror start stumble month panic right must lock dress |
  149 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+

  151 -- A opaque 'Mnemonic' type.
  152 data Mnemonic (mw :: Nat) = Mnemonic
  153     { mnemonicToEntropy  :: Entropy (EntropySize mw)

  164     deriving stock (Show, Typeable)

  167 -- | This wraps errors from "Cardano.Encoding.BIP39"
  168 data MkMnemonicError csz
  169     = ErrMnemonicWords MnemonicWordsError
  170       -- ^ Wrong number of words in mnemonic.
  171     | ErrEntropy (EntropyError csz)
  172       -- ^ Invalid entropy length or checksum.
  173     | ErrDictionary DictionaryError
  174       -- ^ Invalid word in mnemonic.
  175     deriving stock (Eq, Show)
  176 
  177 deriving instance Eq (EntropyError czs)
  178 deriving instance Eq MnemonicWordsError
  179 deriving instance Eq DictionaryError

  181 -- NFData instances
  182 instance NFData (Mnemonic mw) where
  183     rnf (Mnemonic ent ws) = toNormalForm ent `seq` toNormalForm ws
  184 instance NFData (EntropyError csz) where
  185     rnf (ErrInvalidEntropyLength a b) = rnf a `seq` rnf b
  186     rnf (ErrInvalidEntropyChecksum a b) = toNormalForm a `seq` toNormalForm b
  187 instance NFData MnemonicWordsError where
  188     rnf (ErrWrongNumberOfWords a b) = rnf a `seq` rnf b
  189 instance NFData DictionaryError where
  190     rnf (ErrInvalidDictionaryWord s) = toNormalForm s
  191 instance NFData (MkMnemonicError csz) where
  192     rnf (ErrMnemonicWords e) = rnf e
  193     rnf (ErrEntropy e) = rnf e
  194     rnf (ErrDictionary e) = rnf e

  196 -- | Smart-constructor for the 'Entropy'. Make sure the 'ByteString' comes from a highly random source or use 'genEntropy'.
  197 --
  198 -- __example__:
  199 --
  200 -- >>> mkEntropy @160 bytes
  201 -- Entropy {} :: Entropy 160
  202 --
  203 -- __property__:
  204 --
  205 -- prop> mkEntropy (entropyToBytes ent) == Right ent
  206 --
  207 -- @since 1.0.0
  208 mkEntropy
  209     :: forall (ent :: Nat) csz. (ValidEntropySize ent, ValidChecksumSize ent csz)
  210     => ScrubbedBytes
  211     -> Either (EntropyError csz) (Entropy ent)
  212 mkEntropy = toEntropy
  213 
  214 -- | Generate Entropy of a given size using a cryptographically secure random seed.
  215 --
  216 -- __example:__
  217 --
  218 -- >>> genEntropy @128
  219 -- Entropy {} :: Entropy 128
  220 --
  221 -- @since 1.0.0
  222 genEntropy
  223     :: forall (ent :: Nat) csz. (ValidEntropySize ent, ValidChecksumSize ent csz)
  224     => IO (Entropy ent)
  225 genEntropy =
  226     let
  227         size =
  228             fromIntegral $ natVal @ent Proxy
  229         eitherToIO =
  230             either (throwM . UnexpectedEntropyError) return
  231     in
  232         (eitherToIO . mkEntropy) =<< Crypto.getEntropy (size `div` 8)
  233 
  234 -- | Smart-constructor for 'Mnemonic'. Requires a type application to
  235 -- disambiguate the mnemonic size.
  236 --
  237 -- __example__:
  238 --
  239 -- >>> mkMnemonic @15 sentence
  240 -- Mnemonic {} :: Mnemonic 15
  241 --
  242 -- __property__:
  243 --
  244 -- prop> mkMnemonic (mnemonicToText mnemonic) == Right mnemonic
  245 --
  246 -- @since 1.0.0
  247 mkMnemonic
  248     :: forall (mw :: Nat) (ent :: Nat) csz.
  249      ( ConsistentEntropy ent mw csz
  250      , EntropySize mw ~ ent
  251      )
  252     => [Text]
  253     -> Either (MkMnemonicError csz) (Mnemonic mw)
  254 mkMnemonic wordsm = do
  255     phrase <- left ErrMnemonicWords
  256         $ mnemonicPhrase @mw (toUtf8String <$> wordsm)
  257 
  258     sentence <- left ErrDictionary
  259         $ mnemonicPhraseToMnemonicSentence Dictionary.english phrase
  260 
  261     entropy <- left ErrEntropy
  262         $ wordsToEntropy sentence
  263 
  264     pure Mnemonic
  265         { mnemonicToEntropy  = entropy
  266         , mnemonicToSentence = sentence
  267         }

  269 -- | Convert an Entropy to a corresponding Mnemonic Sentence. Since 'Entropy'
  270 -- and 'Mnemonic' can only be created through smart-constructors, this function
  271 -- cannot fail and is total.
  272 --
  273 -- @since 1.0.0
  274 entropyToMnemonic
  275     :: forall mw ent csz.
  276      ( ValidMnemonicSentence mw
  277      , ValidEntropySize ent
  278      , ValidChecksumSize ent csz
  279      , ent ~ EntropySize mw
  280      , mw ~ MnemonicWords ent
  281      )
  282     => Entropy ent
  283     -> Mnemonic mw
  284 entropyToMnemonic entropy = Mnemonic
  285     { mnemonicToSentence = entropyToWords entropy
  286     , mnemonicToEntropy  = entropy
  287     }

  327 -- | Ease the manipulation of 'Mnemonic' by encapsulating the type constraints inside a constructor.
  328 -- This is particularly useful for functions which do not require anything but a valid 'Mnemonic' without any
  329 -- particular pre-condition on the size of the 'Mnemonic' itself.
  330 --
  331 -- @since 1.0.0
  332 data SomeMnemonic where
  333     SomeMnemonic :: forall mw. KnownNat mw => Mnemonic mw -> SomeMnemonic
  334 
  335 deriving instance Show SomeMnemonic
  336 instance Eq SomeMnemonic where
  337     (SomeMnemonic mwa) == (SomeMnemonic mwb) =
  338         case typeOf mwa `testEquality` typeOf mwb of
  339             Nothing -> False
  340             Just Refl -> mwa == mwb
  341 instance NFData SomeMnemonic where
  342     rnf (SomeMnemonic mnem) = rnf mnem

  344 -- | This class enables caller to parse text list of variable length
  345 -- into mnemonic sentences.
  346 --
  347 -- Note that the given 'Nat's **have** to be valid mnemonic sizes, otherwise the
  348 -- underlying code won't even compile, with not-so-friendly error messages.
  349 class MkSomeMnemonic (sz :: [Nat]) where
  350     -- | Construct a mnemonic from a list of words. This function is particularly useful when the
  351     -- number of words is not necessarily known at runtime. The function is however /ambiguous/ and
  352     -- requires thereby a type application.
  353     --
  354     -- __examples:__
  355     --
  356     -- >>> mkSomeMnemonic @'[ 12 ] [ "test", "child", "burst", "immense", "armed", "parrot", "company", "walk", "dog" ]
  357     -- Left "Invalid number of words: 12 words are expected."
  358     --
  359     -- >>> mkSomeMnemonic @'[ 9, 12, 15 ] [ "test", "child", "burst", "immense", "armed", "parrot", "company", "walk", "dog" ]
  360     -- Right (SomeMnemonic ...)
  361     --
  362     -- @since 1.0.0
  363     mkSomeMnemonic :: [Text] -> Either (MkSomeMnemonicError sz) SomeMnemonic
  364 
  365 -- | Error reported from trying to create a passphrase from a given mnemonic
  366 --
  367 -- @since 1.0.0
  368 newtype MkSomeMnemonicError (sz :: [Nat]) =
  369     MkSomeMnemonicError { getMkSomeMnemonicError :: String }
  370     deriving stock (Eq, Show)
  371 
  372 instance {-# OVERLAPS #-}
  373     ( n ~ EntropySize mw
  374     , csz ~ CheckSumBits n
  375     , ConsistentEntropy n mw csz
  376     , MkSomeMnemonic rest
  377     , NatVals rest
  378     ) =>
  379     MkSomeMnemonic (mw ': rest)
  380   where
  381     mkSomeMnemonic parts = case parseMW of
  382         Left err -> left (promote err) parseRest
  383         Right mw -> Right mw
  384       where
  385         parseMW = left (MkSomeMnemonicError . getMkSomeMnemonicError) $ -- coerce
  386             mkSomeMnemonic @'[mw] parts
  387         parseRest = left (MkSomeMnemonicError . getMkSomeMnemonicError) $ -- coerce
  388             mkSomeMnemonic @rest parts
  389         promote e e' =
  390             let
  391                 sz = fromEnum <$> natVals (Proxy :: Proxy (mw ': rest))
  392                 mw = fromEnum $ natVal (Proxy :: Proxy mw)
  393             in if length parts `notElem` sz
  394                 then MkSomeMnemonicError
  395                     $  "Invalid number of words: "
  396                     <> intercalate ", " (show <$> init sz)
  397                     <> (if length sz > 1 then " or " else "") <> show (last sz)
  398                     <> " words are expected."
  399                 else if length parts == mw then e else e'
  400 
  401 -- | Small helper to collect 'Nat' values from a type-level list
  402 class NatVals (ns :: [Nat]) where
  403     natVals :: Proxy ns -> [Integer]
  404 
  405 instance NatVals '[] where
  406     natVals _ = []
  407 
  408 instance (KnownNat n, NatVals rest) => NatVals (n ': rest) where
  409     natVals _ = natVal (Proxy :: Proxy n) : natVals (Proxy :: Proxy rest)
  410 
  411 instance
  412     ( n ~ EntropySize mw
  413     , csz ~ CheckSumBits n
  414     , ConsistentEntropy n mw csz
  415     ) =>
  416     MkSomeMnemonic (mw ': '[])
  417   where
  418     mkSomeMnemonic parts = do
  419         bimap (MkSomeMnemonicError . pretty) SomeMnemonic (mkMnemonic @mw parts)
  420       where
  421         pretty = \case
  422             ErrMnemonicWords ErrWrongNumberOfWords{} ->
  423                 "Invalid number of words: "
  424                 <> show (natVal (Proxy :: Proxy mw))
  425                 <> " words are expected."
  426             ErrDictionary (ErrInvalidDictionaryWord _) ->
  427                 "Found an unknown word not present in the pre-defined dictionary. \
  428                 \The full dictionary is available here: \
  429                 \https://github.com/input-output-hk/cardano-wallet/tree/master/specifications/mnemonic/english.txt"
  430             ErrEntropy ErrInvalidEntropyChecksum{} ->
  431                 "Invalid entropy checksum: please double-check the last word of \
  432                 \your mnemonic sentence."
  433             ErrEntropy ErrInvalidEntropyLength{} ->
  434                 "Something went wrong when trying to generate the entropy from \
  435                 \the given mnemonic. As a user, there's nothing you can do."
