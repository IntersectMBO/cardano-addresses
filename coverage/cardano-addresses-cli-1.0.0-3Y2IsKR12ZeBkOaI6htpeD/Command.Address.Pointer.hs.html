   48 data Cmd = Cmd
   49     { _slotNum :: Natural
   50     , _transactionIndex :: Natural
   51     , _outputIndex :: Natural
   52     } deriving (Show)
   53 
   54 mod :: (Cmd -> parent) -> Mod CommandFields parent
   55 mod liftCmd = command "pointer" $
   56     info (helper <*> fmap liftCmd parser) $ mempty
   57         <> progDesc "Create a pointer address"
   58         <> header "Create addresses with a pointer that indicate the position \
   59             \of a registered stake address on the chain."
   60         <> footerDoc (Just $ vsep
   61             [ string "The payment address is read from stdin."
   62             , string ""
   63             , string "Example:"
   64             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   65             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   66             , indent 2 $ string ""
   67             , indent 2 $ bold $ string "$ cat root.prv \\"
   68             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/0/0 > addr.prv"
   69             , indent 2 $ string ""
   70             , indent 2 $ bold $ string "$ cat addr.prv \\"
   71             , indent 4 $ bold $ string $ "| "<>progName<>" key public \\"
   72             , indent 4 $ bold $ string $ "| "<>progName<>" address payment --network-tag 0\\"
   73             , indent 4 $ bold $ string $ "| "<>progName<>" address pointer 42 14 0"
   74             , indent 2 $ string "addr1grq8e0smk44luyl897e24gn6qfkx4ax734r6pzq29zcew032pcqqef7zzu"
   75             ])
   76   where
   77     parser = Cmd
   78         <$> argument auto (metavar "SLOT" <> help "A slot number")
   79         <*> argument auto (metavar "TX"   <> help "A transaction index within that slot")
   80         <*> argument auto (metavar "OUT"  <> help "An output index within that transaction")
   81 
   82 run :: Cmd -> IO ()
   83 run Cmd{_slotNum,_transactionIndex,_outputIndex} = do
   84     bytes <- hGetBytes stdin
   85     let ptr = ChainPointer
   86             { slotNum = _slotNum
   87             , transactionIndex = _transactionIndex
   88             , outputIndex = _outputIndex
   89             }
   90     case Shelley.extendAddress (unsafeMkAddress bytes) (Right ptr) of
   91         Left (ErrInvalidAddressStyle msg) -> fail msg
   92         Left (ErrInvalidAddressType  msg) -> fail msg
   93         Right addr -> B8.hPutStr stdout $ T.encodeUtf8 $ bech32 addr
