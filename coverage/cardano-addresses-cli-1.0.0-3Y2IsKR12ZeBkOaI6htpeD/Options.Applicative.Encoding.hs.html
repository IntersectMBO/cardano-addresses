   28 -- | Parse an 'Encoding' from the command-line, as set of non-overlapping flags.
   29 encodingOpt :: HumanReadablePart -> Parser Encoding
   30 encodingOpt hrp = base16Flag <|> base58Flag <|> bech32Flag
   31   where
   32     base16Flag = flag'  EBase16 (long "base16")
