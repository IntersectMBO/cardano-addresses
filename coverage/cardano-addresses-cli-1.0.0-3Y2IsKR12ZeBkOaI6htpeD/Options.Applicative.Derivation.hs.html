   76 --
   77 -- Derivation Path
   78 --
   79 
   80 -- | Represent a user-provided 'DerivationPath'.
   81 newtype DerivationPath = DerivationPath [DerivationIndex]
   82     deriving (Show, Eq)
   83 
   84 derivationPathFromString :: String -> Either String DerivationPath
   85 derivationPathFromString str =
   86     DerivationPath
   87         <$> mapM (derivationIndexFromString . T.unpack) (T.splitOn "/" txt)
   88   where
   89     txt = T.pack str
   90 
   91 derivationPathToString :: DerivationPath -> String
   92 derivationPathToString (DerivationPath xs) =
   93     intercalate "/" $ map derivationIndexToString xs
   94 
   95 castDerivationPath :: DerivationPath -> [Index 'WholeDomain depth]
   96 castDerivationPath (DerivationPath xs) = toEnum . fromEnum <$> xs
   97 
   98 derivationPathArg :: Parser DerivationPath
   99 derivationPathArg = argument (eitherReader derivationPathFromString) $ mempty
  100     <> metavar "DERIVATION-PATH"
  101     <> help
  102         "Slash-separated derivation path. Hardened indexes are marked with a \
  103         \'H' (e.g. 1852H/1815H/0H/0)."
  104     <> completer (listCompleter ["1852H/1815H/0H/", "44H/1815H/0H/"])
  105 
  106 derivationPathOpt :: Parser DerivationPath
  107 derivationPathOpt = option (eitherReader derivationPathFromString) $ mempty
  108     <> metavar "DERIVATION-PATH"
  109     <> long "path"
  110     <> help
  111         "Slash-separated derivation path. Hardened indexes are marked with a \
  112         \'H' (e.g. 1852H/1815H/0H/0)."
  113     <> completer (listCompleter ["1852H/1815H/0H/", "44H/1815H/0H/"])

  115 --
  116 -- Derivation Index
  117 --
  118 
  119 newtype DerivationIndex = DerivationIndex Word32
  120     deriving stock   (Show, Eq)
  121     deriving newtype (Bounded, Enum, Ord)
  122 
  123 -- | Safely cast a 'DerivationIndex' to an 'Integer'.
  124 indexToInteger :: DerivationIndex -> Integer
  125 indexToInteger (DerivationIndex ix) = fromIntegral ix
  126 
  127 -- | Get the first 'DerivationIndex' considered /hardened/.
  128 firstHardened :: DerivationIndex
  129 firstHardened = DerivationIndex 0x80000000
  130 
  131 -- | Smart-constructor for a 'DerivationIndex'
  132 mkDerivationIndex :: Integer -> Either String DerivationIndex
  133 mkDerivationIndex ix
  134     | ix > fromIntegral (maxBound @Word32) =
  135         Left $ show ix <> " is too high to be a derivation index."
  136     | otherwise =
  137         pure $ DerivationIndex $ fromIntegral ix
  138 
  139 -- | Convert a string to a derivation index. String must be followed by a
  140 -- capital /H/ to mark hardened index. For example @0@ refers to the first soft
  141 -- index, whereas @0H@ refers to the first hardened index.
  142 derivationIndexFromString :: String -> Either String DerivationIndex
  143 derivationIndexFromString "" = Left "An empty string is not a derivation index!"
  144 derivationIndexFromString str
  145     | "H" `isSuffixOf` str = do
  146         parseHardenedIndex (init str)
  147     | otherwise = do
  148         parseSoftIndex str
  149   where
  150     parseHardenedIndex txt = do
  151         ix <- left (const msg) $ readEitherSafe txt
  152         mkDerivationIndex $ ix + indexToInteger firstHardened
  153       where
  154         msg = mconcat
  155             [ "Unable to parse hardened index. Hardened indexes are integer "
  156             , "values, between "
  157             , show (indexToInteger (minBound @DerivationIndex))
  158             , " and "
  159             , show (indexToInteger firstHardened)
  160             , " ending with a capital 'H'. For example: \"42H\","
  161             ]
  162 
  163     parseSoftIndex txt = do
  164         ix <- left (const msg) $ readEitherSafe txt
  165         guardSoftIndex ix
  166         mkDerivationIndex ix
  167       where
  168         guardSoftIndex ix
  169             | ix >= indexToInteger firstHardened =
  170                 Left $ mconcat
  171                     [ show ix
  172                     , " is too high to be a soft derivation index. "
  173                     , "Did you mean \""
  174                     , show (ix - indexToInteger firstHardened)
  175                     , "H\"?"
  176                     ]
  177             | otherwise =
  178                 pure ()
  179 
  180         msg = mconcat
  181             [ "Unable to parse soft index. Soft indexes are integer "
  182             , "values, between "
  183             , show (indexToInteger (minBound @DerivationIndex))
  184             , " and "
  185             , show (indexToInteger firstHardened)
  186             , ". For example: \"14\"."
  187             ]
  188 
  189 -- | Convert a 'DerivationIndex' back to string.
  190 derivationIndexToString :: DerivationIndex -> String
  191 derivationIndexToString ix_@(DerivationIndex ix)
  192     | ix_ >= firstHardened = show ix' ++ "H"
  193     | otherwise            = show ix
  194   where
  195     ix' = fromIntegral ix - indexToInteger firstHardened

  206 xpubArg :: String -> Parser XPub
  207 xpubArg helpDoc =
  208     argument (eitherReader reader) $ mempty
  209         <> metavar "XPUB"
  210         <> help helpDoc
  211   where
  212     toBytes = T.encodeUtf8 . T.pack
  213     reader str = do
  214         bytes <- case detectEncoding str of
  215             Just EBase16   -> fromBase16 (toBytes str)
  216             Just EBech32{} -> fromBech32 markCharsRedAtIndices (toBytes str)
  217             Just EBase58   -> fromBase58 (toBytes str)
  218             Nothing        -> Left
  219                 "Couldn't detect input encoding? The key must be encoded as \
  220                 \base16, bech32 or base58."
  221         case xpubFromBytes bytes of
  222             Just xpub -> pure xpub
  223             Nothing   -> Left
  224                 "Failed to convert bytes into a valid extended public key."
