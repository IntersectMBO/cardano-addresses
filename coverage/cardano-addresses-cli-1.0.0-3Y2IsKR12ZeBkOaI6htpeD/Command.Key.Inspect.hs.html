   40 data Cmd = Inspect
   41     deriving (Show)
   42 
   43 mod :: (Cmd -> parent) -> Mod CommandFields parent
   44 mod liftCmd = command "inspect" $
   45     info (helper <*> fmap liftCmd parser) $ mempty
   46         <> progDesc "Show information about a key"
   47         <> footerDoc (Just $ string $ mconcat
   48             [ "The parent key is read from stdin."
   49             ])
   50   where
   51     parser = pure Inspect
   52 
   53 run :: Cmd -> IO ()
   54 run Inspect = do
   55     either inspectXPub inspectXPrv =<< hGetXP__ stdin
   56   where
   57     base16 :: ByteString -> Text
   58     base16 = T.decodeUtf8 . encode EBase16
   59 
   60     inspectXPub :: XPub -> IO ()
   61     inspectXPub xpub = do
   62         BL8.hPutStrLn stdout $ Json.encodePretty $ Json.object
   63             [ "key_type" .= Json.String "public"
   64             , "extended_key" .= Json.String (base16 pub)
   65             , "chain_code" .= Json.String (base16 cc)
   66             ]
   67       where
   68         pub = xpubPublicKey xpub
   69         cc  = xpubChainCode xpub
   70 
   71     inspectXPrv :: XPrv -> IO ()
   72     inspectXPrv xprv =
   73         BL8.hPutStrLn stdout $ Json.encodePretty $ Json.object
   74             [ "key_type" .= Json.String "private"
   75             , "extended_key" .= Json.String (base16 prv)
   76             , "chain_code" .= Json.String (base16 cc)
   77             ]
   78       where
   79         prv = xprvPrivateKey xprv
   80         cc  = xprvChainCode  xprv
