   35 data Cmd
   36     = FromRecoveryPhrase FromRecoveryPhrase.Cmd
   37     | Child Child.Cmd
   38     | Public Public.Cmd
   39     | Inspect Inspect.Cmd
   40     deriving (Show)
   41 
   42 mod :: (Cmd -> parent) -> Mod CommandFields parent
   43 mod liftCmd = command "key" $
   44     info (helper <*> fmap liftCmd parser) $ mempty
   45         <> progDesc "About public/private keys"
   46         <> footerDoc (Just $ vsep
   47             [ string "Example:"
   48             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   49             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   50             , indent 2 $ string ""
   51             , indent 2 $ bold $ string "$ cat root.prv \\"
   52             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H \\"
   53             , indent 4 $ bold $ string "| tee acct.prv \\"
   54             , indent 4 $ bold $ string $ "| "<>progName<>" key public > acct.pub"
   55             , indent 2 $ string ""
   56             , indent 2 $ bold $ string $ "$ "<>progName<>" key inspect <<< $(cat acct.prv)"
   57             , indent 2 $ string "{"
   58             , indent 2 $ string "    \"key_type\": \"private\","
   59             , indent 2 $ string "    \"chain_code\": \"67bef6f80df02c7452e20e76ffb4bb57cae8aac2adf042b21a6b19e4f7b1f511\","
   60             , indent 2 $ string "    \"extended_key\": \"90ead3efad7aacac242705ede323665387f49ed847bed025eb333708ccf6aa54403482a867daeb18f38c57d6cddd7e6fd6aed4a3209f7425a3d1c5d9987a9c5f\""
   61             , indent 2 $ string "}"
   62             , indent 2 $ string ""
   63             , indent 2 $ bold $ string $ "$ "<>progName<>" key inspect <<< $(cat acct.pub)"
   64             , indent 2 $ string "{"
   65             , indent 2 $ string "    \"key_type\": \"public\","
   66             , indent 2 $ string "    \"chain_code\": \"67bef6f80df02c7452e20e76ffb4bb57cae8aac2adf042b21a6b19e4f7b1f511\","
   67             , indent 2 $ string "    \"extended_key\": \"d306350ee88f51fb710252e27f0c40006c58e994761b383e02d400e2be59b3cc\""
   68             , indent 2 $ string "}"
   69             ])
   70   where
   71     parser = subparser $ mconcat
   72         [ FromRecoveryPhrase.mod FromRecoveryPhrase
   73         , Child.mod Child
   74         , Public.mod Public
   75         , Inspect.mod Inspect
   76         ]
   77 
   78 run :: Cmd -> IO ()
   79 run = \case
   80     FromRecoveryPhrase sub -> FromRecoveryPhrase.run sub
   81     Child sub -> Child.run sub
   82     Public sub -> Public.run sub
   83     Inspect sub -> Inspect.run sub
