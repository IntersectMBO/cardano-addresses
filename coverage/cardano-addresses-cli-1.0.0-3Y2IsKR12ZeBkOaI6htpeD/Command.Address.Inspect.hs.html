    6 module Command.Address.Inspect
    7     ( Cmd (..)
    8     , mod

   11 
   12 import Prelude hiding
   13     ( mod )
   14 
   15 import Cardano.Address
   16     ( unsafeMkAddress )
   17 import Cardano.Address.Style.Byron
   18     ( inspectByronAddress )
   19 import Cardano.Address.Style.Icarus
   20     ( inspectIcarusAddress )
   21 import Cardano.Address.Style.Jormungandr
   22     ( inspectJormungandrAddress )
   23 import Cardano.Address.Style.Shelley
   24     ( inspectShelleyAddress )
   25 import Control.Applicative
   26     ( (<|>) )
   27 import Options.Applicative
   28     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   29 import Options.Applicative.Help.Pretty
   30     ( bold, indent, string, vsep )
   31 import System.IO
   32     ( stdin, stdout )
   33 import System.IO.Extra
   34     ( hGetBytes, progName )
   35 
   36 import qualified Data.Aeson.Encode.Pretty as Json
   37 import qualified Data.ByteString.Lazy.Char8 as BL8
   38 
   39 
   40 data Cmd = Inspect
   41     deriving (Show)
   42 
   43 mod :: (Cmd -> parent) -> Mod CommandFields parent
   44 mod liftCmd = command "inspect" $
   45     info (helper <*> fmap liftCmd parser) $ mempty
   46         <> progDesc "Show information about an address"
   47         <> footerDoc (Just $ vsep
   48             [ string "The address is read from stdin."
   49             , string ""
   50             , string "Example:"
   51             , indent 2 $ bold $ string "$ cat addr.prv \\"
   52             , indent 4 $ bold $ string $ "| "<>progName<>" key public \\"
   53             , indent 4 $ bold $ string $ "| "<>progName<>" address payment --network-tag 0 \\"
   54             , indent 4 $ bold $ string $ "| "<>progName<>" address delegation $(cat stake.prv | "<>progName<>" key public) \\"
   55             , indent 4 $ bold $ string $ "| "<>progName<>" address inspect"
   56             , indent 2 $ string "{"
   57             , indent 2 $ string "    \"address_style\": \"Shelley\","
   58             , indent 2 $ string "    \"stake_reference\": \"by value\","
   59             , indent 2 $ string "    \"stake_key_hash\": \"6b542d6da35e6c95d95a33c6f66ec482d3f4caf3ad35e2ede09cf827\","
   60             , indent 2 $ string "    \"spending_key_hash\": \"44bc4f524f49a78a9c8a45b882d8710cb9254b3da6a978d50dc9b870\","
   61             , indent 2 $ string "    \"network_tag\": 0"
   62             , indent 2 $ string "}"
   63             ])
   64   where
   65     parser = pure Inspect
   66 
   67 run :: Cmd -> IO ()
   68 run Inspect = do
   69     bytes <- hGetBytes stdin
   70     case inspect (unsafeMkAddress bytes) of
   71         Nothing   -> fail "Unrecognized address on standard input"
   72         Just json -> BL8.hPutStrLn stdout (Json.encodePretty json)
   73   where
   74     inspect addr =
   75         inspectByronAddress       addr <|>
   76         inspectIcarusAddress      addr <|>
   77         inspectJormungandrAddress addr <|>
   78         inspectShelleyAddress     addr
