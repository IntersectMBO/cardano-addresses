   38 newtype Cmd = Cmd
   39     { networkTag :: NetworkTag
   40     } deriving (Show)
   41 
   42 mod :: (Cmd -> parent) -> Mod CommandFields parent
   43 mod liftCmd = command "payment" $
   44     info (helper <*> fmap liftCmd parser) $ mempty
   45         <> progDesc "Create a payment address"
   46         <> header "Payment addresses carry no delegation rights whatsoever."
   47         <> footerDoc (Just $ vsep
   48             [ string "Example:"
   49             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   50             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   51             , indent 2 $ string ""
   52             , indent 2 $ bold $ string "$ cat root.prv \\"
   53             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/0/0 > addr.prv"
   54             , indent 2 $ string ""
   55             , indent 2 $ bold $ string "$ cat addr.prv \\"
   56             , indent 4 $ bold $ string $ "| "<>progName<>" key public \\"
   57             , indent 4 $ bold $ string $ "| "<>progName<>" address payment --network-tag 0"
   58             , indent 2 $ string "addr1vrcmygdgp7v3mhz78v8kdsfru0y9wysnr9pgvvgmdqx2w0qrg8swg"
   59             ])
   60   where
   61     parser = Cmd
   62         <$> networkTagOpt Shelley
   63 
   64 run :: Cmd -> IO ()
   65 run Cmd{networkTag} = do
   66     xpub <- hGetXPub stdin
   67     case (mkNetworkDiscriminant . fromIntegral . unNetworkTag) networkTag of
   68         Left ErrWrongNetworkTag{} -> do
   69             fail "Invalid network tag. Must be between [0, 15]"
   70         Right discriminant -> do
   71             let addr = Shelley.paymentAddress discriminant (Shelley.liftXPub xpub)
   72             B8.hPutStr stdout $ T.encodeUtf8 $ bech32 addr
