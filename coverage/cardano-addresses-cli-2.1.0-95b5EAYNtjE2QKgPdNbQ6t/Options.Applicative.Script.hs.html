   33 scriptArg :: Parser Script
   34 scriptArg = argument (eitherReader scriptReader) $ mempty
   35     <> metavar "SCRIPT"
   36     -- TODO: Provides a bigger help text explaining how to construct a script
   37     -- address.
   38     <> help "Script string."
   39 
   40 scriptReader :: String -> Either String Script
   41 scriptReader = left prettyErrValidateScript . scriptFromString
   42 
   43 scriptHashArg :: String -> Parser ScriptHash
   44 scriptHashArg helpDoc =
   45     argument (eitherReader scriptHashReader) $ mempty
   46         <> metavar "SCRIPT HASH"
   47         <> help helpDoc
   48 
   49 scriptHashReader :: String -> Either String ScriptHash
   50 scriptHashReader str = do
   51     (_hrp, bytes) <- bech32Reader allowedPrefixes str
   52     case scriptHashFromBytes bytes of
   53         Just scriptHash -> pure scriptHash
   54         Nothing -> Left "Failed to convert bytes into a valid script hash."
   55   where
   56     allowedPrefixes =
   57         [ CIP5.script
   58         ]
