   38 newtype Cmd = Cmd
   39     { xpub :: XPub
   40     } deriving (Show)
   41 
   42 mod :: (Cmd -> parent) -> Mod CommandFields parent
   43 mod liftCmd = command "delegation" $
   44     info (helper <*> fmap liftCmd parser) $ mempty
   45         <> progDesc "Create a delegation address"
   46         <> footerDoc (Just $ vsep
   47             [ string "The payment address is read from stdin."
   48             , string ""
   49             , string "Example:"
   50             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   51             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   52             , indent 2 $ string ""
   53             , indent 2 $ bold $ string "$ cat root.prv \\"
   54             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/2/0 > stake.prv"
   55             , indent 2 $ string ""
   56             , indent 2 $ bold $ string "$ cat root.prv \\"
   57             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/0/0 > addr.prv"
   58             , indent 2 $ string ""
   59             , indent 2 $ bold $ string "$ cat addr.prv \\"
   60             , indent 4 $ bold $ string $ "| "<>progName<>" key public \\"
   61             , indent 4 $ bold $ string $ "| "<>progName<>" address payment --network-tag 0 \\"
   62             , indent 4 $ bold $ string $ "| "<>progName<>" address delegation $(cat stake.prv | "<>progName<>" key public)"
   63             , indent 2 $ string "addr1qpj2d4dqzds5p3mmlu95v9pex2d72cdvyjh2u3dtj4yqesv27k..."
   64             ])
   65   where
   66     parser = Cmd
   67         <$> xpubArg "An extended stake public key."
   68 
   69 run :: Cmd -> IO ()
   70 run Cmd{xpub} = do
   71     bytes <- hGetBytes stdin
   72     case Shelley.extendAddress (unsafeMkAddress bytes) (Left $ Shelley.liftXPub xpub) of
   73         Left (ErrInvalidAddressStyle msg) -> fail msg
   74         Left (ErrInvalidAddressType  msg) -> fail msg
   75         Right addr -> B8.hPutStr stdout $ T.encodeUtf8 $ bech32 addr
