   61 -- | A 'Script' type represents multi signature script. The script embodies conditions
   62 -- that need to be satisfied to make it valid.
   63 --
   64 -- @since 3.0.0
   65 data Script
   66     = RequireSignatureOf !KeyHash
   67     | RequireAllOf ![Script]
   68     | RequireAnyOf ![Script]
   69     | RequireMOf Word8 ![Script]
   70     deriving stock (Generic, Show, Eq)
   71 instance NFData Script
   72 
   73 -- | Validate 'Script'
   74 --
   75 -- @since 3.0.0
   76 validateScript :: Script -> Either ErrValidateScript ()
   77 validateScript = \case
   78     RequireSignatureOf (KeyHash bytes) -> do
   79         when (isNothing (keyHashFromBytes bytes)) $ Left WrongKeyHash
   80 
   81     RequireAllOf script -> do
   82         when (L.null script) $ Left EmptyList
   83         when (hasDuplicate script) $ Left DuplicateSignatures
   84         traverse_ validateScript script
   85 
   86     RequireAnyOf script -> do
   87         when (L.null script) $ Left EmptyList
   88         when (hasDuplicate script) $ Left DuplicateSignatures
   89         traverse_ validateScript script
   90 
   91     RequireMOf m script -> do
   92         when (m == 0) $ Left MZero
   93         when (length script < fromIntegral m) $ Left ListTooSmall
   94         when (hasDuplicate script) $ Left DuplicateSignatures
   95         traverse_ validateScript script
   96   where
   97     hasDuplicate xs = do
   98         length sigs /= length (L.nub sigs)
   99       where
  100         sigs = [ sig | RequireSignatureOf sig <- xs ]
  101 
  102 -- | Possible validation errors when validating a script
  103 data ErrValidateScript
  104     = EmptyList
  105     | ListTooSmall
  106     | MZero
  107     | DuplicateSignatures
  108     | WrongKeyHash
  109     deriving (Eq, Show)
  110 
  111 prettyErrValidateScript
  112     :: ErrValidateScript
  113     -> String
  114 prettyErrValidateScript = \case
  115     EmptyList ->
  116         "The list inside a script is empty."
  117     MZero ->
  118         "The M in at_least cannot be 0."
  119     ListTooSmall ->
  120         "The list inside at_least cannot be less than M."
  121     DuplicateSignatures ->
  122         "The list inside a script has duplicate keys."
  123     WrongKeyHash ->
  124         "The hash of verification key is expected to have "<>show hashSize<>" bytes."

  126 -- | This function realizes what cardano-node's `Api.serialiseToCBOR script` realizes
  127 -- This is basically doing the symbolically following:
  128 -- toCBOR [0,multisigScript]
  129 serialize :: Script -> ByteString
  130 serialize script =
  131     multisigTag <> CBOR.toStrictByteString (toCBOR script)
  132   where
  133     -- | Magic number representing the tag of the native multi-signature script
  134     -- language. For each script language included, a new tag is chosen.
  135     multisigTag :: ByteString
  136     multisigTag = "\00"
  137 
  138     toCBOR :: Script -> CBOR.Encoding
  139     toCBOR = \case
  140         RequireSignatureOf (KeyHash verKeyHash) ->
  141             encodeMultiscriptCtr 0 2 <> CBOR.encodeBytes verKeyHash
  142         RequireAllOf contents ->
  143             encodeMultiscriptCtr 1 2 <> encodeFoldable toCBOR contents
  144         RequireAnyOf contents ->
  145             encodeMultiscriptCtr 2 2 <> encodeFoldable toCBOR contents
  146         RequireMOf m contents -> mconcat
  147             [ encodeMultiscriptCtr 3 3
  148             , CBOR.encodeInt (fromInteger $ toInteger m)
  149             , encodeFoldable toCBOR contents
  150             ]
  151 
  152     encodeMultiscriptCtr :: Word -> Word -> CBOR.Encoding
  153     encodeMultiscriptCtr ctrIndex listLen =
  154         CBOR.encodeListLen listLen <> CBOR.encodeWord ctrIndex
  155 
  156     encodeFoldable :: (Foldable f) => (a -> CBOR.Encoding) -> f a -> CBOR.Encoding
  157     encodeFoldable encode xs = wrapArray len contents
  158       where
  159         (len, contents) = foldl' go (0, mempty) xs
  160         go (!l, !enc) next = (l + 1, enc <> encode next)
  161 
  162         wrapArray :: Word -> CBOR.Encoding -> CBOR.Encoding
  163         wrapArray len' contents'
  164             | len' <= 23 = CBOR.encodeListLen len' <> contents'
  165             | otherwise  = CBOR.encodeListLenIndef <> contents' <> CBOR.encodeBreak
  166 
  167 -- | Computes the hash of a given script, by first serializing it to CBOR.
  168 toScriptHash :: Script -> ScriptHash
  169 toScriptHash = ScriptHash . blake2b224 . serialize
  170 
  171 -- | A 'ScriptHash' type represents script hash. The hash is expected to have size of
  172 -- 28-byte.
  173 --
  174 -- @since 3.0.0
  175 newtype ScriptHash = ScriptHash ByteString
  176     deriving (Generic, Show, Eq)
  177 instance NFData ScriptHash
  178 
  179 -- | Construct an 'ScriptHash' from raw 'ByteString' (28 bytes).
  180 --
  181 -- @since 3.0.0
  182 scriptHashFromBytes :: ByteString -> Maybe ScriptHash
  183 scriptHashFromBytes bytes
  184     | BS.length bytes /= hashSize = Nothing
  185     | otherwise = Just $ ScriptHash bytes
  186 
  187 -- | A 'KeyHash' type represents verification key hash that participate in building
  188 -- multi-signature script. The hash is expected to have size of 28-byte.
  189 --
  190 -- @since 3.0.0
  191 newtype KeyHash = KeyHash ByteString
  192     deriving (Generic, Show, Eq)
  193 instance NFData KeyHash
  194 
  195 -- | Construct an 'KeyHash' from raw 'ByteString' (28 bytes).
  196 --
  197 -- @since 3.0.0
  198 keyHashFromBytes :: ByteString -> Maybe KeyHash
  199 keyHashFromBytes bytes
  200     | BS.length bytes /= hashSize = Nothing
  201     | otherwise = Just $ KeyHash bytes

  207 -- Hash a public key
  208 blake2b224 :: ByteString -> ByteString
  209 blake2b224 =
  210     BA.convert . hash @_ @Blake2b_224
  211 
  212 -- Size, in bytes, of a hash of public key (without the corresponding chain code)
  213 hashSize :: Int
  214 hashSize = hashDigestSize Blake2b_224
