   40 -- | Run 'scriptParser' on string input.
   41 scriptFromString :: String -> Maybe Script
   42 scriptFromString str =
   43     case readP_to_S scriptParser str of
   44          [(multisig, "")] -> Just multisig
   45          _ -> Nothing
   46 
   47 -- | The script embodies combination of signing keys that need to be met to make
   48 -- it valid. We assume here that the script could
   49 -- delivered from standard input. The examples below are self-explanatory:
   50 --
   51 -- 1. requiring signature
   52 -- 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe
   53 --
   54 -- 2. 'any' for signature required
   55 -- any [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   56 --
   57 -- 3. 'all' signatures required
   58 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   59 --
   60 -- 4. 'at_least' 1 signature required
   61 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
   62 --
   63 -- 5. Nested script are supported
   64 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]]
   65 --
   66 -- Parser is insensitive to whitespaces.
   67 --
   68 -- @since 3.0.0
   69 scriptParser :: ReadP Script
   70 scriptParser =
   71     requireAllOfParser <++
   72     requireAnyOfParser <++
   73     requireAtLeastOfParser <++
   74     requireSignatureOfParser
   75 
   76 requireSignatureOfParser :: ReadP Script
   77 requireSignatureOfParser = do
   78     P.skipSpaces
   79     verKeyStr <- P.munch1 (\c -> isDigit c || isLetter c || c == '_')
   80     case detectEncoding verKeyStr of
   81         Just EBase16 -> case fromBase16 (toBytes verKeyStr) of
   82             Left _ -> fail "Invalid Base16-encoded string."
   83             Right keyHash -> return $ toSignature keyHash
   84         Just EBech32{} -> case fromBech32 (T.pack verKeyStr) of
   85             Nothing -> fail "Invalid Bech32-encoded string."
   86             Just keyHash -> return $ toSignature keyHash
   87         Just EBase58 -> case fromBase58 (toBytes verKeyStr) of
   88             Left err -> fail err
   89             Right keyHash -> return $ toSignature keyHash
   90         Nothing ->
   91             fail "Verification key hash must be must be encoded as \
   92                    \base16, bech32 or base58."
   93  where
   94     toBytes = T.encodeUtf8 . T.pack
   95     fromBech32 txt = do
   96         (_, dp) <- either (const Nothing) Just (Bech32.decodeLenient txt)
   97         Bech32.dataPartToBytes dp
   98     toSignature = RequireSignatureOf . KeyHash
   99 
  100 requireAllOfParser :: ReadP Script
  101 requireAllOfParser = do
  102     P.skipSpaces
  103     _identifier <- P.string "all"
  104     RequireAllOf <$> commonPart
  105 
  106 requireAnyOfParser :: ReadP Script
  107 requireAnyOfParser = do
  108     P.skipSpaces
  109     _identifier <- P.string "any"
  110     RequireAnyOf <$> commonPart
  111 
  112 requireAtLeastOfParser :: ReadP Script
  113 requireAtLeastOfParser = do
  114     P.skipSpaces
  115     _identifier <- P.string "at_least"
  116     RequireSomeOf <$> naturalParser <*> commonPart
  117 
  118 naturalParser :: ReadP Word8
  119 naturalParser = do
  120     P.skipSpaces
  121     fromInteger . read <$> P.munch1 isDigit
  122 
  123 commonPart :: ReadP [Script]
  124 commonPart = do
  125     P.skipSpaces
  126     _open <- P.string "["
  127     P.skipSpaces
  128     content <- P.sepBy scriptParser (P.string ",")
  129     P.skipSpaces
  130     _close <- P.string "]"
  131     P.skipSpaces
  132     return content
