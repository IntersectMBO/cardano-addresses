  164 -- This module provides an implementation of:
  165 --
  166 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Shelley master keys from mnemonic sentences
  167 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  168 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  169 --
  170 -- - 'paymentAddress': for constructing payment addresses from a address public key or a script
  171 -- - 'delegationAddress': for constructing delegation addresses from payment credential (public key or script) and stake credential (public key or script)
  172 -- - 'pointerAddress': for constructing delegation addresses from payment credential (public key or script) and chain pointer
  173 -- - 'stakeAddress': for constructing reward accounts from stake credential (public key or script)

  175 -- | A cryptographic key for sequential-scheme address derivation, with
  176 -- phantom-types to disambiguate key types.
  177 --
  178 -- @
  179 -- let rootPrivateKey = Shelley 'RootK XPrv
  180 -- let accountPubKey  = Shelley 'AccountK XPub
  181 -- let addressPubKey  = Shelley 'AddressK XPub
  182 -- @
  183 --
  184 -- @since 2.0.0
  185 newtype Shelley (depth :: Depth) key = Shelley
  186     { getKey :: key
  187         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  188         --
  189         -- @since 1.0.0
  190     }
  191     deriving stock (Generic, Show, Eq)
  192 
  193 deriving instance (Functor (Shelley depth))
  194 instance (NFData key) => NFData (Shelley depth key)

  196 -- | Describe what the keys within an account are used for.
  197 --
  198 -- - UTxOExternal: used for public addresses sent to other parties for receiving money.
  199 -- - UTxOInternal: generated by wallet software to send change back to the wallet.
  200 -- - Stake: used for stake key(s) and delegation.
  201 -- - Multisig: used for keys used in multi-party transactions / scripts.
  202 --
  203 -- @since 3.0.0
  204 data Role
  205     = UTxOExternal
  206     | UTxOInternal
  207     | Stake
  208     | Multisig
  209     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  210 
  211 instance NFData Role
  212 
  213 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  214 -- to assign the wrong index to the corresponding constructor (by swapping
  215 -- around the constructor above for instance).
  216 instance Enum Role where
  217     toEnum = \case
  218         0 -> UTxOExternal
  219         1 -> UTxOInternal
  220         2 -> Stake
  221         3 -> Multisig
  222         _ -> error "Role.toEnum: bad argument"
  223     fromEnum = \case
  224         UTxOExternal -> 0
  225         UTxOInternal -> 1
  226         Stake -> 2
  227         Multisig -> 3

  230 -- Key Derivation
  231 --
  232 -- $keyDerivation
  233 --
  234 -- === Generating a root key from 'SomeMnemonic'
  235 -- > :set -XOverloadedStrings
  236 -- > :set -XTypeApplications
  237 -- > :set -XDataKinds
  238 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  239 -- >
  240 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  241 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  242 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shelley 'RootK XPrv
  243 --
  244 -- === Deriving child keys
  245 --
  246 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  247 --
  248 -- > let accIx = toEnum 0x80000000
  249 -- > let acctK = deriveAccountPrivateKey rootK accIx
  250 -- >
  251 -- > let addIx = toEnum 0x00000014
  252 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  253 --
  254 -- > let stakeK = deriveStakingPrivateKey acctK
  255 
  256 instance Internal.GenMasterKey Shelley where
  257     type SecondFactor Shelley = ScrubbedBytes
  258 
  259     genMasterKeyFromXPrv = liftXPrv
  260     genMasterKeyFromMnemonic fstFactor sndFactor =
  261         Shelley $ generateNew seedValidated sndFactor
  262         where
  263             seed  = someMnemonicToBytes fstFactor
  264             seedValidated = assert
  265                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  266                 seed

  268 instance Internal.HardDerivation Shelley where
  269     type AccountIndexDerivationType Shelley = 'Hardened
  270     type AddressIndexDerivationType Shelley = 'Soft
  271     type WithRole Shelley = Role
  272 
  273     deriveAccountPrivateKey (Shelley rootXPrv) accIx =
  274         let
  275             purposeIx =
  276                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  277             coinTypeIx =
  278                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  279             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  280                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  281             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  282                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  283             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  284                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  285         in
  286             Shelley acctXPrv
  287 
  288     deriveAddressPrivateKey (Shelley accXPrv) role addrIx =
  289         let
  290             roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  291             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  292                 deriveXPrv DerivationScheme2 accXPrv roleCode
  293             addrXPrv = -- lvl5 derivation; soft derivation of address index
  294                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  295         in
  296             Shelley addrXPrv
  297 
  298 instance Internal.SoftDerivation Shelley where
  299     deriveAddressPublicKey (Shelley accXPub) role addrIx =
  300         fromMaybe errWrongIndex $ do
  301             let roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  302             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  303                 deriveXPub DerivationScheme2 accXPub roleCode
  304             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  305                 deriveXPub DerivationScheme2 changeXPub addrIx
  306             return $ Shelley addrXPub
  307       where
  308         errWrongIndex = error $
  309             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  310             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  311             \either a programmer error, or, we may have reached the maximum \
  312             \number of addresses for a given wallet."

  372 -- Re-export from 'Cardano.Address.Derivation' to have it documented specialized in Haddock
  373 --
  374 -- | Derive a staking key for a corresponding 'AccountK'. Note that wallet
  375 -- software are by convention only using one staking key per account, and always
  376 -- the first account (with index 0').
  377 --
  378 -- Deriving staking keys for something else than the initial account is not
  379 -- recommended and can lead to incompatibility with existing wallet softwares
  380 -- (Daedalus, Yoroi, Adalite...).
  381 --
  382 -- @since 2.0.0
  383 deriveStakingPrivateKey
  384     :: Shelley 'AccountK XPrv
  385     -> Shelley 'StakingK XPrv
  386 deriveStakingPrivateKey accXPrv =
  387     let (Shelley stakeXPrv) =
  388             deriveAddressPrivateKey accXPrv Stake (minBound @(Index 'Soft _))
  389     in Shelley stakeXPrv

  468 -- | Analyze an 'Address' to know whether it's a Shelley address or not.
  469 --
  470 -- Throws 'AddrError' if it's not a valid Shelley address, or a ready-to-print
  471 -- string giving details about the 'Address'.
  472 --
  473 -- @since 2.0.0
  474 inspectShelleyAddress
  475     :: (Alternative m, MonadThrow m)
  476     => Maybe XPub
  477     -> Address
  478     -> m Json.Value
  479 inspectShelleyAddress mRootPub addr
  480     | BS.length bytes < 1 + hashSize = throwM (ShWrongInputSize (BS.length bytes))
  481     | otherwise =
  482         let
  483             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  484             addrType = fstByte .&. 0b11110000
  485             network  = fstByte .&. 0b00001111
  486         in
  487             case addrType of
  488                -- 0000: base address: keyhash28,keyhash28
  489                 0b00000000 | BS.length rest == 2 * hashSize ->
  490                     pure $ Json.object
  491                         [ "address_style"     .= Json.String "Shelley"
  492                         , "stake_reference"   .= Json.String "by value"
  493                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  494                         , "stake_key_hash"    .= base16 (BS.drop hashSize rest)
  495                         , "network_tag"       .= network
  496                         ]
  497                -- 0001: base address: scripthash32,keyhash28
  498                 0b00010000 | BS.length rest == hashSize + hashSize ->
  499                     pure $ Json.object
  500                         [ "address_style"     .= Json.String "Shelley"
  501                         , "stake_reference"   .= Json.String "by value"
  502                         , "script_hash"       .= base16 (BS.take hashSize rest)
  503                         , "stake_key_hash"    .= base16 (BS.drop hashSize rest)
  504                         , "network_tag"       .= network
  505                         ]
  506                -- 0010: base address: keyhash28,scripthash32
  507                 0b00100000 | BS.length rest == hashSize + hashSize ->
  508                     pure $ Json.object
  509                         [ "address_style"     .= Json.String "Shelley"
  510                         , "stake_reference"   .= Json.String "by value"
  511                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  512                         , "stake_script_hash" .= base16 (BS.drop hashSize rest)
  513                         , "network_tag"       .= network
  514                         ]
  515                -- 0011: base address: scripthash32,scripthash32
  516                 0b00110000 | BS.length rest == 2 * hashSize ->
  517                     pure $ Json.object
  518                         [ "address_style"     .= Json.String "Shelley"
  519                         , "stake_reference"   .= Json.String "by value"
  520                         , "script_hash"       .= base16 (BS.take hashSize rest)
  521                         , "stake_script_hash" .= base16 (BS.drop hashSize rest)
  522                         , "network_tag"       .= network
  523                         ]
  524                -- 0100: pointer address: keyhash28, 3 variable length uint
  525                -- TODO Could fo something better for pointer and try decoding
  526                --      the pointer
  527                 0b01000000 | BS.length rest > hashSize -> do
  528                     ptr <- getPtr (BS.drop hashSize rest)
  529                     pure $ Json.object
  530                         [ "address_style"     .= Json.String "Shelley"
  531                         , "stake_reference"   .= Json.String "by pointer"
  532                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  533                         , "pointer"           .= ptrToJSON ptr
  534                         , "network_tag"       .= network
  535                         ]
  536                -- 0101: pointer address: scripthash32, 3 variable length uint
  537                -- TODO Could fo something better for pointer and try decoding
  538                --      the pointer
  539                 0b01010000 | BS.length rest > hashSize -> do
  540                     ptr <- getPtr (BS.drop hashSize rest)
  541                     pure $ Json.object
  542                         [ "address_style"     .= Json.String "Shelley"
  543                         , "stake_reference"   .= Json.String "by pointer"
  544                         , "script_hash"       .= base16 (BS.take hashSize rest)
  545                         , "pointer"           .= ptrToJSON ptr
  546                         , "network_tag"       .= network
  547                         ]
  548                -- 0110: enterprise address: keyhash28
  549                 0b01100000 | BS.length rest == hashSize ->
  550                     pure $ Json.object
  551                         [ "address_style"     .= Json.String "Shelley"
  552                         , "stake_reference"   .= Json.String "none"
  553                         , "spending_key_hash" .= base16 (BS.take hashSize rest)
  554                         , "network_tag"       .= network
  555                         ]
  556                -- 0111: enterprise address: scripthash32
  557                 0b01110000 | BS.length rest == hashSize ->
  558                     pure $ Json.object
  559                         [ "address_style"     .= Json.String "Shelley"
  560                         , "stake_reference"   .= Json.String "none"
  561                         , "script_hash"       .= base16 (BS.take hashSize rest)
  562                         , "network_tag"       .= network
  563                         ]
  564                -- 1000: byron address
  565                 0b10000000 ->
  566                     inspectIcarusAddress addr <|> inspectByronAddress mRootPub addr
  567                -- 1110: reward account: keyhash28
  568                 0b11100000 | BS.length rest == hashSize ->
  569                     pure $ Json.object
  570                         [ "address_style"     .= Json.String "Shelley"
  571                         , "stake_reference"   .= Json.String "by value"
  572                         , "stake_key_hash"    .= base16 (BS.take hashSize rest)
  573                         , "network_tag"       .= network
  574                         ]
  575                -- 1111: reward account: scripthash32
  576                 0b11110000 | BS.length rest == hashSize ->
  577                     pure $ Json.object
  578                         [ "address_style"     .= Json.String "Shelley"
  579                         , "stake_reference"   .= Json.String "by value"
  580                         , "script_hash"       .= base16 (BS.take hashSize rest)
  581                         , "network_tag"       .= network
  582                         ]
  583                 _ -> throwM ShUnknownAddrType
  584   where
  585     bytes  = unAddress addr
  586     base16 = T.unpack . T.decodeUtf8 . encode EBase16

  710 -- | Extend an existing payment 'Address' to make it a delegation address.
  711 --
  712 -- @since 2.0.0
  713 extendAddress
  714     :: Address
  715     -> Credential 'StakingK
  716     -> Either ErrExtendAddress Address
  717 extendAddress addr stakeReference = do
  718     when (isNothing (inspectShelleyAddress Nothing addr)) $
  719         Left $ ErrInvalidAddressStyle "Given address isn't a Shelley address"
  720 
  721     let bytes = unAddress addr
  722     let (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  723 
  724     let paymentFirstByte = fstByte .&. 0b11110000
  725     let extendableTypes = addressType <$>
  726             [ EnterpriseAddress CredentialFromKey
  727             , EnterpriseAddress CredentialFromScript
  728             ]
  729     unless (paymentFirstByte `elem` extendableTypes) $ do
  730         Left $ ErrInvalidAddressType "Only payment addresses can be extended"
  731 
  732     case stakeReference of
  733         -- base address: keyhash28,keyhash28    : 00000000 -> 0
  734         -- base address: scripthash32,keyhash28 : 00010000 -> 16
  735         StakingFromKey stakingKey -> do
  736             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  737                 -- 0b01100000 .&. 0b00011111 = 0
  738                 -- 0b01110000 .&. 0b00011111 = 16
  739                 putWord8 $ fstByte .&. 0b00011111
  740                 putByteString rest
  741                 putByteString . blake2b224 $ stakingKey
  742 
  743         -- base address: keyhash28,scripthash32    : 00100000 -> 32
  744         -- base address: scripthash32,scripthash32 : 00110000 -> 48
  745         StakingFromScript (ScriptHash scriptBytes) -> do
  746             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  747                 -- 0b01100000 .&. 0b00111111 = 32
  748                 -- 0b01110000 .&. 0b00111111 = 48
  749                 putWord8 $ fstByte .&. 0b00111111
  750                 putByteString rest
  751                 putByteString scriptBytes
  752 
  753         -- pointer address: keyhash28, 3 variable length uint    : 01000000 -> 64
  754         -- pointer address: scripthash32, 3 variable length uint : 01010000 -> 80
  755         StakingFromPointer pointer -> do
  756             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  757                 -- 0b01100000 .&. 0b01011111 = 64
  758                 -- 0b01110000 .&. 0b01011111 = 80
  759                 putWord8 $ fstByte .&. 0b01011111
  760                 putByteString rest
  761                 putPointer pointer
  762   where
  763     putPointer (ChainPointer a b c) = do
  764         putVariableLengthNat a
  765         putVariableLengthNat b
  766         putVariableLengthNat c

  931 addressType :: AddressType -> Word8
  932 addressType = \case
  933     ByronAddress                                                -> 0b10000000
  934     BaseAddress       CredentialFromKey    CredentialFromKey    -> 0b00000000
  935     BaseAddress       CredentialFromScript CredentialFromKey    -> 0b00010000
  936     BaseAddress       CredentialFromKey    CredentialFromScript -> 0b00100000
  937     BaseAddress       CredentialFromScript CredentialFromScript -> 0b00110000
  938     PointerAddress    CredentialFromKey                         -> 0b01000000
  939     PointerAddress    CredentialFromScript                      -> 0b01010000
  940     EnterpriseAddress CredentialFromKey                         -> 0b01100000
  941     EnterpriseAddress CredentialFromScript                      -> 0b01110000
  942     RewardAccount                          CredentialFromKey    -> 0b11100000
  943     RewardAccount                          CredentialFromScript -> 0b11110000

  945 -- Helper to constructs appropriate address headers. Rest of the payload is left
  946 -- to the caller as a raw 'ByteString'.
  947 constructPayload
  948     :: AddressType
  949     -> NetworkDiscriminant Shelley
  950     -> ByteString
  951     -> Address
  952 constructPayload addrType discrimination bytes = unsafeMkAddress $
  953     invariantSize expectedLength $ BL.toStrict $ runPut $ do
  954         putWord8 firstByte
  955         putByteString bytes
  956   where
  957     firstByte =
  958         let netTagLimit = 16
  959         in addressType addrType + invariantNetworkTag netTagLimit (networkTag @Shelley discrimination)
  960     expectedLength =
  961         let headerSizeBytes = 1
  962         in headerSizeBytes + hashSize
