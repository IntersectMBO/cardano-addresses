  160 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Shelley master keys from mnemonic sentences

  165 -- - 'delegationAddress': for constructing delegation addresses from payment credential (public key or script) and stake credential (public key or script)

  167 -- - 'stakeAddress': for constructing reward accounts from stake credential (public key or script)

  169 -- | A cryptographic key for sequential-scheme address derivation, with
  170 -- phantom-types to disambiguate key types.
  171 --
  172 -- @
  173 -- let rootPrivateKey = Shelley 'RootK XPrv
  174 -- let accountPubKey  = Shelley 'AccountK XPub
  175 -- let addressPubKey  = Shelley 'PaymentK XPub
  176 -- @
  177 --
  178 -- @since 2.0.0
  179 newtype Shelley (depth :: Depth) key = Shelley
  180     { getKey :: key
  181         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  182         --
  183         -- @since 1.0.0
  184     }
  185     deriving stock (Generic, Show, Eq)
  186 
  187 deriving instance (Functor (Shelley depth))
  188 instance (NFData key) => NFData (Shelley depth key)

  190 -- | Describe what the keys within an account are used for.
  191 --
  192 -- - UTxOExternal: used for public addresses sent to other parties for receiving money.
  193 -- - UTxOInternal: generated by wallet software to send change back to the wallet.
  194 -- - Stake: used for stake key(s) and delegation.
  195 -- - Multisig: used for keys used in multi-party transactions / scripts.
  196 --
  197 -- @since 3.0.0
  198 data Role
  199     = UTxOExternal
  200     | UTxOInternal
  201     | Stake
  202     | Multisig
  203     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  204 
  205 instance NFData Role
  206 
  207 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  208 -- to assign the wrong index to the corresponding constructor (by swapping
  209 -- around the constructor above for instance).
  210 instance Enum Role where
  211     toEnum = \case
  212         0 -> UTxOExternal
  213         1 -> UTxOInternal
  214         2 -> Stake
  215         3 -> Multisig
  216         _ -> error "Role.toEnum: bad argument"
  217     fromEnum = \case
  218         UTxOExternal -> 0
  219         UTxOInternal -> 1
  220         Stake -> 2
  221         Multisig -> 3

  224 -- Key Derivation
  225 --
  226 -- $keyDerivation
  227 --
  228 -- === Generating a root key from 'SomeMnemonic'
  229 -- > :set -XOverloadedStrings
  230 -- > :set -XTypeApplications
  231 -- > :set -XDataKinds
  232 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  233 -- >
  234 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  235 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  236 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shelley 'RootK XPrv
  237 --
  238 -- === Deriving child keys
  239 --
  240 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  241 --
  242 -- > let accIx = toEnum 0x80000000
  243 -- > let acctK = deriveAccountPrivateKey rootK accIx
  244 -- >
  245 -- > let addIx = toEnum 0x00000014
  246 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  247 --
  248 -- > let stakeK = deriveDelegationPrivateKey acctK
  249 
  250 instance Internal.GenMasterKey Shelley where
  251     type SecondFactor Shelley = ScrubbedBytes
  252 
  253     genMasterKeyFromXPrv = liftXPrv
  254     genMasterKeyFromMnemonic fstFactor sndFactor =
  255         Shelley $ generateNew seedValidated sndFactor
  256         where
  257             seed  = someMnemonicToBytes fstFactor
  258             seedValidated = assert
  259                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  260                 seed

  262 instance Internal.HardDerivation Shelley where
  263     type AccountIndexDerivationType Shelley = 'Hardened
  264     type AddressIndexDerivationType Shelley = 'Soft
  265     type WithRole Shelley = Role
  266 
  267     deriveAccountPrivateKey (Shelley rootXPrv) accIx =
  268         let
  269             purposeIx =
  270                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  271             coinTypeIx =
  272                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  273             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  274                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  275             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  276                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  277             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  278                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  279         in
  280             Shelley acctXPrv
  281 
  282     deriveAddressPrivateKey (Shelley accXPrv) role addrIx =
  283         let
  284             roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  285             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  286                 deriveXPrv DerivationScheme2 accXPrv roleCode
  287             addrXPrv = -- lvl5 derivation; soft derivation of address index
  288                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  289         in
  290             Shelley addrXPrv
  291 
  292 instance Internal.SoftDerivation Shelley where
  293     deriveAddressPublicKey (Shelley accXPub) role addrIx =
  294         fromMaybe errWrongIndex $ do
  295             let roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  296             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  297                 deriveXPub DerivationScheme2 accXPub roleCode
  298             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  299                 deriveXPub DerivationScheme2 changeXPub addrIx
  300             return $ Shelley addrXPub
  301       where
  302         errWrongIndex = error $
  303             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  304             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  305             \either a programmer error, or, we may have reached the maximum \
  306             \number of addresses for a given wallet."

  366 -- Re-export from 'Cardano.Address.Derivation' to have it documented specialized in Haddock
  367 --
  368 -- | Derive a delegation key for a corresponding 'AccountK'. Note that wallet
  369 -- software are by convention only using one delegation key per account, and always
  370 -- the first account (with index 0').
  371 --
  372 -- Deriving delegation keys for something else than the initial account is not
  373 -- recommended and can lead to incompatibility with existing wallet softwares
  374 -- (Daedalus, Yoroi, Adalite...).
  375 --
  376 -- @since 2.0.0
  377 deriveDelegationPrivateKey
  378     :: Shelley 'AccountK XPrv
  379     -> Shelley 'DelegationK XPrv
  380 deriveDelegationPrivateKey accXPrv =
  381     let (Shelley stakeXPrv) =
  382             deriveAddressPrivateKey accXPrv Stake (minBound @(Index 'Soft _))
  383     in Shelley stakeXPrv

  462 -- | Analyze an 'Address' to know whether it's a Shelley address or not.
  463 --
  464 -- Throws 'AddrError' if it's not a valid Shelley address, or a ready-to-print
  465 -- string giving details about the 'Address'.
  466 --
  467 -- @since 2.0.0
  468 inspectShelleyAddress
  469     :: (Alternative m, MonadThrow m)
  470     => Maybe XPub
  471     -> Address
  472     -> m Json.Value
  473 inspectShelleyAddress mRootPub addr
  474     | BS.length bytes < 1 + credentialHashSize = throwM (ShWrongInputSize (BS.length bytes))
  475     | otherwise =
  476         let
  477             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  478             addrType = fstByte .&. 0b11110000
  479             network  = fstByte .&. 0b00001111
  480         in
  481             case addrType of
  482                -- 0000: base address: keyhash28,keyhash28
  483                 0b00000000 | BS.length rest == 2 * credentialHashSize ->
  484                     pure $ Json.object
  485                         [ "address_style"     .= Json.String "Shelley"
  486                         , "stake_reference"   .= Json.String "by value"
  487                         , "spending_key_hash" .= base16 (BS.take credentialHashSize rest)
  488                         , "stake_key_hash"    .= base16 (BS.drop credentialHashSize rest)
  489                         , "network_tag"       .= network
  490                         ]
  491                -- 0001: base address: scripthash32,keyhash28
  492                 0b00010000 | BS.length rest == credentialHashSize + credentialHashSize ->
  493                     pure $ Json.object
  494                         [ "address_style"     .= Json.String "Shelley"
  495                         , "stake_reference"   .= Json.String "by value"
  496                         , "script_hash"       .= base16 (BS.take credentialHashSize rest)
  497                         , "stake_key_hash"    .= base16 (BS.drop credentialHashSize rest)
  498                         , "network_tag"       .= network
  499                         ]
  500                -- 0010: base address: keyhash28,scripthash32
  501                 0b00100000 | BS.length rest == credentialHashSize + credentialHashSize ->
  502                     pure $ Json.object
  503                         [ "address_style"     .= Json.String "Shelley"
  504                         , "stake_reference"   .= Json.String "by value"
  505                         , "spending_key_hash" .= base16 (BS.take credentialHashSize rest)
  506                         , "stake_script_hash" .= base16 (BS.drop credentialHashSize rest)
  507                         , "network_tag"       .= network
  508                         ]
  509                -- 0011: base address: scripthash32,scripthash32
  510                 0b00110000 | BS.length rest == 2 * credentialHashSize ->
  511                     pure $ Json.object
  512                         [ "address_style"     .= Json.String "Shelley"
  513                         , "stake_reference"   .= Json.String "by value"
  514                         , "script_hash"       .= base16 (BS.take credentialHashSize rest)
  515                         , "stake_script_hash" .= base16 (BS.drop credentialHashSize rest)
  516                         , "network_tag"       .= network
  517                         ]
  518                -- 0100: pointer address: keyhash28, 3 variable length uint
  519                -- TODO Could fo something better for pointer and try decoding
  520                --      the pointer
  521                 0b01000000 | BS.length rest > credentialHashSize -> do
  522                     ptr <- getPtr (BS.drop credentialHashSize rest)
  523                     pure $ Json.object
  524                         [ "address_style"     .= Json.String "Shelley"
  525                         , "stake_reference"   .= Json.String "by pointer"
  526                         , "spending_key_hash" .= base16 (BS.take credentialHashSize rest)
  527                         , "pointer"           .= ptrToJSON ptr
  528                         , "network_tag"       .= network
  529                         ]
  530                -- 0101: pointer address: scripthash32, 3 variable length uint
  531                -- TODO Could fo something better for pointer and try decoding
  532                --      the pointer
  533                 0b01010000 | BS.length rest > credentialHashSize -> do
  534                     ptr <- getPtr (BS.drop credentialHashSize rest)
  535                     pure $ Json.object
  536                         [ "address_style"     .= Json.String "Shelley"
  537                         , "stake_reference"   .= Json.String "by pointer"
  538                         , "script_hash"       .= base16 (BS.take credentialHashSize rest)
  539                         , "pointer"           .= ptrToJSON ptr
  540                         , "network_tag"       .= network
  541                         ]
  542                -- 0110: enterprise address: keyhash28
  543                 0b01100000 | BS.length rest == credentialHashSize ->
  544                     pure $ Json.object
  545                         [ "address_style"     .= Json.String "Shelley"
  546                         , "stake_reference"   .= Json.String "none"
  547                         , "spending_key_hash" .= base16 (BS.take credentialHashSize rest)
  548                         , "network_tag"       .= network
  549                         ]
  550                -- 0111: enterprise address: scripthash32
  551                 0b01110000 | BS.length rest == credentialHashSize ->
  552                     pure $ Json.object
  553                         [ "address_style"     .= Json.String "Shelley"
  554                         , "stake_reference"   .= Json.String "none"
  555                         , "script_hash"       .= base16 (BS.take credentialHashSize rest)
  556                         , "network_tag"       .= network
  557                         ]
  558                -- 1000: byron address
  559                 0b10000000 ->
  560                     inspectIcarusAddress addr <|> inspectByronAddress mRootPub addr
  561                -- 1110: reward account: keyhash28
  562                 0b11100000 | BS.length rest == credentialHashSize ->
  563                     pure $ Json.object
  564                         [ "address_style"     .= Json.String "Shelley"
  565                         , "stake_reference"   .= Json.String "by value"
  566                         , "stake_key_hash"    .= base16 (BS.take credentialHashSize rest)
  567                         , "network_tag"       .= network
  568                         ]
  569                -- 1111: reward account: scripthash32
  570                 0b11110000 | BS.length rest == credentialHashSize ->
  571                     pure $ Json.object
  572                         [ "address_style"     .= Json.String "Shelley"
  573                         , "stake_reference"   .= Json.String "by value"
  574                         , "script_hash"       .= base16 (BS.take credentialHashSize rest)
  575                         , "network_tag"       .= network
  576                         ]
  577                 _ -> throwM ShUnknownAddrType
  578   where
  579     bytes  = unAddress addr
  580     base16 = T.unpack . T.decodeUtf8 . encode EBase16

  702 -- | Extend an existing payment 'Address' to make it a delegation address.
  703 --
  704 -- @since 2.0.0
  705 extendAddress
  706     :: Address
  707     -> Credential 'DelegationK
  708     -> Either ErrExtendAddress Address
  709 extendAddress addr stakeReference = do
  710     when (isNothing (inspectShelleyAddress Nothing addr)) $
  711         Left $ ErrInvalidAddressStyle "Given address isn't a Shelley address"
  712 
  713     let bytes = unAddress addr
  714     let (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  715 
  716     let paymentFirstByte = fstByte .&. 0b11110000
  717     let extendableTypes = addressType <$>
  718             [ EnterpriseAddress CredentialFromKey
  719             , EnterpriseAddress CredentialFromScript
  720             ]
  721     unless (paymentFirstByte `elem` extendableTypes) $ do
  722         Left $ ErrInvalidAddressType "Only payment addresses can be extended"
  723 
  724     case stakeReference of
  725         -- base address: keyhash28,keyhash28    : 00000000 -> 0
  726         -- base address: scripthash32,keyhash28 : 00010000 -> 16
  727         DelegationFromKey delegationKey -> do
  728             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  729                 -- 0b01100000 .&. 0b00011111 = 0
  730                 -- 0b01110000 .&. 0b00011111 = 16
  731                 putWord8 $ fstByte .&. 0b00011111
  732                 putByteString rest
  733                 putByteString . hashCredential . xpubPublicKey . getKey $ delegationKey
  734 
  735         -- base address: keyhash28,scripthash32    : 00100000 -> 32
  736         -- base address: scripthash32,scripthash32 : 00110000 -> 48
  737         DelegationFromScript (ScriptHash scriptBytes) -> do
  738             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  739                 -- 0b01100000 .&. 0b00111111 = 32
  740                 -- 0b01110000 .&. 0b00111111 = 48
  741                 putWord8 $ fstByte .&. 0b00111111
  742                 putByteString rest
  743                 putByteString scriptBytes
  744 
  745         -- pointer address: keyhash28, 3 variable length uint    : 01000000 -> 64
  746         -- pointer address: scripthash32, 3 variable length uint : 01010000 -> 80
  747         DelegationFromPointer pointer -> do
  748             pure $ unsafeMkAddress $ BL.toStrict $ runPut $ do
  749                 -- 0b01100000 .&. 0b01011111 = 64
  750                 -- 0b01110000 .&. 0b01011111 = 80
  751                 putWord8 $ fstByte .&. 0b01011111
  752                 putByteString rest
  753                 putPointer pointer
  754   where
  755     putPointer (ChainPointer a b c) = do
  756         putVariableLengthNat a
  757         putVariableLengthNat b
  758         putVariableLengthNat c

  906     = BaseAddress CredentialType CredentialType

  913 addressType :: AddressType -> Word8
  914 addressType = \case
  915     ByronAddress                                                -> 0b10000000
  916     BaseAddress       CredentialFromKey    CredentialFromKey    -> 0b00000000
  917     BaseAddress       CredentialFromScript CredentialFromKey    -> 0b00010000
  918     BaseAddress       CredentialFromKey    CredentialFromScript -> 0b00100000
  919     BaseAddress       CredentialFromScript CredentialFromScript -> 0b00110000
  920     PointerAddress    CredentialFromKey                         -> 0b01000000
  921     PointerAddress    CredentialFromScript                      -> 0b01010000
  922     EnterpriseAddress CredentialFromKey                         -> 0b01100000
  923     EnterpriseAddress CredentialFromScript                      -> 0b01110000
  924     RewardAccount                          CredentialFromKey    -> 0b11100000
  925     RewardAccount                          CredentialFromScript -> 0b11110000

  927 -- Helper to constructs appropriate address headers. Rest of the payload is left
  928 -- to the caller as a raw 'ByteString'.
  929 constructPayload
  930     :: AddressType
  931     -> NetworkDiscriminant Shelley
  932     -> ByteString
  933     -> Address
  934 constructPayload addrType discrimination bytes = unsafeMkAddress $
  935     invariantSize expectedLength $ BL.toStrict $ runPut $ do
  936         putWord8 firstByte
  937         putByteString bytes
  938   where
  939     firstByte =
  940         let netTagLimit = 16
  941         in addressType addrType + invariantNetworkTag netTagLimit (networkTag @Shelley discrimination)
  942     expectedLength =
  943         let headerSizeBytes = 1
  944         in headerSizeBytes + credentialHashSize
