  106 -- | An opaque type representing an extended private key.
  107 --
  108 -- __Properties:__
  109 --
  110 -- ===== Roundtripping
  111 --
  112 -- @forall xprv. 'xprvFromBytes' ('xprvToBytes' xprv) == 'Just' xprv@
  113 --
  114 -- ===== Chain Code Invariance
  115 --
  116 -- @forall xprv. 'xprvChainCode' xprv == 'xpubChainCode' ('toXPub' xprv)@
  117 --
  118 -- ===== Public Key Signature
  119 --
  120 -- @forall xprv msg. 'verify' ('toXPub' xprv) msg ('sign' xprv msg) == 'True'@
  121 --
  122 -- @since 1.0.0
  123 type XPrv = CC.XPrv
  124 
  125 -- | An opaque type representing an extended public key.
  126 --
  127 -- __Properties:__
  128 --
  129 -- ===== Roundtripping
  130 --
  131 -- @forall xpub. 'xpubFromBytes' ('xpubToBytes' xpub) == 'Just' xpub@
  132 --
  133 -- @since 1.0.0

  141 -- | Construct an 'XPub' from raw 'ByteString' (64 bytes).
  142 --
  143 -- @since 1.0.0
  144 xpubFromBytes :: ByteString -> Maybe XPub
  145 xpubFromBytes = eitherToMaybe . CC.xpub
  146 
  147 -- | Convert an 'XPub' to a raw 'ByteString' (64 bytes).
  148 --
  149 -- @since 1.0.0
  150 xpubToBytes :: XPub -> ByteString
  151 xpubToBytes xpub = xpubPublicKey xpub <> xpubChainCode xpub
  152 
  153 -- | Extract the public key from an 'XPub' as a raw 'ByteString' (32 bytes).
  154 --
  155 -- @since 2.0.0
  156 xpubPublicKey :: XPub -> ByteString
  157 xpubPublicKey (CC.XPub pub _cc) = pub
  158 
  159 -- | Extract the chain code from an 'XPub' as a raw 'ByteString' (32 bytes).
  160 --
  161 -- @since 2.0.0
  162 xpubChainCode :: XPub -> ByteString
  163 xpubChainCode (CC.XPub _pub (CC.ChainCode cc)) = cc
  164 
  165 -- | Construct an 'XPrv' from raw 'ByteString' (96 bytes).
  166 --
  167 -- @since 1.0.0
  168 xprvFromBytes :: ByteString -> Maybe XPrv
  169 xprvFromBytes bytes
  170     | BS.length bytes /= 96 = Nothing
  171     | otherwise = do
  172         let (prv, cc) = BS.splitAt 64 bytes
  173         pub <- ed25519ScalarMult (BS.take 32 prv)
  174         eitherToMaybe $ CC.xprv $ prv <> pub <> cc
  175   where
  176     ed25519ScalarMult :: ByteString -> Maybe ByteString
  177     ed25519ScalarMult bs = do
  178         scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
  179         pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar
  180 
  181 -- From  < xprv | pub | cc >
  182 -- â†³ To  < xprv |     | cc >
  183 --
  184 -- | Convert an 'XPrv' to a raw 'ByteString' (96 bytes).
  185 --
  186 -- @since 1.0.0
  187 xprvToBytes :: XPrv -> ByteString
  188 xprvToBytes xprv =
  189     xprvPrivateKey xprv <> xprvChainCode xprv
  190 
  191 -- | Extract the private key from an 'XPrv' as a raw 'ByteString' (64 bytes).
  192 --
  193 -- @since 2.0.0
  194 xprvPrivateKey :: XPrv -> ByteString
  195 xprvPrivateKey = BS.take 64 . CC.unXPrv
  196 
  197 -- | Extract the chain code from an 'XPrv' as a raw 'ByteString' (32 bytes).
  198 --
  199 -- @since 2.0.0
  200 xprvChainCode :: XPrv -> ByteString
  201 xprvChainCode = BS.drop 96 . CC.unXPrv

  297 -- | Key Depth in the derivation path, according to BIP-0039 / BIP-0044
  298 --
  299 -- @
  300 -- root | purpose' | cointype' | account' | role | address@
  301 -- 0th      1st         2nd        3rd       4th     5th
  302 -- @
  303 --
  304 -- We do not manipulate purpose, cointype and change paths directly, so there
  305 -- are no constructors for these.
  306 --
  307 -- @since 1.0.0
  308 data Depth = RootK | AccountK | PaymentK | DelegationK | ScriptK
  309 
  310 -- | A derivation index, with phantom-types to disambiguate derivation type.
  311 --
  312 -- @
  313 -- let accountIx = Index 'Hardened 'AccountK
  314 -- let addressIx = Index 'Soft 'PaymentK
  315 -- @
  316 --
  317 -- @since 1.0.0
  318 newtype Index (derivationType :: DerivationType) (depth :: Depth) = Index
  319     { getIndex :: Word32 }
  320     deriving stock (Generic, Show, Eq, Ord)
  321 
  322 instance NFData (Index derivationType depth)
  323 
  324 instance Bounded (Index 'Hardened depth) where
  325     minBound = Index 0x80000000
  326     maxBound = Index maxBound
  327 
  328 instance Bounded (Index 'Soft depth) where
  329     minBound = Index minBound
  330     maxBound = let (Index ix) = minBound @(Index 'Hardened _) in Index (ix - 1)
  331 
  332 instance Bounded (Index 'WholeDomain depth) where
  333     minBound = Index minBound
  334     maxBound = Index maxBound
  335 
  336 instance Enum (Index 'Hardened depth) where
  337     fromEnum (Index ix) = fromIntegral ix
  338     toEnum ix
  339         | Index (fromIntegral ix) < minBound @(Index 'Hardened _) =
  340             error "Index@Hardened.toEnum: bad argument"
  341         | otherwise =
  342             Index (fromIntegral ix)
  343 
  344 instance Enum (Index 'Soft depth) where
  345     fromEnum (Index ix) = fromIntegral ix
  346     toEnum ix
  347         | Index (fromIntegral ix) > maxBound @(Index 'Soft _) =
  348             error "Index@Soft.toEnum: bad argument"
  349         | otherwise =
  350             Index (fromIntegral ix)
  351 
  352 instance Enum (Index 'WholeDomain depth) where
  353     fromEnum (Index ix) = fromIntegral ix
  354     toEnum ix
  355         | Index (fromIntegral ix) > maxBound @(Index 'WholeDomain _) =
  356             error "Index@WholeDomain.toEnum: bad argument"
  357         | otherwise =
  358             Index (fromIntegral ix)
  359 
  360 instance Buildable (Index derivationType depth) where
  361     build (Index ix) = fromString (show ix)

  364 -- | Type of derivation that should be used with the given indexes.
  365 --
  366 -- In theory, we should only consider two derivation types: soft and hard.
  367 --
  368 -- However, historically, addresses in Cardano used to be generated across both
  369 -- the soft and the hard domain. We therefore introduce a 'WholeDomain' derivation
  370 -- type that is the exact union of `Hardened` and `Soft`.
  371 --
  372 -- @since 1.0.0
  373 data DerivationType = Hardened | Soft | WholeDomain
  374
