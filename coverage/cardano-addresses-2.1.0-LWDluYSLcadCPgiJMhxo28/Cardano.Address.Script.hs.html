   88 -- | A 'Script' type represents multi signature script. The script embodies conditions
   89 -- that need to be satisfied to make it valid.
   90 --
   91 -- @since 3.0.0
   92 data Script
   93     = RequireSignatureOf !KeyHash
   94     | RequireAllOf ![Script]
   95     | RequireAnyOf ![Script]
   96     | RequireSomeOf Word8 ![Script]
   97     deriving stock (Generic, Show, Eq)
   98 instance NFData Script
   99 
  100 -- | Validate 'Script'
  101 --
  102 -- @since 3.0.0
  103 validateScript :: Script -> Either ErrValidateScript ()
  104 validateScript = \case
  105     RequireSignatureOf (KeyHash bytes) -> do
  106         when (isNothing (keyHashFromBytes bytes)) $ Left WrongKeyHash
  107 
  108     RequireAllOf script -> do
  109         when (L.null script) $ Left EmptyList
  110         when (hasDuplicate script) $ Left DuplicateSignatures
  111         traverse_ validateScript script
  112 
  113     RequireAnyOf script -> do
  114         when (L.null script) $ Left EmptyList
  115         when (hasDuplicate script) $ Left DuplicateSignatures
  116         traverse_ validateScript script
  117 
  118     RequireSomeOf m script -> do
  119         when (m == 0) $ Left MZero
  120         when (length script < fromIntegral m) $ Left ListTooSmall
  121         when (hasDuplicate script) $ Left DuplicateSignatures
  122         traverse_ validateScript script
  123   where
  124     hasDuplicate xs = do
  125         length sigs /= length (L.nub sigs)
  126       where
  127         sigs = [ sig | RequireSignatureOf sig <- xs ]
  128 
  129 -- | Possible validation errors when validating a script
  130 data ErrValidateScript
  131     = EmptyList
  132     | ListTooSmall
  133     | MZero
  134     | DuplicateSignatures
  135     | WrongKeyHash
  136     deriving (Eq, Show)
  137 
  138 prettyErrValidateScript
  139     :: ErrValidateScript
  140     -> String
  141 prettyErrValidateScript = \case
  142     EmptyList ->
  143         "The list inside a script is empty."
  144     MZero ->
  145         "The M in at_least cannot be 0."
  146     ListTooSmall ->
  147         "The list inside at_least cannot be less than M."
  148     DuplicateSignatures ->
  149         "The list inside a script has duplicate keys."
  150     WrongKeyHash ->
  151         "The hash of verification key is expected to have "<>show hashSize<>" bytes."

  153 -- | This function realizes what cardano-node's `Api.serialiseToCBOR script` realizes
  154 -- This is basically doing the symbolically following:
  155 -- toCBOR [0,multisigScript]
  156 serialize :: Script -> ByteString
  157 serialize script =
  158     multisigTag <> CBOR.toStrictByteString (toCBOR script)
  159   where
  160     -- | Magic number representing the tag of the native multi-signature script
  161     -- language. For each script language included, a new tag is chosen.
  162     multisigTag :: ByteString
  163     multisigTag = "\00"
  164 
  165     toCBOR :: Script -> CBOR.Encoding
  166     toCBOR = \case
  167         RequireSignatureOf (KeyHash verKeyHash) ->
  168             encodeMultiscriptCtr 0 2 <> CBOR.encodeBytes verKeyHash
  169         RequireAllOf contents ->
  170             encodeMultiscriptCtr 1 2 <> encodeFoldable toCBOR contents
  171         RequireAnyOf contents ->
  172             encodeMultiscriptCtr 2 2 <> encodeFoldable toCBOR contents
  173         RequireSomeOf m contents -> mconcat
  174             [ encodeMultiscriptCtr 3 3
  175             , CBOR.encodeInt (fromInteger $ toInteger m)
  176             , encodeFoldable toCBOR contents
  177             ]
  178 
  179     encodeMultiscriptCtr :: Word -> Word -> CBOR.Encoding
  180     encodeMultiscriptCtr ctrIndex listLen =
  181         CBOR.encodeListLen listLen <> CBOR.encodeWord ctrIndex
  182 
  183     encodeFoldable :: (Foldable f) => (a -> CBOR.Encoding) -> f a -> CBOR.Encoding
  184     encodeFoldable encode' xs = wrapArray len contents
  185       where
  186         (len, contents) = foldl' go (0, mempty) xs
  187         go (!l, !enc) next = (l + 1, enc <> encode' next)
  188 
  189         wrapArray :: Word -> CBOR.Encoding -> CBOR.Encoding
  190         wrapArray len' contents'
  191             | len' <= 23 = CBOR.encodeListLen len' <> contents'
  192             | otherwise  = CBOR.encodeListLenIndef <> contents' <> CBOR.encodeBreak
  193 
  194 -- | Computes the hash of a given script, by first serializing it to CBOR.
  195 toScriptHash :: Script -> ScriptHash
  196 toScriptHash = ScriptHash . blake2b224 . serialize

  198 -- | A 'ScriptHash' type represents script hash. The hash is expected to have size of
  199 -- 28-byte.

  202 newtype ScriptHash = ScriptHash { unScriptHash :: ByteString }
  203     deriving (Generic, Show, Eq)
  204 instance NFData ScriptHash

  206 -- | Construct an 'ScriptHash' from raw 'ByteString' (28 bytes).
  207 --
  208 -- @since 3.0.0
  209 scriptHashFromBytes :: ByteString -> Maybe ScriptHash
  210 scriptHashFromBytes bytes
  211     | BS.length bytes /= hashSize = Nothing
  212     | otherwise = Just $ ScriptHash bytes

  230 data ErrorKeyHashFromText =
  231       ErrorKeyHashFromTextInvalidString (AbstractEncoding ())
  232     | ErrorKeyHashFromTextWrongEncoding
  233     | ErrorKeyHashFromTextWrongPayload
  234     | ErrorKeyHashFromTextWrongHrp
  235     | ErrorKeyHashFromTextWrongDataPart
  236     deriving (Show, Eq)
  237 
  238 showErr :: ErrorKeyHashFromText -> String
  239 showErr (ErrorKeyHashFromTextInvalidString EBase16) =
  240     "Invalid Base16-encoded string."
  241 showErr (ErrorKeyHashFromTextInvalidString EBech32{}) =
  242     "Invalid Bech32-encoded string."
  243 showErr (ErrorKeyHashFromTextInvalidString EBase58) =
  244     "Verification key hash must be must be encoded as \
  245     \base16 or bech32."
  246 showErr ErrorKeyHashFromTextWrongEncoding =
  247     "Verification key hash must be must be encoded as \
  248     \base16 or bech32."
  249 showErr ErrorKeyHashFromTextWrongPayload =
  250     "Verification key hash must contain exactly 28 bytes."
  251 showErr ErrorKeyHashFromTextWrongHrp =
  252     "Verification key hash must have 'script_vkh' hrp when Bech32-encoded."
  253 showErr ErrorKeyHashFromTextWrongDataPart =
  254     "Verification key hash is Bech32-encoded but has wrong data part."
  255 
  256 -- | Construct a 'KeyHash' from 'Text'. Either hex encoded text or
  257 -- Bech32 encoded text with `script_vkh` hrp is expected. Also
  258 -- binary payload is expected to be composed of 28 bytes.
  259 --
  260 -- @since 3.0.0
  261 keyHashFromText :: Text -> Either ErrorKeyHashFromText KeyHash
  262 keyHashFromText txt =  case detectEncoding str of
  263         Just EBase16 -> case fromBase16 (toBytes str) of
  264             Left _ -> Left $ ErrorKeyHashFromTextInvalidString EBase16
  265             Right bytes -> checkPayload bytes
  266         Just EBech32{} -> fromBech32
  267         Just EBase58 -> Left ErrorKeyHashFromTextWrongEncoding
  268         Nothing -> Left ErrorKeyHashFromTextWrongEncoding
  269  where
  270     str = T.unpack txt
  271     toBytes = T.encodeUtf8 . T.pack
  272     checkPayload bytes =
  273         maybeToRight ErrorKeyHashFromTextWrongPayload (keyHashFromBytes bytes)
  274     fromBech32 = do
  275         (hrp, dp) <- either
  276             (const $ Left $ ErrorKeyHashFromTextInvalidString $ EBech32 ())
  277             Right (Bech32.decodeLenient txt)
  278         case Bech32.humanReadablePartToText hrp of
  279             "script_vkh" -> do
  280                 bytes <- maybeToRight
  281                     ErrorKeyHashFromTextWrongDataPart (Bech32.dataPartToBytes dp)
  282                 checkPayload bytes
  283             _ -> Left ErrorKeyHashFromTextWrongHrp

  289 -- Hash a public key
  290 blake2b224 :: ByteString -> ByteString
  291 blake2b224 =
  292     BA.convert . hash @_ @Blake2b_224
  293 
  294 -- Size, in bytes, of a hash of public key (without the corresponding chain code)
  295 hashSize :: Int
  296 hashSize = hashDigestSize Blake2b_224
  297 
  298 -- | Encode a 'KeyHash' to bech32 'Text', using @script_vkh@ as a human readable prefix.
  299 --
  300 -- @since 3.0.0
  301 bech32 :: KeyHash -> Text
  302 bech32 (KeyHash keyHash) = T.decodeUtf8 $ encode (EBech32 hrp) keyHash
  303   where
  304     hrp = [Bech32.humanReadablePart|script_vkh|]

  330 instance ToJSON Script where
  331     toJSON (RequireSignatureOf keyHash) = String $ bech32 keyHash
  332     toJSON (RequireAllOf content) =
  333         object ["all" .= fmap toJSON content]
  334     toJSON (RequireAnyOf content) =
  335         object ["any" .= fmap toJSON content]
  336     toJSON (RequireSomeOf count scripts) =
  337         object ["some" .= object ["at_least" .= count, "from" .= scripts]]
  338 
  339 instance FromJSON Script where
  340     parseJSON v = parseKey v <|> parseAnyOf v  <|> parseAllOf v <|> parseAtLeast v
  341       where
  342         parseKey = withText "Script KeyHash" $
  343             either (fail . showErr) (pure . RequireSignatureOf) . keyHashFromText
  344         parseAnyOf = withObject "Script AnyOf" $ \o ->
  345             RequireAnyOf <$> o .: "any"
  346         parseAllOf = withObject "Script AllOf" $ \o ->
  347             RequireAllOf <$> o .: "all"
  348         parseAtLeast = withObject "Script SomeOf" $ \o -> do
  349             some <- o .: "some"
  350             RequireSomeOf <$> some .: "at_least" <*> some .: "from"
