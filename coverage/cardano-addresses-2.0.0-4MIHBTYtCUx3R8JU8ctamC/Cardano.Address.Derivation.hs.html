   97 -- | An opaque type representing an extended private key.
   98 --
   99 -- __Properties:__
  100 --
  101 -- ===== Roundtripping
  102 --
  103 -- @forall xprv. 'xprvFromBytes' ('xprvToBytes' xprv) == 'Just' xprv@
  104 --
  105 -- ===== Chain Code Invariance
  106 --
  107 -- @forall xprv. 'xprvChainCode' xprv == 'xpubChainCode' ('toXPub' xprv)@
  108 --
  109 -- ===== Public Key Signature
  110 --
  111 -- @forall xprv msg. 'verify' ('toXPub' xprv) msg ('sign' xprv msg) == 'True'@
  112 --
  113 -- @since 1.0.0
  114 type XPrv = CC.XPrv
  115 
  116 -- | An opaque type representing an extended public key.

  118 -- __Properties:__

  120 -- ===== Roundtripping

  124 -- @since 1.0.0

  132 -- | Construct an 'XPub' from raw 'ByteString' (64 bytes).
  133 --
  134 -- @since 1.0.0
  135 xpubFromBytes :: ByteString -> Maybe XPub
  136 xpubFromBytes = eitherToMaybe . CC.xpub
  137 
  138 -- | Convert an 'XPub' to a raw 'ByteString' (64 bytes).
  139 --
  140 -- @since 1.0.0
  141 xpubToBytes :: XPub -> ByteString
  142 xpubToBytes xpub = xpubPublicKey xpub <> xpubChainCode xpub
  143 
  144 -- | Extract the public key from an 'XPub' as a raw 'ByteString' (32 bytes).
  145 --
  146 -- @since 2.0.0
  147 xpubPublicKey :: XPub -> ByteString
  148 xpubPublicKey (CC.XPub pub _cc) = pub
  149 
  150 -- | Extract the chain code from an 'XPub' as a raw 'ByteString' (32 bytes).
  151 --
  152 -- @since 2.0.0
  153 xpubChainCode :: XPub -> ByteString
  154 xpubChainCode (CC.XPub _pub (CC.ChainCode cc)) = cc
  155 
  156 -- | Construct an 'XPrv' from raw 'ByteString' (96 bytes).
  157 --
  158 -- @since 1.0.0
  159 xprvFromBytes :: ByteString -> Maybe XPrv
  160 xprvFromBytes bytes
  161     | BS.length bytes /= 96 = Nothing
  162     | otherwise = do
  163         let (prv, cc) = BS.splitAt 64 bytes
  164         pub <- ed25519ScalarMult (BS.take 32 prv)
  165         eitherToMaybe $ CC.xprv $ prv <> pub <> cc
  166   where
  167     ed25519ScalarMult :: ByteString -> Maybe ByteString
  168     ed25519ScalarMult bs = do
  169         scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
  170         pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar
  171 
  172 -- From  < xprv | pub | cc >
  173 -- â†³ To  < xprv |     | cc >
  174 --
  175 -- | Convert an 'XPrv' to a raw 'ByteString' (96 bytes).
  176 --
  177 -- @since 1.0.0
  178 xprvToBytes :: XPrv -> ByteString
  179 xprvToBytes xprv =
  180     xprvPrivateKey xprv <> xprvChainCode xprv
  181 
  182 -- | Extract the private key from an 'XPrv' as a raw 'ByteString' (64 bytes).
  183 --
  184 -- @since 2.0.0
  185 xprvPrivateKey :: XPrv -> ByteString
  186 xprvPrivateKey = BS.take 64 . CC.unXPrv
  187 
  188 -- | Extract the chain code from an 'XPrv' as a raw 'ByteString' (32 bytes).
  189 --
  190 -- @since 2.0.0
  191 xprvChainCode :: XPrv -> ByteString
  192 xprvChainCode = BS.drop 96 . CC.unXPrv

  275 -- | Key Depth in the derivation path, according to BIP-0039 / BIP-0044
  276 --
  277 -- @
  278 -- root | purpose' | cointype' | account' | change | address@
  279 -- 0th      1st         2nd        3rd        4th      5th
  280 -- @
  281 --
  282 -- We do not manipulate purpose, cointype and change paths directly, so there
  283 -- are no constructors for these.
  284 --
  285 -- @since 1.0.0
  286 data Depth = RootK | AccountK | AddressK | StakingK | MultisigK
  287 
  288 -- | A derivation index, with phantom-types to disambiguate derivation type.
  289 --
  290 -- @
  291 -- let accountIx = Index 'Hardened 'AccountK
  292 -- let addressIx = Index 'Soft 'AddressK
  293 -- @
  294 --
  295 -- @since 1.0.0
  296 newtype Index (derivationType :: DerivationType) (depth :: Depth) = Index
  297     { getIndex :: Word32 }
  298     deriving stock (Generic, Show, Eq, Ord)
  299 
  300 instance NFData (Index derivationType depth)
  301 
  302 instance Bounded (Index 'Hardened depth) where
  303     minBound = Index 0x80000000
  304     maxBound = Index maxBound
  305 
  306 instance Bounded (Index 'Soft depth) where
  307     minBound = Index minBound
  308     maxBound = let (Index ix) = minBound @(Index 'Hardened _) in Index (ix - 1)
  309 
  310 instance Bounded (Index 'WholeDomain depth) where
  311     minBound = Index minBound
  312     maxBound = Index maxBound
  313 
  314 instance Enum (Index 'Hardened depth) where
  315     fromEnum (Index ix) = fromIntegral ix
  316     toEnum ix
  317         | Index (fromIntegral ix) < minBound @(Index 'Hardened _) =
  318             error "Index@Hardened.toEnum: bad argument"
  319         | otherwise =
  320             Index (fromIntegral ix)
  321 
  322 instance Enum (Index 'Soft depth) where
  323     fromEnum (Index ix) = fromIntegral ix
  324     toEnum ix
  325         | Index (fromIntegral ix) > maxBound @(Index 'Soft _) =
  326             error "Index@Soft.toEnum: bad argument"
  327         | otherwise =
  328             Index (fromIntegral ix)
  329 
  330 instance Enum (Index 'WholeDomain depth) where
  331     fromEnum (Index ix) = fromIntegral ix
  332     toEnum ix
  333         | Index (fromIntegral ix) > maxBound @(Index 'WholeDomain _) =
  334             error "Index@WholeDomain.toEnum: bad argument"
  335         | otherwise =
  336             Index (fromIntegral ix)
  337 
  338 instance Buildable (Index derivationType depth) where
  339     build (Index ix) = fromString (show ix)

  342 -- | Type of derivation that should be used with the given indexes.
  343 --
  344 -- In theory, we should only consider two derivation types: soft and hard.
  345 --
  346 -- However, historically, addresses in Cardano used to be generated across both
  347 -- the soft and the hard domain. We therefore introduce a 'WholeDomain' derivation
  348 -- type that is the exact union of `Hardened` and `Soft`.
  349 --
  350 -- @since 1.0.0
  351 data DerivationType = Hardened | Soft | WholeDomain
  352
