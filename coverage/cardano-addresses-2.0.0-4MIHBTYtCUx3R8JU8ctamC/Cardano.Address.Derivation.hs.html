  102 -- | An opaque type representing an extended private key.
  103 --
  104 -- __Properties:__
  105 --
  106 -- ===== Roundtripping
  107 --
  108 -- @forall xprv. 'xprvFromBytes' ('xprvToBytes' xprv) == 'Just' xprv@
  109 --
  110 -- ===== Chain Code Invariance
  111 --
  112 -- @forall xprv. 'xprvChainCode' xprv == 'xpubChainCode' ('toXPub' xprv)@
  113 --
  114 -- ===== Public Key Signature
  115 --
  116 -- @forall xprv msg. 'verify' ('toXPub' xprv) msg ('sign' xprv msg) == 'True'@
  117 --
  118 -- @since 1.0.0
  119 type XPrv = CC.XPrv
  120 
  121 -- | An opaque type representing an extended public key.
  122 --
  123 -- __Properties:__
  124 --
  125 -- ===== Roundtripping
  126 --
  127 -- @forall xpub. 'xpubFromBytes' ('xpubToBytes' xpub) == 'Just' xpub@
  128 --
  129 -- @since 1.0.0
  130 type XPub = CC.XPub

  132 -- | An opaque type representing a signature made from an 'XPrv'.

  134 -- @since 1.0.0

  137 -- | Construct an 'XPub' from raw 'ByteString' (64 bytes).
  138 --
  139 -- @since 1.0.0
  140 xpubFromBytes :: ByteString -> Maybe XPub
  141 xpubFromBytes = eitherToMaybe . CC.xpub
  142 
  143 -- | Convert an 'XPub' to a raw 'ByteString' (64 bytes).
  144 --
  145 -- @since 1.0.0
  146 xpubToBytes :: XPub -> ByteString
  147 xpubToBytes xpub = xpubPublicKey xpub <> xpubChainCode xpub
  148 
  149 -- | Extract the public key from an 'XPub' as a raw 'ByteString' (32 bytes).
  150 --
  151 -- @since 2.0.0
  152 xpubPublicKey :: XPub -> ByteString
  153 xpubPublicKey (CC.XPub pub _cc) = pub
  154 
  155 -- | Extract the chain code from an 'XPub' as a raw 'ByteString' (32 bytes).
  156 --
  157 -- @since 2.0.0
  158 xpubChainCode :: XPub -> ByteString
  159 xpubChainCode (CC.XPub _pub (CC.ChainCode cc)) = cc
  160 
  161 -- | Construct an 'XPrv' from raw 'ByteString' (96 bytes).
  162 --
  163 -- @since 1.0.0
  164 xprvFromBytes :: ByteString -> Maybe XPrv
  165 xprvFromBytes bytes
  166     | BS.length bytes /= 96 = Nothing
  167     | otherwise = do
  168         let (prv, cc) = BS.splitAt 64 bytes
  169         pub <- ed25519ScalarMult (BS.take 32 prv)
  170         eitherToMaybe $ CC.xprv $ prv <> pub <> cc
  171   where
  172     ed25519ScalarMult :: ByteString -> Maybe ByteString
  173     ed25519ScalarMult bs = do
  174         scalar <- eitherToMaybe $ eitherCryptoError $ Ed25519.scalarDecodeLong bs
  175         pure $ Ed25519.pointEncode $ Ed25519.toPoint scalar
  176 
  177 -- From  < xprv | pub | cc >
  178 -- ↳ To  < xprv |     | cc >
  179 --
  180 -- | Convert an 'XPrv' to a raw 'ByteString' (96 bytes).
  181 --
  182 -- @since 1.0.0
  183 xprvToBytes :: XPrv -> ByteString
  184 xprvToBytes xprv =
  185     xprvPrivateKey xprv <> xprvChainCode xprv
  186 
  187 -- | Extract the private key from an 'XPrv' as a raw 'ByteString' (64 bytes).
  188 --
  189 -- @since 2.0.0
  190 xprvPrivateKey :: XPrv -> ByteString
  191 xprvPrivateKey = BS.take 64 . CC.unXPrv
  192 
  193 -- | Extract the chain code from an 'XPrv' as a raw 'ByteString' (32 bytes).
  194 --
  195 -- @since 2.0.0
  196 xprvChainCode :: XPrv -> ByteString
  197 xprvChainCode = BS.drop 96 . CC.unXPrv
  198 
  199 -- | Derive the 'XPub' associated with an 'XPrv'.

  201 -- @since 1.0.0
  202 toXPub :: HasCallStack => XPrv -> XPub
  203 toXPub = CC.toXPub

  251 -- Generate an XPrv using the legacy method (Byron).

  253 -- The seed needs to be at least 32 bytes, otherwise an asynchronous error is thrown.

  255 -- __internal__

  280 -- | Key Depth in the derivation path, according to BIP-0039 / BIP-0044
  281 --
  282 -- @
  283 -- root | purpose' | cointype' | account' | change | address@
  284 -- 0th      1st         2nd        3rd        4th      5th
  285 -- @
  286 --
  287 -- We do not manipulate purpose, cointype and change paths directly, so there
  288 -- are no constructors for these.
  289 --
  290 -- @since 1.0.0
  291 data Depth = RootK | AccountK | AddressK | StakingK
  292 
  293 -- | Marker for addresses type engaged. We want to handle three cases here.
  294 -- The first two are pertinent to UTxO accounting
  295 -- and the last one handles rewards from participation in staking.
  296 -- (a) external chain is used for addresses that are part of the 'advertised'
  297 --     targets of a given transaction
  298 -- (b) internal change is for addresses used to handle the change of a
  299 --     the transaction within a given wallet
  300 data AccountingStyle
  301     = UTxOExternal
  302     | UTxOInternal
  303     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  304 
  305 instance NFData AccountingStyle
  306 
  307 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  308 -- to assign the wrong index to the corresponding constructor (by swapping
  309 -- around the constructor above for instance).
  310 instance Enum AccountingStyle where
  311     toEnum = \case
  312         0 -> UTxOExternal
  313         1 -> UTxOInternal
  314         _ -> error "AccountingStyle.toEnum: bad argument"
  315     fromEnum = \case
  316         UTxOExternal -> 0
  317         UTxOInternal -> 1
  318 
  319 -- | A derivation index, with phantom-types to disambiguate derivation type.
  320 --
  321 -- @
  322 -- let accountIx = Index 'Hardened 'AccountK
  323 -- let addressIx = Index 'Soft 'AddressK
  324 -- @
  325 --
  326 -- @since 1.0.0
  327 newtype Index (derivationType :: DerivationType) (depth :: Depth) = Index
  328     { getIndex :: Word32 }
  329     deriving stock (Generic, Show, Eq, Ord)
  330 
  331 instance NFData (Index derivationType depth)
  332 
  333 instance Bounded (Index 'Hardened depth) where
  334     minBound = Index 0x80000000
  335     maxBound = Index maxBound
  336 
  337 instance Bounded (Index 'Soft depth) where
  338     minBound = Index minBound
  339     maxBound = let (Index ix) = minBound @(Index 'Hardened _) in Index (ix - 1)
  340 
  341 instance Bounded (Index 'WholeDomain depth) where
  342     minBound = Index minBound
  343     maxBound = Index maxBound
  344 
  345 instance Enum (Index 'Hardened depth) where
  346     fromEnum (Index ix) = fromIntegral ix
  347     toEnum ix
  348         | Index (fromIntegral ix) < minBound @(Index 'Hardened _) =
  349             error "Index@Hardened.toEnum: bad argument"
  350         | otherwise =
  351             Index (fromIntegral ix)
  352 
  353 instance Enum (Index 'Soft depth) where
  354     fromEnum (Index ix) = fromIntegral ix
  355     toEnum ix
  356         | Index (fromIntegral ix) > maxBound @(Index 'Soft _) =
  357             error "Index@Soft.toEnum: bad argument"
  358         | otherwise =
  359             Index (fromIntegral ix)
  360 
  361 instance Enum (Index 'WholeDomain depth) where
  362     fromEnum (Index ix) = fromIntegral ix
  363     toEnum ix
  364         | Index (fromIntegral ix) > maxBound @(Index 'WholeDomain _) =
  365             error "Index@WholeDomain.toEnum: bad argument"
  366         | otherwise =
  367             Index (fromIntegral ix)
  368 
  369 instance Buildable (Index derivationType depth) where
  370     build (Index ix) = fromString (show ix)

  373 -- | Type of derivation that should be used with the given indexes.
  374 --
  375 -- In theory, we should only consider two derivation types: soft and hard.
  376 --
  377 -- However, historically, addresses in Cardano used to be generated across both
  378 -- the soft and the hard domain. We therefore introduce a 'WholeDomain' derivation
  379 -- type that is the exact union of `Hardened` and `Soft`.
  380 --
  381 -- @since 1.0.0
  382 data DerivationType = Hardened | Soft | WholeDomain
  383 
  384 -- | An interface for doing hard derivations from the root private key, /Master Key/

  386 -- @since 1.0.0

  389     type AddressIndexDerivationType key :: DerivationType

  396     -- @since 1.0.0

  406     -- @since 1.0.0

  428 -- | An interface for doing staking derivations from the account private key.
  429 --
  430 -- @since 2.0.0
  431 class StakingDerivation (key :: Depth -> * -> *) where
  432     -- | Derive a staking key for a corresponding 'AccountK'. Note that wallet
