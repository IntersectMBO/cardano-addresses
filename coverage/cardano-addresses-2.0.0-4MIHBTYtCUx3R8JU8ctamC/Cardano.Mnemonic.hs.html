  119 -- $introduction
  120 --
  121 -- We call 'Entropy' an arbitrary sequence of bytes that has been generated through __high quality randomness methods__.
  122 -- The allowed size of an 'Entropy' is @96-256@ bits and is __necessarily a multiple of 32 bits__ (4 bytes).
  123 --
  124 -- We call 'Mnemonic' an 'Entropy' with an appended checksum calculated by taking the first @ent / 32@ bits of the /SHA256/ hash of it,
  125 -- where ent designates the 'Entropy' size in bits.
  126 --
  127 -- The concatenated result is split into groups of @11@ bits, each encoding a number from 0 to 2047 serving as an index into a [known dictionary](https://github.com/input-output-hk/cardano-wallet/tree/master/specifications/mnemonic/english.txt).
  128 -- This makes for a __human-readable sentence__ of English words.
  129 --
  130 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  131 -- | Entropy Size        | Checksum Size | Sentence Length | Example                                                                                                                                         |
  132 -- +=====================+===============+=================+=================================================================================================================================================+
  133 -- | 96  bits (12 bytes) | 3 bits        | 9 words         | test child burst immense armed parrot company walk dog                                                                                          |
  134 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  135 -- | 128 bits (16 bytes) | 4 bits        | 12 words        | test walk nut penalty hip pave soap entry language right filter choice                                                                          |
  136 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  137 -- | 160 bits (20 bytes) | 5 bits        | 15 words        | art forum devote street sure rather head chuckle guard poverty release quote oak craft enemy                                                    |
  138 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  139 -- | 192 bits (24 bytes) | 6 bits        | 18 words        | churn shaft spoon second erode useless thrive burst group seed element sign scrub buffalo jelly grace neck useless                              |
  140 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  141 -- | 224 bits (28 bytes) | 7 bits        | 21 words        | draft ability female child jump maid roof hurt below live topple paper exclude ordinary coach churn sunset emerge blame ketchup much            |
  142 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+
  143 -- | 256 bits (32 bytes) | 8 bits        | 24 words        | excess behave track soul table wear ocean cash stay nature item turtle palm soccer lunch horror start stumble month panic right must lock dress |
  144 -- +---------------------+---------------+-----------------+-------------------------------------------------------------------------------------------------------------------------------------------------+

  146 -- A opaque 'Mnemonic' type.
  147 data Mnemonic (mw :: Nat) = Mnemonic
  148     { mnemonicToEntropy  :: Entropy (EntropySize mw)
  149         -- ^ Convert a 'Mnemonic' back to an 'Entropy'.
  150         --
  151         -- @since 1.0.0
  152     , mnemonicToSentence :: MnemonicSentence mw
  153     } deriving (Eq, Show)

  155 -- This wraps EntropyError of "Cardano.Encoding.BIP39"
  156 newtype MnemonicException csz =
  157     UnexpectedEntropyError (EntropyError csz)
  158     -- ^ Invalid entropy length or checksum
  159     deriving (Show, Typeable)

  161 -- | This wraps errors from "Cardano.Encoding.BIP39"
  162 data MkMnemonicError csz
  163     = ErrMnemonicWords MnemonicWordsError
  164       -- ^ Wrong number of words in mnemonic.
  165     | ErrEntropy (EntropyError csz)
  166       -- ^ Invalid entropy length or checksum.
  167     | ErrDictionary DictionaryError
  168       -- ^ Invalid word in mnemonic.
  169     deriving (Eq, Show)
  170 
  171 deriving instance Eq (EntropyError czs)
  172 deriving instance Eq MnemonicWordsError
  173 deriving instance Eq DictionaryError

  175 -- | Smart-constructor for the 'Entropy'. Make sure the 'ByteString' comes from a highly random source or use 'genEntropy'.
  176 --
  177 -- __example__:
  178 --
  179 -- >>> mkEntropy @160 bytes
  180 -- Entropy {} :: Entropy 160
  181 --
  182 -- __property__:
  183 --
  184 -- prop> mkEntropy (entropyToBytes ent) == Right ent
  185 --
  186 -- @since 1.0.0
  187 mkEntropy
  188     :: forall (ent :: Nat) csz. (ValidEntropySize ent, ValidChecksumSize ent csz)
  189     => ScrubbedBytes
  190     -> Either (EntropyError csz) (Entropy ent)
  191 mkEntropy = toEntropy
  192 
  193 -- | Generate Entropy of a given size using a cryptographically secure random seed.
  194 --
  195 -- __example:__
  196 --
  197 -- >>> genEntropy @128
  198 -- Entropy {} :: Entropy 128
  199 --
  200 -- @since 1.0.0
  201 genEntropy
  202     :: forall (ent :: Nat) csz. (ValidEntropySize ent, ValidChecksumSize ent csz)
  203     => IO (Entropy ent)
  204 genEntropy =
  205     let
  206         size =
  207             fromIntegral $ natVal @ent Proxy
  208         eitherToIO =
  209             either (throwM . UnexpectedEntropyError) return
  210     in
  211         (eitherToIO . mkEntropy) =<< Crypto.getEntropy (size `div` 8)
  212 
  213 -- | Smart-constructor for 'Mnemonic'. Requires a type application to
  214 -- disambiguate the mnemonic size.
  215 --
  216 -- __example__:
  217 --
  218 -- >>> mkMnemonic @15 sentence
  219 -- Mnemonic {} :: Mnemonic 15
  220 --
  221 -- __property__:
  222 --
  223 -- prop> mkMnemonic (mnemonicToText mnemonic) == Right mnemonic
  224 --
  225 -- @since 1.0.0
  226 mkMnemonic
  227     :: forall (mw :: Nat) (ent :: Nat) csz.
  228      ( ConsistentEntropy ent mw csz
  229      , EntropySize mw ~ ent
  230      )
  231     => [Text]
  232     -> Either (MkMnemonicError csz) (Mnemonic mw)
  233 mkMnemonic wordsm = do
  234     phrase <- left ErrMnemonicWords
  235         $ mnemonicPhrase @mw (toUtf8String <$> wordsm)
  236 
  237     sentence <- left ErrDictionary
  238         $ mnemonicPhraseToMnemonicSentence Dictionary.english phrase
  239 
  240     entropy <- left ErrEntropy
  241         $ wordsToEntropy sentence
  242 
  243     pure Mnemonic
  244         { mnemonicToEntropy  = entropy
  245         , mnemonicToSentence = sentence
  246         }
  247 
  248 -- | Convert an Entropy to a corresponding Mnemonic Sentence. Since 'Entropy'
  249 -- and 'Mnemonic' can only be created through smart-constructors, this function
  250 -- cannot fail and is total.
  251 --
  252 -- @since 1.0.0
  253 entropyToMnemonic
  254     :: forall mw ent csz.
  255      ( ValidMnemonicSentence mw
  256      , ValidEntropySize ent
  257      , ValidChecksumSize ent csz
  258      , ent ~ EntropySize mw
  259      , mw ~ MnemonicWords ent
  260      )
  261     => Entropy ent
  262     -> Mnemonic mw
  263 entropyToMnemonic entropy = Mnemonic
  264     { mnemonicToSentence = entropyToWords entropy
  265     , mnemonicToEntropy  = entropy
  266     }

  288 -- | Convert a 'Mnemonic' to a sentence of English mnemonic words.

  290 -- @since 1.0.0

  306 -- | Ease the manipulation of 'Mnemonic' by encapsulating the type constraints inside a constructor.
  307 -- This is particularly useful for functions which do not require anything but a valid 'Mnemonic' without any
  308 -- particular pre-condition on the size of the 'Mnemonic' itself.
  309 --
  310 -- @since 1.0.0
  311 data SomeMnemonic where
  312     SomeMnemonic :: forall mw. KnownNat mw => Mnemonic mw -> SomeMnemonic
  313 
  314 deriving instance Show SomeMnemonic
  315 instance Eq SomeMnemonic where
  316     (SomeMnemonic mwa) == (SomeMnemonic mwb) =
  317         case typeOf mwa `testEquality` typeOf mwb of
  318             Nothing -> False
  319             Just Refl -> mwa == mwb

  321 -- | This class enables caller to parse text list of variable length
  322 -- into mnemonic sentences.
  323 --
  324 -- Note that the given 'Nat's **have** to be valid mnemonic sizes, otherwise the
  325 -- underlying code won't even compile, with not-so-friendly error messages.
  326 class MkSomeMnemonic (sz :: [Nat]) where
  327     -- | Construct a mnemonic from a list of words. This function is particularly useful when the
  328     -- number of words is not necessarily known at runtime. The function is however /ambiguous/ and
  329     -- requires thereby a type application.
  330     --
  331     -- __examples:__
  332     --
  333     -- >>> mkSomeMnemonic @'[ 12 ] [ "test", "child", "burst", "immense", "armed", "parrot", "company", "walk", "dog" ]
  334     -- Left "Invalid number of words: 12 words are expected."
  335     --
  336     -- >>> mkSomeMnemonic @'[ 9, 12, 15 ] [ "test", "child", "burst", "immense", "armed", "parrot", "company", "walk", "dog" ]
  337     -- Right (SomeMnemonic ...)
  338     --
  339     -- @since 1.0.0
  340     mkSomeMnemonic :: [Text] -> Either (MkSomeMnemonicError sz) SomeMnemonic
  341 
  342 -- | Error reported from trying to create a passphrase from a given mnemonic
  343 --
  344 -- @since 1.0.0
  345 newtype MkSomeMnemonicError (sz :: [Nat]) =
  346     MkSomeMnemonicError { getMkSomeMnemonicError :: String }
  347     deriving stock (Eq, Show)
  348 
  349 instance {-# OVERLAPS #-}
  350     ( n ~ EntropySize mw
  351     , csz ~ CheckSumBits n
  352     , ConsistentEntropy n mw csz
  353     , MkSomeMnemonic rest
  354     , NatVals rest
  355     ) =>
  356     MkSomeMnemonic (mw ': rest)
  357   where
  358     mkSomeMnemonic parts = case parseMW of
  359         Left err -> left (promote err) parseRest
  360         Right mw -> Right mw
  361       where
  362         parseMW = left (MkSomeMnemonicError . getMkSomeMnemonicError) $ -- coerce
  363             mkSomeMnemonic @'[mw] parts
  364         parseRest = left (MkSomeMnemonicError . getMkSomeMnemonicError) $ -- coerce
  365             mkSomeMnemonic @rest parts
  366         promote e e' =
  367             let
  368                 sz = fromEnum <$> natVals (Proxy :: Proxy (mw ': rest))
  369                 mw = fromEnum $ natVal (Proxy :: Proxy mw)
  370             in if length parts `notElem` sz
  371                 then MkSomeMnemonicError
  372                     $  "Invalid number of words: "
  373                     <> intercalate ", " (show <$> init sz)
  374                     <> (if length sz > 1 then " or " else "") <> show (last sz)
  375                     <> " words are expected."
  376                 else if length parts == mw then e else e'
  377 
  378 -- | Small helper to collect 'Nat' values from a type-level list
  379 class NatVals (ns :: [Nat]) where
  380     natVals :: Proxy ns -> [Integer]
  381 
  382 instance NatVals '[] where
  383     natVals _ = []
  384 
  385 instance (KnownNat n, NatVals rest) => NatVals (n ': rest) where
  386     natVals _ = natVal (Proxy :: Proxy n) : natVals (Proxy :: Proxy rest)
  387 
  388 instance
  389     ( n ~ EntropySize mw
  390     , csz ~ CheckSumBits n
  391     , ConsistentEntropy n mw csz
  392     ) =>
  393     MkSomeMnemonic (mw ': '[])
  394   where
  395     mkSomeMnemonic parts = do
  396         bimap (MkSomeMnemonicError . pretty) SomeMnemonic (mkMnemonic @mw parts)
  397       where
  398         pretty = \case
  399             ErrMnemonicWords ErrWrongNumberOfWords{} ->
  400                 "Invalid number of words: "
  401                 <> show (natVal (Proxy :: Proxy mw))
  402                 <> " words are expected."
  403             ErrDictionary (ErrInvalidDictionaryWord _) ->
  404                 "Found an unknown word not present in the pre-defined dictionary. \
  405                 \The full dictionary is available here: \
  406                 \https://github.com/input-output-hk/cardano-wallet/tree/master/specifications/mnemonic/english.txt"
  407             ErrEntropy ErrInvalidEntropyChecksum{} ->
  408                 "Invalid entropy checksum: please double-check the last word of \
  409                 \your mnemonic sentence."
  410             ErrEntropy ErrInvalidEntropyLength{} ->
  411                 "Something went wrong when trying to generate the entropy from \
  412                 \the given mnemonic. As a user, there's nothing you can do."

  414 -- $troubleshooting
  415 --
  416 -- - /Natural XX is out of bounds for Int/:
  417 --   This usually occurs when ones is trying to specify an invalid size for an
  418 --   'Entropy' or 'Mnemonic'. For example:
  419 --
  420 --   >>> genEntropy @42
  421 --   error:
  422 --     • Natural CheckSumBits 42 is out of bounds for Int

  424 -- - This could be the case as well when forgetting to use an adequate type application:
