  123 -- This module provides an implementation of:
  124 --
  125 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Jormungandr master keys from mnemonic sentences
  126 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  127 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  128 -- - 'Cardano.Address.PaymentAddress': for constructing payment addresses from a address public key
  129 -- - 'Cardano.Address.DelegationAddress': for constructing payment addresses from a address and stake public keys

  131 -- == Disclaimer
  132 --
  133 -- Beware that this format of address have been specially crafted for the
  134 -- incentivized testnet (ITN). This will soon be __deprecated__ in favor of
  135 -- 'Cardano.Address.Style.Shelley.Shelley'.

  139 -- | A cryptographic key for sequential-scheme address derivation, with
  140 -- phantom-types to disambiguate key types.
  141 --
  142 -- @
  143 -- let rootPrivateKey = Jormungandr 'RootK XPrv
  144 -- let accountPubKey  = Jormungandr 'AccountK XPub
  145 -- let addressPubKey  = Jormungandr 'AddressK XPub
  146 -- @
  147 --
  148 -- @since 2.0.0
  149 newtype Jormungandr (depth :: Depth) key = Jormungandr
  150     { getKey :: key
  151         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  152         --
  153         -- @since 2.0.0
  154     }
  155     deriving stock (Generic, Show, Eq)
  156 
  157 deriving instance (Functor (Jormungandr depth))
  158 instance (NFData key) => NFData (Jormungandr depth key)
  159 
  160 data Role
  161     = UTxOExternal
  162     | UTxOInternal
  163     | Stake
  164     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
  165 
  166 instance NFData Role
  167 
  168 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  169 -- to assign the wrong index to the corresponding constructor (by swapping
  170 -- around the constructor above for instance).
  171 instance Enum Role where
  172     toEnum = \case
  173         0 -> UTxOExternal
  174         1 -> UTxOInternal
  175         2 -> Stake
  176         _ -> error "Role.toEnum: bad argument"
  177     fromEnum = \case
  178         UTxOExternal -> 0
  179         UTxOInternal -> 1
  180         Stake -> 2

  183 -- Key Derivation
  184 --
  185 -- $keyDerivation
  186 --
  187 -- === Generating a root key from 'SomeMnemonic'
  188 -- > :set -XOverloadedStrings
  189 -- > :set -XTypeApplications
  190 -- > :set -XDataKinds
  191 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  192 -- >
  193 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  194 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  195 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Jormungandr 'RootK XPrv
  196 --
  197 -- === Deriving child keys
  198 --
  199 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  200 --
  201 -- > let accIx = toEnum 0x80000000
  202 -- > let acctK = deriveAccountPrivateKey rootK accIx
  203 -- >
  204 -- > let addIx = toEnum 0x00000014
  205 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  206 --
  207 -- > let stakeK = deriveStakingPrivateKey acctK
  208 
  209 instance Internal.GenMasterKey Jormungandr where
  210     type SecondFactor Jormungandr = ScrubbedBytes
  211 
  212     genMasterKeyFromXPrv = liftXPrv
  213     genMasterKeyFromMnemonic fstFactor sndFactor =
  214         Jormungandr $ generateNew seedValidated sndFactor
  215         where
  216             seed  = someMnemonicToBytes fstFactor
  217             seedValidated = assert
  218                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  219                 seed

  221 instance Internal.HardDerivation Jormungandr where
  222     type AccountIndexDerivationType Jormungandr = 'Hardened
  223     type AddressIndexDerivationType Jormungandr = 'Soft
  224     type WithRole Jormungandr = Role
  225 
  226     deriveAccountPrivateKey (Jormungandr rootXPrv) accIx =
  227         let
  228             purposeIx =
  229                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  230             coinTypeIx =
  231                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  232             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  233                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  234             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  235                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  236             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  237                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  238         in
  239             Jormungandr acctXPrv
  240 
  241     deriveAddressPrivateKey (Jormungandr accXPrv) role addrIx =
  242         let
  243             roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  244             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  245                 deriveXPrv DerivationScheme2 accXPrv roleCode
  246             addrXPrv = -- lvl5 derivation; soft derivation of address index
  247                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  248         in
  249             Jormungandr addrXPrv
  250 
  251 instance Internal.SoftDerivation Jormungandr where
  252     deriveAddressPublicKey (Jormungandr accXPub) role addrIx =
  253         fromMaybe errWrongIndex $ do
  254             let roleCode = toEnum @(Index 'Soft _) $ fromEnum role
  255             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  256                 deriveXPub DerivationScheme2 accXPub roleCode
  257             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  258                 deriveXPub DerivationScheme2 changeXPub addrIx
  259             return $ Jormungandr addrXPub
  260       where
  261         errWrongIndex = error $
  262             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  263             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  264             \either a programmer error, or, we may have reached the maximum \
  265             \number of addresses for a given wallet."

  325 -- Re-export from 'Cardano.Address.Derivation' to have it documented specialized in Haddock
  326 --
  327 -- | Derive a staking key for a corresponding 'AccountK'. Note that wallet
  328 -- software are by convention only using one staking key per account, and always
  329 -- the first account (with index 0').
  330 --
  331 -- Deriving staking keys for something else than the initial account is not
  332 -- recommended and can lead to incompatibility with existing wallet softwares
  333 -- (Daedalus, Yoroi, Adalite...).
  334 --
  335 -- @since 2.0.0
  336 deriveStakingPrivateKey
  337     :: Jormungandr 'AccountK XPrv
  338     -> Jormungandr 'StakingK XPrv
  339 deriveStakingPrivateKey accXPrv =
  340     let (Jormungandr stakeXPrv) =
  341             deriveAddressPrivateKey accXPrv Stake (minBound @(Index 'Soft _))
  342     in Jormungandr stakeXPrv

  364 instance Internal.PaymentAddress Jormungandr where
  365     paymentAddress discrimination k = unsafeMkAddress $
  366         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  367             putWord8 (invariantNetworkTag 255 firstByte)
  368             putByteString (xpubPublicKey $ getKey k)
  369       where
  370           firstByte = networkTag @Jormungandr discrimination
  371           expectedLength = 1 + publicKeySize
  372 
  373 instance Internal.DelegationAddress Jormungandr where
  374     delegationAddress discrimination paymentKey stakingKey = unsafeMkAddress $
  375         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  376             putWord8 (invariantNetworkTag 255 $ NetworkTag $ firstByte + 1)
  377             putByteString . xpubPublicKey . getKey $ paymentKey
  378             putByteString . xpubPublicKey . getKey $ stakingKey
  379       where
  380           (NetworkTag firstByte) = networkTag @Jormungandr discrimination
  381           expectedLength = 1 + 2*publicKeySize

  383 -- | Analyze an 'Address' to know whether it's a Jormungandr address or not.
  384 --
  385 -- Throws 'JormungandrAddrError' if it's not a valid Shelley address, or a ready-to-print
  386 -- string giving details about the 'Jormungandr'.
  387 --
  388 -- @since 2.0.0
  389 inspectJormungandrAddress :: MonadThrow m => Address -> m Json.Value
  390 inspectJormungandrAddress addr
  391     | BS.length bytes < 1 + publicKeySize = throwM (JoWrongInputSize (BS.length bytes))
  392     | otherwise =
  393         let
  394             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  395             addrType = fstByte .&. 0b01111111
  396             network  = (fstByte .&. 0b10000000) `shiftR` 7
  397             size = publicKeySize
  398         in
  399             case addrType of
  400                 0x03 | BS.length rest == size ->
  401                     pure $ Json.object
  402                     [ "address_style"   .= Json.String "Jormungandr"
  403                     , "address_type"    .= Json.String "single"
  404                     , "stake_reference" .= Json.String "none"
  405                     , "spending_key"    .= base16 (BS.take size rest)
  406                     , "network_tag"     .= network
  407                     ]
  408                 0x04 | BS.length rest == 2 * size ->
  409                     pure $ Json.object
  410                     [ "address_style"   .= Json.String "Jormungandr"
  411                     , "address_type"    .= Json.String "group"
  412                     , "stake_reference" .= Json.String "by value"
  413                     , "spending_key"    .= base16 (BS.take size rest)
  414                     , "stake_key"       .= base16 (BS.drop size rest)
  415                     , "network_tag"     .= network
  416                     ]
  417                 0x05 | BS.length rest == size ->
  418                     pure $ Json.object
  419                     [ "address_style"   .= Json.String "Jormungandr"
  420                     , "address_type"    .= Json.String "account"
  421                     , "stake_reference" .= Json.String "none"
  422                     , "account_key"     .= base16 (BS.take size rest)
  423                     , "network_tag"     .= network
  424                     ]
  425                 0x06 | BS.length rest == size ->
  426                     pure $ Json.object
  427                     [ "address_style"   .= Json.String "Jormungandr"
  428                     , "address_type"    .= Json.String "multisig"
  429                     , "stake_reference" .= Json.String "none"
  430                     , "merkle_root"     .= base16 (BS.take size rest)
  431                     , "network_tag"     .= network
  432                     ]
  433                 _ -> throwM JoUnknownAddrType
  434   where
  435     bytes = unAddress addr
  436     base16 = T.unpack . T.decodeUtf8 . encode EBase16

  513 -- Purpose is a constant set to 1852' (or 0x8000073c) following the BIP-44
  514 -- extension for Cardano:
  515 --
  516 -- https://github.com/input-output-hk/implementation-decisions/blob/e2d1bed5e617f0907bc5e12cf1c3f3302a4a7c42/text/1852-hd-chimeric.md
  517 --
  518 -- It indicates that the subtree of this node is used according to this
  519 -- specification.
  520 --
  521 -- Hardened derivation is used at this level.
  522 purposeIndex :: Word32
  523 purposeIndex = 0x8000073c
  524 
  525 -- One master node (seed) can be used for unlimited number of independent
  526 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  527 -- same space for various cryptocoins has some disadvantages.
  528 --
  529 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  530 -- addresses across cryptocoins and improving privacy issues.
  531 --
  532 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  533 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  534 -- Lovelace.
  535 --
  536 -- Hardened derivation is used at this level.
  537 coinTypeIndex :: Word32
  538 coinTypeIndex = 0x80000717

  540 -- The minimum seed length for 'generateKeyFromSeed' and
  541 -- 'unsafeGenerateKeyFromSeed'.
  542 minSeedLengthBytes :: Int
