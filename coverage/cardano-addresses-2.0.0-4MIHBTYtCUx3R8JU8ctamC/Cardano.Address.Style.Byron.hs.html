  120 -- $overview
  121 --
  122 -- This module provides an implementation of:
  123 --
  124 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Byron master keys from mnemonic sentences
  125 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical derivation of parent to child keys
  126 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  127 --
  128 -- We call 'Byron' addresses the old address type used by Daedalus in the early
  129 -- days of Cardano. Using this type of addresses and underlying key scheme is
  130 -- now considered __deprecated__ because of some security implications.
  131 --
  132 -- The internals of the 'Byron' does not matter for the reader, but basically
  133 -- contains what is necessary to perform key derivation and generate addresses
  134 -- from a 'Byron' type.
  135 --
  136 -- == Deprecation Notice
  137 --
  138 -- Unless you have good reason to do so (like writing backward-compatible code
  139 -- with an existing piece), any new implementation __should use__ the
  140 -- 'Cardano.Address.Style.Icarus.Icarus' style for key and addresses.
  141 
  142 
  143 -- | Material for deriving HD random scheme keys, which can be used for making
  144 -- addresses.
  145 --
  146 -- @since 1.0.0
  147 data Byron (depth :: Depth) key = Byron
  148     { getKey :: key
  149     -- ^ The raw private or public key.
  150     --
  151     -- @since 1.0.0
  152     , derivationPath :: DerivationPath depth
  153     -- ^ The address derivation indices for the level of this key.
  154     --
  155     -- @since 1.0.0
  156     , payloadPassphrase :: ScrubbedBytes
  157     -- ^ Used for encryption of the derivation path payload within an address.
  158     --
  159     -- @since 1.0.0
  160     } deriving stock (Generic)
  161 {-# DEPRECATED Byron "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  162 {-# DEPRECATED getKey "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  171 -- | The hierarchical derivation indices for a given level/depth.
  172 --
  173 -- @since 1.0.0
  174 type family DerivationPath (depth :: Depth) :: * where
  175     -- The root key is generated from the seed.
  176     DerivationPath 'RootK =
  177         ()
  178     -- The account key is generated from the root key and account index.
  179     DerivationPath 'AccountK =
  180         Index 'WholeDomain 'AccountK
  181     -- The address key is generated from the account key and address index.
  182     DerivationPath 'AddressK =
  183         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  184 {-# DEPRECATED DerivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  193 -- > import Cardano.Mnemonic ( mkSomeMnemonic )

  199 --
  200 -- > let accIx = toEnum 0x80000000
  201 -- > let acctK = deriveAccountPrivateKey rootK accIx
  202 -- >
  203 -- > let addIx = toEnum 0x80000014
  204 -- > let addrK = deriveAddressPrivateKey acctK addIx

  206 instance Internal.GenMasterKey Byron where
  207     type SecondFactor Byron = ()
  208 
  209     genMasterKeyFromXPrv xprv =
  210         liftXPrv (toXPub xprv) () xprv
  211     genMasterKeyFromMnemonic (SomeMnemonic mw) () =
  212         liftXPrv (toXPub xprv) () xprv
  213       where
  214         xprv = generate (hashSeed seedValidated)
  215         seed  = entropyToBytes $ mnemonicToEntropy mw
  216         seedValidated = assert
  217             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  218             seed

  220 instance Internal.HardDerivation Byron where
  221     type AddressIndexDerivationType Byron = 'WholeDomain
  222     type AccountIndexDerivationType Byron = 'WholeDomain
  223     type WithRole Byron = ()

  225     deriveAccountPrivateKey rootXPrv accIx = Byron
  226         { getKey = deriveXPrv DerivationScheme1 (getKey rootXPrv) accIx
  227         , derivationPath = accIx
  228         , payloadPassphrase = payloadPassphrase rootXPrv
  229         }

  231     deriveAddressPrivateKey accXPrv () addrIx = Byron
  232         { getKey = deriveXPrv DerivationScheme1 (getKey accXPrv) addrIx
  233         , derivationPath = (derivationPath accXPrv, addrIx)
  234         , payloadPassphrase = payloadPassphrase accXPrv
  235         }

  296 -- | Analyze an 'Address' to know whether it's a Byron address or not.
  297 -- Throws 'ByronAddrError' if the address isn't a byron address, or return a
  298 -- structured JSON that gives information about an address.
  299 --
  300 -- @since 2.0.0
  301 inspectByronAddress :: forall m. MonadThrow m => Maybe XPub -> Address -> m Json.Value
  302 inspectByronAddress mRootPub addr = do
  303     payload <- either (throwM . BrDeserialiseError) pure
  304         $ CBOR.deserialiseCbor CBOR.decodeAddressPayload bytes
  305 
  306     (root, attrs) <- either (throwM . BrDeserialiseError) pure
  307         $ CBOR.deserialiseCbor decodePayload payload
  308 
  309     path <- do
  310         attr <- maybe (throwM BrMissingExpectedDerivationPath) pure
  311             (find ((== 1) . fst) attrs)
  312         case mRootPub of
  313             Nothing ->
  314                 pure $ toJSON $ T.unpack $ T.decodeUtf8 $ encode EBase16 $ snd attr
  315             Just rootPub ->
  316                 decryptPath rootPub attr
  317 
  318     ntwrk <- either (throwM . BrDeserialiseError) pure
  319         $ CBOR.deserialiseCbor CBOR.decodeProtocolMagicAttr payload
  320 
  321     pure $ Json.object
  322         [ "address_style"   .= Json.String "Byron"
  323         , "stake_reference" .= Json.String "none"
  324         , "address_root"    .= T.unpack (T.decodeUtf8 $ encode EBase16 root)
  325         , "derivation_path" .= path
  326         , "network_tag"     .= maybe Json.Null toJSON ntwrk
  327         ]
  328   where
  329     bytes :: ByteString
  330     bytes = unAddress addr
  331 
  332     decodePayload :: forall s. CBOR.Decoder s (ByteString, [(Word8, ByteString)])
  333     decodePayload = do
  334         _ <- CBOR.decodeListLenCanonicalOf 3
  335         root <- CBOR.decodeBytes
  336         (root,) <$> CBOR.decodeAllAttributes
  337 
  338     decryptPath :: XPub -> (Word8, ByteString) -> m Json.Value
  339     decryptPath rootPub attr = do
  340         let pwd = hdPassphrase rootPub
  341         path <- either (const (throwM BrFailedToDecryptPath)) pure
  342             $ CBOR.deserialiseCbor (CBOR.decodeDerivationPathAttr pwd [attr]) mempty
  343         case path of
  344             Nothing -> throwM BrFailedToDecryptPath
  345             Just (acctIx, addrIx) -> pure $ Json.object
  346                 [ "account_index" .= prettyIndex acctIx
  347                 , "address_index" .= prettyIndex addrIx
  348                 ]
  349 
  350     prettyIndex :: Word32 -> String
  351     prettyIndex ix
  352         | ix >= firstHardened = show (ix - firstHardened) <> "H"
  353         | otherwise = show ix
  354       where
  355         firstHardened = 0x80000000

  357 instance Internal.PaymentAddress Byron where
  358     paymentAddress discrimination k = unsafeMkAddress
  359         $ CBOR.toStrictByteString
  360         $ CBOR.encodeAddress (getKey k) attrs
  361       where
  362         (acctIx, addrIx) = bimap word32 word32 $ derivationPath k
  363         pwd = payloadPassphrase k
  364         NetworkTag magic = networkTag @Byron discrimination
  365         attrs = case addressDiscrimination @Byron discrimination of
  366             RequiresNetworkTag ->
  367                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  368                 , CBOR.encodeProtocolMagicAttr magic
  369                 ]
  370             RequiresNoTag ->
  371                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  372                 ]

  418 -- | Backdoor for generating a new key from a raw 'XPrv'.
  419 --
  420 -- Note that the @depth@ is left open so that the caller gets to decide what type
  421 -- of key this is. This is mostly for testing, in practice, seeds are used to
  422 -- represent root keys, and one should 'genMasterKeyFromXPrv'
  423 --
  424 -- The first argument is a type-family 'DerivationPath' and its type depends on

  428 --
  429 -- >>> liftXPrv rootPrv () prv
  430 -- _ :: Byron RootK XPrv

  434 --
  435 -- >>> liftXPrv rootPrv (minBound, minBound) prv
  436 -- _ :: Byron AddressK XPrv

  440     :: XPub -- ^ A root public key
  441     -> DerivationPath depth
  442     -> XPrv
  443     -> Byron depth XPrv
  444 liftXPrv rootPub derivationPath getKey = Byron
  445     { getKey
  446     , derivationPath
  447     , payloadPassphrase = hdPassphrase rootPub
  448     }

  476 -- The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  477 minSeedLengthBytes :: Int
  478 minSeedLengthBytes = 16
  479 
  480 -- Hash the seed entropy (generated from mnemonic) used to initiate a HD
  481 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  482 -- minimum for 'generate' (32 bytes).
  483 --
  484 -- Note that our current implementation deviates from BIP-39 because we use a
  485 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  486 --
  487 -- There are two methods of hashing the seed entropy, for different use cases.
  488 --
  489 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  490 --    Blake2b_256, inside a double CBOR serialization sandwich.
  491 --
  492 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  493 --    Blake2b_256, without any serialization.
  494 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  495 hashSeed = serialize . blake2b256 . serialize
  496   where
  497     serialize = BA.convert . cbor . BA.convert
  498     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  499 
  500 -- Hash a byte string through blake2b 256
  501 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  502 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  503 
  504 -- Derive a symmetric key for encrypting and authenticating the address
  505 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  506 -- is employed.
  507 hdPassphrase :: XPub -> ScrubbedBytes
  508 hdPassphrase masterKey =
  509     PBKDF2.generate
  510     (PBKDF2.prfHMAC SHA512)
  511     (PBKDF2.Parameters 500 32)
  512     (xpubToBytes masterKey)
  513     ("address-hashing" :: ByteString)
