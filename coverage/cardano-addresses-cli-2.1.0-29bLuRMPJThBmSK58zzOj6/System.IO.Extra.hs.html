   72 -- | Read some bytes from the console, and decode them if the encoding is recognized.
   73 hGetBytes :: Handle -> IO ByteString
   74 hGetBytes h = do
   75     raw <- B8.filter noNewline <$> B8.hGetContents h
   76     case detectEncoding (T.unpack $ T.decodeUtf8 raw) of
   77         Just (EBase16  ) -> decode fromBase16 raw
   78         Just (EBech32{}) -> decode (fromBech32 markCharsRedAtIndices) raw
   79         Just (EBase58  ) -> decode fromBase58 raw
   80         Nothing          -> fail
   81             "Couldn't detect input encoding? Data on stdin must be encoded as \
   82             \bech16, bech32 or base58."
   83   where
   84     decode :: (bin -> Either String bin) -> bin -> IO bin
   85     decode from = either fail pure . from
   86 
   87 -- | Read some English mnemonic words from the console, or fail.
   88 hGetSomeMnemonic :: Handle -> IO SomeMnemonic
   89 hGetSomeMnemonic h = do
   90     wrds <- T.words . T.filter noNewline . T.decodeUtf8 <$> B8.hGetContents h
   91     case mkSomeMnemonic @'[ 9, 12, 15, 18, 21, 24 ] wrds of
   92         Left (MkSomeMnemonicError e) -> fail e
   93         Right mw -> pure mw
   94 
   95 -- | Read an encoded private key from the console, or fail.
   96 hGetXPrv :: Handle -> IO XPrv
   97 hGetXPrv h = do
   98     bytes <- hGetBytes h
   99     case xprvFromBytes bytes of
  100         Nothing  -> fail "Couldn't convert bytes into extended private key."
  101         Just key -> pure key
  102 
  103 -- | Read an encoded public key from the console, or fail.
  104 hGetXPub :: Handle -> IO XPub
  105 hGetXPub h = do
  106     bytes <- hGetBytes h
  107     case xpubFromBytes bytes of
  108         Nothing  -> fail "Couldn't convert bytes into extended public key."
  109         Just key -> pure key
  110 
  111 -- | Read either an encoded public or private key from the console, or fail.
  112 hGetXP__ :: Handle -> IO (Either XPub XPrv)
  113 hGetXP__ h = do
  114     bytes <- hGetBytes h
  115     case (xpubFromBytes bytes, xprvFromBytes bytes) of
  116         (Just xpub,         _) -> pure (Left  xpub)
  117         (_        , Just xprv) -> pure (Right xprv)
  118         _                      -> fail
  119             "Couldn't convert bytes into neither extended public or private keys."

  126 -- | Print bytes to the console with the given encoding.
  127 hPutBytes :: Handle -> ByteString -> Encoding -> IO ()
  128 hPutBytes h bytes =
  129     B8.hPutStr h . flip encode bytes

  135 noNewline :: Char -> Bool
  136 noNewline = (`notElem` ['\n', '\r'])
  137 
  138 -- | Fail with a colored red error message.
  139 prettyIOException :: IOException -> IO a
  140 prettyIOException e = do
  141     B8.hPutStrLn stderr $ T.encodeUtf8 $ T.pack $ show e
  142     exitFailure
  143 
  144 -- | Mark all characters from a given string as red (in a console).
  145 markCharsRedAtIndices :: Integral i => [i] -> String -> String
  146 markCharsRedAtIndices ixs = go 0 (sort $ nub ixs)
  147   where
  148     go _c [] [] = mempty
  149     go c (i:is) (s:ss)
  150         | c == i    = red ++ s:def ++ go (c + 1) is ss
  151         | otherwise = s : go (c + 1) (i:is) ss
  152     go _ [] ss = ss
  153     go _ _ [] = [] -- NOTE: Really an error case.
  154 
  155     red = setSGRCode [SetColor Foreground Vivid Red]
  156     def = setSGRCode [Reset]
