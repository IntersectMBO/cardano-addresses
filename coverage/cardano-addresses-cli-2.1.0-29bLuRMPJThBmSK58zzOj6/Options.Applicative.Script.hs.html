    1 {-# LANGUAGE DataKinds #-}

    8 module Options.Applicative.Script
    9     (
   10     -- ** Applicative Parser
   11       scriptArg
   12     ) where
   13 
   14 import Prelude
   15 
   16 import Cardano.Script
   17     ( Script (..) )
   18 import Cardano.ScriptParser
   19     ( scriptParser )
   20 import Options.Applicative
   21     ( Parser, argument, eitherReader, help, metavar )
   22 import Text.ParserCombinators.ReadP
   23     ( readP_to_S )
   24 
   25 
   26 scriptFromString :: String -> Either String Script
   27 scriptFromString str =
   28     case readP_to_S scriptParser str of
   29          [(multisig,_rest)] ->
   30                  Right multisig
   31          _ ->
   32              Left "Parsing of the script failed."
   33 
   34 scriptArg :: Parser Script
   35 scriptArg = argument (eitherReader scriptFromString) $ mempty
   36     <> metavar "SCRIPT"
   37     <> help
   38         "Script string."
