   44 newtype Cmd = Cmd
   45     { networkTag :: NetworkTag
   46     } deriving (Show)
   47 
   48 mod :: (Cmd -> parent) -> Mod CommandFields parent
   49 mod liftCmd = command "stake" $
   50     info (helper <*> fmap liftCmd parser) $ mempty
   51         <> progDesc "Create a stake address"
   52         <> header "Create a stake address \
   53             \that references a staking key (1-1)."
   54         <> footerDoc (Just $ vsep
   55             [ string "The public key is read from stdin."
   56             , string ""
   57             , string "Example:"
   58             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   59             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   60             , indent 2 $ string ""
   61             , indent 2 $ bold $ string "$ cat root.prv \\"
   62             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/2/0 > stake.prv"
   63             , indent 2 $ string ""
   64             , indent 2 $ bold $ string "$ cat stake.prv \\"
   65             , indent 4 $ bold $ string $ "| "<>progName<>" key public \\"
   66             , indent 4 $ bold $ string $ "| "<>progName<>" address stake --network-tag testnet"
   67             , indent 2 $ string "stake_test1uzp7swuxjx7wmpkkvat8kpgrmjl8ze0dj9lytn25qv2tm4g6n5c35"
   68             ])
   69   where
   70     parser = Cmd
   71         <$> networkTagOpt Shelley
   72 
   73 run :: Cmd -> IO ()
   74 run Cmd{networkTag} = do
   75     xpub <- hGetXPub stdin
   76     case (mkNetworkDiscriminant . fromIntegral . unNetworkTag) networkTag of
   77         Left e -> die (fmt $ build e)
   78         Right discriminant -> do
   79             let addr = Shelley.stakeAddress discriminant (Shelley.liftXPub xpub)
   80             B8.hPutStr stdout $ T.encodeUtf8 $ bech32With hrp addr
   81   where
   82     hrp | networkTag == shelleyTestnet = [humanReadablePart|stake_test|]
   83         | otherwise = [humanReadablePart|stake|]
