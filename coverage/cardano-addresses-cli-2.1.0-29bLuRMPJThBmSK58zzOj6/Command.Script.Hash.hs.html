    8 module Command.Script.Hash
    9     ( Cmd
   10     , mod
   11     , run
   12 
   13     ) where
   14 
   15 import Cardano.Script
   16     ( Script (..), toScriptHash, validateScript )
   17 import Codec.Binary.Bech32.TH
   18     ( humanReadablePart )
   19 import Options.Applicative
   20     ( CommandFields, Mod, command, footerDoc, header, helper, info, progDesc )
   21 import Options.Applicative.Encoding
   22     ( Encoding, encodingOpt )
   23 import Options.Applicative.Help.Pretty
   24     ( bold, indent, string, vsep )
   25 import Options.Applicative.Script
   26     ( scriptArg )
   27 import Prelude hiding
   28     ( mod )
   29 import System.IO
   30     ( stdout )
   31 import System.IO.Extra
   32     ( hPutBytes, progName )
   33 
   34 
   35 data Cmd = Cmd
   36     { encoding :: Encoding
   37     , script :: Script
   38     } deriving (Show)
   39 
   40 mod :: (Cmd -> parent) -> Mod CommandFields parent
   41 mod liftCmd = command "hash" $
   42     info (helper <*> fmap liftCmd parser) $ mempty
   43         <> progDesc "Create a script hash"
   44         <> header "Create a script hash \
   45             \that can be used in stake or payment address."
   46         <> footerDoc (Just $ vsep
   47             [ string "The script is taken as argument."
   48             , string ""
   49             , string "Example:"
   50             , indent 2 $ bold $ string "$ cat script.txt"
   51             , indent 2 $ bold $ string "$ all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f333]"
   52             , indent 2 $ bold $ string "$ cat script.txt \\"
   53             , indent 4 $ bold $ string $ "| "<>progName<>" script hash --base16 $(cat script.txt)\\"
   54             , indent 2 $ string "a015ae61075e25c3d9250bdcbc35c6557272127927ecf2a2d716e29f"
   55             ])
   56   where
   57     parser = Cmd
   58         <$> encodingOpt [humanReadablePart|script_hash|]
   59         <*> scriptArg
   60 
   61 run :: Cmd -> IO ()
   62 run Cmd{encoding, script} =
   63     if (validateScript script) then
   64         hPutBytes stdout (toScriptHash script) encoding
   65     else
   66         fail "The script is invalid."
