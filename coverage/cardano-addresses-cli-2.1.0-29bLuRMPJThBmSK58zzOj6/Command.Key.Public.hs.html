    8 module Command.Key.Public
    9     ( Cmd (..)
   10     , mod
   11     , run
   12     ) where
   13 
   14 import Prelude hiding
   15     ( mod )
   16 
   17 import Cardano.Address.Derivation
   18     ( toXPub, xpubToBytes )
   19 import Codec.Binary.Bech32.TH
   20     ( humanReadablePart )
   21 import Options.Applicative
   22     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   23 import Options.Applicative.Encoding
   24     ( Encoding, encodingOpt )
   25 import Options.Applicative.Help.Pretty
   26     ( string )
   27 import System.IO
   28     ( stdin, stdout )
   29 import System.IO.Extra
   30     ( hGetXPrv, hPutBytes )
   31 
   32 
   33 newtype Cmd = Public
   34     { encoding :: Encoding
   35     } deriving (Show)
   36 
   37 mod :: (Cmd -> parent) -> Mod CommandFields parent
   38 mod liftCmd = command "public" $
   39     info (helper <*> fmap liftCmd parser) $ mempty
   40         <> progDesc "Get the public counterpart of a private key"
   41         <> footerDoc (Just $ string $ mconcat
   42             [ "The private key is read from stdin."
   43             ])
   44   where
   45     parser = Public
   46         <$> encodingOpt [humanReadablePart|xpub|]
   47 
   48 run :: Cmd -> IO ()
   49 run Public{encoding} = do
   50     xprv <- hGetXPrv stdin
   51     let xpub = toXPub xprv
   52     hPutBytes stdout (xpubToBytes xpub) encoding
