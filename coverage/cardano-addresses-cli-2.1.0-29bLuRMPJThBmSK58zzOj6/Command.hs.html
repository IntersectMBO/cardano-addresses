   95 -- | Run a given command
   96 run :: CLI -> IO ()
   97 run = handle prettyIOException . \case
   98     RecoveryPhrase sub -> RecoveryPhrase.run sub
   99     Key sub -> Key.run sub
  100     Address sub -> Address.run sub
  101     Script sub -> Script.run sub
  102     Version -> Version.run
  103 
  104 -- | Parse command line options and arguments
  105 parse :: IO CLI
  106 parse = customExecParser (prefs showHelpOnEmpty) cli

  108 -- | Enable ANSI colors on Windows and correct output buffering
  109 setup :: IO ()
  110 setup =
  111     mapM_ hSetup [stderr, stdout, stdin]
  112   where
  113     hSetup :: Handle -> IO ()
  114     hSetup h = do
  115       void $ hSupportsANSIWithoutEmulation h
  116       hSetBuffering h NoBuffering
  117 
  118 -- | Force the locale text encoding to UTF-8. This is needed because the CLI
  119 -- prints UTF-8 characters regardless of the @LANG@ environment variable or any
  120 -- other settings.
  121 --
  122 -- On Windows the current console code page is changed to UTF-8.
  123 withUtf8Encoding :: IO a -> IO a
  124 withUtf8Encoding action = withCP65001 (setUtf8EncodingHandles >> action)
  125   where
  126     setUtf8EncodingHandles :: IO ()
  127     setUtf8EncodingHandles = do
  128         utf8' <- mkTextEncoding "UTF-8//TRANSLIT"
  129         mapM_ (`hSetEncoding` utf8') [stdin, stdout, stderr]
