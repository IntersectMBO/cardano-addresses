   32 
   33 delegationCredentialArg  :: String -> Parser (Credential 'DelegationK)
   34 delegationCredentialArg helpDoc = argument (eitherReader reader) $ mempty
   35     <> metavar "KEY || KEY HASH || SCRIPT HASH"
   36     <> help helpDoc
   37   where
   38     reader :: String -> Either String (Credential 'DelegationK)
   39     reader str =
   40        (DelegationFromKey . liftXPub <$> xpubReader allowedPrefixesForXPub str)
   41        `orElse`
   42        (DelegationFromKeyHash <$> keyhashReader (Delegation, allowedPrefixesForKeyHash) str)
   43        `orElse`
   44        (DelegationFromScript <$> scriptHashReader str)
   45        `orElse`
   46        Left "Couldn't parse delegation credentials. Neither a public key, a public key hash nor a script hash."
   47 
   48     -- TODO: Allow non-extended keys here.
   49     allowedPrefixesForXPub =
   50         [ CIP5.stake_xvk
   51         ]
   52     allowedPrefixesForKeyHash =
   53         [ CIP5.stake_vkh
   54         ]
