    7 module Command.Key.WalletId

   11     ) where

   14     ( mod )
   15 
   16 import Cardano.Address.Derivation
   17     ( hashWalletId, toXPub, xprvFromBytes, xpubToBytes )
   18 import Codec.Binary.Encoding
   19     ( AbstractEncoding (..) )
   20 import Control.Monad
   21     ( when )

   23     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )

   33 
   34 data Cmd = WalletId
   35     deriving (Show)
   36 
   37 mod :: (Cmd -> parent) -> Mod CommandFields parent
   38 mod liftCmd = command "walletid" $
   39     info (helper <*> fmap liftCmd parser) $ mempty
   40         <> progDesc "Shows the cardano-wallet wallet ID for a given key"
   41         <> footerDoc (Just $ string $ mconcat
   42             [ "A wallet ID is a 40-digit hexadecimal string derived "
   43             , "from the walletâ€™s mnemonic. It is used by the cardano-wallet "
   44             , "server to refer to specific wallets.\n\n"
   45             , "This key can be either an extended root key "
   46             , "(full multi-account wallets), or an extended account key "
   47             , "(single-account wallets).\n\n"
   48             , "Private or public extended keys are accepted -- they will have "
   49             , "the same wallet ID.\n\n"
   50             , "The bech32-encoded key is read from standard input.\n"
   51             ])
   52   where
   53     parser = pure WalletId
   54 
   55 run :: Cmd -> IO ()
   56 run WalletId = do
   57     (hrp, bytes) <- hGetBech32 stdin allowedPrefixes
   58     guardBytes hrp bytes
   59     hPutBytes stdout (hashWalletId $ payloadToHash hrp bytes) EBase16
   60   where
   61     allowedPrefixes =
   62         [ CIP5.root_xsk
   63         , CIP5.root_xvk
   64         , CIP5.root_shared_xsk
   65         , CIP5.root_shared_xvk
   66         , CIP5.acct_xvk
   67         , CIP5.acct_xsk
   68         , CIP5.acct_shared_xvk
   69         , CIP5.acct_shared_xsk
   70         ]
   71 
   72     payloadToHash hrp bs
   73         | hrp `elem` [CIP5.root_xsk, CIP5.root_shared_xsk, CIP5.acct_xsk, CIP5.acct_shared_xsk] =
   74               case xprvFromBytes bs of
   75                   Just xprv ->  xpubToBytes . toXPub $ xprv
   76                   Nothing -> "96-byte extended private key is invalid due to 'scalarDecodeLong' failure from cryptonite"
   77         | otherwise =
   78               bs
   79 
   80     guardBytes hrp bytes
   81         | hrp `elem` [CIP5.root_xsk, CIP5.root_shared_xsk, CIP5.acct_xsk, CIP5.acct_shared_xsk] = do
   82             when (BS.length bytes /= 96) $
   83                 fail "data should be a 96-byte private key."
   84 
   85         | otherwise = do
   86             when (BS.length bytes /= 64) $
   87                 fail "data should be a 32-byte public key with a 32-byte chain-code appended."
