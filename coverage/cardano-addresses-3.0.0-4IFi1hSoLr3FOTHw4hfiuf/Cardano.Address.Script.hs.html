   74 -- | A 'Script' type represents multi signature script. The script embodies conditions
   75 -- that need to be satisfied to make it valid.
   76 --
   77 -- @since 3.0.0
   78 data Script
   79     = RequireSignatureOf !KeyHash
   80     | RequireAllOf ![Script]
   81     | RequireAnyOf ![Script]
   82     | RequireSomeOf Word8 ![Script]
   83     deriving stock (Generic, Show, Eq)
   84 instance NFData Script
   85 
   86 -- | This function realizes what cardano-node's `Api.serialiseToCBOR script` realizes
   87 -- This is basically doing the symbolically following:
   88 -- toCBOR [0,multisigScript]
   89 --
   90 -- @since 3.0.0
   91 serializeScript :: Script -> ByteString
   92 serializeScript script =
   93     multisigTag <> CBOR.toStrictByteString (toCBOR script)
   94   where
   95     -- | Magic number representing the tag of the native multi-signature script
   96     -- language. For each script language included, a new tag is chosen.
   97     multisigTag :: ByteString
   98     multisigTag = "\00"
   99 
  100     toCBOR :: Script -> CBOR.Encoding
  101     toCBOR = \case
  102         RequireSignatureOf (KeyHash verKeyHash) ->
  103             encodeMultiscriptCtr 0 2 <> CBOR.encodeBytes verKeyHash
  104         RequireAllOf contents ->
  105             encodeMultiscriptCtr 1 2 <> encodeFoldable toCBOR contents
  106         RequireAnyOf contents ->
  107             encodeMultiscriptCtr 2 2 <> encodeFoldable toCBOR contents
  108         RequireSomeOf m contents -> mconcat
  109             [ encodeMultiscriptCtr 3 3
  110             , CBOR.encodeInt (fromInteger $ toInteger m)
  111             , encodeFoldable toCBOR contents
  112             ]
  113 
  114     encodeMultiscriptCtr :: Word -> Word -> CBOR.Encoding
  115     encodeMultiscriptCtr ctrIndex listLen =
  116         CBOR.encodeListLen listLen <> CBOR.encodeWord ctrIndex
  117 
  118     encodeFoldable :: (Foldable f) => (a -> CBOR.Encoding) -> f a -> CBOR.Encoding
  119     encodeFoldable encode' xs = wrapArray len contents
  120       where
  121         (len, contents) = foldl' go (0, mempty) xs
  122         go (!l, !enc) next = (l + 1, enc <> encode' next)
  123 
  124         wrapArray :: Word -> CBOR.Encoding -> CBOR.Encoding
  125         wrapArray len' contents'
  126             | len' <= 23 = CBOR.encodeListLen len' <> contents'
  127             | otherwise  = CBOR.encodeListLenIndef <> contents' <> CBOR.encodeBreak

  129 -- | Computes the hash of a given script, by first serializing it to CBOR.
  130 --
  131 -- @since 3.0.0
  132 toScriptHash :: Script -> ScriptHash
  133 toScriptHash = ScriptHash . hashCredential . serializeScript

  135 -- | A 'ScriptHash' type represents script hash. The hash is expected to have size of
  136 -- 28-byte.
  137 --
  138 -- @since 3.0.0
  139 newtype ScriptHash = ScriptHash { unScriptHash :: ByteString }
  140     deriving (Generic, Show, Eq)
  141 instance NFData ScriptHash
  142 
  143 -- | Construct an 'ScriptHash' from raw 'ByteString' (28 bytes).
  144 --
  145 -- @since 3.0.0
  146 scriptHashFromBytes :: ByteString -> Maybe ScriptHash
  147 scriptHashFromBytes bytes
  148     | BS.length bytes /= credentialHashSize = Nothing
  149     | otherwise = Just $ ScriptHash bytes

  151 -- | A 'KeyHash' type represents verification key hash that participate in building
  152 -- multi-signature script. The hash is expected to have size of 28-byte.
  153 --
  154 -- @since 3.0.0
  155 newtype KeyHash = KeyHash { unKeyHash :: ByteString }
  156     deriving (Generic, Show, Eq)
  157 instance NFData KeyHash
  158 
  159 -- | Construct an 'KeyHash' from raw 'ByteString' (28 bytes).
  160 --
  161 -- @since 3.0.0
  162 keyHashFromBytes :: ByteString -> Maybe KeyHash
  163 keyHashFromBytes bytes
  164     | BS.length bytes /= credentialHashSize = Nothing
  165     | otherwise = Just $ KeyHash bytes

  167 -- | Encode a 'KeyHash' to bech32 'Text', using @script_vkh@ as a human readable prefix.
  168 --
  169 -- @since 3.0.0
  170 keyHashToText :: KeyHash -> Text
  171 keyHashToText (KeyHash keyHash) =
  172     T.decodeUtf8 $ encode (EBech32 CIP5.script_vkh) keyHash
  173 
  174 -- | Construct a 'KeyHash' from 'Text'. Either hex encoded text or
  175 -- Bech32 encoded text with `script_vkh` hrp is expected. Also
  176 -- binary payload is expected to be composed of 28 bytes.
  177 --
  178 -- @since 3.0.0
  179 keyHashFromText :: Text -> Either ErrKeyHashFromText KeyHash
  180 keyHashFromText txt = do
  181     (hrp, dp) <- either
  182         (const $ Left ErrKeyHashFromTextInvalidString)
  183         Right
  184         (Bech32.decodeLenient txt)
  185 
  186     (guardHrp hrp $> dp)
  187         >>= maybeToRight ErrKeyHashFromTextWrongDataPart . Bech32.dataPartToBytes
  188         >>= maybeToRight ErrKeyHashFromTextWrongPayload . keyHashFromBytes
  189  where
  190     guardHrp hrp
  191         | hrp == CIP5.script_vkh = Right ()
  192         | otherwise = Left ErrKeyHashFromTextWrongHrp
  193 
  194 -- Possible errors when deserializing a key hash from text.
  195 --
  196 -- @since 3.0.0
  197 data ErrKeyHashFromText
  198     = ErrKeyHashFromTextInvalidString
  199     | ErrKeyHashFromTextWrongPayload
  200     | ErrKeyHashFromTextWrongHrp
  201     | ErrKeyHashFromTextWrongDataPart
  202     deriving (Show, Eq)
  203 
  204 -- Possible errors when deserializing a key hash from text.
  205 --
  206 -- @since 3.0.0
  207 prettyErrKeyHashFromText :: ErrKeyHashFromText -> String
  208 prettyErrKeyHashFromText = \case
  209     ErrKeyHashFromTextInvalidString ->
  210         "Invalid encoded string: must be bech32-encoded."
  211     ErrKeyHashFromTextWrongPayload ->
  212         "Verification key hash must contain exactly 28 bytes."
  213     ErrKeyHashFromTextWrongHrp ->
  214         "Invalid human-readable prefix: must be 'script_vkh'."
  215     ErrKeyHashFromTextWrongDataPart ->
  216         "Verification key hash is Bech32-encoded but has an invalid data part."

  246 instance ToJSON Script where
  247     toJSON (RequireSignatureOf keyHash) =
  248         String $ keyHashToText keyHash
  249     toJSON (RequireAllOf content) =
  250         object ["all" .= fmap toJSON content]
  251     toJSON (RequireAnyOf content) =
  252         object ["any" .= fmap toJSON content]
  253     toJSON (RequireSomeOf count scripts) =
  254         object ["some" .= object ["at_least" .= count, "from" .= scripts]]
  255 
  256 instance FromJSON Script where
  257     parseJSON v = parseKey v <|> parseAnyOf v  <|> parseAllOf v <|> parseAtLeast v
  258       where
  259         parseKey = withText "Script KeyHash" $
  260             either
  261                 (fail . prettyErrKeyHashFromText)
  262                 (pure . RequireSignatureOf)
  263                 . keyHashFromText
  264         parseAnyOf = withObject "Script AnyOf" $ \o ->
  265             RequireAnyOf <$> o .: "any"
  266         parseAllOf = withObject "Script AllOf" $ \o ->
  267             RequireAllOf <$> o .: "all"
  268         parseAtLeast = withObject "Script SomeOf" $ \o -> do
  269             some <- o .: "some"
  270             RequireSomeOf <$> some .: "at_least" <*> some .: "from"
