   52 -- | Run 'scriptParser' on string input.
   53 --
   54 -- @since 3.0.0
   55 scriptFromString :: String -> Either ErrValidateScript Script
   56 scriptFromString str =
   57     case readP_to_S scriptParser str of
   58          [(script, "")] -> validateScript script $> script
   59          _ -> Left Malformed
   60 
   61 -- | Validate a 'Script', semantically
   62 --
   63 -- @since 3.0.0
   64 validateScript :: Script -> Either ErrValidateScript ()
   65 validateScript = \case
   66     RequireSignatureOf (KeyHash bytes) -> do
   67         when (BS.length bytes /= credentialHashSize) $ Left WrongKeyHash
   68 
   69     RequireAllOf script -> do
   70         when (L.null script) $ Left EmptyList
   71         when (hasDuplicate script) $ Left DuplicateSignatures
   72         traverse_ validateScript script
   73 
   74     RequireAnyOf script -> do
   75         when (L.null script) $ Left EmptyList
   76         when (hasDuplicate script) $ Left DuplicateSignatures
   77         traverse_ validateScript script
   78 
   79     RequireSomeOf m script -> do
   80         when (m == 0) $ Left MZero
   81         when (length script < fromIntegral m) $ Left ListTooSmall
   82         when (hasDuplicate script) $ Left DuplicateSignatures
   83         traverse_ validateScript script
   84   where
   85     hasDuplicate xs = do
   86         length sigs /= length (L.nub sigs)
   87       where
   88         sigs = [ sig | RequireSignatureOf sig <- xs ]
   89 
   90 -- | Possible validation errors when validating a script
   91 --
   92 -- @since 3.0.0
   93 data ErrValidateScript
   94     = EmptyList
   95     | ListTooSmall
   96     | MZero
   97     | DuplicateSignatures
   98     | WrongKeyHash
   99     | Malformed
  100     deriving (Eq, Show)
  101 
  102 -- | Pretty-print a validation error.
  103 --
  104 -- @since 3.0.0
  105 prettyErrValidateScript
  106     :: ErrValidateScript
  107     -> String
  108 prettyErrValidateScript = \case
  109     EmptyList ->
  110         "The list inside a script is empty."
  111     MZero ->
  112         "The M in at_least cannot be 0."
  113     ListTooSmall ->
  114         "The list inside at_least cannot be less than M."
  115     DuplicateSignatures ->
  116         "The list inside a script has duplicate keys."
  117     WrongKeyHash ->
  118         "The hash of verification key is expected to have "<>show credentialHashSize<>" bytes."
  119     Malformed ->
  120         "Parsing of the script failed. The script should be composed of nested \
  121         \lists, and the verification keys should be either encoded as bech32."
  122

  124 -- | The script embodies combination of signing keys that need to be met to make
  125 -- it valid. We assume here that the script could
  126 -- delivered from standard input. The examples below are self-explanatory:
  127 --
  128 -- 1. requiring signature
  129 -- 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe
  130 --
  131 -- 2. 'any' for signature required
  132 -- any [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
  133 --
  134 -- 3. 'all' signatures required
  135 -- all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
  136 --
  137 -- 4. 'at_least' 1 signature required
  138 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]
  139 --
  140 -- 5. Nested script are supported
  141 -- at_least 1 [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3fe, all [3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1, 3c07030e36bfffe67e2e2ec09e5293d384637cd2f004356ef320f3f1]]
  142 --
  143 -- Parser is insensitive to whitespaces.
  144 --
  145 -- @since 3.0.0
  146 scriptParser :: ReadP Script
  147 scriptParser =
  148     requireAllOfParser <++
  149     requireAnyOfParser <++
  150     requireAtLeastOfParser <++
  151     requireSignatureOfParser
  152 
  153 requireSignatureOfParser :: ReadP Script
  154 requireSignatureOfParser = do
  155     P.skipSpaces
  156     verKeyStr <- P.munch1 (\c -> isDigit c || isLetter c || c == '_')
  157     case fromBech32 (T.pack verKeyStr) of
  158         Nothing -> fail "Invalid Bech32-encoded string."
  159         Just keyHash -> return $ toSignature keyHash
  160  where
  161     toSignature = RequireSignatureOf . KeyHash
  162     fromBech32 txt = do
  163         (_, dp) <- either (const Nothing) Just (Bech32.decodeLenient txt)
  164         Bech32.dataPartToBytes dp
  165 
  166 requireAllOfParser :: ReadP Script
  167 requireAllOfParser = do
  168     P.skipSpaces
  169     _identifier <- P.string "all"
  170     RequireAllOf <$> commonPart
  171 
  172 requireAnyOfParser :: ReadP Script
  173 requireAnyOfParser = do
  174     P.skipSpaces
  175     _identifier <- P.string "any"
  176     RequireAnyOf <$> commonPart
  177 
  178 requireAtLeastOfParser :: ReadP Script
  179 requireAtLeastOfParser = do
  180     P.skipSpaces
  181     _identifier <- P.string "at_least"
  182     RequireSomeOf <$> naturalParser <*> commonPart
  183 
  184 naturalParser :: ReadP Word8
  185 naturalParser = do
  186     P.skipSpaces
  187     fromInteger . read <$> P.munch1 isDigit
  188 
  189 commonPart :: ReadP [Script]
  190 commonPart = do
  191     P.skipSpaces
  192     _open <- P.string "["
  193     P.skipSpaces
  194     content <- P.sepBy scriptParser (P.string ",")
  195     P.skipSpaces
  196     _close <- P.string "]"
  197     P.skipSpaces
  198     return content
