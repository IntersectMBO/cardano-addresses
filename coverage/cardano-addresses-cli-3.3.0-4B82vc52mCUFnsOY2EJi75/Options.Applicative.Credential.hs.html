   29 delegationCredentialArg  :: String -> Parser (Credential 'DelegationK)
   30 delegationCredentialArg helpDoc = argument (eitherReader reader) $ mempty
   31     <> metavar "KEY || SCRIPT HASH"
   32     <> help helpDoc
   33   where
   34     reader :: String -> Either String (Credential 'DelegationK)
   35     reader str =
   36        (DelegationFromKey . liftXPub <$> xpubReader allowedPrefixes str)
   37        `orElse`
   38        (DelegationFromScript <$> scriptHashReader str)
   39        `orElse`
   40        Left "Couldn't parse delegation credentials. Neither a public key nor a script hash."
   41 
   42     -- TODO: Allow non-extended keys here.
   43     allowedPrefixes =
   44         [ CIP5.stake_xvk
   45         ]
   46 
   47     orElse :: Either e result -> Either e result -> Either e result
   48     orElse _1st@Right{} _2nd = _1st
   49     orElse _1st@Left{}  _2nd = _2nd
