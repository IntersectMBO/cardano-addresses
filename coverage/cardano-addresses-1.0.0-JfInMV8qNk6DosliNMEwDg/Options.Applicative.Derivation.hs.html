   52 --
   53 -- Derivation Path
   54 --
   55 
   56 -- | Represent a user-provided 'DerivationPath'.
   57 newtype DerivationPath = DerivationPath [DerivationIndex]
   58     deriving (Show, Eq)
   59 
   60 derivationPathFromString :: String -> Either String DerivationPath
   61 derivationPathFromString str =
   62     DerivationPath
   63         <$> mapM (derivationIndexFromString . T.unpack) (T.splitOn "/" txt)
   64   where
   65     txt = T.pack str
   66 
   67 derivationPathToString :: DerivationPath -> String
   68 derivationPathToString (DerivationPath xs) =
   69     intercalate "/" $ map derivationIndexToString xs
   70 
   71 castDerivationPath :: DerivationPath -> [Index 'WholeDomain depth]
   72 castDerivationPath (DerivationPath xs) = toEnum . fromEnum <$> xs
   73 
   74 derivationPathArg :: Parser DerivationPath
   75 derivationPathArg = argument (eitherReader derivationPathFromString) $ mempty
   76     <> metavar "DERIVATION-PATH"
   77     <> help
   78         "Slash-separated derivation path. Hardened indexes are marked with a \
   79         \'H' (44H/1815H/0H/0)."

   81 --
   82 -- Derivation Index
   83 --
   84 
   85 newtype DerivationIndex = DerivationIndex Word32
   86     deriving stock   (Show, Eq)
   87     deriving newtype (Bounded, Enum, Ord)
   88 
   89 -- | Safely cast a 'DerivationIndex' to an 'Integer'.
   90 indexToInteger :: DerivationIndex -> Integer
   91 indexToInteger (DerivationIndex ix) = fromIntegral ix
   92 
   93 -- | Get the first 'DerivationIndex' considered /hardened/.
   94 firstHardened :: DerivationIndex
   95 firstHardened = DerivationIndex 0x80000000
   96 
   97 -- | Smart-constructor for a 'DerivationIndex'
   98 mkDerivationIndex :: Integer -> Either String DerivationIndex
   99 mkDerivationIndex ix
  100     | ix > fromIntegral (maxBound @Word32) =
  101         Left $ show ix <> " is too high to be a derivation index."
  102     | otherwise =
  103         pure $ DerivationIndex $ fromIntegral ix
  104 
  105 -- | Convert a string to a derivation index. String must be followed by a
  106 -- capital /H/ to mark hardened index. For example @0@ refers to the first soft
  107 -- index, whereas @0H@ refers to the first hardened index.
  108 derivationIndexFromString :: String -> Either String DerivationIndex
  109 derivationIndexFromString "" = Left "An empty string is not a derivation index!"
  110 derivationIndexFromString str
  111     | "H" `isSuffixOf` str = do
  112         parseHardenedIndex (init str)
  113     | otherwise = do
  114         parseSoftIndex str
  115   where
  116     parseHardenedIndex txt = do
  117         ix <- left (const msg) $ readEitherSafe txt
  118         mkDerivationIndex $ ix + indexToInteger firstHardened
  119       where
  120         msg = mconcat
  121             [ "Unable to parse hardened index. Hardened indexes are integer "
  122             , "values, between "
  123             , show (indexToInteger (minBound @DerivationIndex))
  124             , " and "
  125             , show (indexToInteger firstHardened)
  126             , " ending with a capital 'H'. For example: \"42H\","
  127             ]
  128 
  129     parseSoftIndex txt = do
  130         ix <- left (const msg) $ readEitherSafe txt
  131         guardSoftIndex ix
  132         mkDerivationIndex ix
  133       where
  134         guardSoftIndex ix
  135             | ix >= indexToInteger firstHardened =
  136                 Left $ mconcat
  137                     [ show ix
  138                     , " is too high to be a soft derivation index. "
  139                     , "Did you mean \""
  140                     , show (ix - indexToInteger firstHardened)
  141                     , "H\"?"
  142                     ]
  143             | otherwise =
  144                 pure ()
  145 
  146         msg = mconcat
  147             [ "Unable to parse soft index. Soft indexes are integer "
  148             , "values, between "
  149             , show (indexToInteger (minBound @DerivationIndex))
  150             , " and "
  151             , show (indexToInteger firstHardened)
  152             , ". For example: \"14\"."
  153             ]
  154 
  155 -- | Convert a 'DerivationIndex' back to string.
  156 derivationIndexToString :: DerivationIndex -> String
  157 derivationIndexToString ix_@(DerivationIndex ix)
  158     | ix_ >= firstHardened = show ix' ++ "H"
  159     | otherwise            = show ix
  160   where
  161     ix' = fromIntegral ix - indexToInteger firstHardened
