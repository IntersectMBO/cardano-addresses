    7 module Command.Key.FromRecoveryPhrase
    8     ( Cmd (..)
    9     , mod
   10     , run
   11     ) where
   12 
   13 import Prelude hiding
   14     ( mod )
   15 
   16 import Cardano.Address.Derivation
   17     ( xprvToBytes )
   18 import Codec.Binary.Bech32.TH
   19     ( humanReadablePart )
   20 import Options.Applicative
   21     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   22 import Options.Applicative.Encoding
   23     ( Encoding, encodingOpt )
   24 import Options.Applicative.Help.Pretty
   25     ( string )
   26 import Options.Applicative.Style
   27     ( Style, generateRootKey, styleArg )
   28 import System.IO
   29     ( stdin, stdout )
   30 import System.IO.Extra
   31     ( hGetSomeMnemonic, hPutBytes )
   32 
   33 
   34 data Cmd = FromRecoveryPhrase
   35     { encoding :: Encoding
   36     , style :: Style
   37     } deriving (Show)
   38 
   39 mod :: (Cmd -> parent) -> Mod CommandFields parent
   40 mod liftCmd = command "from-recovery-phrase" $
   41     info (helper <*> fmap liftCmd parser) $ mempty
   42         <> progDesc "Convert a recovery phrase to an extended private key."
   43         <> footerDoc (Just $ string $ mconcat
   44             [ "The recovery phrase is read from stdin."
   45             , "\n\n"
   46             , "Example:\n\n"
   47             , "  $ cardano-address recovery-phrase generate \\\n"
   48             , "  | cardano-address key from-recovery-phrase icarus \\\n"
   49             ])
   50   where
   51     parser = FromRecoveryPhrase
   52         <$> encodingOpt [humanReadablePart|xprv|]
   53         <*> styleArg
   54 
   55 run :: Cmd -> IO ()
   56 run FromRecoveryPhrase{encoding,style} = do
   57     someMnemonic <- hGetSomeMnemonic stdin
   58     rootK <- generateRootKey someMnemonic style
   59     hPutBytes stdout (xprvToBytes rootK) encoding
