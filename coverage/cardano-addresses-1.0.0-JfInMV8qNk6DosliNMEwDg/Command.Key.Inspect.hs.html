    6 module Command.Key.Inspect
    7     ( Cmd (..)
    8     , mod
    9     , run
   10     ) where
   11 
   12 import Prelude hiding
   13     ( mod )
   14 
   15 import Cardano.Address.Derivation
   16     ( XPrv, XPub, xprvChainCode, xprvPrivateKey, xpubChainCode, xpubPublicKey )
   17 import Options.Applicative
   18     ( CommandFields, Mod, command, footerDoc, helper, info, progDesc )
   19 import Options.Applicative.Encoding
   20     ( AbstractEncoding (..) )
   21 import Options.Applicative.Help.Pretty
   22     ( string )
   23 import System.IO
   24     ( stdin, stdout )
   25 import System.IO.Extra
   26     ( hGetXP__, hPutBold, hPutBytes )
   27 
   28 import qualified Data.ByteString.Char8 as B8
   29 
   30 
   31 data Cmd = Inspect
   32     deriving (Show)
   33 
   34 mod :: (Cmd -> parent) -> Mod CommandFields parent
   35 mod liftCmd = command "inspect" $
   36     info (helper <*> fmap liftCmd parser) $ mempty
   37         <> progDesc "Show information about a key"
   38         <> footerDoc (Just $ string $ mconcat
   39             [ "The parent key is read from stdin."
   40             ])
   41   where
   42     parser = pure Inspect
   43 
   44 run :: Cmd -> IO ()
   45 run Inspect = do
   46     either inspectXPub inspectXPrv =<< hGetXP__ stdin
   47   where
   48     newline = B8.hPutStrLn stdout ""
   49 
   50     inspectXPub :: XPub -> IO ()
   51     inspectXPub xpub = do
   52         hPutBold stdout "key type:     " *> B8.hPutStr stdout "public"    *> newline
   53         hPutBold stdout "extended key: " *> hPutBytes  stdout pub EBase16 *> newline
   54         hPutBold stdout "chain code:   " *> hPutBytes  stdout cc  EBase16 *> newline
   55       where
   56         pub = xpubPublicKey xpub
   57         cc  = xpubChainCode xpub
   58 
   59     inspectXPrv :: XPrv -> IO ()
   60     inspectXPrv xprv = do
   61         hPutBold stdout "key type:     " *> B8.hPutStr stdout "private"   *> newline
   62         hPutBold stdout "extended key: " *> hPutBytes  stdout prv EBase16 *> newline
   63         hPutBold stdout "chain code:   " *> hPutBytes  stdout cc  EBase16 *> newline
   64       where
   65         prv = xprvPrivateKey xprv
   66         cc  = xprvChainCode  xprv
