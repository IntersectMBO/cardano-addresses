   38 newtype Cmd = Cmd
   39     { networkTag :: NetworkTag
   40     } deriving (Show)
   41 
   42 mod :: (Cmd -> parent) -> Mod CommandFields parent
   43 mod liftCmd = command "payment" $
   44     info (helper <*> fmap liftCmd parser) $ mempty
   45         <> progDesc "Create a payment address"
   46         <> header "Payment addresses carry no delegation rights whatsoever."
   47         <> footerDoc (Just $ string $ mconcat
   48             [ "Example:\n"
   49             , "  $ cardano-address recovery-phrase generate --size 15 \\\n"
   50             , "  | cardano-address key from-recovery-phrase Shelley > root.prv\n"
   51             , "\n"
   52             , "  $ cat root.prv \\\n"
   53             , "  | cardano-address key child 1852H/1815H/0H/0/0 > addr.prv\n"
   54             , "\n"
   55             , "  $ cat addr.prv \\\n"
   56             , "  | cardano-address key public \\\n"
   57             , "  | cardano-address address payment --network-tag 0\n"
   58             , "\n"
   59             , "  addr1vrcmygdgp7v3mhz78v8kdsfru0y9wysnr9pgvvgmdqx2w0qrg8swg"
   60             ])
   61   where
   62     parser = Cmd
   63         <$> networkTagOpt Shelley
   64 
   65 run :: Cmd -> IO ()
   66 run Cmd{networkTag} = do
   67     xpub <- hGetXPub stdin
   68     case (mkNetworkDiscriminant . fromIntegral . unNetworkTag) networkTag of
   69         Left ErrWrongNetworkTag{} -> do
   70             fail "Invalid network tag. Must be between [0, 15]"
   71         Right discriminant -> do
   72             let addr = Shelley.paymentAddress discriminant (Shelley.liftXPub xpub)
   73             B8.hPutStr stdout $ T.encodeUtf8 $ bech32 addr
