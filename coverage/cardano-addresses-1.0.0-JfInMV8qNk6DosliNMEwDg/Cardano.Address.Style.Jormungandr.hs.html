  111 -- $overview
  112 --
  113 -- This module provides an implementation of:
  114 --
  115 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Jormungandr master keys from mnemonic sentences
  116 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical hard derivation of parent to child keys
  117 -- - 'Cardano.Address.Derivation.SoftDerivation': for hierarchical soft derivation of parent to child keys
  118 -- - 'Cardano.Address.PaymentAddress': for constructing payment addresses from a address public key
  119 -- - 'Cardano.Address.DelegationAddress': for constructing payment addresses from a address and stake public keys
  120 --
  121 -- == Disclaimer
  122 --
  123 -- Beware that this format of address have been specially crafted for the
  124 -- incentivized testnet (ITN). This will soon be __deprecated__ in favor of
  125 -- 'Cardano.Address.Style.Shelley.Shelley'.
  126 --
  127 -- __New integrations should not use this address style__.

  129 -- | A cryptographic key for sequential-scheme address derivation, with
  130 -- phantom-types to disambiguate key types.
  131 --
  132 -- @
  133 -- let rootPrivateKey = Jormungandr 'RootK XPrv
  134 -- let accountPubKey  = Jormungandr 'AccountK XPub
  135 -- let addressPubKey  = Jormungandr 'AddressK XPub
  136 -- @
  137 --
  138 -- @since 2.0.0
  139 newtype Jormungandr (depth :: Depth) key = Jormungandr
  140     { getKey :: key
  141         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  142         --
  143         -- @since 2.0.0
  144     }
  145     deriving stock (Generic, Show, Eq)
  146 
  147 deriving instance (Functor (Jormungandr depth))
  148 instance (NFData key) => NFData (Jormungandr depth key)

  151 -- Key Derivation
  152 --
  153 -- $keyDerivation
  154 --
  155 -- === Generating a root key from 'SomeMnemonic'
  156 -- > :set -XOverloadedStrings
  157 -- > :set -XTypeApplications
  158 -- > :set -XDataKinds
  159 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  160 -- >
  161 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  162 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  163 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Jormungandr 'RootK XPrv
  164 --
  165 -- === Deriving child keys
  166 --
  167 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'\/0\/14@
  168 --
  169 -- > import Cardano.Address.Derivation ( AccountingStyle(..) )
  170 -- >
  171 -- > let accIx = toEnum 0x80000000
  172 -- > let acctK = deriveAccountPrivateKey rootK accIx
  173 -- >
  174 -- > let addIx = toEnum 0x00000014
  175 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  176 --
  177 -- > let stakeK = deriveStakingPrivateKey acctK
  178 
  179 instance Internal.GenMasterKey Jormungandr where
  180     type SecondFactor Jormungandr = ScrubbedBytes
  181 
  182     genMasterKeyFromXPrv = liftXPrv
  183     genMasterKeyFromMnemonic fstFactor sndFactor =
  184         Jormungandr $ generateNew seedValidated sndFactor
  185         where
  186             seed  = someMnemonicToBytes fstFactor
  187             seedValidated = assert
  188                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  189                 seed

  191 instance Internal.HardDerivation Jormungandr where
  192     type AccountIndexDerivationType Jormungandr = 'Hardened
  193     type AddressIndexDerivationType Jormungandr = 'Soft
  194     type WithAccountStyle Jormungandr = AccountingStyle
  195 
  196     deriveAccountPrivateKey (Jormungandr rootXPrv) accIx =
  197         let
  198             purposeIx =
  199                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  200             coinTypeIx =
  201                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  202             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  203                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  204             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  205                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  206             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  207                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  208         in
  209             Jormungandr acctXPrv
  210 
  211     deriveAddressPrivateKey (Jormungandr accXPrv) accountingStyle addrIx =
  212         let
  213             changeCode =
  214                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  215             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  216                 deriveXPrv DerivationScheme2 accXPrv changeCode
  217             addrXPrv = -- lvl5 derivation; soft derivation of address index
  218                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  219         in
  220             Jormungandr addrXPrv
  221 
  222 instance Internal.SoftDerivation Jormungandr where
  223     deriveAddressPublicKey (Jormungandr accXPub) accountingStyle addrIx =
  224         fromMaybe errWrongIndex $ do
  225             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  226             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  227                 deriveXPub DerivationScheme2 accXPub changeCode
  228             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  229                 deriveXPub DerivationScheme2 changeXPub addrIx
  230             return $ Jormungandr addrXPub
  231       where
  232         errWrongIndex = error $
  233             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  234             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  235             \either a programmer error, or, we may have reached the maximum \
  236             \number of addresses for a given wallet."
  237 
  238 instance Internal.StakingDerivation Jormungandr where
  239     deriveStakingPrivateKey (Jormungandr accXPrv) =
  240         let
  241             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  242                 deriveXPrv DerivationScheme2 accXPrv (toEnum @(Index 'Soft _) 2)
  243             stakeXPrv = -- lvl5 derivation; soft derivation of address index
  244                 deriveXPrv DerivationScheme2 changeXPrv (minBound @(Index 'Soft _))
  245         in
  246             Jormungandr stakeXPrv

  343 instance Internal.PaymentAddress Jormungandr where
  344     paymentAddress discrimination k = unsafeMkAddress $
  345         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  346             putWord8 (invariantNetworkTag 255 firstByte)
  347             putByteString (xpubPublicKey $ getKey k)
  348       where
  349           firstByte = networkTag @Jormungandr discrimination
  350           expectedLength = 1 + publicKeySize
  351 
  352 instance Internal.DelegationAddress Jormungandr where
  353     delegationAddress discrimination paymentKey stakingKey = unsafeMkAddress $
  354         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  355             putWord8 (invariantNetworkTag 255 $ NetworkTag $ firstByte + 1)
  356             putByteString . xpubPublicKey . getKey $ paymentKey
  357             putByteString . xpubPublicKey . getKey $ stakingKey
  358       where
  359           (NetworkTag firstByte) = networkTag @Jormungandr discrimination
  360           expectedLength = 1 + 2*publicKeySize

  362 -- | Analyze an 'Address' to know whether it's a Jormungandr address or not.
  363 --
  364 -- Returns 'Nothing' if it's not a valid Shelley address, or a ready-to-print
  365 -- string giving details about the 'Jormungandr'.
  366 --
  367 -- @since 2.0.0
  368 inspectJormungandrAddress :: Address -> Maybe String
  369 inspectJormungandrAddress addr
  370     | BS.length bytes < 1 + publicKeySize = Nothing
  371     | otherwise =
  372         let
  373             (fstByte, rest) = first BS.head $ BS.splitAt 1 bytes
  374             addrType = fstByte .&. 0b01111111
  375             network  = (fstByte .&. 0b10000000) `shiftR` 7
  376             size = publicKeySize
  377         in
  378             case addrType of
  379                 0x03 | BS.length rest == size ->
  380                     Just $ unlines
  381                     [ "address style:    " <> "Jormungandr"
  382                     , "address type:     " <> "single"
  383                     , "stake reference:  " <> "none"
  384                     , "spending key:     " <> base16 (BS.take size rest)
  385                     , "network tag:      " <> show network
  386                     ]
  387                 0x04 | BS.length rest == 2 * size ->
  388                     Just $ unlines
  389                     [ "address style:    " <> "Jormungandr"
  390                     , "address type:     " <> "group"
  391                     , "stake reference:  " <> "by value"
  392                     , "spending key:     " <> base16 (BS.take size rest)
  393                     , "stake key:        " <> base16 (BS.drop size rest)
  394                     , "network tag:      " <> show network
  395                     ]
  396                 0x05 | BS.length rest == size ->
  397                     Just $ unlines
  398                     [ "address style:    " <> "Jormungandr"
  399                     , "address type:     " <> "account"
  400                     , "stake reference:  " <> "none"
  401                     , "account key:      " <> base16 (BS.take size rest)
  402                     , "network tag:      " <> show network
  403                     ]
  404                 0x06 | BS.length rest == size ->
  405                     Just $ unlines
  406                     [ "address style:    " <> "Jormungandr"
  407                     , "address type:     " <> "multisig"
  408                     , "stake reference:  " <> "none"
  409                     , "merkle root:      " <> base16 (BS.take size rest)
  410                     , "network tag:      " <> show network
  411                     ]
  412                 _ ->
  413                     Nothing
  414   where
  415     bytes = unAddress addr
  416     base16 = T.unpack . T.decodeUtf8 . encode EBase16

  493 -- Purpose is a constant set to 1852' (or 0x8000073c) following the BIP-44
  494 -- extension for Cardano:
  495 --
  496 -- https://github.com/input-output-hk/implementation-decisions/blob/e2d1bed5e617f0907bc5e12cf1c3f3302a4a7c42/text/1852-hd-chimeric.md
  497 --
  498 -- It indicates that the subtree of this node is used according to this
  499 -- specification.
  500 --
  501 -- Hardened derivation is used at this level.
  502 purposeIndex :: Word32
  503 purposeIndex = 0x8000073c
  504 
  505 -- One master node (seed) can be used for unlimited number of independent
  506 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  507 -- same space for various cryptocoins has some disadvantages.
  508 --
  509 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  510 -- addresses across cryptocoins and improving privacy issues.
  511 --
  512 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  513 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  514 -- Lovelace.
  515 --
  516 -- Hardened derivation is used at this level.
  517 coinTypeIndex :: Word32
  518 coinTypeIndex = 0x80000717
