   41 newtype Cmd = Cmd
   42     {  networkTag :: NetworkTag
   43     } deriving (Show)
   44 
   45 mod :: (Cmd -> parent) -> Mod CommandFields parent
   46 mod liftCmd = command "stake" $
   47     info (helper <*> fmap liftCmd parser) $ mempty
   48         <> progDesc "Create a stake address"
   49         <> header "Create a stake address \
   50             \that references a delegation key (1-1)."
   51         <> footerDoc (Just $ vsep
   52             [ string "The public key is read from stdin."
   53             , string ""
   54             , string "Example:"
   55             , indent 2 $ bold $ string $ "$ "<>progName<>" recovery-phrase generate --size 15 \\"
   56             , indent 4 $ bold $ string $ "| "<>progName<>" key from-recovery-phrase Shelley > root.prv"
   57             , indent 2 $ string ""
   58             , indent 2 $ bold $ string "$ cat root.prv \\"
   59             , indent 4 $ bold $ string $ "| "<>progName<>" key child 1852H/1815H/0H/2/0 > stake.prv"
   60             , indent 2 $ string ""
   61             , indent 2 $ bold $ string "$ cat stake.prv \\"
   62             , indent 4 $ bold $ string $ "| "<>progName<>" key public --with-chain-code \\"
   63             , indent 4 $ bold $ string $ "| "<>progName<>" address stake --network-tag testnet"
   64             , indent 2 $ string "stake_test1uzp7swuxjx7wmpkkvat8kpgrmjl8ze0dj9lytn25qv2tm4g6n5c35"
   65             ])
   66   where
   67     parser = Cmd
   68         <$> networkTagOpt Shelley
   69 
   70 run :: Cmd -> IO ()
   71 run Cmd{networkTag} = do
   72     discriminant <- fromNetworkTag networkTag
   73     (hrp, bytes) <- hGetBech32 stdin allowedPrefixes
   74     addr <- stakeAddressFromBytes discriminant bytes hrp
   75     hPutBytes stdout (unAddress addr) (EBech32 stakeHrp)
   76   where
   77     stakeHrp
   78         | networkTag == shelleyTestnet = CIP5.stake_test
   79         | otherwise = CIP5.stake
   80 
   81     -- TODO: Also allow `XXX_vk` prefixes. We don't need the chain code to
   82     -- construct a payment credential. This will however need some additional
   83     -- abstraction over `xpubFromBytes` but I've done enough yake-shaving at
   84     -- this stage, so leaving this as an item for later.
   85     allowedPrefixes =
   86         [ CIP5.stake_xvk
   87         , CIP5.stake_vkh
   88         , CIP5.script
   89         ]
   90 
   91     stakeAddressFromBytes discriminant bytes hrp
   92         | hrp == CIP5.script = do
   93             case scriptHashFromBytes bytes of
   94                 Nothing ->
   95                     fail "Couldn't convert bytes into script hash."
   96                 Just h  -> do
   97                     let credential = DelegationFromScript h
   98                     pure $ unsafeFromRight $ Shelley.stakeAddress discriminant credential
   99 
  100         | hrp == CIP5.stake_vkh = do
  101             case keyHashFromBytes (Delegation, bytes) of
  102                 Nothing  ->
  103                     fail "Couldn't convert bytes into delegation key hash."
  104                 Just keyhash -> do
  105                     let credential = DelegationFromKeyHash keyhash
  106                     pure $ unsafeFromRight $ Shelley.stakeAddress discriminant credential
  107 
  108         | otherwise = do
  109             case xpubFromBytes bytes of
  110                 Nothing  ->
  111                     fail "Couldn't convert bytes into extended public key."
  112                 Just key -> do
  113                     let credential = DelegationFromKey $ Shelley.liftXPub key
  114                     pure $ unsafeFromRight $ Shelley.stakeAddress discriminant credential
